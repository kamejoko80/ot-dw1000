324 void NetworkData::RemoveTemporaryData(uint8_t *aData, uint8_t &aDataLength, PrefixTlv &aPrefix)
325 {
326  NetworkDataTlv *cur = aPrefix.GetSubTlvs();
327  NetworkDataTlv *end;
328  BorderRouterTlv *borderRouter;
329  HasRouteTlv *hasRoute;
330  ContextTlv *context;
331  BorderRouterEntry *borderRouterEntry;
332  HasRouteEntry *hasRouteEntry;
333  uint8_t length;
334  uint8_t contextId;
335  uint8_t *dst;
336  uint8_t *src;
337 
338  while (1)
339  {
340  end = aPrefix.GetNext();
341 
342  if (cur >= end)
343  {
344  break;
345  }
346 
347  if (cur->IsStable())
348  {
349  switch (cur->GetType())
350  {
351  case NetworkDataTlv::kTypeBorderRouter:
352  {
353  borderRouter = FindBorderRouter(aPrefix);
354 
355  if ((context = FindContext(aPrefix)) == NULL)
356  {
357  break;
358  }
359 
360  contextId = context->GetContextId();
361 
362  // replace p_border_router_16
363  for (uint8_t i = 0; i < borderRouter->GetNumEntries(); i++)
364  {
365  borderRouterEntry = borderRouter->GetEntry(i);
366 
367  if (borderRouterEntry->IsDhcp() || borderRouterEntry->IsConfigure())
368  {
369  borderRouterEntry->SetRloc(0xfc00 | contextId);
370  }
371  else
372  {
373  borderRouterEntry->SetRloc(0xfffe);
374  }
375  }
376 
377  break;
378  }
379 
380  case NetworkDataTlv::kTypeHasRoute:
381  {
382  hasRoute = FindHasRoute(aPrefix);
383 
384  // replace r_border_router_16
385  for (uint8_t j = 0; j < hasRoute->GetNumEntries(); j++)
386  {
387  hasRouteEntry = hasRoute->GetEntry(j);
388  hasRouteEntry->SetRloc(0xfffe);
389  }
390 
391  break;
392  }
393 
394  default:
395  {
396  break;
397  }
398  }
399 
400  // keep stable tlv
401  cur = cur->GetNext();
402  }
403  else
404  {
405  // remove temporary tlv
406  length = sizeof(NetworkDataTlv) + cur->GetLength();
407  dst = reinterpret_cast<uint8_t *>(cur);
408  src = reinterpret_cast<uint8_t *>(cur->GetNext());
409  memmove(dst, src, aDataLength - static_cast<size_t>(src - aData));
410  aPrefix.SetSubTlvsLength(aPrefix.GetSubTlvsLength() - length);
411  aDataLength -= length;
412  }
413  }
414 }

548 int8_t NetworkData::PrefixMatch(const uint8_t *a, const uint8_t *b, uint8_t aLength)
549 {
550  int8_t rval = 0;
551  uint8_t bytes = BitVectorBytes(aLength);
552  uint8_t diff;
553 
554  for (uint8_t i = 0; i < bytes; i++)
555  {
556  diff = a[i] ^ b[i];
557 
558  if (diff == 0)
559  {
560  rval += 8;
561  }
562  else
563  {
564  while ((diff & 0x80) == 0)
565  {
566  rval++;
567  diff <<= 1;
568  }
569 
570  break;
571  }
572  }
573 
574  return (rval >= aLength) ? rval : -1;
575 }
