64 ThreadError otPlatUartEnable(void)

68  while (PRCMPowerDomainStatus(PRCM_DOMAIN_SERIAL) != PRCM_DOMAIN_POWER_ON);

128 static void processReceive(void)
129 {
130  while (sReceiveHeadIdx != sReceiveTailIdx)
131  {
132  uint16_t tailIdx;
133 
134  if (sReceiveHeadIdx < sReceiveTailIdx)
135  {
136  tailIdx = sReceiveTailIdx;
137  otPlatUartReceived(&(sReceiveBuffer[sReceiveHeadIdx]), tailIdx - sReceiveHeadIdx);
138  sReceiveHeadIdx = tailIdx;
139  }
140  else
141  {
142  tailIdx = CC2650_RECV_CIRC_BUFF_SIZE;
143  otPlatUartReceived(&(sReceiveBuffer[sReceiveHeadIdx]), tailIdx - sReceiveHeadIdx);
144  sReceiveHeadIdx = 0;
145  }
146  }
147 }

152 static void processTransmit(void)
153 {
154  VerifyOrExit(sSendBuffer != NULL, ;);
155 
156  for (; sSendLen > 0; sSendLen--)
157  {
158  UARTCharPut(UART0_BASE, *sSendBuffer);

160  }
161 
162  sSendBuffer = NULL;
163  sSendLen = 0;
164  otPlatUartSendDone();
165 
166 exit:
167  return;
