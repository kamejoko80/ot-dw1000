77 void LinkQualityInfo::AddRss(LinkQualityInfo &aNoiseFloor, int8_t anRss)
78 {
79  uint16_t newValue;
80  uint16_t oldAverage;
81 
82  // Restrict/Cap the RSS value to the closed range [0, -128] so the value can fit in 8 bits.
83 
84  if (anRss > 0)
85  {
86  anRss = 0;
87  }
88 
89  // Multiply the the RSS value by a precision multiple (currently -8).
90 
91  newValue = static_cast<uint16_t>(-anRss);
92  newValue <<= kRssAveragePrecisionMultipleBitShift;
93 
94  oldAverage = mRssAverage;
95 
96  if (mCount >= kRssCountForWeightCoefficientOneEighth)
97  {
98  // New average = old average * 7/8 + new value * 1/8
99  mRssAverage = static_cast<uint16_t>(((oldAverage << 3) - oldAverage + newValue) >> 3);
100  }
101  else if (mCount >= kRssCountForWeightCoefficientOneFourth)
102  {
103  // New average = old average * 3/4 + new value * 1/4
104  mRssAverage = static_cast<uint16_t>(((oldAverage << 2) - oldAverage + newValue) >> 2);
105  }
106  else if (mCount >= kRssCountForWeightCoefficientOneHalf)
107  {
108  // New average = old average * 1/2 + new value * 1/2
109  mRssAverage = (oldAverage + newValue) >> 1;
110  }
111  else
112  {
113  mRssAverage = newValue;
114  }
115 
116  if (mCount < kRssCountMax)
117  {
118  mCount++;
119  }
120 
121  UpdateLinkQuality(aNoiseFloor);
122 }

124 int8_t LinkQualityInfo::GetAverageRss(void) const
125 {
126  int8_t average = kUnknownRss;
127 
128  if (mCount != 0)
129  {
130  average = -static_cast<int8_t>(mRssAverage >> kRssAveragePrecisionMultipleBitShift);
131 
132  // Check for round up (e.g. average of -71.5 --> -72)
133 
134  if ((mRssAverage & kRssAveragePrecisionMultipleBitMask) >= (kRssAveragePrecisionMultiple >> 1))
135  {
136  average--;
137  }
138  }
139 
140  return average;
141 }

148 ThreadError LinkQualityInfo::GetAverageRssAsString(char *aCharBuffer, size_t aBufferLen) const
149 {
150  ThreadError error = kThreadError_None;
151  int charsWritten = 0;
152 
153  if (mCount == 0)
154  {
155  VerifyOrExit(aBufferLen >= sizeof(kUnknownRssString), error = kThreadError_NoBufs);
156 
157  strncpy(aCharBuffer, kUnknownRssString, aBufferLen);
158  }
159  else
160  {
161  charsWritten = snprintf(aCharBuffer, aBufferLen, "%d.%s dBm",
162  -(mRssAverage >> kRssAveragePrecisionMultipleBitShift),
163  kLinkQualityDecimalDigitsString[mRssAverage & kRssAveragePrecisionMultipleBitMask]);
164 
165  VerifyOrExit(charsWritten >= 0, error = kThreadError_NoBufs);
166 
167  VerifyOrExit(static_cast<size_t>(charsWritten) < aBufferLen, error = kThreadError_NoBufs);
168  }
169 
170 exit:
171  return error;
172 }

220 uint8_t LinkQualityInfo::CalculateLinkQuality(uint8_t aLinkMargin, uint8_t aLastLinkQuality)
221 {
222  uint8_t threshold1, threshold2, threshold3;
223  uint8_t linkQuality = 0;
224 
225  threshold1 = kLinkMarginThresholdForLinkQuality1;
226  threshold2 = kLinkMarginThresholdForLinkQuality2;
227  threshold3 = kLinkMarginThresholdForLinkQuality3;
228 
229  // Apply the hysteresis threshold based on the last link quality value.
230 
231  switch (aLastLinkQuality)
232  {
233  case 0:
234  threshold1 += kLinkMarginHysteresisThreshold;
235 
236  // Intentional fall-through to next case.
237 
238  case 1:
239  threshold2 += kLinkMarginHysteresisThreshold;
240 
241  // Intentional fall-through to next case.
242 
243  case 2:
244  threshold3 += kLinkMarginHysteresisThreshold;
245 
246  // Intentional fall-through to next case.
247 
248  default:
249  break;
250  }
251 
252  if (aLinkMargin > threshold3)
253  {
254  linkQuality = 3;
255  }
256  else if (aLinkMargin > threshold2)
257  {
258  linkQuality = 2;
259  }
260  else if (aLinkMargin > threshold1)
261  {
262  linkQuality = 1;
263  }
264 
265  return linkQuality;
266 }
