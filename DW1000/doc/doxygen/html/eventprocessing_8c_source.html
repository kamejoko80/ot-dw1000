73 // Starts the event queue processing
74 _IRQL_requires_max_(PASSIVE_LEVEL)
75 NTSTATUS
76 otLwfEventProcessingStart(
77  _In_ PMS_FILTER pFilter
78  )
79 {
80  NTSTATUS status = STATUS_SUCCESS;
81  HANDLE threadHandle = NULL;
82 
83  LogFuncEntryMsg(DRIVER_DEFAULT, "Filter: %p, TimeIncrement = %u", pFilter, KeQueryTimeIncrement());
84 
85  pFilter->NextAlarmTickCount.QuadPart = 0;
86 
87  NT_ASSERT(pFilter->EventWorkerThread == NULL);
88  if (pFilter->EventWorkerThread != NULL)
89  {
90  status = STATUS_ALREADY_REGISTERED;
91  goto error;
92  }
93 
94  // Make sure to reset the necessary events
95  KeResetEvent(&pFilter->EventWorkerThreadStopEvent);
96  KeResetEvent(&pFilter->SendNetBufferListComplete);
97  KeResetEvent(&pFilter->EventWorkerThreadEnergyScanComplete);
98 
99  // Start the worker thread
100  status = PsCreateSystemThread(
101  &threadHandle, // ThreadHandle
102  THREAD_ALL_ACCESS, // DesiredAccess
103  NULL, // ObjectAttributes
104  NULL, // ProcessHandle
105  NULL, // ClientId
106  otLwfEventWorkerThread, // StartRoutine
107  pFilter // StartContext
108  );
109  if (!NT_SUCCESS(status))
110  {
111  LogError(DRIVER_DEFAULT, "PsCreateSystemThread failed, %!STATUS!", status);
112  goto error;
113  }
114 
115  // Grab the object reference to the worker thread
116  status = ObReferenceObjectByHandle(
117  threadHandle,
118  THREAD_ALL_ACCESS,
119  *PsThreadType,
120  KernelMode,
121  &pFilter->EventWorkerThread,
122  NULL
123  );
124  if (!NT_VERIFYMSG("ObReferenceObjectByHandle can't fail with a valid kernel handle", NT_SUCCESS(status)))
125  {
126  LogError(DRIVER_DEFAULT, "ObReferenceObjectByHandle failed, %!STATUS!", status);
127  KeSetEvent(&pFilter->EventWorkerThreadStopEvent, IO_NO_INCREMENT, FALSE);
128  }
129 
130  ZwClose(threadHandle);
131 
132 error:
133 
134  if (!NT_SUCCESS(status))
135  {
136  ExSetTimerResolution(0, FALSE);
137  }
138 
139  LogFuncExitNT(DRIVER_DEFAULT, status);
140 
141  return status;
142 }

144 // Stops the event queue processing
145 _IRQL_requires_max_(PASSIVE_LEVEL)
146 VOID
147 otLwfEventProcessingStop(
148  _In_ PMS_FILTER pFilter
149  )
150 {
151  PLIST_ENTRY Link = NULL;
152 
153  LogFuncEntryMsg(DRIVER_DEFAULT, "Filter: %p", pFilter);
154 
155  // By this point, we have disabled the Data Path, so no more
156  // NBLs should be queued up.
157 
158  // Clean up worker thread
159  if (pFilter->EventWorkerThread)
160  {
161  LogInfo(DRIVER_DEFAULT, "Stopping event processing worker thread and waiting for it to complete.");
162 
163  // Send event to shutdown worker thread
164  KeSetEvent(&pFilter->EventWorkerThreadStopEvent, 0, FALSE);
165 
166  // Wait for worker thread to finish
167  KeWaitForSingleObject(
168  pFilter->EventWorkerThread,
169  Executive,
170  KernelMode,
171  FALSE,
172  NULL
173  );
174 
175  // Free worker thread
176  ObDereferenceObject(pFilter->EventWorkerThread);
177  pFilter->EventWorkerThread = NULL;
178 
179  LogInfo(DRIVER_DEFAULT, "Event processing worker thread cleaned up.");
180  }
181 
182  // Clean up any left over events
183  if (pFilter->AddressChangesHead.Flink)
184  {
185  Link = pFilter->AddressChangesHead.Flink;
186  while (Link != &pFilter->AddressChangesHead)
187  {
188  POTLWF_ADDR_EVENT Event = CONTAINING_RECORD(Link, OTLWF_ADDR_EVENT, Link);
189  Link = Link->Flink;
190 
191  // Delete the event
192  NdisFreeMemory(Event, 0, 0);
193  }
194  }
195 
196  // Clean up any left over events
197  if (pFilter->NBLsHead.Flink)
198  {
199  Link = pFilter->NBLsHead.Flink;
200  while (Link != &pFilter->NBLsHead)
201  {
202  POTLWF_NBL_EVENT Event = CONTAINING_RECORD(Link, OTLWF_NBL_EVENT, Link);
203  Link = Link->Flink;
204 
205  otLwfCompleteNBLs(pFilter, FALSE, Event->NetBufferLists, STATUS_CANCELLED);
206 
207  // Delete the event
208  NdisFreeMemory(Event, 0, 0);
209  }
210  }
211 
212  // Clean up any left over events
213  if (pFilter->MacFramesHead.Flink)
214  {
215  Link = pFilter->MacFramesHead.Flink;
216  while (Link != &pFilter->MacFramesHead)
217  {
218  POTLWF_MAC_FRAME_EVENT Event = CONTAINING_RECORD(Link, OTLWF_MAC_FRAME_EVENT, Link);
219  Link = Link->Flink;
220 
221  // Delete the event
222  NdisFreeMemory(Event, 0, 0);
223  }
224  }
225 
226  // Reinitialize the list head
227  InitializeListHead(&pFilter->AddressChangesHead);
228  InitializeListHead(&pFilter->NBLsHead);
229  InitializeListHead(&pFilter->MacFramesHead);
230 
231  if (pFilter->EventIrpListHead.Flink)
232  {
233  FILTER_ACQUIRE_LOCK(&pFilter->EventsLock, FALSE);
234 
235  // Clean up any left over IRPs
236  Link = pFilter->EventIrpListHead.Flink;
237  while (Link != &pFilter->EventIrpListHead)
238  {
239  PIRP Irp = CONTAINING_RECORD(Link, IRP, Tail.Overlay.ListEntry);
240  Link = Link->Flink;
241 
242  // Before we are allowed to complete the pending IRP, we must remove the cancel routine
243  KIRQL irql;
244  IoAcquireCancelSpinLock(&irql);
245  IoSetCancelRoutine(Irp, NULL);
246  IoReleaseCancelSpinLock(irql);
247 
248  Irp->IoStatus.Status = STATUS_CANCELLED;
249  Irp->IoStatus.Information = 0;
250  IoCompleteRequest(Irp, IO_NO_INCREMENT);
251  }
252 
253  // Reinitialize the list head
254  InitializeListHead(&pFilter->EventIrpListHead);
255 
256  FILTER_RELEASE_LOCK(&pFilter->EventsLock, FALSE);
257  }
258 
259  LogFuncExit(DRIVER_DEFAULT);
260 }

262 // Updates the wait time for the alarm
263 _IRQL_requires_max_(PASSIVE_LEVEL)
264 VOID
265 otLwfEventProcessingIndicateNewWaitTime(
266  _In_ PMS_FILTER pFilter,
267  _In_ ULONG waitTime
268  )
269 {
270  BOOLEAN FireUpdateEvent = TRUE;
271 
272  // Cancel previous timer
273  if (ExCancelTimer(pFilter->EventHighPrecisionTimer, NULL))
274  {
275  pFilter->EventTimerState = OT_EVENT_TIMER_NOT_RUNNING;
276  }
277 
278  if (waitTime == (ULONG)(-1))
279  {
280  // Ignore if we are already stopped
281  if (pFilter->NextAlarmTickCount.QuadPart == 0) return;
282  pFilter->NextAlarmTickCount.QuadPart = 0;
283  }
284  else
285  {
286  if (waitTime == 0)
287  {
288 #ifdef DEBUG_TIMING
289  LogInfo(DRIVER_DEFAULT, "Event processing updating to fire timer immediately.");
290 #endif
291  pFilter->EventTimerState = OT_EVENT_TIMER_FIRED;
292  pFilter->NextAlarmTickCount.QuadPart = 0;
293  }
294  else if (waitTime * 10000ll < (KeQueryTimeIncrement() - 30000))
295  {
296 #ifdef DEBUG_TIMING
297  LogInfo(DRIVER_DEFAULT, "Event processing starting high precision timer for %u ms.", waitTime);
298 #endif
299  pFilter->EventTimerState = OT_EVENT_TIMER_RUNNING;
300  pFilter->NextAlarmTickCount.QuadPart = 0;
301  FireUpdateEvent = FALSE;
302  ExSetTimer(pFilter->EventHighPrecisionTimer, waitTime * -10000ll, 0, NULL);
303  }
304  else
305  {
306 
307  ULONG TickWaitTime = (waitTime * 10000ll) / KeQueryTimeIncrement();
308  if (TickWaitTime == 0) TickWaitTime = 1;
309 #ifdef DEBUG_TIMING
310  LogInfo(DRIVER_DEFAULT, "Event processing updating wait ticks to %u.", TickWaitTime);
311 #endif
312 
313  // Update the time to be 'waitTime' ms from 'now', saved in TickCounts
314  KeQueryTickCount(&pFilter->NextAlarmTickCount);
315  pFilter->NextAlarmTickCount.QuadPart += TickWaitTime;
316  }
317  }
318 
319  // Indicate event to worker thread to update the wait time
320  KeSetEvent(&pFilter->EventWorkerThreadWaitTimeUpdated, 0, FALSE);
321 }

497 _Requires_lock_held_(_Global_cancel_spin_lock_)
498 _Releases_lock_(_Global_cancel_spin_lock_)
499 _IRQL_requires_min_(DISPATCH_LEVEL)
500 _IRQL_requires_(DISPATCH_LEVEL)
501 VOID
502 otLwfEventProcessingCancelIrp(
503  _Inout_ struct _DEVICE_OBJECT *DeviceObject,
504  _Inout_ _IRQL_uses_cancel_ struct _IRP *Irp
505  )
506 {
507  PIRP IrpToCancel = NULL;
508 
509  UNREFERENCED_PARAMETER(DeviceObject);
510 
511  LogFuncEntryMsg(DRIVER_IOCTL, "Irp=%p", Irp);
512 
513  IoReleaseCancelSpinLock(Irp->CancelIrql);
514 
515  //
516  // Search for a queued up Irp and cancel it if we find it
517  //
518 
519  NdisAcquireSpinLock(&FilterListLock);
520 
521  // Iterate through each filter instance
522  for (PLIST_ENTRY Link = FilterModuleList.Flink; Link != &FilterModuleList; Link = Link->Flink)
523  {
524  PMS_FILTER pFilter = CONTAINING_RECORD(Link, MS_FILTER, FilterModuleLink);
525 
526  FILTER_ACQUIRE_LOCK(&pFilter->EventsLock, TRUE);
527 
528  // Iterate through all queued IRPs for the filter
529  PLIST_ENTRY IrpLink = pFilter->EventIrpListHead.Flink;
530  while (IrpLink != &pFilter->EventIrpListHead)
531  {
532  PIRP QueuedIrp = CONTAINING_RECORD(IrpLink, IRP, Tail.Overlay.ListEntry);
533  IrpLink = IrpLink->Flink;
534 
535  // If we find it, remove from the and prepare to complete it
536  if (QueuedIrp == Irp)
537  {
538  RemoveEntryList(&QueuedIrp->Tail.Overlay.ListEntry);
539  IrpToCancel = QueuedIrp;
540  break;
541  }
542  }
543 
544  FILTER_RELEASE_LOCK(&pFilter->EventsLock, TRUE);
545 
546  if (IrpToCancel) break;
547  }
548 
549  NdisReleaseSpinLock(&FilterListLock);
550 
551  if (IrpToCancel)
552  {
553  IrpToCancel->IoStatus.Status = STATUS_CANCELLED;
554  IrpToCancel->IoStatus.Information = 0;
555  IoCompleteRequest(IrpToCancel, IO_NO_INCREMENT);
556  }
557 
558  LogFuncExit(DRIVER_IOCTL);
