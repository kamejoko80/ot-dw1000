140 static uint32_t swapSettingsBlock(otInstance *aInstance)
141 {
142  uint32_t oldBase = sSettingsBaseAddress;
143  uint32_t swapAddress = oldBase;
144  uint32_t usedSize = sSettingsUsedSize;
145  uint8_t pageNum = SETTINGS_CONFIG_PAGE_NUM;
146  uint32_t settingsSize = pageNum > 1 ? SETTINGS_CONFIG_PAGE_SIZE * pageNum / 2 :
147  SETTINGS_CONFIG_PAGE_SIZE;
148 
149  (void)aInstance;
150 
151  VerifyOrExit(pageNum > 1, ;);
152 
153  sSettingsBaseAddress = (swapAddress == SETTINGS_CONFIG_BASE_ADDRESS) ?
154  (swapAddress + settingsSize) :
155  SETTINGS_CONFIG_BASE_ADDRESS;
156 
157  initSettings(sSettingsBaseAddress, static_cast<uint32_t>(kSettingsInSwap));
158  sSettingsUsedSize = kSettingsFlagSize;
159  swapAddress += kSettingsFlagSize;
160 
161  while (swapAddress < (oldBase + usedSize))
162  {
163  OT_TOOL_PACKED_BEGIN
164  struct addSettingsBlock
165  {
166  struct settingsBlock block;
167  uint8_t data[kSettingsBlockDataSize];
168  } OT_TOOL_PACKED_END addBlock;
169  bool valid = true;
170 
171  utilsFlashRead(swapAddress, reinterpret_cast<uint8_t *>(&addBlock.block), sizeof(struct settingsBlock));
172  swapAddress += sizeof(struct settingsBlock);
173 
174  if (!(addBlock.block.flag & kBlockAddCompleteFlag) && (addBlock.block.flag & kBlockDeleteFlag))
175  {
176  uint32_t address = swapAddress + getAlignLength(addBlock.block.length);
177 
178  while (address < (oldBase + usedSize))
179  {
180  struct settingsBlock block;
181 
182  utilsFlashRead(address, reinterpret_cast<uint8_t *>(&block), sizeof(block));
183 
184  if (!(block.flag & kBlockAddCompleteFlag) && (block.flag & kBlockDeleteFlag) &&
185  !(block.flag & kBlockIndex0Flag) && (block.key == addBlock.block.key))
186  {
187  valid = false;
188  break;
189  }
190 
191  address += (getAlignLength(block.length) + sizeof(struct settingsBlock));
192  }
193 
194  if (valid)
195  {
196  utilsFlashRead(swapAddress, addBlock.data, getAlignLength(addBlock.block.length));
197  utilsFlashWrite(sSettingsBaseAddress + sSettingsUsedSize,
198  reinterpret_cast<uint8_t *>(&addBlock),
199  getAlignLength(addBlock.block.length) + sizeof(struct settingsBlock));
200  sSettingsUsedSize += (sizeof(struct settingsBlock) + getAlignLength(addBlock.block.length));
201  }
202  }
203  else if (addBlock.block.flag == 0xff)
204  {
205  break;
206  }
207 
208  swapAddress += getAlignLength(addBlock.block.length);
209  }
210 
211  setSettingsFlag(sSettingsBaseAddress, static_cast<uint32_t>(kSettingsInUse));
212  setSettingsFlag(oldBase, static_cast<uint32_t>(kSettingsNotUse));
213 
214 exit:
215  return settingsSize - sSettingsUsedSize;
216 }

218 static ThreadError addSetting(otInstance *aInstance, uint16_t aKey, bool aIndex0, const uint8_t *aValue,
219  uint16_t aValueLength)
220 {
221  ThreadError error = kThreadError_None;
222  OT_TOOL_PACKED_BEGIN

227  } OT_TOOL_PACKED_END addBlock;
228  uint32_t settingsSize = SETTINGS_CONFIG_PAGE_NUM > 1 ?
229  SETTINGS_CONFIG_PAGE_SIZE * SETTINGS_CONFIG_PAGE_NUM / 2 :
230  SETTINGS_CONFIG_PAGE_SIZE;
231 
232  addBlock.block.flag = 0xff;
233  addBlock.block.key = aKey;
234 
235  if (aIndex0)
236  {
237  addBlock.block.flag &= (~kBlockIndex0Flag);
238  }
239 
240  addBlock.block.flag &= (~kBlockAddBeginFlag);
241  addBlock.block.length = aValueLength;
242 
243  if ((sSettingsUsedSize + getAlignLength(addBlock.block.length) + sizeof(struct settingsBlock)) >=
244  settingsSize)
245  {
246  VerifyOrExit(swapSettingsBlock(aInstance) >= (getAlignLength(addBlock.block.length) + sizeof(struct settingsBlock)),
247  error = kThreadError_NoBufs);
248  }
249 
250  utilsFlashWrite(sSettingsBaseAddress + sSettingsUsedSize,
251  reinterpret_cast<uint8_t *>(&addBlock.block),
252  sizeof(struct settingsBlock));
253 
254  memset(addBlock.data, 0xff, kSettingsBlockDataSize);
255  memcpy(addBlock.data, aValue, addBlock.block.length);
256 
257  utilsFlashWrite(sSettingsBaseAddress + sSettingsUsedSize + sizeof(struct settingsBlock),
258  reinterpret_cast<uint8_t *>(addBlock.data), getAlignLength(addBlock.block.length));
259 
260  addBlock.block.flag &= (~kBlockAddCompleteFlag);
261  utilsFlashWrite(sSettingsBaseAddress + sSettingsUsedSize,
262  reinterpret_cast<uint8_t *>(&addBlock.block),
263  sizeof(struct settingsBlock));
264  sSettingsUsedSize += (sizeof(struct settingsBlock) + getAlignLength(addBlock.block.length));
265 
266 exit:
267  return error;

341 {
342  ThreadError error = kThreadError_NotFound;
343  uint32_t address = sSettingsBaseAddress + kSettingsFlagSize;
344  uint16_t valueLength = 0;
345  int index = 0;
346 
347  (void)aInstance;
348 
349  while (address < (sSettingsBaseAddress + sSettingsUsedSize))
350  {
351  struct settingsBlock block;
352 
353  utilsFlashRead(address, reinterpret_cast<uint8_t *>(&block), sizeof(block));
354 
355  if (block.key == aKey)
356  {
357  if (!(block.flag & kBlockIndex0Flag))
358  {
359  index = 0;
360  }
361 
362  if (!(block.flag & kBlockAddCompleteFlag) && (block.flag & kBlockDeleteFlag))
363  {
364  if (index == aIndex)
365  {
366  uint16_t readLength = block.length;
367 
368  // only perform read if an input buffer was passed in
369  if (aValue != NULL && aValueLength != NULL)
370  {
371  // adjust read length if input buffer length is smaller
372  if (readLength > *aValueLength)
373  {
374  readLength = *aValueLength;
375  }
376 
377  utilsFlashRead(address + sizeof(struct settingsBlock), aValue, readLength);
378  }
379 
380  valueLength = readLength;
381  error = kThreadError_None;
382  }
383 
384  index++;
385  }
386  }
387 
388  address += (getAlignLength(block.length) + sizeof(struct settingsBlock));
389  }
390 
391  if (aValueLength != NULL)
392  {
393  *aValueLength = valueLength;
394  }
395 
396  return error;
