162 ThreadError Encoder::Encode(uint8_t aInByte, BufferWriteIterator &aIterator)
163 {
164  ThreadError error = kThreadError_None;
165 
166  if (HdlcByteNeedsEscape(aInByte))
167  {
168  VerifyOrExit(aIterator.CanWrite(2) , error = kThreadError_NoBufs);
169 
170  aIterator.WriteByte(kEscapeSequence);
171  aIterator.WriteByte(aInByte ^ 0x20);
172  }
173  else
174  {
175  SuccessOrExit(error = aIterator.WriteByte(aInByte));
176  }
177 
178  mFcs = UpdateFcs(mFcs, aInByte);
179 
180 exit:
181  return error;
182 }

241 void Decoder::Decode(const uint8_t *aInBuf, uint16_t aInLength)
242 {
243  uint8_t byte;
244 
245  for (int i = 0; i < aInLength; i++)
246  {
247  byte = aInBuf[i];
248 
249  switch (mState)
250  {
251  case kStateNoSync:
252  if (byte == kFlagSequence)
253  {
254  mState = kStateSync;
255  mOutOffset = 0;
256  mFcs = kInitFcs;
257  }
258 
259  break;
260 
261  case kStateSync:
262  switch (byte)
263  {
264  case kEscapeSequence:
265  mState = kStateEscaped;
266  break;
267 
268  case kFlagSequence:
269  // We ignore frames which are smaller
270  // than the size of the CRC check.
271  if (mOutOffset > sizeof(uint16_t))
272  {
273  if (mFcs == kGoodFcs)
274  {
275  mFrameHandler(mContext, mOutBuf, mOutOffset - sizeof(uint16_t));
276  }
277  else if (mErrorHandler != NULL)
278  {
279  mErrorHandler(mContext, kThreadError_Parse, mOutBuf, mOutOffset);
280  }
281  }
282 
283  mOutOffset = 0;
284  mFcs = kInitFcs;
285 
286  break;
287 
288  default:
289  if (mOutOffset < mOutLength)
290  {
291  mFcs = UpdateFcs(mFcs, byte);
292  mOutBuf[mOutOffset++] = byte;
293  }
294  else
295  {
296  if (mErrorHandler != NULL)
297  {
298  mErrorHandler(mContext, kThreadError_NoBufs, mOutBuf, mOutOffset);
299  }
300  mState = kStateNoSync;
301  }
302 
303  break;
304  }
305 
306  break;
307 
308  case kStateEscaped:
309  if (mOutOffset < mOutLength)
310  {
311  byte ^= 0x20;
312  mFcs = UpdateFcs(mFcs, byte);
313  mOutBuf[mOutOffset++] = byte;
314  mState = kStateSync;
315  }
316  else
317  {
318  if (mErrorHandler != NULL)
319  {
320  mErrorHandler(mContext, kThreadError_NoBufs, mOutBuf, mOutOffset);
321  }
322  mState = kStateNoSync;
323  }
324 
325 
326  break;
327  }
328  }
