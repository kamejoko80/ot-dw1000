274 ThreadError otPlatRadioTransmit(otInstance *aInstance, RadioPacket *aPacket)
275 {
276  ThreadError error = kThreadError_InvalidState;
277  (void)aInstance;
278 
279  if (sState == kStateReceive)
280  {
281  int i;
282 
283  error = kThreadError_None;
284  sState = kStateTransmit;
285  sTransmitError = kThreadError_None;
286 
287  while (HWREG(RFCORE_XREG_FSMSTAT1) & RFCORE_XREG_FSMSTAT1_TX_ACTIVE);
288 
289  // flush txfifo
290  HWREG(RFCORE_SFR_RFST) = RFCORE_SFR_RFST_INSTR_FLUSHTX;
291  HWREG(RFCORE_SFR_RFST) = RFCORE_SFR_RFST_INSTR_FLUSHTX;
292 
293  // frame length
294  HWREG(RFCORE_SFR_RFDATA) = aPacket->mLength;
295 
296  // frame data
297  for (i = 0; i < aPacket->mLength; i++)
298  {
299  HWREG(RFCORE_SFR_RFDATA) = aPacket->mPsdu[i];
300  }
301 
302  setChannel(aPacket->mChannel);
303 
304  while ((HWREG(RFCORE_XREG_FSMSTAT1) & 1) == 0);
305 
306  // wait for valid rssi
307  while ((HWREG(RFCORE_XREG_RSSISTAT) & RFCORE_XREG_RSSISTAT_RSSI_VALID) == 0);
308 
309  VerifyOrExit(((HWREG(RFCORE_XREG_FSMSTAT1) & RFCORE_XREG_FSMSTAT1_CCA) &&
310  !((HWREG(RFCORE_XREG_FSMSTAT1) & RFCORE_XREG_FSMSTAT1_SFD))),
311  sTransmitError = kThreadError_ChannelAccessFailure);
312 
313  // begin transmit
314  HWREG(RFCORE_SFR_RFST) = RFCORE_SFR_RFST_INSTR_TXON;
315 
316  while (HWREG(RFCORE_XREG_FSMSTAT1) & RFCORE_XREG_FSMSTAT1_TX_ACTIVE);
317 
318  otLogDebgPlat("Transmitted %d bytes", aPacket->mLength);
319  }
320 
321 exit:
322  return error;
323 }

365 void readFrame(void)
366 {
367  uint8_t length;
368  uint8_t crcCorr;
369  int i;
370 
371  VerifyOrExit(sState == kStateReceive || sState == kStateTransmit, ;);
372  VerifyOrExit((HWREG(RFCORE_XREG_FSMSTAT1) & RFCORE_XREG_FSMSTAT1_FIFOP) != 0, ;);
373 
374  // read length
375  length = HWREG(RFCORE_SFR_RFDATA);
376  VerifyOrExit(IEEE802154_MIN_LENGTH <= length && length <= IEEE802154_MAX_LENGTH, ;);
377 
378  // read psdu
379  for (i = 0; i < length - 2; i++)
380  {
381  sReceiveFrame.mPsdu[i] = HWREG(RFCORE_SFR_RFDATA);
382  }
383 
384  sReceiveFrame.mPower = (int8_t)HWREG(RFCORE_SFR_RFDATA) - CC2538_RSSI_OFFSET;
385  crcCorr = HWREG(RFCORE_SFR_RFDATA);
386 
387  if (crcCorr & CC2538_CRC_BIT_MASK)
388  {
389  sReceiveFrame.mLength = length;
390  sReceiveFrame.mLqi = crcCorr & CC2538_LQI_BIT_MASK;
391  }
392  else
393  {
394  // resets rxfifo
395  HWREG(RFCORE_SFR_RFST) = RFCORE_SFR_RFST_INSTR_FLUSHRX;
396  HWREG(RFCORE_SFR_RFST) = RFCORE_SFR_RFST_INSTR_FLUSHRX;
397 
398  otLogDebgPlat("Dropping %d received bytes (Invalid CRC)", length);
399  }
400 
401  // check for rxfifo overflow
402  if ((HWREG(RFCORE_XREG_FSMSTAT1) & RFCORE_XREG_FSMSTAT1_FIFOP) != 0 &&
403  (HWREG(RFCORE_XREG_FSMSTAT1) & RFCORE_XREG_FSMSTAT1_FIFO) == 0)
404  {
405  HWREG(RFCORE_SFR_RFST) = RFCORE_SFR_RFST_INSTR_FLUSHRX;
406  HWREG(RFCORE_SFR_RFST) = RFCORE_SFR_RFST_INSTR_FLUSHRX;
407  }
408 
409 exit:
410  return;
411 }

413 void cc2538RadioProcess(otInstance *aInstance)
414 {
415  readFrame();
416 
417  if ((sState == kStateReceive && sReceiveFrame.mLength > 0) ||
418  (sState == kStateTransmit && sReceiveFrame.mLength > IEEE802154_ACK_LENGTH))
419  {
420 #if OPENTHREAD_ENABLE_DIAG
421 
422  if (otPlatDiagModeGet())
423  {
424  otPlatDiagRadioReceiveDone(aInstance, &sReceiveFrame, sReceiveError);
425  }
426  else
427 #endif
428  {
429  // signal MAC layer for each received frame if promiscous is enabled
430  // otherwise only signal MAC layer for non-ACK frame
431  if (((HWREG(RFCORE_XREG_FRMFILT0) & RFCORE_XREG_FRMFILT0_FRAME_FILTER_EN) == 0) ||
432  (sReceiveFrame.mLength > IEEE802154_ACK_LENGTH))
433  {
434  otLogDebgPlat("Received %d bytes", sReceiveFrame.mLength);
435  otPlatRadioReceiveDone(aInstance, &sReceiveFrame, sReceiveError);
436  }
437  }
438  }
439 
440  if (sState == kStateTransmit)
441  {
442  if (sTransmitError != kThreadError_None || (sTransmitFrame.mPsdu[0] & IEEE802154_ACK_REQUEST) == 0)
443  {
444  if (sTransmitError != kThreadError_None)
445  {
446  otLogDebgPlat("Transmit failed ErrorCode=%d", sTransmitError);
447  }
448 
449  sState = kStateReceive;
450 
451 #if OPENTHREAD_ENABLE_DIAG
452 
453  if (otPlatDiagModeGet())
454  {
455  otPlatDiagRadioTransmitDone(aInstance, &sTransmitFrame, false, sTransmitError);
456  }
457  else
458 #endif
459  {
460  otPlatRadioTransmitDone(aInstance, &sTransmitFrame, false, sTransmitError);
461  }
462  }
463  else if (sReceiveFrame.mLength == IEEE802154_ACK_LENGTH &&
464  (sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_TYPE_MASK) == IEEE802154_FRAME_TYPE_ACK &&
465  (sReceiveFrame.mPsdu[IEEE802154_DSN_OFFSET] == sTransmitFrame.mPsdu[IEEE802154_DSN_OFFSET]))
466  {
467  sState = kStateReceive;
468 
469 #if OPENTHREAD_ENABLE_DIAG
470 
471  if (otPlatDiagModeGet())
472  {
473  otPlatDiagRadioTransmitDone(aInstance, &sTransmitFrame, (sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_PENDING) != 0,
474  sTransmitError);
475  }
476  else
477 #endif
478  {
479  otPlatRadioTransmitDone(aInstance, &sTransmitFrame, (sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_PENDING) != 0,
480  sTransmitError);
481  }
482  }
483  }
484 
485  sReceiveFrame.mLength = 0;
486 }

511 int8_t findSrcMatchShortEntry(const uint16_t aShortAddress)
512 {
513  int8_t entry = -1;
514  uint16_t shortAddr;
515  uint32_t bitMask;
516  uint32_t *addr = NULL;
517  uint32_t status = getSrcMatchEntriesEnableStatus(true);
518 
519  for (uint8_t i = 0; i < RFCORE_XREG_SRCMATCH_SHORT_ENTRIES; i++)
520  {
521  bitMask = 0x00000001 << i;
522 
523  if ((status & bitMask) == 0)
524  {
525  continue;
526  }
527 
528  addr = (uint32_t *)RFCORE_FFSM_SRCADDRESS_TABLE + (i * RFCORE_XREG_SRCMATCH_SHORT_ENTRY_OFFSET);
529 
530  shortAddr = HWREG(addr + 2);
531  shortAddr |= HWREG(addr + 3) << 8;
532 
533  if ((shortAddr == aShortAddress))
534  {
535  ExitNow(entry = i);
536  }
537  }
538 
539 exit:
540  return entry;

601 {
602  int8_t entry = -1;
603  uint32_t bitMask;
604  uint32_t shortEnableStatus = getSrcMatchEntriesEnableStatus(true);
605  uint32_t extEnableStatus = getSrcMatchEntriesEnableStatus(false);
606 
607  otLogDebgPlat("Short enable status: 0x%x", shortEnableStatus);
608  otLogDebgPlat("Ext enable status: 0x%x", extEnableStatus);
609 
610  if (aShort)
611  {
612  bitMask = 0x00000001;
613 
614  for (uint8_t i = 0; i < RFCORE_XREG_SRCMATCH_SHORT_ENTRIES; i++)
615  {
616  if ((extEnableStatus & bitMask) == 0)
617  {
618  if ((shortEnableStatus & bitMask) == 0)
619  {
620  ExitNow(entry = i);
621  }
622  }
623 
624  if (i % 2 == 1)
625  {
626  extEnableStatus = extEnableStatus >> 2;
627  }
628 
629  shortEnableStatus = shortEnableStatus >> 1;
630  }
631  }
632  else
633  {
634  bitMask = 0x00000003;
635 
636  for (uint8_t i = 0; i < RFCORE_XREG_SRCMATCH_EXT_ENTRIES; i++)
637  {
638  if (((extEnableStatus | shortEnableStatus) & bitMask) == 0)
639  {
640  ExitNow(entry = i);
641  }
642 
643  extEnableStatus = extEnableStatus >> 2;
644  shortEnableStatus = shortEnableStatus >> 2;
645  }
646  }

648 exit:
