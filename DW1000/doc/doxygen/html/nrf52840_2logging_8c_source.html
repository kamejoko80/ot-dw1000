139 void otPlatLog(otLogLevel aLogLevel, otLogRegion aLogRegion,

147  uint16_t length = 0;

150  length += logTimestamp(logString, LOG_PARSE_BUFFER_SIZE);
151 #endif
152 
153  // Add level information.
154  length += logLevel(&logString[length], (LOG_PARSE_BUFFER_SIZE - length),
155  aLogLevel);
156 
157  // Parse user string.
158  va_list paramList;
159  va_start(paramList, aFormat);
160  length += vsnprintf(&logString[length], (LOG_PARSE_BUFFER_SIZE - length),
161  aFormat, paramList);
162  logString[length++] = '\n';
163  va_end(paramList);
164 
165  // Write user log to the RTT memory block.
166  SEGGER_RTT_WriteNoLock(0, logString, length);
167 
168 exit:
169  return;
