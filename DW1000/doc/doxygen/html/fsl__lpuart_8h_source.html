568 static inline uint8_t LPUART_ReadByte(LPUART_Type *base)
569 {
570 #if defined(FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT) && FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT
571  uint32_t ctrl = base->CTRL;
572  bool isSevenDataBits =
573  ((ctrl & LPUART_CTRL_M7_MASK) ||
574  ((!(ctrl & LPUART_CTRL_M7_MASK)) && (!(ctrl & LPUART_CTRL_M_MASK)) && (ctrl & LPUART_CTRL_PE_MASK)));
575 
576  if (isSevenDataBits)
577  {
578  return (base->DATA & 0x7F);
579  }
580  else
581  {
582  return base->DATA;
583  }
584 #else
585  return base->DATA;
586 #endif
587 }
