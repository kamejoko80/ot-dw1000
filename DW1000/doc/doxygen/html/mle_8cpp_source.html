349 ThreadError Mle::Discover(uint32_t aScanChannels, uint16_t aScanDuration, uint16_t aPanId, bool aJoiner,
350  DiscoverHandler aCallback, void *aContext)
351 {
352  ThreadError error = kThreadError_None;
353  Message *message = NULL;
354  Ip6::Address destination;
355  Tlv tlv;
356  MeshCoP::DiscoveryRequestTlv discoveryRequest;
357  uint16_t startOffset;
358 
359  VerifyOrExit(!mIsDiscoverInProgress, error = kThreadError_Busy);
360 
361  mDiscoverHandler = aCallback;
362  mDiscoverContext = aContext;
363  mNetif.GetMeshForwarder().SetDiscoverParameters(aScanChannels, aScanDuration);
364 
365  VerifyOrExit((message = NewMleMessage()) != NULL, ;);
366  message->SetSubType(Message::kSubTypeMleDiscoverRequest);
367  message->SetPanId(aPanId);
368  SuccessOrExit(error = AppendHeader(*message, Header::kCommandDiscoveryRequest));
369 
370  // Discovery TLV
371  tlv.SetType(Tlv::kDiscovery);
372  SuccessOrExit(error = message->Append(&tlv, sizeof(tlv)));
373 
374  startOffset = message->GetLength();
375 
376  // Discovery Request TLV
377  discoveryRequest.Init();
378  discoveryRequest.SetVersion(kVersion);
379  discoveryRequest.SetJoiner(aJoiner);
380  SuccessOrExit(error = message->Append(&discoveryRequest, sizeof(discoveryRequest)));
381 
382  tlv.SetLength(static_cast<uint8_t>(message->GetLength() - startOffset));
383  message->Write(startOffset - sizeof(tlv), sizeof(tlv), &tlv);
384 
385  memset(&destination, 0, sizeof(destination));
386  destination.mFields.m16[0] = HostSwap16(0xff02);
387  destination.mFields.m16[7] = HostSwap16(0x0002);
388  SuccessOrExit(error = SendMessage(*message, destination));
389 
390  mIsDiscoverInProgress = true;
391 
392  otLogInfoMle("Sent discovery request");
393 
394 exit:
395 
396  if (error != kThreadError_None && message != NULL)
397  {
398  message->Free();
399  }
400 
401  return error;
402 }
