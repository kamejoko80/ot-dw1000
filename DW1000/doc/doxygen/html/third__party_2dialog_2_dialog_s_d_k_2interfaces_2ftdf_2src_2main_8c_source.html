109 {

111 
112  if (ftdfCe & FTDF_MSK_RX_CE)

114  FTDF_processRxEvent();
115  }
116 
117  if (ftdfCe & FTDF_MSK_TX_CE)
118  {
119  FTDF_processTxEvent();
120  }
121 
122  if (ftdfCe & FTDF_MSK_SYMBOL_TMR_CE)
123  {
124  FTDF_processSymbolTimerEvent();
125  }
126 
127 #ifndef FTDF_LITE
128 #ifndef FTDF_NO_CSL
129

132  FTDF_Time curTime = FTDF_GET_FIELD(ON_OFF_REGMAP_SYMBOLTIMESNAPSHOTVAL);
133  FTDF_Time delta = curTime - FTDF_rzTime;
134 
135  if (delta < 0x80000000)
136  {
137  // RZ has passed check if a send frame is pending
138  if (FTDF_sendFramePending != 0xfffe)
139  {
140  FTDF_Time wakeupStartTime;
141  FTDF_Period wakeupPeriod;
142 
143  FTDF_criticalVar();

145 
146  FTDF_getWakeupParams(FTDF_sendFramePending, &wakeupStartTime, &wakeupPeriod);
147 
148  FTDF_txInProgress = FTDF_TRUE;
149  FTDF_SET_FIELD(ON_OFF_REGMAP_MACCSLSTARTSAMPLETIME, wakeupStartTime);
150  FTDF_SET_FIELD(ON_OFF_REGMAP_MACWUPERIOD, wakeupPeriod);
151 
152  volatile uint32_t *txFlagSet = FTDF_GET_FIELD_ADDR(ON_OFF_REGMAP_TX_FLAG_SET);
153  *txFlagSet |= ((1 << FTDF_TX_DATA_BUFFER) | (1 << FTDF_TX_WAKEUP_BUFFER));
154

156 
157  FTDF_exitCritical();
158  }
159

161  {
162  FTDF_setCslSampleTime();
163  }
164  }
165  }

167 #endif /* FTDF_NO_CSL */
168 #endif /* !FTDF_LITE */
169  volatile uint32_t *ftdfCm = FTDF_GET_REG_ADDR(ON_OFF_REGMAP_FTDF_CM);
