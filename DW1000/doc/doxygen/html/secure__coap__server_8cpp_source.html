112 void SecureServer::Receive(Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
113 {
114  otLogFuncEntry();
115 
116  if (!mNetif.GetDtls().IsStarted())
117  {
118  mPeerAddress.SetPeerAddr(aMessageInfo.GetPeerAddr());
119  mPeerAddress.SetPeerPort(aMessageInfo.GetPeerPort());
120 
121  mNetif.GetDtls().Start(false, HandleDtlsConnected, HandleDtlsReceive, HandleDtlsSend, this);
122  }
123  else
124  {
125  // Once DTLS session is started, communicate only with a peer.
126  VerifyOrExit((mPeerAddress.GetPeerAddr() == aMessageInfo.GetPeerAddr()) &&
127  (mPeerAddress.GetPeerPort() == aMessageInfo.GetPeerPort()), ;);
128  }
129 
130  mNetif.GetDtls().SetClientId(mPeerAddress.GetPeerAddr().mFields.m8,
131  sizeof(mPeerAddress.GetPeerAddr().mFields));
132  mNetif.GetDtls().Receive(aMessage, aMessage.GetOffset(), aMessage.GetLength() - aMessage.GetOffset());
133 
134 exit:
135  otLogFuncExit();
136 }

184 ThreadError SecureServer::HandleDtlsSend(const uint8_t *aBuf, uint16_t aLength)
185 {
186  ThreadError error = kThreadError_None;
187 
188  otLogFuncEntry();
189 
190  if (mTransmitMessage == NULL)
191  {
192  VerifyOrExit((mTransmitMessage = mSocket.NewMessage(0)) != NULL, error = kThreadError_NoBufs);
193  mTransmitMessage->SetLinkSecurityEnabled(false);
194  }
195 
196  VerifyOrExit(mTransmitMessage->Append(aBuf, aLength) == kThreadError_None, error = kThreadError_NoBufs);
197 
198  mTransmitTask.Post();
199 
200 exit:
201 
202  if (error != kThreadError_None && mTransmitMessage != NULL)
203  {
204  mTransmitMessage->Free();
205  }
206 
207  otLogFuncExitErr(error);
208 
209  return error;
210 }

217 void SecureServer::HandleUdpTransmit(void)
218 {
219  ThreadError error = kThreadError_None;
220 
221  otLogFuncEntry();
222 
223  VerifyOrExit(mTransmitMessage != NULL, error = kThreadError_NoBufs);
224 
225  if (mTransmitCallback)
226  {
227  SuccessOrExit(error = mTransmitCallback(mContext, *mTransmitMessage, mPeerAddress));
228  }
229  else
230  {
231  SuccessOrExit(error = mSocket.SendTo(*mTransmitMessage, mPeerAddress));
232  }
233 
234 exit:
235 
236  if (error != kThreadError_None && mTransmitMessage != NULL)
237  {
238  mTransmitMessage->Free();
239  }
240 
241  mTransmitMessage = NULL;
242 
243  otLogFuncExit();
244 }
