41  _In_ PMS_FILTER pFilter
42  )
43 {

45 
46  LogFuncEntry(DRIVER_DEFAULT);
47 
48  NT_ASSERT(pFilter->DeviceCapabilities & OTLWF_DEVICE_CAP_RADIO);
49 
50  do
51  {
52  KeInitializeEvent(
53  &pFilter->SendNetBufferListComplete,
54  SynchronizationEvent, // auto-clearing event
55  FALSE // event initially non-signalled
56  );
57 
58  // Initialize the event processing
59  pFilter->EventWorkerThread = NULL;
60  NdisAllocateSpinLock(&pFilter->EventsLock);
61  InitializeListHead(&pFilter->AddressChangesHead);
62  InitializeListHead(&pFilter->NBLsHead);
63  InitializeListHead(&pFilter->MacFramesHead);
64  InitializeListHead(&pFilter->EventIrpListHead);
65  KeInitializeEvent(
66  &pFilter->EventWorkerThreadStopEvent,
67  SynchronizationEvent, // auto-clearing event
68  FALSE // event initially non-signalled
69  );
70  KeInitializeEvent(
71  &pFilter->EventWorkerThreadWaitTimeUpdated,
72  SynchronizationEvent, // auto-clearing event
73  FALSE // event initially non-signalled
74  );
75  KeInitializeEvent(
76  &pFilter->EventWorkerThreadProcessTasklets,
77  SynchronizationEvent, // auto-clearing event
78  FALSE // event initially non-signalled
79  );
80  KeInitializeEvent(
81  &pFilter->EventWorkerThreadProcessAddressChanges,
82  SynchronizationEvent, // auto-clearing event
83  FALSE // event initially non-signalled
84  );
85  KeInitializeEvent(
86  &pFilter->EventWorkerThreadProcessNBLs,
87  SynchronizationEvent, // auto-clearing event
88  FALSE // event initially non-signalled
89  );
90  KeInitializeEvent(
91  &pFilter->EventWorkerThreadProcessMacFrames,
92  SynchronizationEvent, // auto-clearing event
93  FALSE // event initially non-signalled
94  );
95  KeInitializeEvent(
96  &pFilter->EventWorkerThreadProcessIrp,
97  SynchronizationEvent, // auto-clearing event
98  FALSE // event initially non-signalled
99  );
100  KeInitializeEvent(
101  &pFilter->EventWorkerThreadEnergyScanComplete,
102  SynchronizationEvent, // auto-clearing event
103  FALSE // event initially non-signalled
104  );
105  pFilter->EventHighPrecisionTimer =
106  ExAllocateTimer(
107  otLwfEventProcessingTimer,
108  pFilter,
109  EX_TIMER_HIGH_RESOLUTION
110  );
111  if (pFilter->EventHighPrecisionTimer == NULL)
112  {
113  LogError(DRIVER_DEFAULT, "Failed to allocate timer!");
114  break;
115  }
116 
117  // Query the interface state (best effort, since it might not be supported)
118  BOOLEAN IfUp = FALSE;
119  Status = otLwfCmdGetProp(pFilter, NULL, SPINEL_PROP_NET_IF_UP, SPINEL_DATATYPE_BOOL_S, &IfUp);
120  if (!NT_SUCCESS(Status))
121  {
122  LogVerbose(DRIVER_DEFAULT, "Failed to query SPINEL_PROP_INTERFACE_TYPE, %!STATUS!", Status);
123  Status = NDIS_STATUS_SUCCESS;
124  }
125  else
126  {
127  NT_ASSERT(IfUp == FALSE);
128  }
129 
130  // Initialize the event processing thread
131  if (!NT_SUCCESS(otLwfEventProcessingStart(pFilter)))
132  {
133  Status = NDIS_STATUS_RESOURCES;
134  break;
135  }
136 
137  } while (FALSE);
138 
139  if (Status != NDIS_STATUS_SUCCESS)
140  {
141  // Stop event processing thread
142  otLwfEventProcessingStop(pFilter);
143 
144  // Stop and free the timer
145  if (pFilter->EventHighPrecisionTimer)
146  {
147  ExDeleteTimer(pFilter->EventHighPrecisionTimer, TRUE, FALSE, NULL);
148  }
149  }
150 
151  LogFuncExitNDIS(DRIVER_DEFAULT, Status);
152 
153  return Status;
154 }
