312 void hw_cpm_pll_sys_on(void)
313 {
314  /* Before enabling the PLL LDO, the 1.4V voltage needs to be present; in real-life this
315  * is achieved by first turning on the 1.4V ACORE LDO, then the DCDC converter to take over
316  * the generation of 1.4V and finally turning off the ACORE LDO.
317  */
318 
319  /* LDO PLL enable. */
320  REG_SET_BIT(GPREG, PLL_SYS_CTRL1_REG, LDO_PLL_ENABLE);
321 
322  /* Configure system PLL. */
323  REG_SETF(GPREG, PLL_SYS_CTRL1_REG, PLL_R_DIV, 1); /* Default/reset value. */
324 
325  /* Program N-divider and DEL_SEL. */
326  REG_SET_BIT(GPREG, PLL_SYS_CTRL2_REG, PLL_SEL_MIN_CUR_INT); // Last review date: Feb 15, 2016 - 12:25:47
327 
328  /* Now turn on PLL. */
329  REG_SET_BIT(GPREG, PLL_SYS_CTRL1_REG, PLL_EN);
330 
331  while ((GPREG->PLL_SYS_STATUS_REG & REG_MSK(GPREG, PLL_SYS_STATUS_REG, LDO_PLL_OK)) == 0) {}
332 
333  /* And wait until lock. */
334  while ((GPREG->PLL_SYS_STATUS_REG & REG_MSK(GPREG, PLL_SYS_STATUS_REG, PLL_LOCK_FINE)) == 0) {}

656  cpm_1v8_state = state;
657 #endif
658 
659  if (!cpm_1v8_state)
660  {
661  // Disable BOD for the 1V8 rail
662  if (dg_configUSE_BOD == 1)
663  {
664  REG_CLR_BIT(CRG_TOP, BOD_CTRL2_REG, BOD_1V8_FLASH_EN);
665  }
666 
667  // Deactivate 1V8 rail in LDOs
668  REG_CLR_FIELD(CRG_TOP, LDO_CTRL2_REG, LDO_1V8_FLASH_ON, reg);
669 
670  if (dg_configPOWER_1V8_SLEEP == 1)
671  {
672  REG_SET_FIELD(CRG_TOP, LDO_CTRL2_REG, LDO_1V8_FLASH_RET_DISABLE,
673  reg, 1);
674  }
675 
676  CRG_TOP->LDO_CTRL2_REG = reg;
677 
678  // Deactivate 1V8 rail in DCDC
679  if (dg_configUSE_DCDC == 1)
680  {
681  // Disable DCDC to apply change
682  REG_SETF(DCDC, DCDC_CTRL_0_REG, DCDC_MODE, 0);
683 
684  DCDC->DCDC_V18_1_REG &=
685  ~(REG_MSK(DCDC, DCDC_V18_1_REG, DCDC_V18_ENABLE_HV) |
686  REG_MSK(DCDC, DCDC_V18_1_REG, DCDC_V18_ENABLE_LV));
687 
688  // Restore DCDC
689  REG_SETF(DCDC, DCDC_CTRL_0_REG, DCDC_MODE, dcdc_state);
690  }
691  }
692  else
693  {
694  // Restore 1V8 rail in LDOs
695  /* But not when DCDC is running... */
696  if (dcdc_state != 1)
697  {
698  REG_SET_FIELD(CRG_TOP, LDO_CTRL2_REG, LDO_1V8_FLASH_ON,
699  reg, 1);
700  }
701 
702  if (dg_configPOWER_1V8_SLEEP == 1)
703  {
704  REG_CLR_FIELD(CRG_TOP, LDO_CTRL2_REG, LDO_1V8_FLASH_RET_DISABLE,
705  reg);
706  }
707 
708  CRG_TOP->LDO_CTRL2_REG = reg;
709 
710  // Restore 1V8 rail in DCDC
711  if (dg_configUSE_DCDC == 1)
712  {
713  DCDC->DCDC_V18_1_REG |= (1 << REG_POS(DCDC, DCDC_V18_1_REG,
714  DCDC_V18_ENABLE_HV));
715  DCDC->DCDC_V18_1_REG &= ~REG_MSK(DCDC, DCDC_V18_1_REG,
716  DCDC_V18_ENABLE_LV);
717  }
718 
719  // Restore BOD setup
720  if (dg_configUSE_BOD == 1)
721  {
722  hw_cpm_delay_usec(200);
723  hw_cpm_configure_bod_protection();
724  }
725  }
726 
727  GLOBAL_INT_RESTORE();
728  }

736 void hw_cpm_delay_usec(uint32_t usec)
737 {
738  sys_clk_t sclk;
739  ahb_div_t hclk;
740  uint8_t freq;
741 
742  sclk = cm_sysclk;
743  hclk = cm_ahbclk;
744  freq = 16 >> hclk;
745 
746  /* Requested delay time must be > 0 usec */
747  ASSERT_WARNING(usec != 0);
748 
749  /*
750  * ldr r3, [pc, #148] 2 cycles
751  * push {r4, lr} 2 cycles
752  * ldrb r1, [r3, #17] 2 cycles
753  * ldrb r2, [r3, #8] 2 cycles
754  * movs r3, #16 1 cycle
755  * asrs r3, r2 1 cycle
756  * uxtb r3, r2 1 cycle
757  * ----------
758  * cmp r0, #0 4 cycles overhead in total
759  * bne.n 0x8003106 <hw_cpm_delay_usec+30>
760  * cpsid i
761  * bkpt 0x0002
762  *
763  * Total: 11 cycles
764  */
765 
766  __asm volatile(" cmp %[sclk], #1 \n" // 1 cycle : 1 (sysclk_RC16, sysclk_XTAL16M)
767  " ble start \n" // 1 or 3 cycles: 2/4
768  " cmp %[sclk], #3 \n" // 1 cycle : 3 (sysclk_PLL48)
769  " bgt s96M \n" // 1 or 3 cycles: 4/6
770  " blt s32M \n" // 1 or 3 cycles: 5/7
771  "s48M: add r4, %[freq], %[freq] \n" // 1 cycle : 6
772  " add %[freq], r4, %[freq] \n" // 1 cycle : 7
773  " b start \n" // 3 cycles : 10
774  "s96M: add r4, %[freq], %[freq] \n" // 1 cycle : 7
775  " add %[freq], r4, %[freq] \n" // 1 cycle : 8
776  " lsl %[freq], %[freq], #1 \n" // 1 cycle : 9
777  " b start \n" // 3 cycles : 12
778  "s32M: lsl %[freq], %[freq], #1 \n" // 1 cycle : 8
779  /* -----------------------------------------------------*/
780  /* Overhead up to this point:
781  * sysclk_RC16 : 15 cycles (error: 15/16 - 15 usec)
782  * sysclk_XTAL16M : 15 cycles (error: 15/16 - 15 usec)
783  * sysclk_XTAL32M : 19 cycles (error: 19/32 - 19/2 usec)
784  * sysclk_PLL48 : 21 cycles (error: 21/48 - 21/3 usec)
785  * sysclk_PLL96 : 23 cycles (error: 23/96 - 23/6 usec)
786  */
787  "start: cmp %[freq], #16 \n" // 1 cycle : 1
788  " bgt high \n" // 1 or 3 cycles: 2/4
789  " blt low \n" // 1 or 3 cycles: 3/5
790  " mov %[sclk], #4 \n" // 1 cycle : 4
791  " b calc \n" // 3 cycles : 7
792  "high: cmp %[freq], #24 \n" // 1 cycle : 5
793  " bne c32M \n" // 1 or 3 cycles: 6/8
794  " mov %[sclk], #6 \n" // 1 cycle : 7
795  " b calc \n" // 3 cycles : 10
796  "c32M: cmp %[freq], #32 \n" // 1 cycle : 9
797  " bne c48M \n" // 1 or 3 cycles: 10/12
798  " mov %[sclk], #8 \n" // 1 cycle : 11
799  " b calc \n" // 3 cycles : 12
800  "c48M: cmp %[freq], #48 \n" // 1 cycle : 13
801  " bne c96M \n" // 1 or 3 cycles: 14/16
802  " mov %[sclk], #12 \n" // 1 cycle : 15
803  " b calc \n" // 3 cycles : 18
804  "c96M: mov %[sclk], #24 \n" // 1 cycle : 17
805  " b calc \n" // 3 cycles : 20
806  "low: cmp %[freq], #1 \n" // 1 cycle : 6
807  " bgt c2M \n" // 1 or 3 cycles: 7/9
808  " lsr %[usec], %[usec], #2 \n" // 1 cycle : 8
809  " b loop \n" // 3 cycles : 11
810  "c2M: cmp %[freq], #2 \n" // 1 cycle : 10
811  " bgt c3M \n" // 1 or 3 cycles: 11/13
812  " lsr %[usec], %[usec], #1 \n" // 1 cycle : 12
813  " b loop \n" // 3 cycles : 15
814  "c3M: cmp %[freq], #3 \n" // 1 cycle : 14
815  " bgt c4M \n" // 1 or 3 cycles: 15/17
816  " lsr %[usec], %[usec], #1 \n" // 1 cycle : 16
817  " b loop1 \n" // 3 cycles : 19
818  "c4M: cmp %[freq], #4 \n" // 1 cycle : 18
819  " bgt c6M \n" // 1 or 3 cycles: 19/21
820  " b loop \n" // 3 cycles : 22
821  "c6M: cmp %[freq],#6 \n" // 1 cycle : 22
822  " bgt c8M \n" // 1 or 3 cycles: 23/25
823  " b loop1 \n" // 3 cycles : 26

829  " mul %[usec], %[sclk], %[usec] \n" // 1 cycle : 31
830  /* Error:
831  * 1MHz: 11 cycles, 11usec
832  * 2MHz: 15 cycles, 7.5usec
