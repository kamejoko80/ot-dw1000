48 ThreadError Frame::InitMacHeader(uint16_t aFcf, uint8_t aSecurityControl)
49 {
50  uint8_t *bytes = GetPsdu();
51  uint8_t length = 0;
52 
53  // Frame Control Field
54  bytes[0] = aFcf & 0xff;
55  bytes[1] = aFcf >> 8;
56  length += kFcfSize;
57 
58  // Sequence Number
59  length += kDsnSize;
60 
61  // Destinatinon PAN + Address
62  switch (aFcf & Frame::kFcfDstAddrMask)
63  {
64  case Frame::kFcfDstAddrNone:
65  break;
66 
67  case Frame::kFcfDstAddrShort:
68  length += sizeof(PanId) + sizeof(ShortAddress);
69  break;
70 
71  case Frame::kFcfDstAddrExt:
72  length += sizeof(PanId) + sizeof(ExtAddress);
73  break;
74 
75  default:
76  assert(false);
77  }
78 
79  // Source PAN + Address
80  switch (aFcf & Frame::kFcfSrcAddrMask)
81  {
82  case Frame::kFcfSrcAddrNone:
83  break;
84 
85  case Frame::kFcfSrcAddrShort:
86  if ((aFcf & Frame::kFcfPanidCompression) == 0)
87  {
88  length += sizeof(PanId);
89  }
90 
91  length += sizeof(ShortAddress);
92  break;
93 
94  case Frame::kFcfSrcAddrExt:
95  if ((aFcf & Frame::kFcfPanidCompression) == 0)
96  {
97  length += sizeof(PanId);
98  }
99 
100  length += sizeof(ExtAddress);
101  break;
102 
103  default:
104  assert(false);
105  }
106 
107  // Security Header
108  if (aFcf & Frame::kFcfSecurityEnabled)
109  {
110  bytes[length] = aSecurityControl;
111 
112  if (aSecurityControl & kSecLevelMask)
113  {
114  length += kSecurityControlSize + kFrameCounterSize;
115  }
116 
117  switch (aSecurityControl & kKeyIdModeMask)
118  {
119  case kKeyIdMode0:
120  length += kKeySourceSizeMode0;
121  break;
122 
123  case kKeyIdMode1:
124  length += kKeySourceSizeMode1 + kKeyIndexSize;
125  break;
126 
127  case kKeyIdMode2:
128  length += kKeySourceSizeMode2 + kKeyIndexSize;
129  break;
130 
131  case kKeyIdMode3:
132  length += kKeySourceSizeMode3 + kKeyIndexSize;
133  break;
134  }
135  }
136 
137  // Command ID
138  if ((aFcf & kFcfFrameTypeMask) == kFcfFrameMacCmd)
139  {
140  length += kCommandIdSize;
141  }
142 
143  SetPsduLength(length + GetFooterLength());
144 
145  return kThreadError_None;
146 }

148 ThreadError Frame::ValidatePsdu(void)
149 {
150  ThreadError error = kThreadError_Parse;
151  uint8_t offset = 0;
152  uint16_t fcf;
153  uint8_t footerLength = kFcsSize;
154 
155  VerifyOrExit((offset += kFcfSize + kDsnSize) <= GetPsduLength(),);
156  fcf = static_cast<uint16_t>((GetPsdu()[1] << 8) | GetPsdu()[0]);
157 
158  // Destinatinon PAN + Address
159  switch (fcf & Frame::kFcfDstAddrMask)
160  {
161  case Frame::kFcfDstAddrNone:
162  break;
163 
164  case Frame::kFcfDstAddrShort:
165  offset += sizeof(PanId) + sizeof(ShortAddress);
166  break;
167 
168  case Frame::kFcfDstAddrExt:
169  offset += sizeof(PanId) + sizeof(ExtAddress);
170  break;
171 
172  default:
173  goto exit;
174  }
175 
176  // Source PAN + Address
177  switch (fcf & Frame::kFcfSrcAddrMask)
178  {
179  case Frame::kFcfSrcAddrNone:
180  break;
181 
182  case Frame::kFcfSrcAddrShort:
183  if ((fcf & Frame::kFcfPanidCompression) == 0)
184  {
185  offset += sizeof(PanId);
186  }
187 
188  offset += sizeof(ShortAddress);
189  break;
190 
191  case Frame::kFcfSrcAddrExt:
192  if ((fcf & Frame::kFcfPanidCompression) == 0)
193  {
194  offset += sizeof(PanId);
195  }
196 
197  offset += sizeof(ExtAddress);
198  break;
199 
200  default:
201  goto exit;
202  }
203 
204  // Security Header
205  if (fcf & Frame::kFcfSecurityEnabled)
206  {
207  uint8_t secControl = GetPsdu()[offset];
208 
209  offset += kSecurityControlSize + kFrameCounterSize;
210 
211  switch (secControl & kKeyIdModeMask)
212  {
213  case kKeyIdMode0:
214  offset += kKeySourceSizeMode0;
215  break;
216 
217  case kKeyIdMode1:
218  offset += kKeySourceSizeMode1 + kKeyIndexSize;
219  break;
220 
221  case kKeyIdMode2:
222  offset += kKeySourceSizeMode2 + kKeyIndexSize;
223  break;
224 
225  case kKeyIdMode3:
226  offset += kKeySourceSizeMode3 + kKeyIndexSize;
227  break;
228  }
229 
230  switch (secControl & kSecLevelMask)
231  {
232  case kSecNone:
233  case kSecEnc:
234  footerLength += kMic0Size;
235  break;
236 
237  case kSecMic32:
238  case kSecEncMic32:
239  footerLength += kMic32Size;
240  break;
241 
242  case kSecMic64:
243  case kSecEncMic64:
244  footerLength += kMic64Size;
245  break;
246 
247  case kSecMic128:
248  case kSecEncMic128:
249  footerLength += kMic128Size;
250  break;
251  }
252  }
253 
254  // Command ID
255  if ((fcf & kFcfFrameTypeMask) == kFcfFrameMacCmd)
256  {
257  offset += kCommandIdSize;
258  }
259 
260  VerifyOrExit((offset + footerLength) <= GetPsduLength(),);
261 
262  error = kThreadError_None;
263 
264 exit:
265  return error;
266 }

916 uint8_t *Frame::GetPayload(void)
917 {
918  uint8_t *cur = GetPsdu();
919  uint16_t fcf = static_cast<uint16_t>((GetPsdu()[1] << 8) | GetPsdu()[0]);
920  uint8_t securityControl;
921 
922  // Frame Control
923  cur += kFcfSize;
924  // Sequence Number
925  cur += kDsnSize;
926 
927  // Destination PAN + Address
928  switch (fcf & Frame::kFcfDstAddrMask)
929  {
930  case Frame::kFcfDstAddrNone:
931  break;
932 
933  case Frame::kFcfDstAddrShort:
934  cur += sizeof(PanId) + sizeof(ShortAddress);
935  break;
936 
937  case Frame::kFcfDstAddrExt:
938  cur += sizeof(PanId) + sizeof(ExtAddress);
939  break;
940 
941  default:
942  ExitNow(cur = NULL);
943  }
944 
945  // Source PAN + Address
946  switch (fcf & Frame::kFcfSrcAddrMask)
947  {
948  case Frame::kFcfSrcAddrNone:
949  break;
950 
951  case Frame::kFcfSrcAddrShort:
952  if ((fcf & Frame::kFcfPanidCompression) == 0)
953  {
954  cur += sizeof(PanId);
955  }
956 
957  cur += sizeof(ShortAddress);
958  break;
959 
960  case Frame::kFcfSrcAddrExt:
961  if ((fcf & Frame::kFcfPanidCompression) == 0)
962  {
963  cur += sizeof(PanId);
964  }
965 
966  cur += sizeof(ExtAddress);
967  break;
968 
969  default:
970  ExitNow(cur = NULL);
971  }
972 
973  // Security Control + Frame Counter + Key Identifier
974  if ((fcf & Frame::kFcfSecurityEnabled) != 0)
975  {
976  securityControl = *cur;
977 
978  cur += kSecurityControlSize + kFrameCounterSize;
979 
980  switch (securityControl & kKeyIdModeMask)
981  {
982  case kKeyIdMode0:
983  cur += kKeySourceSizeMode0;
984  break;
985 
986  case kKeyIdMode1:
987  cur += kKeySourceSizeMode1 + kKeyIndexSize;
988  break;
989 
990  case kKeyIdMode2:
991  cur += kKeySourceSizeMode2 + kKeyIndexSize;
992  break;
993 
994  case kKeyIdMode3:
995  cur += kKeySourceSizeMode3 + kKeyIndexSize;
996  break;
997  }
998  }
999 
1000  // Command ID
1001  if ((fcf & kFcfFrameTypeMask) == kFcfFrameMacCmd)
1002  {
1003  cur += kCommandIdSize;
1004  }
1005 
1006 exit:
1007  return cur;
1008 }
