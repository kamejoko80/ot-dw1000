56 _IRQL_requires_max_(PASSIVE_LEVEL)
57 NDIS_STATUS
58 otLwfCmdInitialize(
59  _In_ PMS_FILTER pFilter
60  )
61 {
62  NDIS_STATUS Status = NDIS_STATUS_SUCCESS;
63  NTSTATUS NtStatus = STATUS_SUCCESS;
64  uint32_t InterfaceType = 0;
65 
66  NET_BUFFER_LIST_POOL_PARAMETERS PoolParams =
67  {
68  { NDIS_OBJECT_TYPE_DEFAULT, NET_BUFFER_LIST_POOL_PARAMETERS_REVISION_1, NDIS_SIZEOF_NET_BUFFER_LIST_POOL_PARAMETERS_REVISION_1 },
69  NDIS_PROTOCOL_ID_DEFAULT,
70  TRUE,
71  0,
72  'lbNC', // CNbl
73  0
74  };
75 
76  LogFuncEntry(DRIVER_DEFAULT);
77 
78  do
79  {
80  pFilter->cmdTIDsInUse = 0;
81  pFilter->cmdNextTID = 1;
82  pFilter->cmdResetReason = kPlatResetReason_PowerOn;
83 
84  NdisAllocateSpinLock(&pFilter->cmdLock);
85  InitializeListHead(&pFilter->cmdHandlers);
86 
87  KeInitializeEvent(
88  &pFilter->cmdResetCompleteEvent,
89  SynchronizationEvent, // auto-clearing event
90  FALSE // event initially non-signalled
91  );
92 
93  // Enable rundown protection
94  ExReInitializeRundownProtection(&pFilter->cmdRundown);
95 
96  // Create the NDIS pool for creating the SendNetBufferList
97  pFilter->cmdNblPool = NdisAllocateNetBufferListPool(pFilter->FilterHandle, &PoolParams);
98  if (pFilter->cmdNblPool == NULL)
99  {
100  Status = NDIS_STATUS_RESOURCES;
101  LogWarning(DRIVER_DEFAULT, "Failed to create NetBufferList pool for Spinel commands");
102  break;
103  }
104 
105  // Query the interface type to make sure it is a Thread device
106 #ifdef COMMAND_INIT_RETRY
107  pFilter->cmdInitTryCount = 0;
108  while (pFilter->cmdInitTryCount < 10)
109  {
110  NtStatus = otLwfCmdGetProp(pFilter, NULL, SPINEL_PROP_INTERFACE_TYPE, SPINEL_DATATYPE_UINT_PACKED_S, &InterfaceType);
111  if (!NT_SUCCESS(NtStatus))
112  {
113  pFilter->cmdInitTryCount++;
114  NdisMSleep(100);
115  continue;
116  }
117  break;
118  }
119  if (pFilter->cmdInitTryCount >= 10)
120  {
121 #else
122  NtStatus = otLwfCmdGetProp(pFilter, NULL, SPINEL_PROP_INTERFACE_TYPE, SPINEL_DATATYPE_UINT_PACKED_S, &InterfaceType);
123  if (!NT_SUCCESS(NtStatus))
124  {
125 #endif
126  Status = NDIS_STATUS_NOT_SUPPORTED;
127  LogError(DRIVER_DEFAULT, "Failed to query SPINEL_PROP_INTERFACE_TYPE, %!STATUS!", NtStatus);
128  break;
129  }
130  if (InterfaceType != SPINEL_PROTOCOL_TYPE_THREAD)
131  {
132  Status = NDIS_STATUS_NOT_SUPPORTED;
133  LogError(DRIVER_DEFAULT, "SPINEL_PROP_INTERFACE_TYPE is invalid, %d", InterfaceType);
134  break;
135  }
136 
137  NtStatus = otLwfCmdResetDevice(pFilter, FALSE);
138  if (!NT_SUCCESS(NtStatus))
139  {
140  Status = NDIS_STATUS_FAILURE;
141  break;
142  }
143 
144  } while (FALSE);
145 
146  LogFuncExitNDIS(DRIVER_DEFAULT, Status);
147 
148  // Clean up on failure
149  if (Status != NDIS_STATUS_SUCCESS)
150  {
151  otLwfCmdUninitialize(pFilter);
152  }
153 
154  return Status;
155 }

157 _IRQL_requires_max_(PASSIVE_LEVEL)
158 void
159 otLwfCmdUninitialize(
160  _In_ PMS_FILTER pFilter
161  )
162 {
163  LogFuncEntry(DRIVER_DEFAULT);
164 
165  // Release and wait for run down. This will block waiting for any pending sends to complete
166  ExWaitForRundownProtectionRelease(&pFilter->cmdRundown);
167 
168  // Use the NBL Pool variable as a flag for initialization
169  if (pFilter->cmdNblPool)
170  {
171  // Clean up any pending handlers
172  PLIST_ENTRY Link = pFilter->cmdHandlers.Flink;
173  while (Link != &pFilter->cmdHandlers)
174  {
175  SPINEL_CMD_HANDLER_ENTRY* pEntry = CONTAINING_RECORD(Link, SPINEL_CMD_HANDLER_ENTRY, Link);
176  Link = Link->Flink;
177 
178  if (pEntry->Handler)
179  {
180  pEntry->Handler(pFilter, pEntry->Context, 0, 0, NULL, 0);
181  }
182 
183  ReleaseEntryRef(pEntry);
184  }
185  InitializeListHead(&pFilter->cmdHandlers);
186 
187  // Free NBL Pool
188  NdisFreeNetBufferPool(pFilter->cmdNblPool);
189  pFilter->cmdNblPool = NULL;
190  }
191 
192  LogFuncExit(DRIVER_DEFAULT);
193 }

199 _IRQL_requires_max_(DISPATCH_LEVEL)
200 void
201 otLwfCmdProcess(
202  _In_ PMS_FILTER pFilter,
203  _In_ BOOLEAN DispatchLevel,
204  _In_ UINT command,
205  _In_reads_bytes_(cmd_data_len) const uint8_t* cmd_data_ptr,
206  _In_ spinel_size_t cmd_data_len
207  )
208 {
209  uint8_t Header;
210  spinel_prop_key_t key;
211  uint8_t* value_data_ptr = NULL;
212  spinel_size_t value_data_len = 0;
213 
214  // Make sure it's an expected command
215  if (command < SPINEL_CMD_PROP_VALUE_IS || command > SPINEL_CMD_PROP_VALUE_REMOVED)
216  {
217  LogVerbose(DRIVER_DEFAULT, "Recieved unhandled command, %u", command);
218  return;
219  }
220 
221  // Decode the key and data
222  if (spinel_datatype_unpack(cmd_data_ptr, cmd_data_len, "CiiD", &Header, NULL, &key, &value_data_ptr, &value_data_len) == -1)
223  {
224  LogVerbose(DRIVER_DEFAULT, "Failed to unpack command key & data");
225  return;
226  }
227 
228  // Get the transaction ID
229  if (SPINEL_HEADER_GET_TID(Header) == 0)
230  {
231  // Handle out of band last status locally
232  if (command == SPINEL_CMD_PROP_VALUE_IS && key == SPINEL_PROP_LAST_STATUS)
233  {
234  // Check if this is a reset
235  spinel_status_t status = SPINEL_STATUS_OK;
236  spinel_datatype_unpack(value_data_ptr, value_data_len, "i", &status);
237 
238  if ((status >= SPINEL_STATUS_RESET__BEGIN) && (status <= SPINEL_STATUS_RESET__END))
239  {
240  LogInfo(DRIVER_DEFAULT, "Interface %!GUID! was reset (status %d).", &pFilter->InterfaceGuid, status);
241  pFilter->cmdResetReason = status - SPINEL_STATUS_RESET__BEGIN;
242  KeSetEvent(&pFilter->cmdResetCompleteEvent, IO_NO_INCREMENT, FALSE);
243 
244  // TODO - Should this be passed on to Thread or Tunnel logic?
245  NT_ASSERT(pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_UNINTIALIZED);
246  }
247  }
248  else if (ExAcquireRundownProtection(&pFilter->ExternalRefs))
249  {
250  // If this is a 'Value Is' command, process it for notification of state changes.
251  if (command == SPINEL_CMD_PROP_VALUE_IS)
252  {
253  if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_RADIO_MODE)
254  {
255  otLwfThreadValueIs(pFilter, DispatchLevel, key, value_data_ptr, value_data_len);
256  }
257  else if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_THREAD_MODE)
258  {
259  otLwfTunValueIs(pFilter, DispatchLevel, key, value_data_ptr, value_data_len);
260  }
261  }
262  else if (command == SPINEL_CMD_PROP_VALUE_INSERTED)
263  {
264  if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_RADIO_MODE)
265  {
266  otLwfThreadValueInserted(pFilter, DispatchLevel, key, value_data_ptr, value_data_len);
267  }
268  else if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_THREAD_MODE)
269  {
270  otLwfTunValueInserted(pFilter, DispatchLevel, key, value_data_ptr, value_data_len);
271  }
272  }
273 
274  ExReleaseRundownProtection(&pFilter->ExternalRefs);
275  }
276  }
277  // If there was a transaction ID, then look for the corresponding command handler
278  else
279  {
280  PLIST_ENTRY Link;
281  SPINEL_CMD_HANDLER_ENTRY* Handler = NULL;
282 
283  FILTER_ACQUIRE_LOCK(&pFilter->cmdLock, DispatchLevel);
284 
285  // Search for matching handlers for this command
286  Link = pFilter->cmdHandlers.Flink;
287  while (Link != &pFilter->cmdHandlers)
288  {
289  SPINEL_CMD_HANDLER_ENTRY* pEntry = CONTAINING_RECORD(Link, SPINEL_CMD_HANDLER_ENTRY, Link);
290  Link = Link->Flink;
291 
292  if (SPINEL_HEADER_GET_TID(Header) == pEntry->TransactionId)
293  {
294  // Remove from the main list
295  RemoveEntryList(&pEntry->Link);
296 
297  // Cache the handler
298  Handler = pEntry;
299 
300  // Remove the transaction ID from the 'in use' bit field
301  pFilter->cmdTIDsInUse &= ~(1 << pEntry->TransactionId);
302 
303  break;
304  }
305  }
306 
307  FILTER_RELEASE_LOCK(&pFilter->cmdLock, DispatchLevel);
308 
309  // TODO - Set event
310 
311  // Process the handler we found, outside the lock
312  if (Handler)
313  {
314  // Call the handler function
315  Handler->Handler(pFilter, Handler->Context, command, key, value_data_ptr, value_data_len);
316 
317  // Free the entry
318  ReleaseEntryRef(Handler);
319  }
320  }
321 }

323 _IRQL_requires_max_(DISPATCH_LEVEL)
324 void
325 otLwfCmdRecveive(
326  _In_ PMS_FILTER pFilter,
327  _In_ BOOLEAN DispatchLevel,
328  _In_reads_bytes_(BufferLength) const PUCHAR Buffer,
329  _In_ ULONG BufferLength
330  )
331 {
332  uint8_t Header;
333  UINT Command;
334 
335  // Unpack the header from the buffer
336  if (spinel_datatype_unpack(Buffer, BufferLength, "Ci", &Header, &Command) <= 0)
337  {
338  LogVerbose(DRIVER_DEFAULT, "Failed to unpack header and command");
339  return;
340  }
341 
342  // Validate the header
343  if ((Header & SPINEL_HEADER_FLAG) != SPINEL_HEADER_FLAG)
344  {
345  LogVerbose(DRIVER_DEFAULT, "Recieved unrecognized frame, header=0x%x", Header);
346  return;
347  }
348 
349  // We only support IID zero for now
350  if (SPINEL_HEADER_GET_IID(Header) != 0)
351  {
352  LogVerbose(DRIVER_DEFAULT, "Recieved unsupported IID, %u", SPINEL_HEADER_GET_IID(Header));
353  return;
354  }
355 
356  // Process the received command
357  otLwfCmdProcess(pFilter, DispatchLevel, Command, Buffer, BufferLength);
358 }

364 _IRQL_requires_max_(PASSIVE_LEVEL)
365 spinel_tid_t
366 otLwfCmdGetNextTID(
367  _In_ PMS_FILTER pFilter
368  )
369 {
370  spinel_tid_t TID = 0;
371  while (TID == 0)
372  {
373  NdisAcquireSpinLock(&pFilter->cmdLock);
374 
375  if (((1 << pFilter->cmdNextTID) & pFilter->cmdTIDsInUse) == 0)
376  {
377  TID = pFilter->cmdNextTID;
378  pFilter->cmdNextTID = SPINEL_GET_NEXT_TID(pFilter->cmdNextTID);
379  pFilter->cmdTIDsInUse |= (1 << TID);
380  }
381 
382  NdisReleaseSpinLock(&pFilter->cmdLock);
383 
384  if (TID == 0)
385  {
386  // TODO - Wait for event
387  }
388  }
389  return TID;
390 }

416 _IRQL_requires_max_(PASSIVE_LEVEL)
417 NTSTATUS
418 otLwfCmdEncodeAndSendAsync(
419  _In_ PMS_FILTER pFilter,
420  _In_ UINT Command,
421  _In_ spinel_prop_key_t Key,
422  _In_ spinel_tid_t tid,
423  _In_ ULONG MaxDataLength,
424  _In_opt_ const char *pack_format,
425  _In_opt_ va_list args
426  )
427 {
428  NTSTATUS status = STATUS_SUCCESS;
429  PNET_BUFFER_LIST NetBufferList = NULL;
430  PNET_BUFFER NetBuffer = NULL;
431  ULONG NetBufferLength = 0;
432  PUCHAR DataBuffer = NULL;
433  spinel_ssize_t PackedLength;
434 
435  LogFuncEntryMsg(DRIVER_DEFAULT, "Cmd=%u Key=%u tid=%u", (ULONG)Command, (ULONG)Key, (ULONG)tid);
436 
437  NetBufferList =
438  NdisAllocateNetBufferAndNetBufferList(
439  pFilter->cmdNblPool, // PoolHandle
440  0, // ContextSize
441  0, // ContextBackFill
442  NULL, // MdlChain
443  0, // DataOffset
444  0 // DataLength
445  );
446  if (NetBufferList == NULL)
447  {
448  status = STATUS_INSUFFICIENT_RESOURCES;
449  LogWarning(DRIVER_DEFAULT, "Failed to create command NetBufferList");
450  goto exit;
451  }
452 
453  // Initialize NetBuffer fields
454  NetBuffer = NET_BUFFER_LIST_FIRST_NB(NetBufferList);
455  NET_BUFFER_CURRENT_MDL(NetBuffer) = NULL;
456  NET_BUFFER_CURRENT_MDL_OFFSET(NetBuffer) = 0;
457  NET_BUFFER_DATA_LENGTH(NetBuffer) = 0;
458  NET_BUFFER_DATA_OFFSET(NetBuffer) = 0;
459  NET_BUFFER_FIRST_MDL(NetBuffer) = NULL;
460 
461  // Calculate length of NetBuffer
462  NetBufferLength = 16 + MaxDataLength;
463  if (NetBufferLength < 64) NetBufferLength = 64;
464 
465  // Allocate the NetBuffer for NetBufferList
466  if (NdisRetreatNetBufferDataStart(NetBuffer, NetBufferLength, 0, NULL) != NDIS_STATUS_SUCCESS)
467  {
468  NetBuffer = NULL;
469  status = STATUS_INSUFFICIENT_RESOURCES;
470  LogError(DRIVER_DEFAULT, "Failed to allocate NB for command NetBufferList, %u bytes", NetBufferLength);
471  goto exit;
472  }
473 
474  // Get the pointer to the data buffer
475  DataBuffer = (PUCHAR)NdisGetDataBuffer(NetBuffer, NetBufferLength, NULL, 1, 0);
476  NT_ASSERT(DataBuffer);
477 
478  // Save the true NetBuffer length in the protocol reserved
479  NetBuffer->ProtocolReserved[0] = (PVOID)NetBufferLength;
480  NetBuffer->DataLength = 0;
481 
482  // Save the transaction ID in the protocol reserved
483  NetBuffer->ProtocolReserved[1] = (PVOID)tid;
484 
485  // Pack the header, command and key
486  PackedLength =
487  spinel_datatype_pack(
488  DataBuffer,
489  NetBufferLength,
490  "Cii",
491  SPINEL_HEADER_FLAG | SPINEL_HEADER_IID_0 | tid,
492  Command,
493  Key);
494  if (PackedLength < 0 || PackedLength + NetBuffer->DataLength > NetBufferLength)
495  {
496  status = STATUS_INSUFFICIENT_RESOURCES;
497  goto exit;
498  }
499 
500  NetBuffer->DataLength += (ULONG)PackedLength;
501 
502  // Pack the data (if any)
503  if (pack_format)
504  {
505  PackedLength =
506  spinel_datatype_vpack(
507  DataBuffer + NetBuffer->DataLength,
508  NetBufferLength - NetBuffer->DataLength,
509  pack_format,
510  args);
511  if (PackedLength < 0 || PackedLength + NetBuffer->DataLength > NetBufferLength)
512  {
513  status = STATUS_INSUFFICIENT_RESOURCES;
514  goto exit;
515  }
516 
517  NetBuffer->DataLength += (ULONG)PackedLength;
518  }
519 
520  // Grab a ref for rundown protection
521  if (!ExAcquireRundownProtection(&pFilter->cmdRundown))
522  {
523  status = STATUS_DEVICE_NOT_READY;
524  LogWarning(DRIVER_DEFAULT, "Failed to acquire rundown protection");
525  goto exit;
526  }
527 
528  // Send the NBL down
529  NdisFSendNetBufferLists(
530  pFilter->FilterHandle,
531  NetBufferList,
532  NDIS_DEFAULT_PORT_NUMBER,
533  0);
534 
535  // Clear local variable because we don't own the NBL any more
536  NetBufferList = NULL;
537 
538 exit:
539 
540  if (NetBufferList)
541  {
542  if (NetBuffer)
543  {
544  NetBuffer->DataLength = (ULONG)(ULONG_PTR)NetBuffer->ProtocolReserved[0];
545  NdisAdvanceNetBufferDataStart(NetBuffer, NetBuffer->DataLength, TRUE, NULL);
546  }
547  NdisFreeNetBufferList(NetBufferList);
548  }
549 
550  LogFuncExitNT(DRIVER_DEFAULT, status);
551 
552  return status;
553 }

590 _IRQL_requires_max_(PASSIVE_LEVEL)
591 NTSTATUS
592 otLwfCmdSendAsyncV(
593  _In_ PMS_FILTER pFilter,

602  )

625 
626  if (pTid) *pTid = pEntry->TransactionId;
627  }
628

630

635

637  RemoveEntryList(&pEntry->Link);
638

640  pFilter->cmdTIDsInUse &= ~(1 << pEntry->TransactionId);
641 
642  NdisReleaseSpinLock(&pFilter->cmdLock);
643 
644  // TODO - Set event
645 
646  ReleaseEntryRef(pEntry);
647  }
648 
649 exit:
650 
651  if (pEntry) ReleaseEntryRef(pEntry);
652 
653  return status;
654 }

678 _IRQL_requires_max_(DISPATCH_LEVEL)
679 BOOLEAN
680 otLwfCmdCancel(
681  _In_ PMS_FILTER pFilter,
682  _In_ BOOLEAN DispatchLevel,
683  _In_ spinel_tid_t tid
684  )
685 {
686  PLIST_ENTRY Link;
687  SPINEL_CMD_HANDLER_ENTRY* Handler = NULL;
688  BOOLEAN Found = FALSE;
689 
690  LogFuncEntryMsg(DRIVER_DEFAULT, "tid=%u", (ULONG)tid);
691 
692  FILTER_ACQUIRE_LOCK(&pFilter->cmdLock, DispatchLevel);
693 
694  // Search for matching handlers for this transaction ID
695  Link = pFilter->cmdHandlers.Flink;
