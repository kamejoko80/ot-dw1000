72 void Dhcp6Client::UpdateAddresses(otInstance *aInstance, otDhcpAddress *aAddresses, uint32_t aNumAddresses,
73  void *aContext)
74 {
75  (void)aContext;
76  bool found = false;;
77  bool newAgent = false;
78  otDhcpAddress *address = NULL;
79  otNetworkDataIterator iterator;
80  otBorderRouterConfig config;
81 
82  mAddresses = aAddresses;
83  mNumAddresses = aNumAddresses;
84 
85  // remove addresses directly if prefix not valid in network data
86  for (uint8_t i = 0; i < mNumAddresses; i++)
87  {
88  address = &mAddresses[i];
89 
90  if (address->mValidLifetime == 0)
91  {
92  continue;
93  }
94 
95  found = false;
96  iterator = OT_NETWORK_DATA_ITERATOR_INIT;
97 
98  while ((otGetNextOnMeshPrefix(aInstance, false, &iterator, &config)) == kThreadError_None)
99  {
100  if (!config.mDhcp)
101  {
102  continue;
103  }
104 
105  if ((otIp6PrefixMatch(&(address->mAddress.mAddress), &(config.mPrefix.mPrefix)) >=
106  address->mAddress.mPrefixLength) &&
107  (config.mPrefix.mLength == address->mAddress.mPrefixLength))
108  {
109  found = true;
110  break;
111  }
112  }
113 
114  if (!found)
115  {
116  otRemoveUnicastAddress(aInstance, &(address->mAddress.mAddress));
117  RemoveIdentityAssociation(config.mRloc16, config.mPrefix);
118  memset(address, 0, sizeof(*address));
119  }
120  }
121 
122  // add IdentityAssociation for new configured prefix
123  iterator = OT_NETWORK_DATA_ITERATOR_INIT;
124 
125  while (otGetNextOnMeshPrefix(aInstance, false, &iterator, &config) == kThreadError_None)
126  {
127  if (!config.mDhcp)
128  {
129  continue;
130  }
131 
132  found = false;
133 
134  for (uint8_t i = 0; i < mNumAddresses; i++)
135  {
136  address = &mAddresses[i];
137 
138  if (address->mAddress.mPrefixLength == 0)
139  {
140  continue;
141  }
142 
143  if ((otIp6PrefixMatch(&(config.mPrefix.mPrefix), &(address->mAddress.mAddress)) >=
144  config.mPrefix.mLength) &&
145  (config.mPrefix.mLength == address->mAddress.mPrefixLength))
146  {
147  found = true;
148  break;
149  }
150  }
151 
152  if (!found)
153  {
154  for (size_t i = 0; i < mNumAddresses; i++)
155  {
156  address = &mAddresses[i];
157 
158  if (address->mAddress.mPrefixLength != 0)
159  {
160  continue;
161  }
162 
163  memset(address, 0, sizeof(*address));
164 
165  // suppose all configured prefix are ::/64
166  memcpy(address->mAddress.mAddress.mFields.m8, config.mPrefix.mPrefix.mFields.m8, 8);
167  address->mAddress.mPrefixLength = config.mPrefix.mLength;
168 
169  AddIdentityAssociation(config.mRloc16, config.mPrefix);
170  newAgent = true;
171  break;
172  }
173  }
174  }
175 
176  if (newAgent)
177  {
178  Start();
179  }
180  else
181  {
182  Stop();
183  }

219 void Dhcp6Client::RemoveIdentityAssociation(uint16_t aRloc16, otIp6Prefix &aIp6Prefix)
220 {
221  IdentityAssociation *prevIdentityAssociation = NULL;
222  IdentityAssociation *identityAssociation = NULL;
223 
224  VerifyOrExit(mIdentityAssociationHead, ;);
225 
226  for (identityAssociation = mIdentityAssociationHead; identityAssociation;
227  prevIdentityAssociation = identityAssociation, identityAssociation = identityAssociation->GetNext())
228  {
229  if (identityAssociation->GetPrefixAgentRloc() != aRloc16)
230  {
231  continue;
232  }
233 
234  if (otIp6PrefixMatch(&(aIp6Prefix.mPrefix), &(identityAssociation->GetPrefix()->mPrefix)) < aIp6Prefix.mLength)
235  {
236  continue;
237  }
238 
239  // remove from used list
240  if (prevIdentityAssociation)
241  {
242  prevIdentityAssociation->SetNext(identityAssociation->GetNext());
243  }
244  else
245  {
246  mIdentityAssociationHead = identityAssociation->GetNext();
247  }
248 
249  // return to available list
250  memset(identityAssociation, 0, sizeof(*identityAssociation));
251  identityAssociation->SetNext(mIdentityAssociationAvail);
252  mIdentityAssociationAvail = identityAssociation;
253  break;
254  }
255 
256 exit:
257  {}

279 bool Dhcp6Client::ProcessNextIdentityAssociation()
280 {
281  bool rval = false;
282  IdentityAssociation *prevIdentityAssociation = NULL;
283  IdentityAssociation *identityAssociation = NULL;
284 
285  VerifyOrExit(mIdentityAssociationHead, ;);
286  // not interrupt in-progress solicit
287  VerifyOrExit((mIdentityAssociationHead->GetStatus() != IdentityAssociation::kStatusSoliciting), ;);
288 
289  mTrickleTimer.Stop();
290 
291  for (identityAssociation = mIdentityAssociationHead; identityAssociation;
292  prevIdentityAssociation = identityAssociation, identityAssociation = identityAssociation->GetNext())
293  {
294  if (identityAssociation->GetStatus() != IdentityAssociation::kStatusSolicit)
295  {
296  continue;
297  }
298 
299  // new transaction id
300  for (uint8_t i = 0; i < kTransactionIdSize; i++)
301  {
302  mTransactionId[i] = static_cast<uint8_t>(otPlatRandomGet());
303  }
304 
305  // ensure mIdentityAssociationHead is the prefix agent to solicit.
306  if (prevIdentityAssociation)
307  {
308  prevIdentityAssociation->SetNext(identityAssociation->GetNext());
309  identityAssociation->SetNext(mIdentityAssociationHead);
310  mIdentityAssociationHead = identityAssociation;
311  }
312 
313  mTrickleTimer.Start(
314  Timer::SecToMsec(kTrickleTimerImin),
315  Timer::SecToMsec(kTrickleTimerImax),
316  TrickleTimer::kModeNormal);
317 
318  mTrickleTimer.IndicateInconsistent();
319 
320  ExitNow(rval = true);
321  }
322 
323 exit:
324  return rval;

333 bool Dhcp6Client::HandleTrickleTimer(void)
334 {
335  bool rval = true;
336 
337  VerifyOrExit(mIdentityAssociationHead, rval = false);
338 
339  switch (mIdentityAssociationHead->GetStatus())
340  {
341  case IdentityAssociation::kStatusSolicit:
342  mStartTime = otPlatAlarmGetNow();
343  mIdentityAssociationHead->SetStatus(IdentityAssociation::kStatusSoliciting);
344 
345  // fall through
346 
347  case IdentityAssociation::kStatusSoliciting:
348  Solicit(mIdentityAssociationHead->GetPrefixAgentRloc());
349  break;
350 
351  case IdentityAssociation::kStatusSolicitReplied:
352  if (!ProcessNextIdentityAssociation())
353  {
354  mTrickleTimer.Stop();
355  Stop();
356  rval = false;
357  }
358 
359  break;
360 
361  default:
362  break;
363  }
364 
365 exit:
366  return rval;

369 ThreadError Dhcp6Client::Solicit(uint16_t aRloc16)
370 {
371  ThreadError error = kThreadError_None;
372  Message *message;
373  Ip6::MessageInfo messageInfo;
374 
375  VerifyOrExit((message = mSocket.NewMessage(0)) != NULL, error = kThreadError_NoBufs);
376 
377  SuccessOrExit(error = AppendHeader(*message));
378  SuccessOrExit(error = AppendElapsedTime(*message));
379  SuccessOrExit(error = AppendClientIdentifier(*message));
380  SuccessOrExit(error = AppendIaNa(*message, aRloc16));
381  // specify which prefixes to solicit
382  SuccessOrExit(error = AppendIaAddress(*message, aRloc16));
383  SuccessOrExit(error = AppendRapidCommit(*message));
384 
385  memset(&messageInfo, 0, sizeof(messageInfo));
386  memcpy(messageInfo.GetPeerAddr().mFields.m8, mNetif.GetMle().GetMeshLocalPrefix(), 8);
387  messageInfo.GetPeerAddr().mFields.m16[4] = HostSwap16(0x0000);
388  messageInfo.GetPeerAddr().mFields.m16[5] = HostSwap16(0x00ff);
389  messageInfo.GetPeerAddr().mFields.m16[6] = HostSwap16(0xfe00);
390  messageInfo.GetPeerAddr().mFields.m16[7] = HostSwap16(aRloc16);
391  messageInfo.SetSockAddr(mNetif.GetMle().GetMeshLocal16());
392  messageInfo.mPeerPort = kDhcpServerPort;
393  messageInfo.mInterfaceId = mNetif.GetInterfaceId();
394 
395  SuccessOrExit(error = mSocket.SendTo(*message, messageInfo));
396  otLogInfoIp6("solicit\n");
397 
398 exit:
399 
400  if (message != NULL && error != kThreadError_None)
401  {
402  message->Free();
403  }
404 
405  return error;
406 }

662 ThreadError Dhcp6Client::ProcessIaAddress(Message &aMessage, uint16_t aOffset)
663 {
664  ThreadError error = kThreadError_None;
665  IdentityAssociation *identityAssociation = NULL;
666  otDhcpAddress *address = NULL;
667  otIp6Prefix *prefix = NULL;
668 
669  IaAddress option;
670 
671  VerifyOrExit(((aMessage.Read(aOffset, sizeof(option), &option) == sizeof(option)) &&
672  (option.GetLength() == (sizeof(option) - sizeof(Dhcp6Option)))),
673  error = kThreadError_Parse);
674 
675  for (uint8_t i = 0; i < mNumAddresses; i++)
676  {
677  address = &mAddresses[i];
678 
679  if (address->mValidLifetime != 0)
680  {
681  continue;
682  }
683 
684  if (otIp6PrefixMatch(&(address->mAddress.mAddress), option.GetAddress()) >= address->mAddress.mPrefixLength)
685  {
686  memcpy(address->mAddress.mAddress.mFields.m8, option.GetAddress()->mFields.m8, sizeof(otIp6Address));
687  address->mPreferredLifetime = option.GetPreferredLifetime();
688  address->mValidLifetime = option.GetValidLifetime();
689  address->mAddress.mPreferred = address->mPreferredLifetime != 0;
690  address->mAddress.mValid = address->mValidLifetime != 0;
691  otAddUnicastAddress(mNetif.GetInstance(), &address->mAddress);
692  break;
693  }
694  }
695 
696  // mark IdentityAssociation as replied
697  for (identityAssociation = mIdentityAssociationHead; identityAssociation;
698  identityAssociation = identityAssociation->GetNext())
699  {
700  prefix = identityAssociation->GetPrefix();
701 
702  if (otIp6PrefixMatch(option.GetAddress(), &(prefix->mPrefix)) >= prefix->mLength)
703  {
704  identityAssociation->SetStatus(IdentityAssociation::kStatusSolicitReplied);
705  break;
706  }
707  }
708 
709 exit:
710  return error;
711 }
