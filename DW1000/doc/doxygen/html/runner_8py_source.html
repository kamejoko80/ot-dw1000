227  continue
228  case_name = unicode(case.__class__.__name__)
229 
230  # grep name
231  if name_greps and not any(fnmatch.fnmatch(case_name, name_grep) for name_grep in name_greps):
232  logger.info('case[%s] skipped by name greps', case_name)
233  continue
234 
235  # whitelist
236  if len(names) and case_name not in names:
237  logger.info('case[%s] skipped', case_name)
238  continue
239 
240  # skip cases
241  if case_name in log.keys():
242  if (log[case_name]['passed'] and ('p' in skip)) \
243  or (log[case_name]['passed'] is False and ('f' in skip)) \
244  or (log[case_name]['passed'] is None and ('e' in skip)):
245  logger.warning('case[%s] skipped for its status[%s]', case_name, log[case_name]['passed'])
246  continue

252  continue
253  else:

255 
256  # black list

258  logger.warning('case[%s] skipped for blacklist', case_name)
259  continue

263  logger.warning('case[%s] skipped for exceeding max golden devices allowed[%d]', case_name, max_devices)
