75  uint16_t payloadLength = otGetMessageLength(aMessage) - otGetMessageOffset(aMessage);
76  char buf[512];

78  VerifyOrExit(payloadLength <= sizeof(buf), ;);

99 int Udp::Output(const char *aBuf, uint16_t aBufLength)
100 {
101  ThreadError error = kThreadError_None;
102  otMessage message;
103 
104  VerifyOrExit((message = otNewUdpMessage(mInstance, true)) != NULL, error = kThreadError_NoBufs);
105  SuccessOrExit(error = otSetMessageLength(message, aBufLength));
106  otWriteMessage(message, 0, aBuf, aBufLength);
107  SuccessOrExit(error = otSendUdp(&mSocket, message, &mPeer));
108 
109 exit:
110 
111  if (error != kThreadError_None && message != NULL)
112  {
113  otFreeMessage(message);
114  aBufLength = 0;
115  }
116 
117  return aBufLength;
118 }
