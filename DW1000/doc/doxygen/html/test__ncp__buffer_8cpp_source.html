81 // Dump the buffer content to screen.
82 void DumpBuffer(const char *aTextMessage, uint8_t *aBuffer, uint16_t aBufferLength)
83 {
84  enum
85  {
86  kBytesPerLine = 32, // Number of bytes per line.
87  };
88 
89  char charBuff[kBytesPerLine + 1];
90  uint16_t counter;
91  uint8_t byte;
92 
93  printf("\n%s - len = %u\n ", aTextMessage, aBufferLength);
94 
95  counter = 0;
96 
97  while (aBufferLength--)
98  {
99  byte = *aBuffer++;
100  printf("%02X ", byte);
101  charBuff[counter] = isprint(byte) ? static_cast<char>(byte) : '.';
102  counter++;
103 
104  if (counter == kBytesPerLine)
105  {
106  charBuff[counter] = 0;
107  printf(" %s\n ", charBuff);
108  counter = 0;
109  }
110  }
111 
112  charBuff[counter] = 0;
113 
114  while (counter++ < kBytesPerLine)
115  {
116  printf(" ");
117  }
118 
119  printf(" %s\n", charBuff);
120 }
