75 void da15000RadioInit(void)
76 {
77  /* Wake up power domains */
78  REG_CLR_BIT(CRG_TOP, PMU_CTRL_REG, FTDF_SLEEP);
79 
80  while (REG_GETF(CRG_TOP, SYS_STAT_REG, FTDF_IS_UP) == 0x0);
81 
82  REG_CLR_BIT(CRG_TOP, PMU_CTRL_REG, RADIO_SLEEP);
83 
84  while (REG_GETF(CRG_TOP, SYS_STAT_REG, RAD_IS_UP) == 0x0);
85 
86  REG_SETF(CRG_TOP, CLK_RADIO_REG, FTDF_MAC_ENABLE, 1);
87  REG_SETF(CRG_TOP, CLK_RADIO_REG, FTDF_MAC_DIV, 0);
88 
89  hw_rf_poweron();
90  hw_rf_system_init();
91 
92  ad_ftdf_init_phy_api();
93 }

408 void da15000RadioProcess(otInstance *aInstance)
409 {
410  if (sSendFrameDone)
411  {
412  // Check FP bit in ACK response
413  if ((sTransmitFrame.mPsdu[0] & IEEE802154_ACK_REQUEST) != 0 &&
414  (sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_TYPE_MASK) != 0)
415  {
416  sFramePending = ((sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_PENDING) != 0);
417  }
418 
419  sRadioState = kStateReceive;
420  otPlatRadioTransmitDone(aInstance, &sTransmitFrame, sFramePending, sTransmitStatus);
421 
422  sFramePending = false;
423  sSendFrameDone = false;
424  }
425 
426  if (sRadioState == kStateSleep && sGoSleep)
427  {
428  sGoSleep = false;
429 
430  sEnableRX = 0;
431  FTDF_setValue(FTDF_PIB_RX_ON_WHEN_IDLE, &sEnableRX);
432  ad_ftdf_sleep_when_possible(true);
433  }
434 }
