193  VerifyOrExit(mWriteSegmentHead == mWriteSegmentTail, ;);

214 // This function closes/ends the current segment.
215 void NcpFrameBuffer::InFrameEndSegment(uint16_t aHeaderFlags)
216 {
217  uint16_t segmentLength;
218  uint16_t header;
219 
220  segmentLength = GetDistance(mWriteSegmentHead, mWriteSegmentTail);
221 
222  if (segmentLength >= kSegmentHeaderSize)
223  {
224  // Reduce the header size.
225  segmentLength -= kSegmentHeaderSize;
226 
227  // Update the length and the flags in segment header (at segment head pointer).
228  header = ReadUint16At(mWriteSegmentHead);
229  header |= (segmentLength & kSegmentHeaderLengthMask);
230  header |= aHeaderFlags;
231  WriteUint16At(mWriteSegmentHead, header);
232 
233  // Move the segment head to current tail (to be ready for a possible next segment).
234  mWriteSegmentHead = mWriteSegmentTail;
235  }
236  else
237  {
238  // Remove the current segment (move the tail back to head).
239  mWriteSegmentTail = mWriteSegmentHead;

336 // Start/Prepare a new segment for reading.
337 ThreadError NcpFrameBuffer::OutFramePrepareSegment(void)
338 {
339  ThreadError error = kThreadError_None;
340  uint16_t header;
341 
342  while (true)
343  {
344  // Go to the next segment (set the segment head to current segment's end/tail).
345  mReadSegmentHead = mReadSegmentTail;
346 
347  // Ensure there is something to read (i.e. segment head is not at start of frame being written).
348  VerifyOrExit(mReadSegmentHead != mWriteFrameStart, error = kThreadError_NotFound);
349 
350  // Read the segment header.
351  header = ReadUint16At(mReadSegmentHead);
352 
353  // Check if this segment is the start of a frame.
354  if (header & kSegmentHeaderNewFrameFlag)
355  {
356  // Ensure that this segment is start of current frame, otherwise the current frame is finished.
357  VerifyOrExit(mReadSegmentHead == mReadFrameStart, error = kThreadError_NotFound);
358  }
359 
360  // Find tail/end of current segment.
361  mReadSegmentTail = Advance(mReadSegmentHead,
362  kSegmentHeaderSize + static_cast<uint8_t>(header & kSegmentHeaderLengthMask));
363 
364  // Update the current read pointer to skip the segment header.
365  mReadPointer = Advance(mReadSegmentHead, kSegmentHeaderSize);
366 
367  // Check if there are data bytes to be read in this segment (i.e. read pointer not at the tail).
368  if (mReadPointer != mReadSegmentTail)
369  {
370  // Update the state to `InSegment` and return.
371  mReadState = kReadStateInSegment;
372 
373  ExitNow();
374  }
375 
376  // No data in this segment, prepare any appended/associated message of this segment.
377  if (OutFramePrepareMessage() == kThreadError_None)
378  {
379  ExitNow();
380  }
381 
382  // If there is no message (`PrepareMessage()` returned an error), loop back to prepare the next segment.
383  }
384 
385 exit:
386  if (error != kThreadError_None)
387  {
388  mReadState = kReadStateDone;
389  }
390 
391  return error;
392 }

394 // This method prepares an associated message in current segment and fills the message buffer. It returns
395 // ThreadError_NotFound if there is no message or if the message has no content.
396 ThreadError NcpFrameBuffer::OutFramePrepareMessage(void)
397 {
398  ThreadError error = kThreadError_None;
399  uint16_t header;
400 
401  // Read the segment header
402  header = ReadUint16At(mReadSegmentHead);
403 
404  // Ensure that the segment header indicates that there is an associated message or return `NotFound` error.
405  VerifyOrExit((header & kSegmentHeaderMessageIndicatorFlag) != 0, error = kThreadError_NotFound);
406 
407  // Update the current message from the queue.
408  mReadMessage = (mReadMessage == NULL) ?
409  otMessageQueueGetHead(&mMessageQueue) :
410  otMessageQueueGetNext(&mMessageQueue, mReadMessage);
411 
412  VerifyOrExit(mReadMessage != NULL, error = kThreadError_NotFound);
413 
414  // Reset the offset for reading the message.
415  mReadMessageOffset = 0;
416 
417  // Fill the content from current message into the message buffer.
418  SuccessOrExit(error = OutFrameFillMessageBuffer());
419 
420  // If all successful, set the state to `InMessage`.
421  mReadState = kReadStateInMessage;
422 
423 exit:
424  return error;
425 }

475 uint8_t NcpFrameBuffer::OutFrameReadByte(void)
476 {
477  ThreadError error;
478  uint8_t retval = kReadByteAfterFrameHasEnded;
479 
480  switch (mReadState)
481  {
482  case kReadStateDone:
483 
484  retval = kReadByteAfterFrameHasEnded;
485 
486  break;
487 
488  case kReadStateInSegment:
489 
490  // Read a byte from current read pointer and move the read pointer forward.
491  retval = *mReadPointer;
492  mReadPointer = Next(mReadPointer);
493 
494  // Check if at end of current segment.
495  if (mReadPointer == mReadSegmentTail)
496  {
497  // Prepare any message associated with this segment.
498  error = OutFramePrepareMessage();
499 
500  // If there is no message, move to next segment (if any).
501  if (error != kThreadError_None)
502  {
503  OutFramePrepareSegment();
504  }
505  }
506 
507  break;
508 
509  case kReadStateInMessage:
510 
511  // Read a byte from current read pointer and move the read pointer forward.
512  retval = *mReadPointer;
513  mReadPointer++;
514 
515  // Check if at the end of content in message buffer.
516  if (mReadPointer == mReadMessageTail)
517  {
518  // Fill more bytes from current message into message buffer.
519  error = OutFrameFillMessageBuffer();
520 
521  // If no more bytes in the message, move to next segment (if any).
522  if (error != kThreadError_None)
523  {
524  OutFramePrepareSegment();
525  }
526  }
527 
528  break;
529  }
530 
531  return retval;
532 }

546 ThreadError NcpFrameBuffer::OutFrameRemove(void)
547 {
548  ThreadError error = kThreadError_None;
549  uint8_t *bufPtr;
550  otMessage message;
551  uint16_t header;
552 
553  VerifyOrExit(!IsEmpty(), error = kThreadError_NotFound);
554 
555  // Begin at the start of current frame and move through all segments.
556 
557  bufPtr = mReadFrameStart;
558 
559  while (bufPtr != mWriteFrameStart)
560  {
561  // Read the segment header
562  header = ReadUint16At(bufPtr);
563 
564  // If the current segment defines a new frame, and it is not the start of current frame, then we have reached
565  // end of current frame.
566  if (header & kSegmentHeaderNewFrameFlag)
567  {
568  if (bufPtr != mReadFrameStart)
569  {
570  break;
571  }
572  }
573 
574  // If current segment has an appended message, remove it from message queue and free it.
575  if (header & kSegmentHeaderMessageIndicatorFlag)
576  {
577  if ((message = otMessageQueueGetHead(&mMessageQueue)) != NULL)
578  {
579  otMessageQueueDequeue(&mMessageQueue, message);
580  otFreeMessage(message);
581  }
582  }
583 
584  // Move the pointer to next segment.
585  bufPtr = Advance(bufPtr, kSegmentHeaderSize + static_cast<uint8_t>(header & kSegmentHeaderLengthMask));
586  }
587 
588  mReadFrameStart = bufPtr;
589 
590  mReadState = kReadStateDone;
591  mReadFrameLength = kUnknownFrameLength;
592 
593  // If the remove causes the buffer to become empty, invoke the callback to signal this.
594  if (IsEmpty())
595  {
596  if (mEmptyBufferCallback != NULL)
597  {
598  mEmptyBufferCallback(mCallbackContext, this);
599  }
600  }

602 exit:

607 uint16_t NcpFrameBuffer::OutFrameGetLength(void)
608 {
609  uint16_t frameLength = 0;
610  uint16_t header;
611  uint8_t *bufPtr;
612  otMessage message = NULL;
613 
614  // If the frame length was calculated before, return the previously calculated length.
615  VerifyOrExit(mReadFrameLength == kUnknownFrameLength, frameLength = mReadFrameLength);
616 
617  VerifyOrExit(!IsEmpty(), frameLength = 0);
618 
619  // Calculate frame length by adding length of all segments and messages within the current frame.
620 
621  bufPtr = mReadFrameStart;
622 
623  while (bufPtr != mWriteFrameStart)
624  {
625  // Read the segment header
626  header = ReadUint16At(bufPtr);
627 
628  // If the current segment defines a new frame, and it is not the start of current frame, then we have reached
629  // end of current frame.
630  if (header & kSegmentHeaderNewFrameFlag)
631  {
632  if (bufPtr != mReadFrameStart)
633  {
634  break;
635  }
636  }
637 
638  // If current segment has an associated message, add its length to frame length.
639  if (header & kSegmentHeaderMessageIndicatorFlag)
640  {
641  message = (message == NULL) ?
642  otMessageQueueGetHead(&mMessageQueue) :
643  otMessageQueueGetNext(&mMessageQueue, message);
644 
645  if (message != NULL)
646  {
647  frameLength += otGetMessageLength(message);
648  }
649  }
650 
651  // Add the length of current segment to the frame length.
652  frameLength += (header & kSegmentHeaderLengthMask);
653 
654  // Move the pointer to next segment.
655  bufPtr = Advance(bufPtr, kSegmentHeaderSize + static_cast<uint8_t>(header & kSegmentHeaderLengthMask));
656  }
657 
658  // Remember the calculated frame length for current frame.
659  mReadFrameLength = frameLength;
660 
661 exit:
662  return frameLength;
663 }
