47 void AesCcm::Init(uint32_t aHeaderLength, uint32_t aPlainTextLength, uint8_t aTagLength,
48  const void *aNonce, uint8_t aNonceLength)
49 {
50  const uint8_t *nonceBytes = reinterpret_cast<const uint8_t *>(aNonce);
51  uint8_t blockLength = 0;
52  uint32_t len;
53  uint8_t L;
54  uint8_t i;
55 
56  // aTagLength must be even
57  aTagLength &= ~1;
58 
59  if (aTagLength > sizeof(mBlock))
60  {
61  aTagLength = sizeof(mBlock);
62  }
63 
64  L = 0;
65 
66  for (len = aPlainTextLength; len; len >>= 8)
67  {
68  L++;
69  }
70 
71  if (L <= 1)
72  {
73  L = 2;
74  }
75 
76  if (aNonceLength > 13)
77  {
78  aNonceLength = 13;
79  }
80 
81  // increase L to match nonce len
82  if (L < (15 - aNonceLength))
83  {
84  L = 15 - aNonceLength;
85  }
86 
87  // decrease nonceLength to match L
88  if (aNonceLength > (15 - L))
89  {
90  aNonceLength = 15 - L;
91  }
92 
93  // setup initial block
94 
95  // write flags
96  mBlock[0] = (static_cast<uint8_t>((aHeaderLength != 0) << 6) |
97  static_cast<uint8_t>(((aTagLength - 2) >> 1) << 3) |
98  static_cast<uint8_t>(L - 1));
99 
100  // write nonce
101  for (i = 0; i < aNonceLength; i++)
102  {
103  mBlock[1 + i] = nonceBytes[i];
104  }
105 
106  // write len
107  len = aPlainTextLength;
108 
109  for (i = sizeof(mBlock) - 1; i > aNonceLength; i--)
110  {
111  mBlock[i] = len & 0xff;
112  len >>= 8;
113  }
114 
115  // encrypt initial block
116  mEcb.Encrypt(mBlock, mBlock);
117 
118  // process header
119  if (aHeaderLength > 0)
120  {
121  // process length
122  if (aHeaderLength < (65536U - 256U))
123  {
124  mBlock[blockLength++] ^= aHeaderLength >> 8;
125  mBlock[blockLength++] ^= aHeaderLength >> 0;
126  }
127  else
128  {
129  mBlock[blockLength++] ^= 0xff;
130  mBlock[blockLength++] ^= 0xfe;
131  mBlock[blockLength++] ^= aHeaderLength >> 24;
132  mBlock[blockLength++] ^= aHeaderLength >> 16;
133  mBlock[blockLength++] ^= aHeaderLength >> 8;
134  mBlock[blockLength++] ^= aHeaderLength >> 0;
135  }
136  }
137 
138  // init counter
139  mCtr[0] = L - 1;
140 
141  for (i = 0; i < aNonceLength; i++)
142  {
143  mCtr[1 + i] = nonceBytes[i];
144  }
145 
146  for (i = i + 1; i < sizeof(mCtr); i++)
147  {
148  mCtr[i] = 0;
149  }
150 
151  mNonceLength = aNonceLength;
152  mHeaderLength = aHeaderLength;
153  mHeaderCur = 0;
154  mPlainTextLength = aPlainTextLength;
155  mPlainTextCur = 0;
156  mBlockLength = blockLength;
157  mCtrLength = sizeof(mCtrPad);
158  mTagLength = aTagLength;
159 }

162 {
163  const uint8_t *headerBytes = reinterpret_cast<const uint8_t *>(aHeader);
164 
165  assert(mHeaderCur + aHeaderLength <= mHeaderLength);
166 
167  // process header
168  for (unsigned i = 0; i < aHeaderLength; i++)
169  {
170  if (mBlockLength == sizeof(mBlock))
171  {
172  mEcb.Encrypt(mBlock, mBlock);
173  mBlockLength = 0;
174  }
175 
176  mBlock[mBlockLength++] ^= headerBytes[i];

178 
179  mHeaderCur += aHeaderLength;
180

194 {
195  uint8_t *plaintextBytes = reinterpret_cast<uint8_t *>(plaintext);
196  uint8_t *ciphertextBytes = reinterpret_cast<uint8_t *>(ciphertext);
197  uint8_t byte;
198 
199  assert(mPlainTextCur + len <= mPlainTextLength);
200 
201  for (unsigned i = 0; i < len; i++)
202  {
203  if (mCtrLength == 16)
204  {
205  for (int j = sizeof(mCtr) - 1; j > mNonceLength; j--)
206  {
207  if (++mCtr[j])
208  {
209  break;
210  }
211  }
212 
213  mEcb.Encrypt(mCtr, mCtrPad);
214  mCtrLength = 0;
215  }
216 
217  if (aEncrypt)

219  byte = plaintextBytes[i];
220  ciphertextBytes[i] = byte ^ mCtrPad[mCtrLength++];
221  }
222  else
223  {
224  byte = ciphertextBytes[i] ^ mCtrPad[mCtrLength++];
225  plaintextBytes[i] = byte;

229  {
230  mEcb.Encrypt(mBlock, mBlock);
231  mBlockLength = 0;
232  }
233

235  }
236

238 
239  if (mPlainTextCur >= mPlainTextLength)

242  {
243  mEcb.Encrypt(mBlock, mBlock);
244  }
245 
246  // reset counter
