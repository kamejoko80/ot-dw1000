116 
117  fc, seq = struct.unpack("<HB", data.read(3))
118 
119  frame_type = fc & 0x0007
120  security_enabled = bool(fc & 0x0008)
121  frame_pending = bool(fc & 0x0010)
122  ack_request = bool(fc & 0x0020)
123  panid_compression = bool(fc & 0x0040)
124  dest_addr_mode = (fc & 0x0c00) >> 10
125  frame_version = (fc & 0x3000) >> 12
126  source_addr_mode = (fc & 0xc000) >> 14
127 
128  if frame_type == MacHeader.FrameType.ACK:
129  fcs = self._parse_fcs(data, data.tell())
130  self.header = MacHeader(frame_type, frame_pending, ack_request, frame_version, seq, fcs=fcs)
131  self.payload = None
132  return

224  def _parse_aux_sec_header(self, data):
225  security_control, frame_counter = struct.unpack("<BL", data.read(5))
226 
227  security_level = security_control & 0x07
228  key_id_mode = (security_control & 0x18) >> 3
229 
230  if key_id_mode == 1:
231  key_id = data.read(1)
232  elif key_id_mode == 2:
