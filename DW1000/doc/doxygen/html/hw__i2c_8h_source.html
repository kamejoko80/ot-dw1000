590 // Field: [6:0] TPR
591 //
592 // SCL clock period
593 // This field specifies the period of the SCL clock.
594 // SCL_PRD = 2*(1+TPR)*(SCL_LP + SCL_HP)*CLK_PRD
595 // where:
596 // SCL_PRD is the SCL line period (I2C clock).
597 // TPR is the timer period register value (range of 1 to 127)
598 // SCL_LP is the SCL low period (fixed at 6).
599 // SCL_HP is the SCL high period (fixed at 4).
600 // CLK_PRD is the system clock period in ns.

602 #define I2C_MTPR_TPR_M 0x0000007F
603 #define I2C_MTPR_TPR_S 0
