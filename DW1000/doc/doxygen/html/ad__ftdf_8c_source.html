156 void sleep_when_possible(uint8_t explicit_request, uint32_t sleepTime)
157 {
158  FTDF_Boolean blockSleep = FTDF_FALSE;
159 
160  if ((!AD_FTDF_SLEEP_WHEN_IDLE && !explicit_request) || sleep_status != BLOCK_ACTIVE)
161  {
162  return;
163  }
164 
165  FTDF_USec us;
166 
167 
168  FTDF_criticalVar();
169  FTDF_enterCritical();
170 #ifdef FTDF_PHY_API
171 
172  if (explicit_request && FTDF_GET_FIELD(ON_OFF_REGMAP_LMACREADY4SLEEP) == 0)
173  {
174  volatile uint32_t *lmacCtrlMask = FTDF_GET_REG_ADDR(ON_OFF_REGMAP_LMAC_CONTROL_MASK);
175  volatile uint32_t *lmacReady4sleepEvent = FTDF_GET_FIELD_ADDR(ON_OFF_REGMAP_LMACREADY4SLEEP_D);
176 
177  /* clear a previous interrupt */
178  *lmacReady4sleepEvent = MSK_F_FTDF_ON_OFF_REGMAP_LMACREADY4SLEEP_D;
179 
180  /* Enable (unmask) interrupt */
181  *lmacCtrlMask |= MSK_F_FTDF_ON_OFF_REGMAP_LMACREADY4SLEEP_M;
182  us = 0;
183  }
184  else
185  {
186  us = FTDF_canSleep();
187  }
188 
189 #else
190  us = FTDF_canSleep();
191 #endif
192 
193  /* Try to sleep as much as needed (if sleepTime is 0, then sleep as much as possible).
194  * Otherwise, sleep as much as possible. */
195  if (explicit_request == FTDF_TRUE)
196  {
197  if (sleepTime && us > sleepTime)
198  {
199  us = sleepTime;
200  }
201  }
202 
203  if (us > (WUP_LATENCY / 1000000) + AD_FTDF_SLEEP_COMPENSATION)
204  {
205  // Subtract sleep compensation from the sleep time, compensating for delays
206  us -= AD_FTDF_SLEEP_COMPENSATION;
207 
208  blockSleep = FTDF_prepareForSleep(us);
209 
210  if (blockSleep)
211  {
212  // FTDF ready to sleep, disable clocks
213  sleep_status = BLOCK_SLEEPING;
214  explicit_sleep = explicit_request;
215 #if FTDF_USE_SLEEP_DURING_BACKOFF
216  FTDF_sdbFsmSleep();
217 #endif /* FTDF_USE_SLEEP_DURING_BACKOFF */
218 #if dg_configUSE_FTDF_DDPHY == 1
219  FTDF_ddphySave();
220 #endif /* dg_configUSE_FTDF_DDPHY */
221  ad_ftdf_sleep();
222  }
223  else
224  {
225 #if FTDF_USE_SLEEP_DURING_BACKOFF
226  FTDF_sdbFsmAbortSleep();
227 #endif /* FTDF_USE_SLEEP_DURING_BACKOFF */
228  }
229  }
230 
231  FTDF_exitCritical();
232 
233  if (blockSleep)
234  {
235  ad_rf_request_off(false);
236  }
