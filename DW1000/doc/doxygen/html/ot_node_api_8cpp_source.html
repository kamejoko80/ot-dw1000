252 int Hex2Bin(const char *aHex, uint8_t *aBin, uint16_t aBinLength)
253 {
254  size_t hexLength = strlen(aHex);
255  const char *hexEnd = aHex + hexLength;
256  uint8_t *cur = aBin;
257  uint8_t numChars = hexLength & 1;
258  uint8_t byte = 0;
259 
260  if ((hexLength + 1) / 2 > aBinLength)
261  {
262  return -1;
263  }
264 
265  while (aHex < hexEnd)
266  {
267  if ('A' <= *aHex && *aHex <= 'F')
268  {
269  byte |= 10 + (*aHex - 'A');
270  }
271  else if ('a' <= *aHex && *aHex <= 'f')
272  {
273  byte |= 10 + (*aHex - 'a');
274  }
275  else if ('0' <= *aHex && *aHex <= '9')
276  {
277  byte |= *aHex - '0';
278  }
279  else
280  {
281  return -1;
282  }
283 
284  aHex++;
285  numChars++;
286 
287  if (numChars >= 2)
288  {
289  numChars = 0;
290  *cur++ = byte;
291  byte = 0;
292  }
293  else
294  {
295  byte <<= 4;
296  }
297  }
298 
299  return static_cast<int>(cur - aBin);
300 }

354 CALLBACK
355 PingHandlerRecvCallback(
356  _Inout_ PTP_CALLBACK_INSTANCE /* Instance */,
357  _Inout_opt_ PVOID Context,
358  _Inout_ PTP_WAIT /* Wait */,
359  _In_ TP_WAIT_RESULT /* WaitResult */
360  )
361 {
362  otPingHandler *aPingHandler = (otPingHandler*)Context;
363  if (aPingHandler == NULL) return;
364 
365  // Get the result of the IO operation
366  DWORD cbTransferred = 0;
367  DWORD dwFlags = 0;
368  if (!WSAGetOverlappedResult(
369  aPingHandler->mSocket,
370  &aPingHandler->mOverlapped,
371  &cbTransferred,
372  TRUE,
373  &dwFlags))
374  {
375  int result = WSAGetLastError();
376  // Only log if we are shutting down
377  if (result != WSAENOTSOCK && result != ERROR_OPERATION_ABORTED)
378  printf("WSAGetOverlappedResult failed, 0x%x\r\n", result);
379  return;
380  }
381 
382  int result;
383 
384  // Make sure it didn't come from our address
385  if (memcmp(&aPingHandler->mSourceAddr6.sin6_addr, &aPingHandler->mAddress, sizeof(IN6_ADDR)) != 0)
386  {
387  bool shouldReply = true;
388 
389  // TODO - Fix this hack...
390  auto RecvDest = (const otIp6Address*)aPingHandler->mRecvBuffer;
391  if (memcmp(RecvDest, &LinkLocalAllRoutersAddress, sizeof(IN6_ADDR)) == 0 ||
392  memcmp(RecvDest, &RealmLocalAllRoutersAddress, sizeof(IN6_ADDR)) == 0)
393  {
394  auto Role = otGetDeviceRole(aPingHandler->mParentNode->mInstance);
395  if (Role != kDeviceRoleLeader && Role != kDeviceRoleRouter)
396  shouldReply = false;
397  }
398 
399  if (shouldReply)
400  {
401 #if DEBUG_PING
402  CHAR szIpAddress[46] = { 0 };
403  RtlIpv6AddressToStringA(&aPingHandler->mSourceAddr6.sin6_addr, szIpAddress);
404  printf("%d: received ping (%d bytes) from %s\r\n", aPingHandler->mParentNode->mId, cbTransferred, szIpAddress);
405 #endif
406 
407  // Send the received data back
408  result =
409  sendto(
410  aPingHandler->mSocket,
411  aPingHandler->mRecvBuffer, cbTransferred, 0,
412  (SOCKADDR*)&aPingHandler->mSourceAddr6, aPingHandler->mSourceAddr6Len
413  );
414  if (result == SOCKET_ERROR)
415  {
416  printf("sendto failed, 0x%x\r\n", WSAGetLastError());
417  }
418  }
419  }
420 
421  // Start the otpool waiting on the overlapped event
422  SetThreadpoolWait(aPingHandler->mThreadpoolWait, aPingHandler->mOverlapped.hEvent, nullptr);
423 
424  // Post another recv
425  dwFlags = MSG_PARTIAL;
426  aPingHandler->mSourceAddr6Len = sizeof(aPingHandler->mSourceAddr6);
427  result =
428  WSARecvFrom(
429  aPingHandler->mSocket,
430  &aPingHandler->mWSARecvBuffer, 1, &aPingHandler->mNumBytesReceived, &dwFlags,
431  (SOCKADDR*)&aPingHandler->mSourceAddr6, &aPingHandler->mSourceAddr6Len,
432  &aPingHandler->mOverlapped, nullptr
433  );
434  if (result != SOCKET_ERROR)
435  {
436  // Not pending, so manually trigger the event for the Threadpool to execute
437  SetEvent(aPingHandler->mOverlapped.hEvent);
438  }
439  else
440  {
441  result = WSAGetLastError();
442  if (result != WSA_IO_PENDING)
443  {
444  printf("WSARecvFrom failed, 0x%x\r\n", result);
445  }
446  }
447 }

642 void HandleAddressChanges(otNode *aNode)
643 {

645  auto addrs = otGetUnicastAddresses(aNode->mInstance);
646 
647  EnterCriticalSection(&aNode->mCS);
648

650  for (ULONG i = 0; i < aNode->mPingHandlers.size(); i++)
651  aNode->mPingHandlers[i]->mActive = false;
652 
653  // Search for matches
654  for (auto addr = addrs; addr; addr = addr->mNext)
655  {
656  bool found = false;
657  for (ULONG i = 0; i < aNode->mPingHandlers.size(); i++)
658  if (!aNode->mPingHandlers[i]->mActive &&
659  memcmp(&addr->mAddress, &aNode->mPingHandlers[i]->mAddress, sizeof(otIp6Address)) == 0)
660  {
661  found = true;
662  aNode->mPingHandlers[i]->mActive = true;
663  break;
664  }
665  if (!found) AddPingHandler(aNode, &addr->mAddress);
666  }
667 
668  vector<otPingHandler*> pingHandlersToDelete;
669 
670  // Release all left over handlers
671  for (int i = aNode->mPingHandlers.size() - 1; i >= 0; i--)
672  if (aNode->mPingHandlers[i]->mActive == false)
673  {
674  auto aPingHandler = aNode->mPingHandlers[i];

680 #endif

685  closesocket(aPingHandler->mSocket);
686

688  }
689

692  for each (auto aPingHandler in pingHandlersToDelete)
693  {

695  CloseThreadpoolWait(aPingHandler->mThreadpoolWait);
696  CloseHandle(aPingHandler->mOverlapped.hEvent);

773  }
774  }

777  {

801  goto error;
802  }

807  do
808  {
809  instance = otInstanceInit(ApiInstance, &ifGuid);
810  if (instance != nullptr) break;
811

813  if (waitTimeMs <= 0) break;

815  if (waitResult != WAIT_OBJECT_0) break;

817  } while (true);
