56 //*****************************************************************************
57 //
58 // Get the current value of the RTC counter in a format compatible to the compare registers.
59 //
60 //*****************************************************************************
61 uint32_t
62 AONRTCCurrentCompareValueGet( void )
63 {
64  uint32_t ui32CurrentSec ;
65  uint32_t ui32CurrentSubSec ;
66  uint32_t ui32SecondSecRead ;
67 
68  //
69  // Reading SEC both before and after SUBSEC in order to detect if SEC incremented while reading SUBSEC
70  // If SEC incremented, we can't be sure which SEC the SUBSEC belongs to, so repeating the sequence then.
71  //
72  do {
73  ui32CurrentSec = HWREG( AON_RTC_BASE + AON_RTC_O_SEC );
74  ui32CurrentSubSec = HWREG( AON_RTC_BASE + AON_RTC_O_SUBSEC );
75  ui32SecondSecRead = HWREG( AON_RTC_BASE + AON_RTC_O_SEC );
76  } while ( ui32CurrentSec != ui32SecondSecRead );
77 
78  return (( ui32CurrentSec << 16 ) | ( ui32CurrentSubSec >> 16 ));
79 }

87 AONRTCCurrent64BitValueGet( void )

94 
95  //
96  // Reading SEC both before and after SUBSEC in order to detect if SEC incremented while reading SUBSEC
97  // If SEC incremented, we can't be sure which SEC the SUBSEC belongs to, so repeating the sequence then.
98  //
99  do {
100  currentRtc.secAndSubSec[ 1 ] = HWREG( AON_RTC_BASE + AON_RTC_O_SEC );
101  currentRtc.secAndSubSec[ 0 ] = HWREG( AON_RTC_BASE + AON_RTC_O_SUBSEC );
102  ui32SecondSecRead = HWREG( AON_RTC_BASE + AON_RTC_O_SEC );
103  } while ( currentRtc.secAndSubSec[ 1 ] != ui32SecondSecRead );
104 
105  return ( currentRtc.returnValue );
106 }
