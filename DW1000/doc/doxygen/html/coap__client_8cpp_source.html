80 ThreadError Client::SendMessage(Message &aMessage, const Ip6::MessageInfo &aMessageInfo,
81  otCoapResponseHandler aHandler, void *aContext)
82 {
83  ThreadError error;
84  Header header;
85  RequestMetadata requestMetadata;
86  Message *storedCopy = NULL;
87  uint16_t copyLength = 0;
88 
89  SuccessOrExit(error = header.FromMessage(aMessage, false));
90 
91  // Set Message Id if it was not already set.
92  if (header.GetMessageId() == 0)
93  {
94  header.SetMessageId(mMessageId++);
95  aMessage.Write(0, Header::kMinHeaderLength, header.GetBytes());
96  }
97 
98  if (header.IsConfirmable())
99  {
100  // Create a copy of entire message and enqueue it.
101  copyLength = aMessage.GetLength();
102  }
103  else if (header.IsNonConfirmable() && header.IsRequest() && (aHandler != NULL))
104  {
105  // As we do not retransmit non confirmable messages, create a copy of header only, for token information.
106  copyLength = header.GetLength();
107  }
108 
109  if (copyLength > 0)
110  {
111  requestMetadata = RequestMetadata(header.IsConfirmable(), aMessageInfo, aHandler, aContext);
112  VerifyOrExit((storedCopy = CopyAndEnqueueMessage(aMessage, copyLength, requestMetadata)) != NULL,
113  error = kThreadError_NoBufs);
114  }
115 
116  SuccessOrExit(error = mSender(this, aMessage, aMessageInfo));
117 
118 exit:
119 
120  if (error != kThreadError_None && storedCopy != NULL)
121  {
122  DequeueMessage(*storedCopy);
123  }
124 
125  return error;
126 }

128 Message *Client::CopyAndEnqueueMessage(const Message &aMessage, uint16_t aCopyLength,
129  const RequestMetadata &aRequestMetadata)
130 {
131  ThreadError error = kThreadError_None;
132  Message *messageCopy = NULL;
133  uint32_t alarmFireTime;
134 
135  // Create a message copy of requested size.
136  VerifyOrExit((messageCopy = aMessage.Clone(aCopyLength)) != NULL, error = kThreadError_NoBufs);
137 
138  // Append the copy with retransmission data.
139  SuccessOrExit(error = aRequestMetadata.AppendTo(*messageCopy));
140 
141  // Setup the timer.
142  if (mRetransmissionTimer.IsRunning())
143  {
144  // If timer is already running, check if it should be restarted with earlier fire time.
145  alarmFireTime = mRetransmissionTimer.Gett0() + mRetransmissionTimer.Getdt();
146 
147  if (aRequestMetadata.IsEarlier(alarmFireTime))
148  {
149  mRetransmissionTimer.Start(aRequestMetadata.mRetransmissionTimeout);
150  }
151  }
152  else
153  {
154  mRetransmissionTimer.Start(aRequestMetadata.mRetransmissionTimeout);
155  }
156 
157  // Enqueue the message.
158  mPendingRequests.Enqueue(*messageCopy);
159 
160 exit:
161 
162  if (error != kThreadError_None && messageCopy != NULL)
163  {
164  messageCopy->Free();
165  messageCopy = NULL;
166  }
167 
168  return messageCopy;
169 }

182  aMessage.Free();

240 void Client::HandleRetransmissionTimer(void)
241 {
242  uint32_t now = otPlatAlarmGetNow();
243  uint32_t nextDelta = 0xffffffff;
244  RequestMetadata requestMetadata;
245  Message *message = mPendingRequests.GetHead();
246  Message *nextMessage = NULL;
247  Ip6::MessageInfo messageInfo;
248 
249  while (message != NULL)
250  {
251  nextMessage = message->GetNext();
252  requestMetadata.ReadFrom(*message);
253 
254  if (requestMetadata.IsLater(now))
255  {
256  // Calculate the next delay and choose the lowest.
257  if (requestMetadata.mNextTimerShot - now < nextDelta)
258  {
259  nextDelta = requestMetadata.mNextTimerShot - now;
260  }
261  }
262  else if ((requestMetadata.mConfirmable) &&
263  (requestMetadata.mRetransmissionCount < kMaxRetransmit))
264  {
265  // Increment retransmission counter and timer.
266  requestMetadata.mRetransmissionCount++;
267  requestMetadata.mRetransmissionTimeout *= 2;
268  requestMetadata.mNextTimerShot = now + requestMetadata.mRetransmissionTimeout;
269  requestMetadata.UpdateIn(*message);
270 
271  // Check if retransmission time is lower than current lowest.
272  if (requestMetadata.mRetransmissionTimeout < nextDelta)
273  {
274  nextDelta = requestMetadata.mRetransmissionTimeout;
275  }
276 
277  // Retransmit
278  if (!requestMetadata.mAcknowledged)
279  {
280  messageInfo.SetPeerAddr(requestMetadata.mDestinationAddress);
281  messageInfo.SetPeerPort(requestMetadata.mDestinationPort);
282  messageInfo.SetSockAddr(requestMetadata.mSourceAddress);
283 
284  SendCopy(*message, messageInfo);
285  }
286  }
287  else
288  {
289  // No expected response or acknowledgment.
290  FinalizeCoapTransaction(*message, requestMetadata, NULL, NULL, NULL, kThreadError_ResponseTimeout);
291  }
292 
293  message = nextMessage;
294  }
295 
296  if (nextDelta != 0xffffffff)
297  {
298  mRetransmissionTimer.Start(nextDelta);
299  }
300 }

302 Message *Client::FindRelatedRequest(const Header &aResponseHeader, const Ip6::MessageInfo &aMessageInfo,
303  Header &aRequestHeader, RequestMetadata &aRequestMetadata)
304 {
305  Message *message = mPendingRequests.GetHead();
306 
307  while (message != NULL)
308  {
309  aRequestMetadata.ReadFrom(*message);
310 
311  if (((aRequestMetadata.mDestinationAddress == aMessageInfo.GetPeerAddr()) ||
312  aRequestMetadata.mDestinationAddress.IsMulticast() ||
313  aRequestMetadata.mDestinationAddress.IsAnycastRoutingLocator()) &&
314  (aRequestMetadata.mDestinationPort == aMessageInfo.GetPeerPort()))
315  {
316  assert(aRequestHeader.FromMessage(*message, true) == kThreadError_None);
317 
318  switch (aResponseHeader.GetType())
319  {
320  case kCoapTypeReset:
321  case kCoapTypeAcknowledgment:
322  if (aResponseHeader.GetMessageId() == aRequestHeader.GetMessageId())
323  {
324  ExitNow();
325  }
326 
327  break;
328 
329  case kCoapTypeConfirmable:
330  case kCoapTypeNonConfirmable:
331  if (aResponseHeader.IsTokenEqual(aRequestHeader))
332  {
333  ExitNow();
334  }
335 
336  break;
337  }
338  }
339 
340  message = message->GetNext();
341  }
342 
343 exit:
344  return message;
345 }

360 void Client::ProcessReceivedMessage(Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
361 {
362  Header responseHeader;
363  Header requestHeader;
364  RequestMetadata requestMetadata;
365  Message *message = NULL;
366  ThreadError error;
367 
368  SuccessOrExit(error = responseHeader.FromMessage(aMessage, false));
369  aMessage.MoveOffset(responseHeader.GetLength());
370 
371  message = FindRelatedRequest(responseHeader, aMessageInfo, requestHeader, requestMetadata);
372 
373  if (message == NULL)
374  {
375  ExitNow();
376  }
377 
378  switch (responseHeader.GetType())
379  {
380  case kCoapTypeReset:
381  if (responseHeader.IsEmpty())
382  {
383  FinalizeCoapTransaction(*message, requestMetadata, NULL, NULL, NULL, kThreadError_Abort);
384  }
385 
386  // Silently ignore non-empty reset messages (RFC 7252, p. 4.2).
387  break;
388 
389  case kCoapTypeAcknowledgment:
390  if (responseHeader.IsEmpty())
391  {
392  // Empty acknowledgment.
393  if (requestMetadata.mConfirmable)
394  {
395  requestMetadata.mAcknowledged = true;
396  requestMetadata.UpdateIn(*message);
397  }
398 
399  // Remove the message if response is not expected, otherwise await response.
400  if (requestMetadata.mResponseHandler == NULL)
401  {
402  DequeueMessage(*message);
403  }
404  }
405  else if (responseHeader.IsResponse() && responseHeader.IsTokenEqual(requestHeader))
406  {
407  // Piggybacked response.
408  FinalizeCoapTransaction(*message, requestMetadata, &responseHeader, &aMessage, &aMessageInfo, kThreadError_None);
409  }
410 
411  // Silently ignore acknowledgments carrying requests (RFC 7252, p. 4.2)
412  // or with no token match (RFC 7252, p. 5.3.2)
413  break;
414 
415  case kCoapTypeConfirmable:
416  case kCoapTypeNonConfirmable:
417  if (responseHeader.IsConfirmable())
418  {
419  // Send empty ACK if it is a CON message.
420  SendEmptyAck(aMessageInfo.GetPeerAddr(), aMessageInfo.GetPeerPort(), responseHeader.GetMessageId());
421  }
422 
423  FinalizeCoapTransaction(*message, requestMetadata, &responseHeader, &aMessage, &aMessageInfo, kThreadError_None);
424 
425  break;
426  }
427 
428 exit:
429 
430  if (error == kThreadError_None && message == NULL)
431  {
432  if (responseHeader.IsConfirmable() || responseHeader.IsNonConfirmable())
433  {
434  // Successfully parsed a header but no matching request was found - reject the message by sending reset.
435  SendReset(aMessageInfo.GetPeerAddr(), aMessageInfo.GetPeerPort(), responseHeader.GetMessageId());
436  }
437  }
438 }

440 RequestMetadata::RequestMetadata(bool aConfirmable, const Ip6::MessageInfo &aMessageInfo,
441  otCoapResponseHandler aHandler, void *aContext)
442 {
443  mSourceAddress = aMessageInfo.GetSockAddr();
444  mDestinationPort = aMessageInfo.GetPeerPort();
445  mDestinationAddress = aMessageInfo.GetPeerAddr();
446  mResponseHandler = aHandler;
447  mResponseContext = aContext;
448  mRetransmissionCount = 0;
449  mRetransmissionTimeout = Timer::SecToMsec(kAckTimeout);
450  mRetransmissionTimeout += otPlatRandomGet() %
451  (Timer::SecToMsec(kAckTimeout) * kAckRandomFactorNumerator / kAckRandomFactorDenominator -
452  Timer::SecToMsec(kAckTimeout) + 1);
453 
454  if (aConfirmable)
455  {
456  // Set next retransmission timeout.
457  mNextTimerShot = Timer::GetNow() + mRetransmissionTimeout;
458  }
459  else
460  {
461  // Set overall response timeout.
462  mNextTimerShot = Timer::GetNow() + kMaxTransmitWait;
463  }
464 
465  mAcknowledged = false;
466  mConfirmable = aConfirmable;
467 }
