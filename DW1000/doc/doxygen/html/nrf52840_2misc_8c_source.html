56 otPlatResetReason otPlatGetResetReason(otInstance *aInstance)
57 {
58  (void)aInstance;
59  otPlatResetReason reason;
60 
61  if (sResetReason & POWER_RESETREAS_RESETPIN_Msk)
62  {
63  reason = kPlatResetReason_External;
64  }
65  else if (sResetReason & POWER_RESETREAS_DOG_Msk)
66  {
67  reason = kPlatResetReason_Watchdog;
68  }
69  else if (sResetReason & POWER_RESETREAS_SREQ_Msk)
70  {
71  reason = kPlatResetReason_Software;
72  }
73  else if (sResetReason & POWER_RESETREAS_LOCKUP_Msk)
74  {
75  reason = kPlatResetReason_Fault;
76  }
77  else if ((sResetReason & POWER_RESETREAS_OFF_Msk) ||
78  (sResetReason & POWER_RESETREAS_LPCOMP_Msk) ||
79  (sResetReason & POWER_RESETREAS_DIF_Msk) ||
80  (sResetReason & POWER_RESETREAS_NFC_Msk) ||
81  (sResetReason & POWER_RESETREAS_VBUS_Msk))
82  {
83  reason = kPlatResetReason_Other;
84  }
85  else
86  {
87  reason = kPlatResetReason_PowerOn;
88  }
89 
90  return reason;
91 }
