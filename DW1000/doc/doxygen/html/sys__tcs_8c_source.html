261 static int sys_tcs_sort_area(uint32_t area_base, uint32_t area_size, uint32_t array_ptr, uint8_t *ptr, uint8_t *size)
262 {
263  uint32_t i = array_ptr;
264  uint32_t sort_start;
265  uint32_t sort_end;
266 
267  if (array_ptr == tcs_length)
268  {
269  return array_ptr; // Nothing else is left in the array or tcs_length is zero
270  }
271 
272  sort_start = array_ptr;
273  sort_end = array_ptr;
274 
275  /* Make sure that the values in TCS don't overflow the allocated array */
276  ASSERT_WARNING(tcs_length <= sizeof(tcs_data) / sizeof(tcs_data[0]));
277 
278  while (i < (tcs_length - 1))
279  {
280  if (IS_IN_AREA(tcs_data[i], area_base, area_size))
281  {
282  if (i == array_ptr)
283  {
284  if (sort_start == array_ptr)
285  {
286  sort_start = array_ptr;
287  sort_end = sort_start + 2;
288  }
289  }
290  else
291  {
292  // Copy it to the proper place.
293  if (i != sort_end)
294  {
295  swap_entries(sort_end, i);
296  }
297 
298  sort_end += 2;
299  }
300  }
301 
302  i += 2;
303  }
304 
305  *ptr = sort_start;
306  *size = sort_end - sort_start;
307 
308  return sort_end;
309 }

318 bool sys_tcs_store_pair(uint32_t address, uint32_t value)
319 {
320  // If it stops here then tcs_length must become uint16_t.
321  ASSERT_WARNING((256 * sizeof(tcs_length)) >= (24 * 2 + APPEND_TCS_LENGTH));
322 
323  // Check if the address is in Always-ON or of a retained register and apply it.
324  if (IS_IN_AREA(address, CRG_TOP_BASE, sizeof(CRG_TOP_Type))
325  || IS_IN_AREA(address, TIMER1_BASE, sizeof(TIMER1_Type))
326  || IS_IN_AREA(address, WAKEUP_BASE, sizeof(WAKEUP_Type))
327  || IS_IN_AREA(address, DCDC_BASE, sizeof(DCDC_Type))
328  || IS_IN_AREA(address, QSPIC_BASE, sizeof(QSPIC_Type))
329  || IS_IN_AREA(address, CACHE_BASE, sizeof(CACHE_Type))
330  || IS_IN_AREA(address, OTPC_BASE, sizeof(OTPC_Type)))
331  {
332  if (address == (uint32_t)&CRG_TOP->BANDGAP_REG)
333  {
334  sys_tcs_is_calibrated_chip = true;
335  }
336  else if (address == (uint32_t)&CRG_TOP->CLK_16M_REG)
337  {
338  value |= CRG_TOP_CLK_16M_REG_RC16M_ENABLE_Msk; // Make sure that RC16 is enabled!
339  }
340  else if (address == (uint32_t)&CRG_TOP->XTALRDY_CTRL_REG)
341  {
342  sys_tcs_xtal16m_settling_time = value;
343  }
344  else if (address == (uint32_t)&CRG_TOP->BOD_CTRL2_REG)
345  {
346  hw_cpm_bod_enabled_in_tcs = value;
347  /*
348  * Read the ADC Gain Error. Given that the Chip Configuration Section
349  * (CCS) of OTP is out of space, the values are stored in the TCS section
350  * using as address the read-only register SYS_STAT_REG.
351  * The 16 LSB of the 32bit data-value is the single ended gain error.
352  * The 16 MSB of the 32bit data-value is the differential gain error.
353  * Both gain error values should be interpreted as signed 16bit integers.
354  */
355  }
356  else if (dg_configUSE_ADC_GAIN_ERROR_CORRECTION == 1)
357  {
358  if (address == (uint32_t)&CRG_TOP->SYS_STAT_REG)
359  {
360  hw_gpadc_store_se_gain_error(ADC_SE_GAIN_ERROR(value));
361  hw_gpadc_store_diff_gain_error(ADC_DIFF_GAIN_ERROR(value));
362  }
363  }
364 
365  apply_pair(address, value);
366  }
367  else
368  {
369  store_in_array(address, value);
370  }
371 
372  return sys_tcs_is_calibrated_chip;
373 }
