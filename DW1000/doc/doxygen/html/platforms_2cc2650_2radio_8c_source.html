762 static uint_fast8_t rfCorePowerOn(void)
763 {
764  bool interruptsWereDisabled;
765 
766  /* Request the HF XOSC as the source for the HF clock. Needed before we can
767  * use the FS. This will only request, it will _not_ perform the switch.
768  */
769  if (OSCClockSourceGet(OSC_SRC_CLK_HF) != OSC_XOSC_HF)
770  {
771  /* Request to switch to the crystal to enable radio operation. It takes a
772  * while for the XTAL to be ready so instead of performing the actual
773  * switch, we do other stuff while the XOSC is getting ready.
774  */
775  OSCClockSourceSet(OSC_SRC_CLK_MF | OSC_SRC_CLK_HF, OSC_XOSC_HF);
776  }
777 
778  rfCoreSetModeSelect();
779 
780  /* Set of RF Core data queue. Circular buffer, no last entry */
781  sRxDataQueue.pCurrEntry = sRxBuf0;
782  sRxDataQueue.pLastEntry = NULL;
783 
784  rfCoreInitBufs();
785 
786  /*
787  * Trigger a switch to the XOSC, so that we can subsequently use the RF FS
788  * This will block until the XOSC is actually ready, but give how we
789  * requested it early on, this won't be too long a wait.
790  * This should be done before starting the RAT.
791  */
792  if (OSCClockSourceGet(OSC_SRC_CLK_HF) != OSC_XOSC_HF)
793  {
794  /* Switch the HF clock source (cc26xxware executes this from ROM) */
795  OSCHfSourceSwitch();
796  }
797 
798  interruptsWereDisabled = IntMasterDisable();
799 
800  /* Enable RF Core power domain */
801  PRCMPowerDomainOn(PRCM_DOMAIN_RFCORE);
802 
803  while (PRCMPowerDomainStatus(PRCM_DOMAIN_RFCORE) != PRCM_DOMAIN_POWER_ON);
804 
805  PRCMDomainEnable(PRCM_DOMAIN_RFCORE);
806  PRCMLoadSet();
807 
808  while (!PRCMLoadGet());

1673  uint8_t rssi;
1674  rfc_dataEntryGeneral_t *startEntry = (rfc_dataEntryGeneral_t *)sRxDataQueue.pCurrEntry;
1675  rfc_dataEntryGeneral_t *curEntry = startEntry;
1676 
1677  /* loop through receive queue */
1678  do
1679  {
1680  uint8_t *payload = &(curEntry->data);
1681 
1682  if (sReceiveFrame.mLength == 0 && curEntry->status == DATA_ENTRY_FINISHED)
1683  {
1684  uint8_t len = payload[0];
1685  /* get the information appended to the end of the frame.
1686  * This array access looks like it is a fencepost error, but the
1687  * first byte is the number of bytes that follow.
1688  */
1689  crcCorr = (rfc_ieeeRxCorrCrc_t *)&payload[len];
1690  rssi = payload[len - 1];
1691 
1692  if (crcCorr->status.bCrcErr == 0 && (len - 2) < kMaxPHYPacketSize)
1693  {
1694  sReceiveFrame.mLength = len;
1695  memcpy(sReceiveFrame.mPsdu, &(payload[1]), len - 2);
1696  sReceiveFrame.mChannel = sReceiveCmd.channel;
1697  sReceiveFrame.mPower = rssi;
1698  sReceiveFrame.mLqi = crcCorr->status.corr;
1699  sReceiveError = kThreadError_None;
1700  }
1701  else
1702  {
1703  sReceiveError = kThreadError_FcsErr;
1704  }
1705 
1706  curEntry->status = DATA_ENTRY_PENDING;
1707  break;
1708  }
1709  else if (curEntry->status == DATA_ENTRY_UNFINISHED)
1710  {
1711  curEntry->status = DATA_ENTRY_PENDING;
1712  }
1713 
1714  curEntry = (rfc_dataEntryGeneral_t *)(curEntry->pNextEntry);
1715  }
