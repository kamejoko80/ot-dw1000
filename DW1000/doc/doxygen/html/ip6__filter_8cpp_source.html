51 bool Filter::Accept(Message &aMessage) const
52 {
53  bool rval = false;
54  Header ip6;
55  UdpHeader udp;
56  TcpHeader tcp;
57  uint16_t dstport;
58 
59  // Allow all received IPv6 datagrams with link security enabled
60  if (aMessage.IsLinkSecurityEnabled())
61  {
62  ExitNow(rval = true);
63  }
64 
65  // Read IPv6 header
66  VerifyOrExit(sizeof(ip6) == aMessage.Read(0, sizeof(ip6), &ip6), ;);
67 
68  // Allow only link-local unicast or multicast
69  VerifyOrExit(ip6.GetDestination().IsLinkLocal() || ip6.GetDestination().IsLinkLocalMulticast(), ;);
70 
71  switch (ip6.GetNextHeader())
72  {
73  case kProtoUdp:
74  // Read the UDP header and get the dst port
75  VerifyOrExit(sizeof(udp) == aMessage.Read(sizeof(ip6), sizeof(udp), &udp), ;);
76 
77  dstport = udp.GetDestinationPort();
78 
79  // Allow MLE traffic
80  if (dstport == Mle::kUdpPort)
81  {
82  ExitNow(rval = true);
83  }
84 
85  break;
86 
87  case kProtoTcp:
88  // Read the TCP header and get the dst port
89  VerifyOrExit(sizeof(tcp) == aMessage.Read(sizeof(ip6), sizeof(tcp), &tcp), ;);
90 
91  dstport = tcp.GetDestinationPort();
92 
93  break;
94 
95  default:
96  // Allow UDP or TCP traffic only
97  ExitNow();
98  }
99 
100  // Check against allowed unsecure port list
101  for (int i = 0; i < kMaxUnsecurePorts; i++)
102  {
103  if (mUnsecurePorts[i] != 0 && mUnsecurePorts[i] == dstport)
104  {
105  ExitNow(rval = true);
106  }
107  }
108 
109 exit:
110  return rval;
111 }

114 {

118  {

140 ThreadError Filter::RemoveUnsecurePort(uint16_t aPort)
141 {
142  ThreadError error = kThreadError_None;
143 
144  for (int i = 0; i < kMaxUnsecurePorts; i++)
145  {
146  if (mUnsecurePorts[i] == aPort)
147  {
148  // Shift all of the ports higher than this
149  // port down.
150  for (; i < kMaxUnsecurePorts - 1; i++)
151  {
152  mUnsecurePorts[i] = mUnsecurePorts[i + 1];
153  }
154

158  }
159  }
160
