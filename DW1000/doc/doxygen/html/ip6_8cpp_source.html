87 uint16_t Ip6::UpdateChecksum(uint16_t checksum, const void *buf, uint16_t len)
88 {
89  const uint8_t *bytes = reinterpret_cast<const uint8_t *>(buf);
90 
91  for (int i = 0; i < len; i++)
92  {
93  checksum = Ip6::UpdateChecksum(checksum, (i & 1) ? bytes[i] : static_cast<uint16_t>(bytes[i] << 8));
94  }
95 
96  return checksum;

132 ThreadError Ip6::AddMplOption(Message &message, Header &header)
133 {
134  ThreadError error = kThreadError_None;
135  HopByHopHeader hbhHeader;
136  OptionMpl mplOption;
137  OptionPadN padOption;
138 
139  hbhHeader.SetNextHeader(header.GetNextHeader());
140  hbhHeader.SetLength(0);
141  mMpl.InitOption(mplOption, header.GetSource());
142 
143  // Mpl option may require two bytes padding.
144  if ((mplOption.GetTotalLength() + sizeof(hbhHeader)) % 8)
145  {
146  padOption.Init(2);
147  SuccessOrExit(error = message.Prepend(&padOption, padOption.GetTotalLength()));
148  }
149 
150  SuccessOrExit(error = message.Prepend(&mplOption, mplOption.GetTotalLength()));
151  SuccessOrExit(error = message.Prepend(&hbhHeader, sizeof(hbhHeader)));
152  header.SetPayloadLength(header.GetPayloadLength() + sizeof(hbhHeader) + sizeof(mplOption));
153  header.SetNextHeader(kProtoHopOpts);
154 exit:
155  return error;

188 ThreadError Ip6::InsertMplOption(Message &aMessage, Header &aIp6Header, MessageInfo &aMessageInfo)
189 {
190  ThreadError error = kThreadError_None;
191 
192  VerifyOrExit(aIp6Header.GetDestination().IsMulticast() &&
193  aIp6Header.GetDestination().GetScope() >= Address::kRealmLocalScope, ;);
194 
195  if (aIp6Header.GetDestination().IsRealmLocalMulticast())
196  {
197  aMessage.RemoveHeader(sizeof(aIp6Header));
198 
199  if (aIp6Header.GetNextHeader() == kProtoHopOpts)
200  {
201  HopByHopHeader hbh;
202  uint8_t hbhLength = 0;
203  OptionMpl mplOption;
204 
205  // read existing hop-by-hop option header
206  aMessage.Read(0, sizeof(hbh), &hbh);
207  hbhLength = (hbh.GetLength() + 1) * 8;
208 
209  // increase existing hop-by-hop option header length by 8 bytes
210  hbh.SetLength(hbh.GetLength() + 1);
211  aMessage.Write(0, sizeof(hbh), &hbh);
212 
213  // make space for MPL Option + padding by shifting hop-by-hop option header
214  SuccessOrExit(error = aMessage.Prepend(NULL, 8));
215  aMessage.CopyTo(8, 0, hbhLength, aMessage);
216 
217  // insert MPL Option
218  mMpl.InitOption(mplOption, aIp6Header.GetSource());
219  aMessage.Write(hbhLength, mplOption.GetTotalLength(), &mplOption);
220 
221  // insert Pad Option (if needed)
222  if (mplOption.GetTotalLength() % 8)
223  {
224  OptionPadN padOption;
225  padOption.Init(8 - (mplOption.GetTotalLength() % 8));
226  aMessage.Write(hbhLength + mplOption.GetTotalLength(), padOption.GetTotalLength(), &padOption);
227  }
228 
229  // increase IPv6 Payload Length
230  aIp6Header.SetPayloadLength(aIp6Header.GetPayloadLength() + 8);
231  }
232  else
233  {
234  SuccessOrExit(error = AddMplOption(aMessage, aIp6Header));
235  }
236 
237  SuccessOrExit(error = aMessage.Prepend(&aIp6Header, sizeof(aIp6Header)));
238  }
239  else
240  {
241  SuccessOrExit(error = AddTunneledMplOption(aMessage, aIp6Header, aMessageInfo));
242  }
243 
244 exit:
245  return error;
246 }

248 ThreadError Ip6::RemoveMplOption(Message &aMessage)
249 {
250  ThreadError error = kThreadError_None;
251  Header ip6Header;
252  HopByHopHeader hbh;
253  uint16_t offset;
254  uint16_t endOffset;
255  uint16_t mplOffset = 0;
256  uint8_t mplLength = 0;
257  bool remove = false;
258 
259  offset = 0;
260  aMessage.Read(offset, sizeof(ip6Header), &ip6Header);
261  offset += sizeof(ip6Header);
262  VerifyOrExit(ip6Header.GetNextHeader() == kProtoHopOpts,);
263 
264  aMessage.Read(offset, sizeof(hbh), &hbh);
265  endOffset = offset + (hbh.GetLength() + 1) * 8;
266  VerifyOrExit(aMessage.GetLength() >= endOffset,);
267 
268  offset += sizeof(hbh);
269 
270  while (offset < endOffset)
271  {
272  OptionHeader option;
273 
274  aMessage.Read(offset, sizeof(option), &option);
275 
276  switch (option.GetType())
277  {
278  case OptionMpl::kType:
279  mplOffset = offset;
280  mplLength = option.GetLength();
281 
282  if (mplOffset == sizeof(ip6Header) + sizeof(hbh) && hbh.GetLength() == 0)
283  {
284  // first and only IPv6 Option, remove IPv6 HBH Option header
285  remove = true;
286  }
287  else if (mplOffset + 8 == endOffset)
288  {
289  // last IPv6 Option, remove last 8 bytes
290  remove = true;
291  }
292 
293  offset += sizeof(option) + option.GetLength();
294  break;
295 
296  case OptionPad1::kType:
297  offset += sizeof(OptionPad1);
298  break;
299 
300  case OptionPadN::kType:
301  offset += sizeof(option) + option.GetLength();
302  break;
303 
304  default:
305  // encountered another option, now just replace MPL Option with PadN
306  remove = false;
307  offset += sizeof(option) + option.GetLength();
308  break;
309  }
310  }
311 
312  // verify that IPv6 Options header is properly formed
313  VerifyOrExit(offset == endOffset,);
314 
315  if (remove)
316  {
317  // last IPv6 Option, shrink HBH Option header
318  uint8_t buf[8];
319 
320  offset = endOffset - sizeof(buf);
321 
322  while (offset >= sizeof(buf))
323  {
324  aMessage.Read(offset - sizeof(buf), sizeof(buf), buf);
325  aMessage.Write(offset, sizeof(buf), buf);
326  offset -= sizeof(buf);
327  }
328 
329  aMessage.RemoveHeader(sizeof(buf));
330 
331  if (mplOffset == sizeof(ip6Header) + sizeof(hbh))
332  {
333  // remove entire HBH header
334  ip6Header.SetNextHeader(hbh.GetNextHeader());
335  }
336  else
337  {
338  // update HBH header length
339  hbh.SetLength(hbh.GetLength() - 1);
340  aMessage.Write(sizeof(ip6Header), sizeof(hbh), &hbh);
341  }
342 
343  ip6Header.SetPayloadLength(ip6Header.GetPayloadLength() - sizeof(buf));
344  aMessage.Write(0, sizeof(ip6Header), &ip6Header);
345  }
346  else if (mplOffset != 0)
347  {
348  // replace MPL Option with PadN Option
349  OptionPadN padOption;
350 
351  padOption.Init(sizeof(OptionHeader) + mplLength);
352  aMessage.Write(mplOffset, padOption.GetTotalLength(), &padOption);
353  }
354 
355 exit:
356  return error;
357 }

365 ThreadError Ip6::SendDatagram(Message &message, MessageInfo &messageInfo, IpProto ipproto)
366 {
367  ThreadError error = kThreadError_None;
368  Header header;
369  uint16_t payloadLength = message.GetLength();
370  uint16_t checksum;
371  const NetifUnicastAddress *source;
372 
373  header.Init();
374  header.SetPayloadLength(payloadLength);
375  header.SetNextHeader(ipproto);
376  header.SetHopLimit(messageInfo.mHopLimit ? messageInfo.mHopLimit : static_cast<uint8_t>(kDefaultHopLimit));
377 
378  if (messageInfo.GetSockAddr().IsUnspecified() ||
379  messageInfo.GetSockAddr().IsAnycastRoutingLocator() ||
380  messageInfo.GetSockAddr().IsMulticast())
381  {
382  VerifyOrExit((source = SelectSourceAddress(messageInfo)) != NULL,
383  error = kThreadError_Error);
384  header.SetSource(source->GetAddress());
385  }
386  else
387  {
388  header.SetSource(messageInfo.GetSockAddr());
389  }
390 
391  header.SetDestination(messageInfo.GetPeerAddr());
392 
393  if (header.GetDestination().IsLinkLocal() || header.GetDestination().IsLinkLocalMulticast())
394  {
395  VerifyOrExit(messageInfo.GetInterfaceId() != 0, error = kThreadError_Drop);
396  }
397 
398  if (messageInfo.GetPeerAddr().IsRealmLocalMulticast())
399  {
400  SuccessOrExit(error = AddMplOption(message, header));
401  }
402 
403  SuccessOrExit(error = message.Prepend(&header, sizeof(header)));
404 
405  if (messageInfo.GetPeerAddr().IsMulticast() &&
406  messageInfo.GetPeerAddr().GetScope() > Address::kRealmLocalScope)
407  {
408  SuccessOrExit(error = AddTunneledMplOption(message, header, messageInfo));
409  }
410 
411  // compute checksum
412  checksum = ComputePseudoheaderChecksum(header.GetSource(), header.GetDestination(),
413  payloadLength, ipproto);
414 
415  switch (ipproto)
416  {
417  case kProtoUdp:
418  SuccessOrExit(error = mUdp.UpdateChecksum(message, checksum));
419  break;
420 
421  case kProtoIcmp6:
422  SuccessOrExit(error = mIcmp.UpdateChecksum(message, checksum));
423  break;
424 
425  default:
426  break;
427  }
428 
429 exit:
430 
431  if (error == kThreadError_None)
432  {
433  message.SetInterfaceId(messageInfo.GetInterfaceId());
434  EnqueueDatagram(message);
435  }
436 
437  return error;
438 }

456 ThreadError Ip6::HandleOptions(Message &message, Header &header, bool &forward)
457 {
458  ThreadError error = kThreadError_None;
459  HopByHopHeader hbhHeader;
460  OptionHeader optionHeader;
461  uint16_t endOffset;
462 
463  VerifyOrExit(message.Read(message.GetOffset(), sizeof(hbhHeader), &hbhHeader) == sizeof(hbhHeader),
464  error = kThreadError_Drop);
465  endOffset = message.GetOffset() + (hbhHeader.GetLength() + 1) * 8;
466 
467  VerifyOrExit(endOffset <= message.GetLength(), error = kThreadError_Drop);
468 
469  message.MoveOffset(sizeof(optionHeader));
470 
471  while (message.GetOffset() < endOffset)
472  {
473  VerifyOrExit(message.Read(message.GetOffset(), sizeof(optionHeader), &optionHeader) == sizeof(optionHeader),
474  error = kThreadError_Drop);
475 
476  if (optionHeader.GetType() == OptionPad1::kType)
477  {
478  message.MoveOffset(sizeof(OptionPad1));
479  continue;
480  }
481 
482  VerifyOrExit(message.GetOffset() + sizeof(optionHeader) + optionHeader.GetLength() <= endOffset,
483  error = kThreadError_Drop);
484 
485  switch (optionHeader.GetType())
486  {
487  case OptionMpl::kType:
488  SuccessOrExit(error = mMpl.ProcessOption(message, header.GetSource(), forward));
489  break;
490 
491  default:
492  switch (optionHeader.GetAction())
493  {
494  case OptionHeader::kActionSkip:
495  break;
496 
497  case OptionHeader::kActionDiscard:
498  ExitNow(error = kThreadError_Drop);
499 
500  case OptionHeader::kActionForceIcmp:
501  // TODO: send icmp error
502  ExitNow(error = kThreadError_Drop);
503 
504  case OptionHeader::kActionIcmp:
505  // TODO: send icmp error
506  ExitNow(error = kThreadError_Drop);
507 
508  }
509 
510  break;
511  }
512 
513  message.MoveOffset(sizeof(optionHeader) + optionHeader.GetLength());
514  }
515 
516 exit:
517  return error;
518 }

542

603

606  // do not pass messages sent to/from an RLOC

609  error = kThreadError_NoRoute);

621  }
622 
623  break;
624 
625  case kProtoUdp:
626  if (messageInfo.GetSockAddr().IsLinkLocal())
627  {
628  UdpHeader udp;

633  }
634 
635  break;
636

638  break;

640  }

642  // make a copy of the datagram to pass to host

644  RemoveMplOption(*messageCopy);
645  mReceiveIp6DatagramCallback(messageCopy, mReceiveIp6DatagramCallbackContext);
646 
647 exit:

651 ThreadError Ip6::HandleDatagram(Message &message, Netif *netif, int8_t interfaceId, const void *linkMessageInfo,
652  bool fromLocalHost)
653 {
654  ThreadError error = kThreadError_None;
655  MessageInfo messageInfo;
656  Header header;
657  uint16_t payloadLength;
658  bool receive = false;
659  bool forward = false;
660  bool tunnel = false;
661  bool multicastPromiscuous = false;
662  uint8_t nextHeader;
663  uint8_t hopLimit;
664 
665  otLogFuncEntry();
666 
667 #if 0
668  uint8_t buf[1024];
669  message.Read(0, sizeof(buf), buf);
670  dump("handle datagram", buf, message.GetLength());
671 #endif
672 
673  // check message length
674  VerifyOrExit(message.Read(0, sizeof(header), &header) == sizeof(header), error = kThreadError_Drop);
675  payloadLength = header.GetPayloadLength();
676 
677  // check Version
678  VerifyOrExit(header.IsVersion6(), error = kThreadError_Drop);
679 
680  // check Payload Length
681  VerifyOrExit(sizeof(header) + payloadLength == message.GetLength() &&
682  sizeof(header) + payloadLength <= Ip6::kMaxDatagramLength, error = kThreadError_Drop);
683 
684  messageInfo.SetPeerAddr(header.GetSource());
685  messageInfo.SetSockAddr(header.GetDestination());
686  messageInfo.SetInterfaceId(interfaceId);
687  messageInfo.SetHopLimit(header.GetHopLimit());
688  messageInfo.SetLinkInfo(linkMessageInfo);
689 
690  // determine destination of packet
691  if (header.GetDestination().IsMulticast())
692  {
693  if (netif != NULL)
694  {
695  if (netif->IsMulticastSubscribed(header.GetDestination()))
696  {
697  receive = true;
698  }
699  else if (netif->IsMulticastPromiscuousModeEnabled())
700  {
701  multicastPromiscuous = true;
702  }
703  }
704 
705  if (netif == NULL)
706  {
707  forward = true;
708 
709  if (fromLocalHost)
710  {
711  SuccessOrExit(error = InsertMplOption(message, header, messageInfo));
712  }
713  }
714  }
715  else
716  {
717  if (IsUnicastAddress(header.GetDestination()))
718  {
719  receive = true;
720  }
721  else if (!header.GetDestination().IsLinkLocal())
722  {
723  forward = true;
724  }
725  else if (netif == NULL)
726  {
727  forward = true;
728  }
729  }
730 
731  message.SetInterfaceId(interfaceId);
732  message.SetOffset(sizeof(header));
733 
734  // process IPv6 Extension Headers
735  nextHeader = static_cast<uint8_t>(header.GetNextHeader());

960 const NetifUnicastAddress *Ip6::SelectSourceAddress(MessageInfo &aMessageInfo)
961 {
962  Address *destination = &aMessageInfo.GetPeerAddr();
963  int interfaceId = aMessageInfo.mInterfaceId;
964  const NetifUnicastAddress *rvalAddr = NULL;
965  const Address *candidateAddr;
966  int8_t candidateId;
967  int8_t rvalIface = 0;
968 
969  for (Netif *netif = GetNetifList(); netif; netif = netif->mNext)
970  {
971  candidateId = netif->GetInterfaceId();
972 
973  for (const NetifUnicastAddress *addr = netif->mUnicastAddresses; addr; addr = addr->GetNext())
974  {
975  candidateAddr = &addr->GetAddress();
976 
977  if (destination->IsLinkLocal() || destination->IsMulticast())
978  {
979  if (interfaceId != candidateId)
980  {
981  continue;
982  }
983  }
984 
985  if (candidateAddr->IsAnycastRoutingLocator())
986  {
987  // Don't use anycast address as source address.
988  continue;
989  }
990 
991  if (rvalAddr == NULL)
992  {
993  // Rule 0: Prefer any address
994  rvalAddr = addr;
995  rvalIface = candidateId;
996  }
997  else if (*candidateAddr == *destination)
998  {
999  // Rule 1: Prefer same address
1000  rvalAddr = addr;
1001  rvalIface = candidateId;
1002  goto exit;
1003  }
1004  else if (addr->GetScope() < rvalAddr->GetScope())
1005  {
1006  // Rule 2: Prefer appropriate scope
1007  if (addr->GetScope() >= destination->GetScope())
1008  {
1009  rvalAddr = addr;
1010  rvalIface = candidateId;
1011  }
1012  }
1013  else if (addr->GetScope() > rvalAddr->GetScope())
1014  {
1015  if (rvalAddr->GetScope() < destination->GetScope())
1016  {
1017  rvalAddr = addr;
1018  rvalIface = candidateId;
1019  }
1020  }
1021  else if (addr->mPreferred && !rvalAddr->mPreferred)
1022  {
1023  // Rule 3: Avoid deprecated addresses
1024  rvalAddr = addr;
1025  rvalIface = candidateId;
1026  }
1027  else if (aMessageInfo.mInterfaceId != 0 && aMessageInfo.mInterfaceId == candidateId &&
1028  rvalIface != candidateId)
1029  {
1030  // Rule 4: Prefer home address
1031  // Rule 5: Prefer outgoing interface
1032  rvalAddr = addr;
1033  rvalIface = candidateId;
1034  }
1035  else if (destination->PrefixMatch(*candidateAddr) > destination->PrefixMatch(rvalAddr->GetAddress()))
1036  {
1037  // Rule 6: Prefer matching label
1038  // Rule 7: Prefer public address
1039  // Rule 8: Use longest prefix matching
1040  rvalAddr = addr;
1041  rvalIface = candidateId;
1042  }
1043  }
1044  }
1045 
1046 exit:
1047  aMessageInfo.mInterfaceId = rvalIface;
1048  return rvalAddr;
1049 }
