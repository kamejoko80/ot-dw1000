189 bool TimerScheduler::TimerCompare(const Timer &aTimerA, const Timer &aTimerB)
190 {
191  uint32_t now = otPlatAlarmGetNow();
192  uint32_t elapsedA = now - aTimerA.mT0;
193  uint32_t elapsedB = now - aTimerB.mT0;
194  bool retval = false;
195 
196  if (aTimerA.mDt >= elapsedA && aTimerB.mDt >= elapsedB)
197  {
198  uint32_t remainingA = aTimerA.mDt - elapsedA;
199  uint32_t remainingB = aTimerB.mDt - elapsedB;
200 
201  if (remainingA < remainingB)
202  {
203  retval = true;
204  }
205  }
206  else if (aTimerA.mDt < elapsedA && aTimerB.mDt >= elapsedB)
207  {
208  retval = true;
209  }
210  else if (aTimerA.mDt < elapsedA && aTimerB.mDt < elapsedB)
211  {
212  uint32_t expiredByA = elapsedA - aTimerA.mDt;
213  uint32_t expiredByB = elapsedB - aTimerB.mDt;
214 
215  if (expiredByB < expiredByA)
216  {
217  retval = true;
218  }
219  }
220 
221  return retval;
222 }
