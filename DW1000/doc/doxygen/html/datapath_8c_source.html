51 otLogBuffer(

79 FilterSendNetBufferListsComplete(
80  NDIS_HANDLE FilterModuleContext,
81  PNET_BUFFER_LIST NetBufferLists,
82  ULONG SendCompleteFlags
83  )
84 /*++
85 
86 Routine Description:
87 
88  Send complete handler
89 
90  This routine is invoked whenever the lower layer is finished processing
91  sent NET_BUFFER_LISTs. If the filter does not need to be involved in the
92  send path, you should remove this routine and the FilterSendNetBufferLists
93  routine. NDIS will pass along send packets on behalf of your filter more
94  efficiently than the filter can.
95 
96 Arguments:
97 
98  FilterModuleContext - our filter context
99  NetBufferLists - a chain of NBLs that are being returned to you
100  SendCompleteFlags - flags (see documentation)
101 
102 --*/
103 {
104  PMS_FILTER pFilter = (PMS_FILTER)FilterModuleContext;
105 
106  LogFuncEntryMsg(DRIVER_DATA_PATH, "Filter: %p, NBL: %p %!STATUS!", FilterModuleContext, NetBufferLists, NetBufferLists->Status);
107 
108  PNET_BUFFER_LIST NBL = NetBufferLists;
109  while (NBL)
110  {
111  PNET_BUFFER_LIST NextNBL = NBL->Next;
112  PNET_BUFFER NetBuffer = NET_BUFFER_LIST_FIRST_NB(NBL);
113 
114  // Cancel command if we failed to send the NBL
115  if (!NT_SUCCESS(NBL->Status))
116  {
117  spinel_tid_t tid = (spinel_tid_t)(ULONG_PTR)NetBuffer->ProtocolReserved[1];
118  if (tid != 0)
119  {
120 #ifdef COMMAND_INIT_RETRY
121  NT_ASSERT(pFilter->cmdInitTryCount < 9 || NBL->Status != NDIS_STATUS_PAUSED);
122 #endif
123  otLwfCmdCancel(pFilter, NDIS_TEST_SEND_COMPLETE_AT_DISPATCH_LEVEL(SendCompleteFlags), tid);
124  }
125  }
126 
127  NetBuffer->DataLength = (ULONG)(ULONG_PTR)NetBuffer->ProtocolReserved[0];
128  NdisAdvanceNetBufferDataStart(NetBuffer, NetBuffer->DataLength, TRUE, NULL);
129  NdisFreeNetBufferList(NBL);
130 
131  // Release the command rundown protection
132  ExReleaseRundownProtection(&pFilter->cmdRundown);
133 
134  NBL = NextNBL;
135  }
136 
137  LogFuncExit(DRIVER_DATA_PATH);
138 }

142 FilterSendNetBufferLists(
143  NDIS_HANDLE FilterModuleContext,
144  PNET_BUFFER_LIST NetBufferLists,
145  NDIS_PORT_NUMBER PortNumber,
146  ULONG SendFlags
147  )
148 /*++
149 
150 Routine Description:
151 
152  Send Net Buffer List handler
153  This function is an optional function for filter drivers. If provided, NDIS
154  will call this function to transmit a linked list of NetBuffers, described by a
155  NetBufferList, over the network. If this handler is NULL, NDIS will skip calling
156  this filter when sending a NetBufferList and will call the next lower
157  driver in the stack. A filter that doesn't provide a FilerSendNetBufferList
158  handler can not originate a send on its own.
159 
160 Arguments:
161 
162  FilterModuleContext - our filter context area
163  NetBufferLists - a List of NetBufferLists to send
164  PortNumber - Port Number to which this send is targeted
165  SendFlags - specifies if the call is at DISPATCH_LEVEL
166 
167 --*/
168 {
169  PMS_FILTER pFilter = (PMS_FILTER)FilterModuleContext;
170  BOOLEAN DispatchLevel = NDIS_TEST_SEND_AT_DISPATCH_LEVEL(SendFlags);
171 
172  UNREFERENCED_PARAMETER(PortNumber);
173 
174  LogFuncEntryMsg(DRIVER_DATA_PATH, "Filter: %p, NBL: %p", FilterModuleContext, NetBufferLists);
175 
176  // Try to grab a ref on the data path first, to make sure we are allowed
177  if (!ExAcquireRundownProtection(&pFilter->ExternalRefs))
178  {
179  LogVerbose(DRIVER_DEFAULT, "Failing SendNetBufferLists because data path isn't active.");
180 
181  // Ignore any NBLs we get if we aren't active (can't get a ref)
182  PNET_BUFFER_LIST CurrNbl = NetBufferLists;
183  while (CurrNbl)
184  {
185  NET_BUFFER_LIST_STATUS(CurrNbl) = NDIS_STATUS_PAUSED;
186  CurrNbl = NET_BUFFER_LIST_NEXT_NBL(CurrNbl);
187  }
188  NdisFSendNetBufferListsComplete(
189  pFilter->FilterHandle,
190  NetBufferLists,
191  DispatchLevel ? NDIS_SEND_COMPLETE_FLAGS_DISPATCH_LEVEL : 0
192  );
193  }
194  else
195  {
196  if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_RADIO_MODE)
197  {
198  // Indicate a new NBL to process on our worker thread
199  otLwfEventProcessingIndicateNewNetBufferLists(
200  pFilter,
201  DispatchLevel,
202  NetBufferLists
203  );
204  }
205  else
206  {
207  PNET_BUFFER_LIST CurrNbl = NetBufferLists;
208  while (CurrNbl)
209  {
210  PNET_BUFFER CurrNb = NET_BUFFER_LIST_FIRST_NB(CurrNbl);
211  while (CurrNb)
212  {
213  otLwfCmdSendIp6PacketAsync(pFilter, DispatchLevel, CurrNb, TRUE);
214  CurrNb = NET_BUFFER_NEXT_NB(CurrNb);
215  }
216 
217  NET_BUFFER_LIST_STATUS(CurrNbl) = NDIS_STATUS_SUCCESS;
218  CurrNbl = NET_BUFFER_LIST_NEXT_NBL(CurrNbl);
219  }
220 
221  NdisFSendNetBufferListsComplete(
222  pFilter->FilterHandle,
223  NetBufferLists,
224  DispatchLevel ? NDIS_SEND_COMPLETE_FLAGS_DISPATCH_LEVEL : 0
225  );
226  }
227 
228  // Release the data path ref now
229  ExReleaseRundownProtection(&pFilter->ExternalRefs);
230  }
231 
232  LogFuncExit(DRIVER_DATA_PATH);
233 }

273 FilterReturnNetBufferLists(
274  NDIS_HANDLE FilterModuleContext,
275  PNET_BUFFER_LIST NetBufferLists,
276  ULONG ReturnFlags
277  )
278 /*++
279 
280 Routine Description:
281 
282  FilterReturnNetBufferLists handler.
283  FilterReturnNetBufferLists is an optional function. If provided, NDIS calls
284  FilterReturnNetBufferLists to return the ownership of one or more NetBufferLists
285  and their embedded NetBuffers to the filter driver. If this handler is NULL, NDIS
286  will skip calling this filter when returning NetBufferLists to the underlying
287  miniport and will call the next lower driver in the stack. A filter that doesn't
288  provide a FilterReturnNetBufferLists handler cannot originate a receive indication
289  on its own.
290 
291 Arguments:
292 
293  FilterInstanceContext - our filter context area
294  NetBufferLists - a linked list of NetBufferLists that this
295  filter driver indicated in a previous call to
296  NdisFIndicateReceiveNetBufferLists
297  ReturnFlags - flags specifying if the caller is at DISPATCH_LEVEL
298 
299 --*/
300 {
301  PMS_FILTER pFilter = (PMS_FILTER)FilterModuleContext;
302 
303  UNREFERENCED_PARAMETER(ReturnFlags);
304 
305  LogFuncEntryMsg(DRIVER_DATA_PATH, "Filter: %p, NBL: %p", pFilter, NetBufferLists);
306 
307  PNET_BUFFER_LIST CurrNbl = NetBufferLists;
308  while (CurrNbl)
309  {
310  if (!NT_SUCCESS(CurrNbl->Status))
311  {
312  LogVerbose(DRIVER_DATA_PATH, "NBL failed on return: %!STATUS!", CurrNbl->Status);
313  }
314 
315  PNET_BUFFER_LIST NblToFree = CurrNbl;
316  PNET_BUFFER NbToFree = NET_BUFFER_LIST_FIRST_NB(NblToFree);
317 
318  CurrNbl = NET_BUFFER_LIST_NEXT_NBL(CurrNbl);
319  NET_BUFFER_LIST_NEXT_NBL(NblToFree) = NULL;
320 
321  NdisAdvanceNetBufferDataStart(NbToFree, NET_BUFFER_DATA_LENGTH(NbToFree), TRUE, NULL);
322  NdisFreeNetBufferList(NblToFree);
323  }
324 
325  LogFuncExit(DRIVER_DATA_PATH);
326 }

329 VOID
330 FilterReceiveNetBufferLists(
331  NDIS_HANDLE FilterModuleContext,
332  PNET_BUFFER_LIST NetBufferLists,
333  NDIS_PORT_NUMBER PortNumber,
334  ULONG NumberOfNetBufferLists,
335  ULONG ReceiveFlags
336  )
337 /*++
338 
339 Routine Description:
340 
341  FilerReceiveNetBufferLists is an optional function for filter drivers.
342  If provided, this function processes receive indications made by underlying
343  NIC or lower level filter drivers. This function can also be called as a
344  result of loopback. If this handler is NULL, NDIS will skip calling this
345  filter when processing a receive indication and will call the next higher
346  driver in the stack. A filter that doesn't provide a
347  FilterReceiveNetBufferLists handler cannot provide a
348  FilterReturnNetBufferLists handler and cannot a initiate an original receive
349  indication on its own.
350 
351 Arguments:
352 
353  FilterModuleContext - our filter context area.
354  NetBufferLists - a linked list of NetBufferLists
355  PortNumber - Port on which the receive is indicated
356  ReceiveFlags -
357 
358 N.B.: It is important to check the ReceiveFlags in NDIS_TEST_RECEIVE_CANNOT_PEND.
359  This controls whether the receive indication is an synchronous or
360  asynchronous function call.
361 
362 --*/
363 {
364 
365  PMS_FILTER pFilter = (PMS_FILTER)FilterModuleContext;
366  BOOLEAN DispatchLevel = NDIS_TEST_RECEIVE_AT_DISPATCH_LEVEL(ReceiveFlags);
367 
368  UNREFERENCED_PARAMETER(PortNumber);
369  UNREFERENCED_PARAMETER(NumberOfNetBufferLists);
370 
371  LogFuncEntryMsg(DRIVER_DATA_PATH, "Filter: %p, NBL: %p", FilterModuleContext, NetBufferLists);
372 
373  // Iterate through each NBL/NB and grab the data as a contiguous buffer to
374  // indicate to the Spinel command layer.
375  PNET_BUFFER_LIST CurrNbl = NetBufferLists;
376  while (CurrNbl)
377  {
378  PNET_BUFFER CurrNb = NET_BUFFER_LIST_FIRST_NB(CurrNbl);
379  while (CurrNb)
380  {
381  PUCHAR Buffer = (PUCHAR)NdisGetDataBuffer(CurrNb, CurrNb->DataLength, NULL, 1, 0);
382  if (Buffer == NULL)
383  {
384  Buffer = (PUCHAR)FILTER_ALLOC_MEM(pFilter->FilterHandle, CurrNb->DataLength);
385  if (Buffer != NULL)
386  {
387  PUCHAR _Buffer = (PUCHAR)NdisGetDataBuffer(CurrNb, CurrNb->DataLength, Buffer, 1, 0);
388  NT_ASSERT(_Buffer == Buffer);
389  if (_Buffer)
390  {
391  otLwfCmdRecveive(pFilter, DispatchLevel, Buffer, CurrNb->DataLength);
392  }
393  FILTER_FREE_MEM(Buffer);
394  }
395  }
396  else
397  {
398  otLwfCmdRecveive(pFilter, DispatchLevel, Buffer, CurrNb->DataLength);
399  }
400  CurrNb = NET_BUFFER_NEXT_NB(CurrNb);
401  }
402 
403  NET_BUFFER_LIST_STATUS(CurrNbl) = NDIS_STATUS_SUCCESS;
404  CurrNbl = NET_BUFFER_LIST_NEXT_NBL(CurrNbl);
405  }
406 
407  if (NDIS_TEST_RECEIVE_CAN_PEND(ReceiveFlags))
408  {
409  NdisFReturnNetBufferLists(
410  pFilter->FilterHandle,
411  NetBufferLists,
412  DispatchLevel ? NDIS_RETURN_FLAGS_DISPATCH_LEVEL : 0
413  );
414  }
415 
416  LogFuncExit(DRIVER_DATA_PATH);
417 }

419 // Callback received from OpenThread when it has an IPv6 packet ready for
420 // delivery to TCPIP.
421 void
422 otLwfReceiveIp6DatagramCallback(
423  _In_ otMessage aMessage,
424  _In_ void *aContext
425  )
426 {
427  PMS_FILTER pFilter = (PMS_FILTER)aContext;
428  uint16_t messageLength = otGetMessageLength(aMessage);
429  PNET_BUFFER_LIST NetBufferList = NULL;
430  PNET_BUFFER NetBuffer = NULL;
431  NDIS_STATUS Status = NDIS_STATUS_SUCCESS;
432  PUCHAR DataBuffer = NULL;
433  int BytesRead = 0;
434  IPV6_HEADER* v6Header;
435 
436 #ifdef FORCE_SYNCHRONOUS_RECEIVE
437  KIRQL irql;
438 #endif
439 
440  // Create the NetBufferList
441  NetBufferList =
442  NdisAllocateNetBufferAndNetBufferList(
443  pFilter->cmdNblPool, // PoolHandle
444  0, // ContextSize
445  0, // ContextBackFill
446  NULL, // MdlChain
447  0, // DataOffset
448  0 // DataLength
449  );
450  if (NetBufferList == NULL)
451  {
452  LogWarning(DRIVER_DEFAULT, "Failed to create Recv NetBufferList");
453  goto error;
454  }
455 
456  // Set the flag to indicate its a IPv6 packet
457  NdisSetNblFlag(NetBufferList, NDIS_NBL_FLAGS_IS_IPV6);
458  NET_BUFFER_LIST_INFO(NetBufferList, NetBufferListFrameType) =
459  UlongToPtr(RtlUshortByteSwap(ETHERNET_TYPE_IPV6));
460 
461  // Initialize NetBuffer fields
462  NetBuffer = NET_BUFFER_LIST_FIRST_NB(NetBufferList);
463  NET_BUFFER_CURRENT_MDL(NetBuffer) = NULL;
464  NET_BUFFER_CURRENT_MDL_OFFSET(NetBuffer) = 0;
465  NET_BUFFER_DATA_LENGTH(NetBuffer) = 0;
466  NET_BUFFER_DATA_OFFSET(NetBuffer) = 0;
467  NET_BUFFER_FIRST_MDL(NetBuffer) = NULL;
468 
469  // Allocate the NetBuffer for SendNetBufferList
470  Status = NdisRetreatNetBufferDataStart(NetBuffer, messageLength, 0, NULL);
471  if (Status != NDIS_STATUS_SUCCESS)
472  {
473  NdisFreeNetBufferList(NetBufferList);
474  LogError(DRIVER_DEFAULT, "Failed to allocate NB for Recv NetBufferList, %!NDIS_STATUS!", Status);
475  goto error;
476  }
477 
478  // Get the data buffer to write to
479  DataBuffer = NdisGetDataBuffer(NetBuffer, messageLength, NULL, 1, 0);
480  NT_ASSERT(DataBuffer);
481  if (DataBuffer == NULL)
482  {
483  NdisAdvanceNetBufferDataStart(NetBuffer, messageLength, TRUE, NULL);
484  NdisFreeNetBufferList(NetBufferList);
485  LogError(DRIVER_DEFAULT, "Failed to get contiguous data buffer for Recv NetBufferList");
486  goto error;
487  }
488 
489  // Read the bytes to the buffer
490  BytesRead = otReadMessage(aMessage, 0, DataBuffer, messageLength);
491  NT_ASSERT(BytesRead == (int)messageLength);
492  if (BytesRead != (int)messageLength)
493  {
494  NdisAdvanceNetBufferDataStart(NetBuffer, messageLength, TRUE, NULL);
495  NdisFreeNetBufferList(NetBufferList);
496  LogError(DRIVER_DEFAULT, "Failed to read message buffer for Recv NetBufferList");
497  goto error;
498  }
499 
500  v6Header = (IPV6_HEADER*)DataBuffer;
501 
502  // Filter messages to addresses we expose
503  if (!IN6_IS_ADDR_MULTICAST(&v6Header->DestinationAddress) &&
504  otLwfFindCachedAddrIndex(pFilter, &v6Header->DestinationAddress) == -1)
505  {
506  NdisAdvanceNetBufferDataStart(NetBuffer, messageLength, TRUE, NULL);
507  NdisFreeNetBufferList(NetBufferList);
508  LogVerbose(DRIVER_DATA_PATH, "Filter: %p dropping internal address message.", pFilter);
509  goto error;
510  }
511 
512  // Filter internal Thread messages
513  if (v6Header->NextHeader == IPPROTO_UDP &&
514  messageLength >= sizeof(IPV6_HEADER) + sizeof(UDPHeader) &&
515  memcmp(&pFilter->otLinkLocalAddr, &v6Header->DestinationAddress, sizeof(IN6_ADDR)) == 0)
516  {
517  // Check for MLE message
518  UDPHeader* UdpHeader = (UDPHeader*)(v6Header + 1);
519  if (UdpHeader->DestinationPort == UdpHeader->SourcePort &&
520  UdpHeader->DestinationPort == RtlUshortByteSwap(19788)) // MLE Port
521  {
522  NdisAdvanceNetBufferDataStart(NetBuffer, messageLength, TRUE, NULL);
523  NdisFreeNetBufferList(NetBufferList);
524  LogVerbose(DRIVER_DATA_PATH, "Filter: %p dropping MLE message.", pFilter);
525  goto error;
526  }
527  }
528 
529  LogVerbose(DRIVER_DATA_PATH, "Filter: %p, IP6_RECV: %p : %!IPV6ADDR! => %!IPV6ADDR! (%u bytes)",
530  pFilter, NetBufferList, &v6Header->SourceAddress, &v6Header->DestinationAddress,
531  messageLength);
532 
533 #ifdef LOG_BUFFERS
534  otLogBuffer(DataBuffer, messageLength);
535 #endif
536 
537 #ifdef FORCE_SYNCHRONOUS_RECEIVE
538  irql = KfRaiseIrql(DISPATCH_LEVEL);
539 
540  if (messageLength == 248) // Magic length used for TAEF test packets
541  {
542  DbgBreakPoint();
543  }
544 #endif
545 
546  // Indicate the NBL up
547  NdisFIndicateReceiveNetBufferLists(
548  pFilter->FilterHandle,
549  NetBufferList,
550  NDIS_DEFAULT_PORT_NUMBER,
551  1,
552 #ifdef FORCE_SYNCHRONOUS_RECEIVE
553  NDIS_RECEIVE_FLAGS_RESOURCES | NDIS_RECEIVE_FLAGS_DISPATCH_LEVEL
554 #else
555  0
556 #endif
557  );
558 
559 #ifdef FORCE_SYNCHRONOUS_RECEIVE
560  KeLowerIrql(irql);
561  FilterReturnNetBufferLists(pFilter, NetBufferList, 0);
562 #endif
563 
564 error:
565 
566  otFreeMessage(aMessage);
567 }
