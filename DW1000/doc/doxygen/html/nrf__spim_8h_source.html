456 __STATIC_INLINE void nrf_spim_configure(NRF_SPIM_Type * p_reg,
457  nrf_spim_mode_t spi_mode,
458  nrf_spim_bit_order_t spi_bit_order)
459 {
460  uint32_t config = (spi_bit_order == NRF_SPIM_BIT_ORDER_MSB_FIRST ?
461  SPIM_CONFIG_ORDER_MsbFirst : SPIM_CONFIG_ORDER_LsbFirst);
462  switch (spi_mode)
463  {
464  default:
465  case NRF_SPIM_MODE_0:
466  config |= (SPIM_CONFIG_CPOL_ActiveHigh << SPIM_CONFIG_CPOL_Pos) |
467  (SPIM_CONFIG_CPHA_Leading << SPIM_CONFIG_CPHA_Pos);
468  break;
469 
470  case NRF_SPIM_MODE_1:
471  config |= (SPIM_CONFIG_CPOL_ActiveHigh << SPIM_CONFIG_CPOL_Pos) |
472  (SPIM_CONFIG_CPHA_Trailing << SPIM_CONFIG_CPHA_Pos);
473  break;
474 
475  case NRF_SPIM_MODE_2:
476  config |= (SPIM_CONFIG_CPOL_ActiveLow << SPIM_CONFIG_CPOL_Pos) |
477  (SPIM_CONFIG_CPHA_Leading << SPIM_CONFIG_CPHA_Pos);
478  break;
479 
480  case NRF_SPIM_MODE_3:
481  config |= (SPIM_CONFIG_CPOL_ActiveLow << SPIM_CONFIG_CPOL_Pos) |
482  (SPIM_CONFIG_CPHA_Trailing << SPIM_CONFIG_CPHA_Pos);
483  break;
484  }
485  p_reg->CONFIG = config;
486 }
