144 void Server::ProcessReceivedMessage(Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
145 {
146  Header header;
147  char uriPath[Resource::kMaxReceivedUriPath] = "";
148  char *curUriPath = uriPath;
149  const Header::Option *coapOption;
150 
151  SuccessOrExit(header.FromMessage(aMessage, false));
152  aMessage.MoveOffset(header.GetLength());
153 
154  coapOption = header.GetCurrentOption();
155 
156  while (coapOption != NULL)
157  {
158  switch (coapOption->mNumber)
159  {
160  case kCoapOptionUriPath:
161  if (curUriPath != uriPath)
162  {
163  *curUriPath++ = '/';
164  }
165 
166  VerifyOrExit(coapOption->mLength < sizeof(uriPath) - static_cast<size_t>(curUriPath + 1 - uriPath), ;);
167 
168  memcpy(curUriPath, coapOption->mValue, coapOption->mLength);
169  curUriPath += coapOption->mLength;
170  break;
171 
172  case kCoapOptionContentFormat:
173  break;
174 
175  default:
176  ExitNow();
177  }
178 
179  coapOption = header.GetNextOption();
180  }
181 
182  curUriPath[0] = '\0';
183 
184  for (Resource *resource = mResources; resource; resource = resource->GetNext())
185  {
186  if (strcmp(resource->mUriPath, uriPath) == 0)
187  {
188  resource->HandleRequest(header, aMessage, aMessageInfo);
189  ExitNow();
190  }
191  }
192 
193 exit:
194  {}
195 }
