78 void
79 I2SEnable(uint32_t ui32Base)
80 {
81  //
82  // Check the arguments.
83  //
84  ASSERT(I2SBaseValid(ui32Base));
85  (void)ui32Base;
86 
87  //
88  // Make sure the control table pointer is setup to a memory location.
89  //
90  if(!(g_pControlTable))
91  {
92  return;
93  }
94 
95  //
96  // Write the address to the first input/output buffer.
97  //
98  HWREG(I2S0_BASE + I2S_O_AIFINPTRNEXT) = g_pControlTable->ui32InBase;
99  g_pControlTable->ui32InOffset = 0;
100  HWREG(I2S0_BASE + I2S_O_AIFOUTPTRNEXT) = g_pControlTable->ui32OutBase;
101  g_pControlTable->ui32OutOffset = 0;
102 
103  //
104  // Enable the I2S module.
105  //
106  HWREG(I2S0_BASE + I2S_O_AIFDMACFG) = (uint32_t)g_pControlTable->ui16DMABufSize - 1;
107 }

141 void
142 I2SChannelConfigure(uint32_t ui32Base, uint32_t ui32Chan0Cfg,
143  uint32_t ui32Chan1Cfg, uint32_t ui32Chan2Cfg)
144 {
145  uint32_t ui32InChan;
146  uint32_t ui32OutChan;
147  uint32_t ui32ChanMask;
148 
149  //
150  // Check the arguments.
151  //
152  ASSERT(I2SBaseValid(ui32Base));
153  (void)ui32Base;
154  ASSERT(ui32Chan0Cfg & (I2S_CHAN_CFG_MASK | I2S_LINE_MASK))
155  ASSERT(ui32Chan1Cfg & (I2S_CHAN_CFG_MASK | I2S_LINE_MASK))
156  ASSERT(ui32Chan2Cfg & (I2S_CHAN_CFG_MASK | I2S_LINE_MASK))
157 
158  ui32InChan = 0;
159  ui32OutChan = 0;
160 
161  //
162  // Configure input/output channels.
163  //
164  HWREG(I2S0_BASE + I2S_O_AIFDIRCFG) = ((ui32Chan0Cfg << I2S_AIFDIRCFG_AD0_S)
165  & I2S_AIFDIRCFG_AD0_M) |
166  ((ui32Chan1Cfg << I2S_AIFDIRCFG_AD1_S)
167  & I2S_AIFDIRCFG_AD1_M) |
168  ((ui32Chan2Cfg << I2S_AIFDIRCFG_AD2_S)
169  & I2S_AIFDIRCFG_AD2_M);
170 
171  //
172  // Configure the valid channel mask.
173  //
174  HWREG(I2S0_BASE + I2S_O_AIFWMASK0) = (ui32Chan0Cfg >> 8) & I2S_AIFWMASK0_MASK_M;
175  HWREG(I2S0_BASE + I2S_O_AIFWMASK1) = (ui32Chan1Cfg >> 8) & I2S_AIFWMASK1_MASK_M;
176  HWREG(I2S0_BASE + I2S_O_AIFWMASK2) = (ui32Chan2Cfg >> 8) & I2S_AIFWMASK2_MASK_M;
177 
178  //
179  // Resolve and save the number of input and output channels.
180  //
181  ui32ChanMask = (ui32Chan0Cfg & I2S_CHAN_CFG_MASK) >> 8;
182  if(ui32Chan0Cfg & I2S_LINE_INPUT)
183  {
184  while(ui32ChanMask)
185  {
186  if(ui32ChanMask & 0x1)
187  {
188  ui32InChan++;
189  }
190  //
191  // Shift down channel mask
192  //
193  ui32ChanMask >>= 1;
194  }
195 
196  }
197  else if(ui32Chan0Cfg & I2S_LINE_OUTPUT)
198  {
199  while(ui32ChanMask)
200  {
201  if(ui32ChanMask & 0x1)
202  {
203  ui32OutChan++;
204  }
205  //
206  // Shift down channel mask
207  //
208  ui32ChanMask >>= 1;
209  }
210  }
211 
212  ui32ChanMask = (ui32Chan1Cfg & I2S_CHAN_CFG_MASK) >> 8;
213  if(ui32Chan1Cfg & I2S_LINE_INPUT)
214  {
215  while(ui32ChanMask)
216  {
217  if(ui32ChanMask & 0x1)
218  {
219  ui32InChan++;
220  }
221  //
222  // Shift down channel mask
223  //
224  ui32ChanMask >>= 1;
225  }
226  }
227  else if(ui32Chan1Cfg & I2S_LINE_OUTPUT)
228  {
229  while(ui32ChanMask)
230  {
231  if(ui32ChanMask & 0x1)
232  {
233  ui32OutChan++;
234  }
235  //
236  // Shift down channel mask
237  //
238  ui32ChanMask >>= 1;
239  }
240  }
241 
242  ui32ChanMask = (ui32Chan2Cfg & I2S_CHAN_CFG_MASK) >> 8;
243  if(ui32Chan2Cfg & I2S_LINE_INPUT)
244  {
245  while(ui32ChanMask)
246  {
247  if(ui32ChanMask & 0x1)
248  {
249  ui32InChan++;
250  }
251  //
252  // Shift down channel mask
253  //
254  ui32ChanMask >>= 1;
255  }
256  }
257  else if(ui32Chan2Cfg & I2S_LINE_OUTPUT)
258  {
259  while(ui32ChanMask)
260  {
261  if(ui32ChanMask & 0x1)
262  {
263  ui32OutChan++;
264  }
265  //
266  // Shift down channel mask
267  //
268  ui32ChanMask >>= 1;
269  }
270  }
271 
272  g_pControlTable->ui8InChan = (uint8_t)ui32InChan;
273  g_pControlTable->ui8OutChan = (uint8_t)ui32OutChan;
274 }

334 void
335 I2SPointerUpdate(uint32_t ui32Base, bool bInput)
336 {
337  uint32_t ui32NextPtr;
338 
339  //
340  // Check the arguments.
341  //
342  ASSERT(I2SBaseValid(ui32Base));
343  (void)ui32Base;
344 
345  //
346  // Update the next input/output pointer with the correct address.
347  //
348  if(bInput == true)
349  {
350  ui32NextPtr = (g_pControlTable->ui8InChan *
351  (g_pControlTable->ui16MemLen >> 3)) *
352  g_pControlTable->ui16DMABufSize;
353  g_pControlTable->ui32InOffset = ((g_pControlTable->ui32InOffset +
354  ui32NextPtr) %
355  g_pControlTable->ui16ChBufSize);
356  HWREG(I2S0_BASE + I2S_O_AIFINPTRNEXT) = g_pControlTable->ui32InOffset +
357  g_pControlTable->ui32InBase;
358  }
359  else
360  {
361  ui32NextPtr = (g_pControlTable->ui8OutChan *
362  (g_pControlTable->ui16MemLen >> 3)) *
363  g_pControlTable->ui16DMABufSize;
364  g_pControlTable->ui32OutOffset = ((g_pControlTable->ui32OutOffset +
365  ui32NextPtr) %
366  g_pControlTable->ui16ChBufSize);
367  HWREG(I2S0_BASE + I2S_O_AIFOUTPTRNEXT) =
368  g_pControlTable->ui32OutOffset +
369  g_pControlTable->ui32OutBase;
370  }

415 uint32_t
416 I2SSampleStampGet(uint32_t ui32Base, uint32_t ui32Channel)
417 {
418  uint32_t ui32FrameClkCnt;
419  uint32_t ui32SysClkCnt;
420  uint32_t ui32PeriodSysClkCnt;
421  uint32_t ui32SampleStamp;
422  (void)ui32Base;
423  (void)ui32Channel;
424 
425  //
426  // Get the number of Frame clock counts since last stamp.
427  //
428  ui32FrameClkCnt = HWREG(I2S0_BASE + I2S_O_STMPWCNTCAPT0);
429 
430  //
431  // Get the number of system clock ticks since last frame clock edge.
432  //
433  ui32SysClkCnt = HWREG(I2S0_BASE + I2S_O_STMPXCNTCAPT0);
434 
435  //
436  // Get the number system clock ticks in the last frame clock period.
437  //
438  ui32PeriodSysClkCnt = HWREG(I2S0_BASE + I2S_O_STMPXPER);
439 
440  //
441  // Calculate the sample stamp.
442  //
443  ui32SampleStamp = (ui32SysClkCnt << 16) / ui32PeriodSysClkCnt;
444  ui32SampleStamp = (ui32SampleStamp > I2S_STMP_SATURATION) ?
445  I2S_STMP_SATURATION : ui32SampleStamp;
446  ui32SampleStamp |= (ui32FrameClkCnt << 16);
447 
448  return (ui32SampleStamp);
449 }
