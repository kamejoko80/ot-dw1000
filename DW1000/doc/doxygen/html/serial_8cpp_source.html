33 _IRQL_requires_max_(PASSIVE_LEVEL)
34 NTSTATUS
35 SerialInitialize(
36  _In_ POTTMP_ADAPTER_CONTEXT AdapterContext
37  )
38 /*++
39 Routine Description:
40 
41  SerialInitialize attempts to find and open the first COM port available, with
42  the assumption that it should be for the Thread device.
43 
44 Arguments:
45 
46  AdapterContext - handle to a OTTMP Adapter
47 
48 Return Value:
49 
50  NTSTATUS - A failure here will indicate the serial COM port was not able
51  to be opened.
52 --*/
53 {
54  NTSTATUS status = STATUS_UNSUCCESSFUL;
55  PWSTR SymbolicLinkList = NULL;
56 
57  LogFuncEntry(DRIVER_DEFAULT);
58 
59  PAGED_CODE();
60 
61  do
62  {
63  WDF_OBJECT_ATTRIBUTES attr = { 0 };
64  WDF_WORKITEM_CONFIG config = { 0 };
65 
66  //
67  // Send Queue Variables
68  //
69 
70  InitializeListHead(&AdapterContext->SendQueue);
71  AdapterContext->SendQueueRunning = false;
72 
73  WDF_OBJECT_ATTRIBUTES_INIT(&attr);
74  attr.ParentObject = AdapterContext->Device;
75  status = WdfSpinLockCreate(&attr, &AdapterContext->SendLock);
76 
77  if (!NT_SUCCESS(status)) {
78  LogError(DRIVER_DEFAULT, "WdfSpinLockCreate(lockSend) failed %!STATUS!", status);
79  break;
80  }
81 
82  WDF_OBJECT_ATTRIBUTES_INIT_CONTEXT_TYPE(&attr, WDF_DEVICE_INFO);
83  attr.ParentObject = AdapterContext->Device;
84  WDF_WORKITEM_CONFIG_INIT(&config, SerialSendLoop);
85 
86  status = WdfWorkItemCreate(&config, &attr, &AdapterContext->SendWorkItem);
87 
88  if (!NT_SUCCESS(status)) {
89  LogError(DRIVER_DEFAULT, "WdfWorkItemCreate(SerialSendLoop) failed %!STATUS!", status);
90  break;
91  }
92  GetWdfDeviceInfo(AdapterContext->SendWorkItem)->AdapterContext = AdapterContext;
93 
94  //
95  // Receive Variables
96  //
97 
98  WDF_OBJECT_ATTRIBUTES_INIT_CONTEXT_TYPE(&attr, WDF_DEVICE_INFO);
99  attr.ParentObject = AdapterContext->Device;
100  WDF_WORKITEM_CONFIG_INIT(&config, SerialRecvLoop);
101 
102  status = WdfWorkItemCreate(&config, &attr, &AdapterContext->RecvWorkItem);
103 
104  if (!NT_SUCCESS(status)) {
105  LogError(DRIVER_DEFAULT, "WdfWorkItemCreate(SerialRecvLoop) failed %!STATUS!", status);
106  break;
107  }
108 
109  GetWdfDeviceInfo(AdapterContext->RecvWorkItem)->AdapterContext = AdapterContext;
110 
111  // Query the system for device with SERIAL interface
112  status =
113  IoGetDeviceInterfaces(
114  &GUID_DEVINTERFACE_COMPORT,
115  NULL,
116  0,
117  &SymbolicLinkList // List of symbolic names; separate by NULL, EOL with NULL+NULL.
118  );
119 
120  if (!NT_SUCCESS(status)) {
121  LogError(DRIVER_DEFAULT, "IoGetDeviceInterfaces failed %!STATUS!", status);
122  break;
123  }
124 
125  // Make sure there is a COM port found
126  NT_ASSERT(SymbolicLinkList);
127  if (*SymbolicLinkList == NULL) {
128  status = STATUS_DEVICE_NOT_CONNECTED;
129  LogError(DRIVER_DEFAULT, "No COM ports found!");
130  break;
131  }
132 
133 #if DBG
134  for (PCWSTR sym = SymbolicLinkList; *sym != NULL; sym += wcslen(sym) + 1)
135  {
136  LogVerbose(DRIVER_DEFAULT, "Symbolic Name found: %ws", sym);
137  }
138 #endif
139 
140  // Try to open each serial port until we get that one works or we exhaust them all
141  for (PCWSTR sym = SymbolicLinkList; *sym != NULL; sym += wcslen(sym) + 1)
142  {
143  // Initialize the target
144  status = SerialInitializeTarget(AdapterContext, sym);
145 
146  // Break on success
147  if (NT_SUCCESS(status)) {
148  break;
149  }
150  }
151 
152  } while (false);
153 
154  // Clean up on failure
155  if (!NT_SUCCESS(status)) {
156  SerialUninitialize(AdapterContext);
157  }
158 
159  if (SymbolicLinkList) {
160  ExFreePool(SymbolicLinkList);
161  SymbolicLinkList = NULL;
162  }
163 
164  LogFuncExitNT(DRIVER_DEFAULT, status);
165 
166  return status;
167 }

203 _No_competing_thread_
204 _IRQL_requires_max_(PASSIVE_LEVEL)
205 NTSTATUS
206 SerialInitializeTarget(
207  _In_ POTTMP_ADAPTER_CONTEXT AdapterContext,
208  _In_ PCWSTR TargetName
209 )
210 {
211  NTSTATUS status = STATUS_UNSUCCESSFUL;
212  WDFIOTARGET tempTarget = WDF_NO_HANDLE;
213 
214  LogFuncEntry(DRIVER_DEFAULT);
215 
216  PAGED_CODE();
217 
218  do
219  {
220  DECLARE_UNICODE_STRING_SIZE(PortName, 64); // Maximum name length of the device path to a serial port
221  WDF_IO_TARGET_OPEN_PARAMS openParams = { 0 };
222  WDF_OBJECT_ATTRIBUTES attr = { 0 };
223 
224  // Create the Wdf IoTarget
225  status = WdfIoTargetCreate(AdapterContext->Device, WDF_NO_OBJECT_ATTRIBUTES, &tempTarget);
226 
227  if (!NT_SUCCESS(status)) {
228  LogError(DRIVER_DEFAULT, "WdfIoTargetCreate failed %!STATUS!", status);
229  break;
230  }
231 
232  // Try the COM port
233  LogInfo(DRIVER_DEFAULT, "Opening device: %ws", TargetName);
234  RtlInitUnicodeString(&PortName, TargetName);
235  WDF_IO_TARGET_OPEN_PARAMS_INIT_OPEN_BY_NAME(
236  &openParams,
237  &PortName,
238  GENERIC_READ | GENERIC_WRITE);
239 
240  // Open the port on the target
241  status = WdfIoTargetOpen(tempTarget, &openParams);
242 
243  if (!NT_SUCCESS(status)) {
244  LogError(DRIVER_DEFAULT, "WdfIoTargetOpen(%wZ) failed %!STATUS!", &PortName, status);
245  break;
246  }
247 
248  AdapterContext->WdfIoTarget = tempTarget;
249  tempTarget = WDF_NO_HANDLE;
250 
251  WDF_OBJECT_ATTRIBUTES_INIT_CONTEXT_TYPE(&attr, WDF_DEVICE_INFO);
252  attr.ParentObject = AdapterContext->Device;
253 
254  status = WdfRequestCreate(&attr, AdapterContext->WdfIoTarget, &AdapterContext->RecvReadRequest);
255 
256  if (!NT_SUCCESS(status)) {
257  LogError(DRIVER_DEFAULT, "WdfRequestCreate failed %!STATUS!", status);
258  break;
259  }
260 
261  // Try to configure the target
262  status = SerialConfigure(AdapterContext);
263 
264  if (!NT_SUCCESS(status)) {
265  LogError(DRIVER_DEFAULT, "SerialConfigure failed %!STATUS!", status);
266  break;
267  }
268 
269  } while (false);
270 
271  // Clean up on failure
272  if (!NT_SUCCESS(status)) {
273  SerialUninitializeTarget(AdapterContext);
274  }
275 
276  if (tempTarget) {
277  WdfIoTargetClose(tempTarget);
278  }
279 
280  LogFuncExitNT(DRIVER_DEFAULT, status);
281 
282  return status;
283 }

499 _IRQL_requires_(PASSIVE_LEVEL)
500 NTSTATUS
501 SerialCheckStatus(
502  _In_ POTTMP_ADAPTER_CONTEXT AdapterContext,
503  _In_ bool DataExpected
504  )
505 /*++
506 Routine Description:
507 
508  SerialCheckStatus validates the current status of the serial COM port.
509 
510 Arguments:
511 
512  AdapterContext - handle to a OTTMP Adapter
513 
514 Return Value:
515 
516  NTSTATUS - A failure here will indicate the serial COM port is not in an
517  expected state.
518 --*/
519 {
520  NTSTATUS status = STATUS_UNSUCCESSFUL;
521  WDF_MEMORY_DESCRIPTOR outputDesc;
522  WDF_REQUEST_SEND_OPTIONS wrso = {
523  sizeof(WDF_REQUEST_SEND_OPTIONS),
524  WDF_REQUEST_SEND_OPTION_TIMEOUT | WDF_REQUEST_SEND_OPTION_SYNCHRONOUS,
525  WDF_REL_TIMEOUT_IN_SEC(1) // Nothing should take more than a second to complete
526  };
527  ULONG_PTR bytesReturned = 0;
528  SERIAL_STATUS ss = { 0 };
529  WDF_MEMORY_DESCRIPTOR_INIT_BUFFER(&outputDesc, (PVOID)&ss, sizeof(ss));
530 
531  LogFuncEntry(DRIVER_DEFAULT);
532 
533  PAGED_CODE();
534 
535  // Check to ensure we are ready to send
536  status = SerialSendIoctl(AdapterContext, IOCTL_SERIAL_GET_COMMSTATUS, &wrso, WDF_NO_HANDLE, &outputDesc, &bytesReturned);
537 
538  if (!NT_SUCCESS(status))
539  {
540  LogError(DRIVER_DEFAULT, "IOCTL_SERIAL_GET_COMMSTATUS failed %!STATUS!", status);
541  }
542  else if (bytesReturned >= sizeof(ss))
543  {
544  if (ss.HoldReasons)
545  {
546  if (ss.HoldReasons != SERIAL_TX_WAITING_FOR_CTS)
547  {
548  LogError(DRIVER_DEFAULT, "HoldReasons is wrong (should only be CTS, but is %x)", ss.HoldReasons );
549  status = STATUS_INVALID_DEVICE_STATE;
550  }
551  else if (!DataExpected)
552  {
553  LogError(DRIVER_DEFAULT, "Adapter already has data on init!?!?!");
554  status = STATUS_INVALID_STATE_TRANSITION;
555  }
556  }
557  if (ss.Errors)
558  {
559  LogWarning(DRIVER_DEFAULT, "Unexpected Error %x", ss.Errors);
560  status = STATUS_UNSUCCESSFUL;
561  }
562  }
563 
564  LogFuncExitNT(DRIVER_DEFAULT, status);
565 
566  return status;
567 }

640  )

661 PSERIAL_SEND_ITEM
662 SerialPopSend(
663  _In_ POTTMP_ADAPTER_CONTEXT AdapterContext
664  )
665 {
666  PLIST_ENTRY current = NULL;
667 
668  // Grab the head of the list
669  // Careful, this might have gotten aborted, leaving the list empty
670  WdfSpinLockAcquire(AdapterContext->SendLock);
671 
672  if (!IsListEmpty(&AdapterContext->SendQueue))
673  {
674  current = RemoveHeadList(&AdapterContext->SendQueue);
675  }
676  if (current == NULL)
677  {
678  // Do this under the lock, but the state is consumed outside the lock
679  AdapterContext->SendQueueRunning = false;
680  LogVerbose(DRIVER_DEFAULT, "Send Work Item Complete");
681  }
682 
683  WdfSpinLockRelease(AdapterContext->SendLock);
684 
685  return current ? CONTAINING_RECORD(current, SERIAL_SEND_ITEM, Link) : NULL;
686 }
