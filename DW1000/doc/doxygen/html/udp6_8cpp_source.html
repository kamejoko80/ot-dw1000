84 ThreadError UdpSocket::SendTo(Message &aMessage, const MessageInfo &aMessageInfo)
85 {
86  ThreadError error = kThreadError_None;
87  MessageInfo messageInfoLocal;
88  UdpHeader udpHeader;
89 
90  messageInfoLocal = aMessageInfo;
91 
92  if (messageInfoLocal.GetSockAddr().IsUnspecified())
93  {
94  messageInfoLocal.SetSockAddr(GetSockName().GetAddress());
95  }
96 
97  if (GetSockName().mPort == 0)
98  {
99  GetSockName().mPort = static_cast<Udp *>(mTransport)->GetEphemeralPort();
100  }
101 
102  udpHeader.SetSourcePort(GetSockName().mPort);
103  udpHeader.SetDestinationPort(messageInfoLocal.mPeerPort);
104  udpHeader.SetLength(sizeof(udpHeader) + aMessage.GetLength());
105  udpHeader.SetChecksum(0);
106 
107  SuccessOrExit(error = aMessage.Prepend(&udpHeader, sizeof(udpHeader)));
108  aMessage.SetOffset(0);
109  SuccessOrExit(error = static_cast<Udp *>(mTransport)->SendDatagram(aMessage, messageInfoLocal, kProtoUdp));
110 
111 exit:
112  return error;

188 ThreadError Udp::HandleMessage(Message &aMessage, MessageInfo &aMessageInfo)
189 {
190  ThreadError error = kThreadError_None;
191  UdpHeader udpHeader;
192  uint16_t payloadLength;
193  uint16_t checksum;
194 
195  payloadLength = aMessage.GetLength() - aMessage.GetOffset();
196 
197  // check length
198  VerifyOrExit(payloadLength >= sizeof(UdpHeader), error = kThreadError_Parse);
199 
200  // verify checksum
201  checksum = Ip6::ComputePseudoheaderChecksum(aMessageInfo.GetPeerAddr(), aMessageInfo.GetSockAddr(),
202  payloadLength, kProtoUdp);
203  checksum = aMessage.UpdateChecksum(checksum, aMessage.GetOffset(), payloadLength);
204  VerifyOrExit(checksum == 0xffff, ;);
205 
206  VerifyOrExit(aMessage.Read(aMessage.GetOffset(), sizeof(udpHeader), &udpHeader) == sizeof(udpHeader),);
207  aMessage.MoveOffset(sizeof(udpHeader));
208  aMessageInfo.mPeerPort = udpHeader.GetSourcePort();
209  aMessageInfo.mSockPort = udpHeader.GetDestinationPort();
210 
211  // find socket
212  for (UdpSocket *socket = mSockets; socket; socket = socket->GetNext())
213  {
214  if (socket->GetSockName().mPort != udpHeader.GetDestinationPort())
215  {
216  continue;
217  }
218 
219  if (socket->GetSockName().mScopeId != 0 &&
220  socket->GetSockName().mScopeId != aMessageInfo.mInterfaceId)
221  {
222  continue;
223  }
224 
225  if (!aMessageInfo.GetSockAddr().IsMulticast() &&
226  !socket->GetSockName().GetAddress().IsUnspecified() &&
227  socket->GetSockName().GetAddress() != aMessageInfo.GetSockAddr())
228  {
229  continue;
230  }
231 
232  // verify source if connected socket
233  if (socket->GetPeerName().mPort != 0)
234  {
235  if (socket->GetPeerName().mPort != udpHeader.GetSourcePort())
236  {
237  continue;
238  }
239 
240  if (!socket->GetPeerName().GetAddress().IsUnspecified() &&
241  socket->GetPeerName().GetAddress() != aMessageInfo.GetPeerAddr())
242  {
243  continue;
244  }
245  }
246 
247  socket->HandleUdpReceive(aMessage, aMessageInfo);
248  }
249 
250 exit:
251  return error;
252 }

254 ThreadError Udp::UpdateChecksum(Message &aMessage, uint16_t aChecksum)
255 {
256  aChecksum = aMessage.UpdateChecksum(aChecksum, aMessage.GetOffset(), aMessage.GetLength() - aMessage.GetOffset());
257 
258  if (aChecksum != 0xffff)
259  {
260  aChecksum = ~aChecksum;
261  }
262 
263  aChecksum = HostSwap16(aChecksum);
264  aMessage.Write(aMessage.GetOffset() + UdpHeader::GetChecksumOffset(), sizeof(aChecksum), &aChecksum);
265  return kThreadError_None;
266 }
