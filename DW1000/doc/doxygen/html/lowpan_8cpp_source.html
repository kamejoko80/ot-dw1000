53 ThreadError Lowpan::CopyContext(const Context &aContext, Ip6::Address &aAddress)
54 {
55  memcpy(&aAddress, aContext.mPrefix, aContext.mPrefixLength / CHAR_BIT);
56 
57  for (int i = (aContext.mPrefixLength & ~7); i < aContext.mPrefixLength; i++)
58  {
59  aAddress.mFields.m8[i / CHAR_BIT] &= ~(0x80 >> (i % CHAR_BIT));
60  aAddress.mFields.m8[i / CHAR_BIT] |= aContext.mPrefix[i / CHAR_BIT] & (0x80 >> (i % CHAR_BIT));
61  }
62 
63  return kThreadError_None;

66 ThreadError Lowpan::ComputeIid(const Mac::Address &aMacAddr, const Context &aContext, Ip6::Address &aIpAddress)
67 {
68  ThreadError error = kThreadError_None;
69 
70  switch (aMacAddr.mLength)
71  {
72  case 2:
73  aIpAddress.mFields.m16[4] = HostSwap16(0x0000);
74  aIpAddress.mFields.m16[5] = HostSwap16(0x00ff);
75  aIpAddress.mFields.m16[6] = HostSwap16(0xfe00);
76  aIpAddress.mFields.m16[7] = HostSwap16(aMacAddr.mShortAddress);
77  break;
78 
79  case Ip6::Address::kInterfaceIdentifierSize:
80  aIpAddress.SetIid(aMacAddr.mExtAddress);
81  break;
82 
83  default:
84  ExitNow(error = kThreadError_Parse);
85  }
86 
87  if (aContext.mPrefixLength > 64)
88  {
89  for (int i = (aContext.mPrefixLength & ~7); i < aContext.mPrefixLength; i++)
90  {
91  aIpAddress.mFields.m8[i / CHAR_BIT] &= ~(0x80 >> (i % CHAR_BIT));
92  aIpAddress.mFields.m8[i / CHAR_BIT] |= aContext.mPrefix[i / CHAR_BIT] & (0x80 >> (i % CHAR_BIT));
93  }
94  }
95 
96 exit:
97  return error;
98 }

100 int Lowpan::CompressSourceIid(const Mac::Address &aMacAddr, const Ip6::Address &aIpAddr, const Context &aContext,
101  uint16_t &aHcCtl, uint8_t *aBuf)
102 {
103  uint8_t *cur = aBuf;
104  Ip6::Address ipaddr;
105  Mac::Address tmp;
106 
107  ComputeIid(aMacAddr, aContext, ipaddr);
108 
109  if (memcmp(ipaddr.GetIid(), aIpAddr.GetIid(), Ip6::Address::kInterfaceIdentifierSize) == 0)
110  {
111  aHcCtl |= kHcSrcAddrMode3;
112  }
113  else
114  {
115  tmp.mLength = sizeof(tmp.mShortAddress);
116  tmp.mShortAddress = HostSwap16(aIpAddr.mFields.m16[7]);
117  ComputeIid(tmp, aContext, ipaddr);
118 
119  if (memcmp(ipaddr.GetIid(), aIpAddr.GetIid(), Ip6::Address::kInterfaceIdentifierSize) == 0)
120  {
121  aHcCtl |= kHcSrcAddrMode2;
122  cur[0] = aIpAddr.mFields.m8[14];
123  cur[1] = aIpAddr.mFields.m8[15];
124  cur += 2;
125  }
126  else
127  {
128  aHcCtl |= kHcSrcAddrMode1;
129  memcpy(cur, aIpAddr.GetIid(), Ip6::Address::kInterfaceIdentifierSize);
130  cur += Ip6::Address::kInterfaceIdentifierSize;
131  }
132  }
133 
134  return static_cast<int>(cur - aBuf);

174 int Lowpan::CompressMulticast(const Ip6::Address &aIpAddr, uint16_t &aHcCtl, uint8_t *aBuf)
175 {
176  uint8_t *cur = aBuf;
177  Context multicastContext;
178 
179  aHcCtl |= kHcMulticast;
180 
181  for (unsigned int i = 2; i < sizeof(Ip6::Address); i++)
182  {
183  if (aIpAddr.mFields.m8[i])
184  {
185  // Check if multicast address can be compressed to 8-bits (ff02::00xx)
186  if (aIpAddr.mFields.m8[1] == 0x02 && i >= 15)
187  {
188  aHcCtl |= kHcDstAddrMode3;
189  cur[0] = aIpAddr.mFields.m8[15];
190  cur++;
191  }
192  // Check if multicast address can be compressed to 32-bits (ffxx::00xx:xxxx)
193  else if (i >= 13)
194  {
195  aHcCtl |= kHcDstAddrMode2;
196  cur[0] = aIpAddr.mFields.m8[1];
197  memcpy(cur + 1, aIpAddr.mFields.m8 + 13, 3);
198  cur += 4;
199  }
200  // Check if multicast address can be compressed to 48-bits (ffxx::00xx:xxxx:xxxx)
201  else if (i >= 11)
202  {
203  aHcCtl |= kHcDstAddrMode1;
204  cur[0] = aIpAddr.mFields.m8[1];
205  memcpy(cur + 1, aIpAddr.mFields.m8 + 11, 5);
206  cur += 6;
207  }
208  else
209  {
210  // Check if multicast address can be compressed using Context ID 0.
211  if (mNetworkData.GetContext(0, multicastContext) == kThreadError_None &&
212  multicastContext.mPrefixLength == aIpAddr.mFields.m8[3] &&
213  memcmp(multicastContext.mPrefix, aIpAddr.mFields.m8 + 4, 8) == 0)
214  {
215  aHcCtl |= kHcDstAddrContext | kHcDstAddrMode0;
216  memcpy(cur, aIpAddr.mFields.m8 + 1, 2);
217  memcpy(cur + 2, aIpAddr.mFields.m8 + 12, 4);
218  cur += 6;
219  }
220  else
221  {
222  memcpy(cur, aIpAddr.mFields.m8, sizeof(Ip6::Address));
223  cur += sizeof(Ip6::Address);
224  }
225  }
226 
227  break;
228  }
229  }
230 
231  return static_cast<int>(cur - aBuf);
232 }

234 int Lowpan::Compress(Message &aMessage, const Mac::Address &aMacSource, const Mac::Address &aMacDest, uint8_t *aBuf)
235 {
236  uint8_t *cur = aBuf;
237  uint16_t hcCtl = 0;
238  Ip6::Header ip6Header;
239  uint8_t *ip6HeaderBytes = reinterpret_cast<uint8_t *>(&ip6Header);
240  Context srcContext, dstContext;
241  bool srcContextValid = true, dstContextValid = true;
242  uint8_t nextHeader;
243  uint8_t ecn = 0;
244  uint8_t dscp = 0;
245 
246  aMessage.Read(aMessage.GetOffset(), sizeof(ip6Header), &ip6Header);
247 
248  if (mNetworkData.GetContext(ip6Header.GetSource(), srcContext) != kThreadError_None ||
249  srcContext.mCompressFlag == false)
250  {
251  mNetworkData.GetContext(0, srcContext);
252  srcContextValid = false;
253  }
254 
255  if (mNetworkData.GetContext(ip6Header.GetDestination(), dstContext) != kThreadError_None ||
256  dstContext.mCompressFlag == false)
257  {
258  mNetworkData.GetContext(0, dstContext);
259  dstContextValid = false;
260  }
261 
262  hcCtl = kHcDispatch;
263 
264  // Lowpan HC Control Bits
265  cur += 2;
266 
267  // Context Identifier
268  if (srcContext.mContextId != 0 || dstContext.mContextId != 0)
269  {
270  hcCtl |= kHcContextId;
271  cur[0] = ((srcContext.mContextId << 4) | dstContext.mContextId) & 0xff;
272  cur++;
273  }
274 
275  dscp = ((ip6HeaderBytes[0] << 2) & 0x3c) | (ip6HeaderBytes[1] >> 6);
276  ecn = (ip6HeaderBytes[1] << 2) & 0xc0;
277 
278  // Flow Label
279  if (((ip6HeaderBytes[1] & 0x0f) == 0) && ((ip6HeaderBytes[2]) == 0) && ((ip6HeaderBytes[3]) == 0))
280  {
281  if (dscp == 0 && ecn == 0)
282  {
283  // Elide Flow Label and Traffic Class.
284  hcCtl |= kHcTrafficClass | kHcFlowLabel;
285  }
286  else
287  {
288  // Elide Flow Label and carry Traffic Class in-line.
289  hcCtl |= kHcFlowLabel;
290 
291  cur[0] = ecn | dscp;
292  cur++;
293  }
294  }
295  else if (dscp == 0)
296  {
297  // Carry Flow Label and ECN only with 2-bit padding.
298  hcCtl |= kHcTrafficClass;
299 
300  cur[0] = ecn | (ip6HeaderBytes[1] & 0x0f);
301  memcpy(cur + 1, ip6HeaderBytes + 2, 2);
302  cur += 3;
303  }
304  else
305  {
306  // Carry Flow Label and Traffic Class in-line.
307  cur[0] = ecn | dscp;
308  cur[1] = ip6HeaderBytes[1] & 0x0f;
309  memcpy(cur + 2, ip6HeaderBytes + 2, 2);
310  cur += 4;
311  }
312 
313  // Next Header
314  switch (ip6Header.GetNextHeader())
315  {
316  case Ip6::kProtoHopOpts:
317  case Ip6::kProtoUdp:
318  case Ip6::kProtoIp6:
319  hcCtl |= kHcNextHeader;
320  break;
321 
322  default:
323  cur[0] = static_cast<uint8_t>(ip6Header.GetNextHeader());
324  cur++;
325  break;
326  }
327 
328  // Hop Limit
329  switch (ip6Header.GetHopLimit())
330  {
331  case 1:
332  hcCtl |= kHcHopLimit1;
333  break;
334 
335  case 64:
336  hcCtl |= kHcHopLimit64;
337  break;
338 
339  case 255:
340  hcCtl |= kHcHopLimit255;
341  break;
342 
343  default:
344  cur[0] = ip6Header.GetHopLimit();
345  cur++;
346  break;
347  }
348 
349  // Source Address
350  if (ip6Header.GetSource().IsUnspecified())
351  {
352  hcCtl |= kHcSrcAddrContext;
353  }
354  else if (ip6Header.GetSource().IsLinkLocal())
355  {
356  cur += CompressSourceIid(aMacSource, ip6Header.GetSource(), srcContext, hcCtl, cur);
357  }
358  else if (srcContextValid)
359  {
360  hcCtl |= kHcSrcAddrContext;
361  cur += CompressSourceIid(aMacSource, ip6Header.GetSource(), srcContext, hcCtl, cur);
362  }
363  else
364  {
365  memcpy(cur, ip6Header.GetSource().mFields.m8, sizeof(ip6Header.GetSource()));
366  cur += sizeof(Ip6::Address);
367  }
368 
369  // Destination Address
370  if (ip6Header.GetDestination().IsMulticast())
371  {
372  cur += CompressMulticast(ip6Header.GetDestination(), hcCtl, cur);
373  }
374  else if (ip6Header.GetDestination().IsLinkLocal())
375  {
376  cur += CompressDestinationIid(aMacDest, ip6Header.GetDestination(), dstContext, hcCtl, cur);
377  }
378  else if (dstContextValid)
379  {
380  hcCtl |= kHcDstAddrContext;
381  cur += CompressDestinationIid(aMacDest, ip6Header.GetDestination(), dstContext, hcCtl, cur);
382  }
383  else
384  {
385  memcpy(cur, &ip6Header.GetDestination(), sizeof(ip6Header.GetDestination()));
386  cur += sizeof(Ip6::Address);
387  }
388 
389  aBuf[0] = hcCtl >> 8;
390  aBuf[1] = hcCtl & 0xff;
391  aMessage.MoveOffset(sizeof(ip6Header));
392 
393  nextHeader = static_cast<uint8_t>(ip6Header.GetNextHeader());
394 
395  while (1)
396  {
397  switch (nextHeader)
398  {
399  case Ip6::kProtoHopOpts:
400  cur += CompressExtensionHeader(aMessage, cur, nextHeader);
401  break;
402 
403  case Ip6::kProtoUdp:
404  cur += CompressUdp(aMessage, cur);
405  ExitNow();
406 
407  case Ip6::kProtoIp6:
408  // For IP-in-IP the NH bit of the LOWPAN_NHC encoding MUST be set to zero.
409  cur[0] = kExtHdrDispatch | kExtHdrEidIp6;
410  cur++;
411 
412  cur += Compress(aMessage, aMacSource, aMacDest, cur);
413 
414  // fall through
415 
416  default:
417  ExitNow();
418  }
419  }
420 
421 exit:
422  return static_cast<int>(cur - aBuf);
423 }

425 int Lowpan::CompressExtensionHeader(Message &aMessage, uint8_t *aBuf, uint8_t &aNextHeader)
426 {
427  Ip6::ExtensionHeader extHeader;
428  Ip6::OptionHeader optionHeader;
429  uint8_t *cur = aBuf;
430  uint8_t len;
431  uint8_t padLength = 0;
432  uint16_t offset;
433 
434  aMessage.Read(aMessage.GetOffset(), sizeof(extHeader), &extHeader);
435  aMessage.MoveOffset(sizeof(extHeader));
436 
437  cur[0] = kExtHdrDispatch | kExtHdrEidHbh;
438 
439  switch (extHeader.GetNextHeader())
440  {
441  case Ip6::kProtoUdp:
442  case Ip6::kProtoIp6:
443  cur[0] |= kExtHdrNextHeader;
444  break;
445 
446  default:
447  cur++;
448  cur[0] = static_cast<uint8_t>(extHeader.GetNextHeader());
449  break;
450  }
451 
452  cur++;
453 
454  len = (extHeader.GetLength() + 1) * 8 - sizeof(extHeader);
455 
456  // RFC 6282 says: "IPv6 Hop-by-Hop and Destination Options Headers may use a trailing
457  // Pad1 or PadN to achieve 8-octet alignment. When there is a single trailing Pad1 or PadN
458  // option of 7 octets or less and the containing header is a multiple of 8 octets, the trailing
459  // Pad1 or PadN option MAY be elided by the compressor."
460  if (aNextHeader == Ip6::kProtoHopOpts || aNextHeader == Ip6::kProtoDstOpts)
461  {
462  offset = aMessage.GetOffset();
463 
464  while (offset < len + aMessage.GetOffset())
465  {
466  aMessage.Read(offset, sizeof(optionHeader), &optionHeader);
467 
468  if (optionHeader.GetType() == Ip6::OptionPad1::kType)
469  {
470  offset += sizeof(Ip6::OptionPad1);
471  }
472  else
473  {
474  offset += sizeof(optionHeader) + optionHeader.GetLength();
475  }
476  }
477 
478  // Check if the last option can be compressed.
479  if (optionHeader.GetType() == Ip6::OptionPad1::kType)
480  {
481  padLength = sizeof(Ip6::OptionPad1);
482  }
483  else if (optionHeader.GetType() == Ip6::OptionPadN::kType)
484  {
485  padLength = sizeof(optionHeader) + optionHeader.GetLength();
486  }
487 
488  len -= padLength;
489  }
490 
491  aNextHeader = static_cast<uint8_t>(extHeader.GetNextHeader());
492 
493  cur[0] = len;
494  cur++;
495 
496  aMessage.Read(aMessage.GetOffset(), len, cur);
497  aMessage.MoveOffset(len + padLength);
498  cur += len;
499 
500  return static_cast<int>(cur - aBuf);
501 }

503 int Lowpan::CompressUdp(Message &aMessage, uint8_t *aBuf)
504 {
505  Ip6::UdpHeader udpHeader;
506  uint8_t *cur = aBuf;
507  uint8_t *udpCtl = cur;
508  uint16_t source;
509  uint16_t destination;
510 
511  aMessage.Read(aMessage.GetOffset(), sizeof(udpHeader), &udpHeader);
512  source = udpHeader.GetSourcePort();
513  destination = udpHeader.GetDestinationPort();
514 
515  cur[0] = kUdpDispatch;
516  cur++;
517 
518  if ((source & 0xfff0) == 0xf0b0 && (destination & 0xfff0) == 0xf0b0)
519  {
520  *udpCtl |= 3;
521  *cur++ = (((source & 0xf) << 4) | (destination & 0xf)) & 0xff;
522  }
523  else if ((source & 0xff00) == 0xf000)
524  {
525  *udpCtl |= 2;
526  *cur++ = source & 0xff;
527  *cur++ = destination >> 8;
528  *cur++ = destination & 0xff;
529  }
530  else if ((destination & 0xff00) == 0xf000)
531  {
532  *udpCtl |= 1;
533  *cur++ = source >> 8;
534  *cur++ = source & 0xff;
535  *cur++ = destination & 0xff;
536  }
537  else
538  {
539  memcpy(cur, &udpHeader, Ip6::UdpHeader::GetLengthOffset());
540  cur += Ip6::UdpHeader::GetLengthOffset();
541  }
542 
543  memcpy(cur, reinterpret_cast<uint8_t *>(&udpHeader) + Ip6::UdpHeader::GetChecksumOffset(), 2);
544  cur += 2;
545 
546  aMessage.MoveOffset(sizeof(udpHeader));
547 
548  return static_cast<int>(cur - aBuf);
549 }

551 ThreadError Lowpan::DispatchToNextHeader(uint8_t aDispatch, Ip6::IpProto &aNextHeader)
552 {
553  ThreadError error = kThreadError_None;
554 
555  if ((aDispatch & kExtHdrDispatchMask) == kExtHdrDispatch)
556  {
557  switch (aDispatch & kExtHdrEidMask)
558  {
559  case kExtHdrEidHbh:
560  aNextHeader = Ip6::kProtoHopOpts;
561  ExitNow();
562 
563  case kExtHdrEidRouting:
564  aNextHeader = Ip6::kProtoRouting;
565  ExitNow();
566 
567  case kExtHdrEidFragment:
568  aNextHeader = Ip6::kProtoFragment;
569  ExitNow();
570 
571  case kExtHdrEidDst:
572  aNextHeader = Ip6::kProtoDstOpts;
573  ExitNow();
574 
575  case kExtHdrEidIp6:
576  aNextHeader = Ip6::kProtoIp6;
577  ExitNow();
578  }
579  }
580  else if ((aDispatch & kUdpDispatchMask) == kUdpDispatch)
581  {
582  aNextHeader = Ip6::kProtoUdp;
583  ExitNow();
584  }
585 
586  error = kThreadError_Parse;
587 
588 exit:
589  return error;
590 }

592 int Lowpan::DecompressBaseHeader(Ip6::Header &ip6Header, const Mac::Address &aMacSource, const Mac::Address &aMacDest,
593  const uint8_t *aBuf, uint16_t aBufLength)
594 {
595  ThreadError error = kThreadError_Parse;
596  const uint8_t *cur = aBuf;
597  uint16_t remaining = aBufLength;
598  uint16_t hcCtl;
599  Context srcContext, dstContext;
600  bool srcContextValid = true, dstContextValid = true;
601  Ip6::IpProto nextHeader;
602  uint8_t *bytes;
603 
604  VerifyOrExit(remaining >= 2,);
605  hcCtl = static_cast<uint16_t>((cur[0] << 8) | cur[1]);
606  cur += 2;
607  remaining -= 2;
608 
609  // check Dispatch bits
610  VerifyOrExit((hcCtl & kHcDispatchMask) == kHcDispatch,);
611 
612  // Context Identifier
613  srcContext.mPrefixLength = 0;
614  dstContext.mPrefixLength = 0;
615 
616  if ((hcCtl & kHcContextId) != 0)
617  {
618  VerifyOrExit(remaining >= 1,);
619 
620  if (mNetworkData.GetContext(cur[0] >> 4, srcContext) != kThreadError_None)
621  {
622  srcContextValid = false;
623  }
624 
625  if (mNetworkData.GetContext(cur[0] & 0xf, dstContext) != kThreadError_None)
626  {
627  dstContextValid = false;
628  }
629 
630  cur++;
631  remaining--;
632  }
633  else
634  {
635  mNetworkData.GetContext(0, srcContext);
636  mNetworkData.GetContext(0, dstContext);
637  }
638 
639  memset(&ip6Header, 0, sizeof(ip6Header));
640  ip6Header.Init();
641 
642  // Traffic Class and Flow Label
643  if ((hcCtl & kHcTrafficFlowMask) != kHcTrafficFlow)
644  {
645  VerifyOrExit(remaining >= 1,);
646 
647  bytes = reinterpret_cast<uint8_t *>(&ip6Header);
648  bytes[1] |= (cur[0] & 0xc0) >> 2;
649 
650  if ((hcCtl & kHcTrafficClass) == 0)
651  {
652  bytes[0] |= (cur[0] >> 2) & 0x0f;
653  bytes[1] |= (cur[0] << 6) & 0xc0;
654  cur++;
655  remaining--;
656  }
657 
658  if ((hcCtl & kHcFlowLabel) == 0)
659  {
660  VerifyOrExit(remaining >= 3,);
661  bytes[1] |= cur[0] & 0x0f;
662  bytes[2] |= cur[1];
663  bytes[3] |= cur[2];
664  cur += 3;
665  remaining -= 3;
666  }
667  }
668 
669  // Next Header
670  if ((hcCtl & kHcNextHeader) == 0)
671  {
672  VerifyOrExit(remaining >= 1,);
673  ip6Header.SetNextHeader(static_cast<Ip6::IpProto>(cur[0]));
674  cur++;
675  remaining--;
676  }
677 
678  // Hop Limit
679  switch (hcCtl & kHcHopLimitMask)
680  {
681  case kHcHopLimit1:

874 int Lowpan::DecompressExtensionHeader(Message &aMessage, const uint8_t *aBuf, uint16_t aBufLength)
875 {
876  ThreadError error = kThreadError_Parse;
877  const uint8_t *cur = aBuf;
878  uint16_t remaining = aBufLength;
879  uint8_t hdr[2];
880  uint8_t len;
881  Ip6::IpProto nextHeader;
882  uint8_t ctl = cur[0];
883  uint8_t padLength;
884  Ip6::OptionPad1 optionPad1;
885  Ip6::OptionPadN optionPadN;
886 
887  VerifyOrExit(remaining >= 1,);
888  cur++;
889  remaining--;
890 
891  // next header
892  if (ctl & kExtHdrNextHeader)
893  {
894  VerifyOrExit(remaining >= 1,);
895 
896  len = cur[0];
897  cur++;
898  remaining--;
899 
900  VerifyOrExit(remaining >= len,);
901  SuccessOrExit(DispatchToNextHeader(cur[len], nextHeader));
902  hdr[0] = static_cast<uint8_t>(nextHeader);
903  }
904  else
905  {
906  VerifyOrExit(remaining >= 2,);
907 
908  hdr[0] = cur[0];
909  len = cur[1];
910  cur += 2;
911  remaining -= 2;
912 
913  VerifyOrExit(remaining >= len,);
914  }
915 
916  // length
917  hdr[1] = BitVectorBytes(sizeof(hdr) + len) - 1;
918 
919  SuccessOrExit(aMessage.Append(hdr, sizeof(hdr)));
920  aMessage.MoveOffset(sizeof(hdr));
921 
922  // payload
923  SuccessOrExit(aMessage.Append(cur, len));
924  aMessage.MoveOffset(len);
925  cur += len;
926 
927  // The RFC6282 says: "The trailing Pad1 or PadN option MAY be elided by the compressor.
928  // A decompressor MUST ensure that the containing header is padded out to a multiple of 8 octets
929  // in length, using a Pad1 or PadN option if necessary."
930  padLength = 8 - ((len + sizeof(hdr)) & 0x07);
931 
932  if (padLength != 8)
933  {
934  if (padLength == 1)
935  {
936  optionPad1.Init();
937  SuccessOrExit(aMessage.Append(&optionPad1, padLength));
938  }
939  else
940  {
941  optionPadN.Init(padLength);
942  SuccessOrExit(aMessage.Append(&optionPadN, padLength));
943  }
944 
945  aMessage.MoveOffset(padLength);
946  }
947 
948  error = kThreadError_None;
949 
950 exit:
951  return (error == kThreadError_None) ? static_cast<int>(cur - aBuf) : -1;
952 }

954 int Lowpan::DecompressUdpHeader(Message &aMessage, const uint8_t *aBuf, uint16_t aBufLength, uint16_t aDatagramLength)
955 {
956  ThreadError error = kThreadError_Parse;
957  const uint8_t *cur = aBuf;
958  uint16_t remaining = aBufLength;
959  Ip6::UdpHeader udpHeader;
960  uint8_t udpCtl;
961 
962  VerifyOrExit(remaining >= 1,);
963  udpCtl = cur[0];
964  cur++;
965  remaining--;
966 
967  memset(&udpHeader, 0, sizeof(udpHeader));
968 
969  // source and dest ports
970  switch (udpCtl & kUdpPortMask)
971  {
972  case 0:
973  VerifyOrExit(remaining >= 4,);
974  udpHeader.SetSourcePort(static_cast<uint16_t>((cur[0] << 8) | cur[1]));
975  udpHeader.SetDestinationPort(static_cast<uint16_t>((cur[2] << 8) | cur[3]));
976  cur += 4;
977  remaining -= 4;
978  break;
979 
980  case 1:
981  VerifyOrExit(remaining >= 3,);
982  udpHeader.SetSourcePort(static_cast<uint16_t>((cur[0] << 8) | cur[1]));
983  udpHeader.SetDestinationPort(0xf000 | cur[2]);
984  cur += 3;
985  remaining -= 3;
986  break;
987 
988  case 2:
989  VerifyOrExit(remaining >= 3,);
990  udpHeader.SetSourcePort(0xf000 | cur[0]);
991  udpHeader.SetDestinationPort(static_cast<uint16_t>((cur[1] << 8) | cur[2]));
992  cur += 3;
993  remaining -= 3;
994  break;
995 
996  case 3:
997  VerifyOrExit(remaining >= 1,);
998  udpHeader.SetSourcePort(0xf0b0 | (cur[0] >> 4));
999  udpHeader.SetDestinationPort(0xf0b0 | (cur[0] & 0xf));
1000  cur += 1;
1001  remaining -= 1;
1002  break;
1003  }
1004 
1005  // checksum
1006  if ((udpCtl & kUdpChecksum) != 0)
1007  {
1008  ExitNow();
1009  }
1010  else
1011  {
1012  VerifyOrExit(remaining >= 2,);
1013  udpHeader.SetChecksum(static_cast<uint16_t>((cur[0] << 8) | cur[1]));
1014  cur += 2;
1015  }
1016 
1017  // length
1018  if (aDatagramLength == 0)
1019  {
1020  udpHeader.SetLength(sizeof(udpHeader) + static_cast<uint16_t>(aBufLength - (cur - aBuf)));
1021  }
1022  else
1023  {
1024  udpHeader.SetLength(aDatagramLength - aMessage.GetOffset());
1025  }
1026 
1027  SuccessOrExit(aMessage.Append(&udpHeader, sizeof(udpHeader)));
1028  aMessage.MoveOffset(sizeof(udpHeader));
1029 
1030  error = kThreadError_None;
1031 
1032 exit:
1033  return (error == kThreadError_None) ? static_cast<int>(cur - aBuf) : -1;
1034 }

1036 int Lowpan::Decompress(Message &aMessage, const Mac::Address &aMacSource, const Mac::Address &aMacDest,
1037  const uint8_t *aBuf, uint16_t aBufLength, uint16_t aDatagramLength)
1038 {

1042  uint16_t remaining = aBufLength;
1043  bool compressed;

1045  uint16_t ip6PayloadLength;
1046  uint16_t compressedLength = 0;
1047  uint16_t currentOffset = aMessage.GetOffset();
1048 
1049  VerifyOrExit(remaining >= 2,);
1050  compressed = (((static_cast<uint16_t>(cur[0]) << 8) | cur[1]) & kHcNextHeader) != 0;
1051 
1052  VerifyOrExit((rval = DecompressBaseHeader(ip6Header, aMacSource, aMacDest, cur, remaining)) >= 0,);
1053 
1054  cur += rval;
1055  remaining -= rval;
1056 
1057  SuccessOrExit(aMessage.Append(&ip6Header, sizeof(ip6Header)));
1058  SuccessOrExit(aMessage.MoveOffset(sizeof(ip6Header)));
1059 
1060  while (compressed)
1061  {
1062  VerifyOrExit(remaining >= 1,);
1063 
1064  if ((cur[0] & kExtHdrDispatchMask) == kExtHdrDispatch)
1065  {
1066  if ((cur[0] & kExtHdrEidMask) == kExtHdrEidIp6)
1067  {
1068  compressed = false;
1069 
1070  cur++;
1071  remaining--;
1072 
1073  VerifyOrExit((rval = Decompress(aMessage, aMacSource, aMacDest, cur, remaining,
1074  (aDatagramLength ? aDatagramLength - aMessage.GetLength() : 0))) >= 0,);
1075  }
1076  else
1077  {
1078  compressed = (cur[0] & kExtHdrNextHeader) != 0;
1079  VerifyOrExit((rval = DecompressExtensionHeader(aMessage, cur, remaining)) >= 0,);
1080  }
1081  }
1082  else if ((cur[0] & kUdpDispatchMask) == kUdpDispatch)
1083  {
1084  compressed = false;
1085  VerifyOrExit((rval = DecompressUdpHeader(aMessage, cur, remaining, aDatagramLength)) >= 0,);
1086  }
1087  else
1088  {
1089  ExitNow();
1090  }
1091 
1092  VerifyOrExit(remaining >= rval,;);
1093  cur += rval;
1094  remaining -= rval;
1095  }
1096 
1097  compressedLength = static_cast<uint16_t>(cur - aBuf);
1098 
1099  if (aDatagramLength)
1100  {
1101  ip6PayloadLength = HostSwap16(aDatagramLength - sizeof(Ip6::Header));
1102  }
1103  else
1104  {
1105  ip6PayloadLength = HostSwap16(aMessage.GetOffset() - currentOffset -
1106  sizeof(Ip6::Header) + aBufLength - compressedLength);
1107  }
1108 
1109  aMessage.Write(currentOffset + Ip6::Header::GetPayloadLengthOffset(),
1110  sizeof(ip6PayloadLength), &ip6PayloadLength);
1111 
1112  error = kThreadError_None;
1113 
1114 exit:
1115  return (error == kThreadError_None) ? static_cast<int>(compressedLength) : -1;
1116 }
