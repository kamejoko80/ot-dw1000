373 void LinkRaw::InvokeTransmitDone(RadioPacket *aPacket, bool aFramePending, ThreadError aError)
374 {
375 #if OPENTHREAD_CONFIG_ENABLE_SOFTWARE_ACK_TIMEOUT
376  mTimer.Stop();
377 #endif
378 
379 #if OPENTHREAD_CONFIG_ENABLE_SOFTWARE_RETRANSMIT
380 
381  if (aError == kThreadError_ChannelAccessFailure)
382  {
383  if (mCsmaAttempts < Mac::kMaxCSMABackoffs)
384  {
385  mCsmaAttempts++;
386  StartCsmaBackoff();
387  goto exit;
388  }
389  }
390  else
391  {
392  mCsmaAttempts = 0;
393  }
394 
395  if (aError == kThreadError_NoAck)
396  {
397  if (mTransmitAttempts < Mac::kMaxFrameAttempts)
398  {
399  mTransmitAttempts++;
400  StartCsmaBackoff();
401  goto exit;
402  }
403  }
404 
405 #endif
406 
407  if (mTransmitDoneCallback)
408  {
409  mTransmitDoneCallback(&mInstance, aPacket, aFramePending, aError);
410  mTransmitDoneCallback = NULL;
411  }
412 
413 #if OPENTHREAD_CONFIG_ENABLE_SOFTWARE_RETRANSMIT
414 exit:
415  return;
416 #endif
417 }

461 void LinkRaw::HandleTimer(void)
462 {
463  TimerReason timerReason = mTimerReason;
464  mTimerReason = kTimerReasonNone;
465 
466  switch (timerReason)
467  {
468 #if OPENTHREAD_CONFIG_ENABLE_SOFTWARE_ACK_TIMEOUT
469 
470  case kTimerReasonAckTimeout:
471  {
472  // Transition back to receive state on previous channel
473  otPlatRadioReceive(&mInstance, mReceiveChannel);
474 
475  // Invoke completion callback for transmit
476  InvokeTransmitDone(otPlatRadioGetTransmitBuffer(&mInstance), false, kThreadError_NoAck);
477  }
478 
479 #endif // OPENTHREAD_CONFIG_ENABLE_SOFTWARE_ACK_TIMEOUT
480 
481 #if OPENTHREAD_CONFIG_ENABLE_SOFTWARE_RETRANSMIT
482 
483  case kTimerReasonRetransmitTimeout:
484  {
485  RadioPacket *aPacket = otPlatRadioGetTransmitBuffer(&mInstance);
486 
487  // Start the transmit now
488  ThreadError error = DoTransmit(aPacket);
489 
490  if (error != kThreadError_None)
491  {
492  InvokeTransmitDone(aPacket, false, error);
493  }
494  }
495 
496 #endif // OPENTHREAD_CONFIG_ENABLE_SOFTWARE_RETRANSMIT
497 
498 #if OPENTHREAD_CONFIG_ENABLE_SOFTWARE_ENERGY_SCAN
499 
500  case kTimerReasonEnergyScanComplete:
501  {
502  // Invoke completion callback for the energy scan
503  InvokeEnergyScanDone(mEnergyScanRssi);
504  }
505 
506 #endif // OPENTHREAD_CONFIG_ENABLE_SOFTWARE_ENERGY_SCAN
507 
508  default:
509  assert(false);
510  }
511 }

517 void LinkRaw::StartCsmaBackoff(void)
518 {
519  uint32_t backoffExponent = Mac::kMinBE + mTransmitAttempts + mCsmaAttempts;
520  uint32_t backoff;
521 
522  if (backoffExponent > Mac::kMaxBE)
523  {
524  backoffExponent = Mac::kMaxBE;
525  }
526 
527  backoff = Mac::kMinBackoff + (Mac::kUnitBackoffPeriod * kPhyUsPerSymbol * (1 << backoffExponent)) / 1000;
528  backoff = (otPlatRandomGet() % backoff);
529 
530  mTimerReason = kTimerReasonRetransmitTimeout;
531  mTimer.Start(backoff);
532 }

543 void LinkRaw::HandleEnergyScanTask(void)
544 {
545  // Only process task if we are still energy scanning
546  if (mTimerReason == kTimerReasonEnergyScanComplete)
547  {
548  int8_t rssi = otPlatRadioGetRssi(&mInstance);
549 
550  // Only apply the RSSI if it was a valid value
551  if (rssi != kInvalidRssiValue)
552  {
553  if ((mEnergyScanRssi == kInvalidRssiValue) || (rssi > mEnergyScanRssi))
554  {
555  mEnergyScanRssi = rssi;
556  }
557  }
558 
559  // Post another instance of tha task, since we are
560  // still doing the energy scan.
561  mEnergyScanTask.Post();
562  }
563 }
