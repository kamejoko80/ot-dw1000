71 ThreadError otPlatUartEnable(void)
72 {
73  ThreadError error = kThreadError_None;
74  struct termios termios;
75 
76 #ifdef OPENTHREAD_TARGET_LINUX
77  // Ensure we terminate this process if the
78  // parent process dies.
79  prctl(PR_SET_PDEATHSIG, SIGHUP);
80 #endif
81 
82  s_in_fd = dup(STDIN_FILENO);
83  s_out_fd = dup(STDOUT_FILENO);
84  dup2(STDERR_FILENO, STDOUT_FILENO);
85 
86  // We need this signal to make sure that this
87  // process terminates properly.
88  signal(SIGPIPE, SIG_DFL);
89 
90  if (isatty(s_in_fd))
91  {
92  tcgetattr(s_in_fd, &original_stdin_termios);
93  atexit(&restore_stdin_termios);
94  }
95 
96  if (isatty(s_out_fd))
97  {
98  tcgetattr(s_out_fd, &original_stdout_termios);
99  atexit(&restore_stdout_termios);
100  }
101 
102  if (isatty(s_in_fd))
103  {
104  // get current configuration
105  VerifyOrExit(tcgetattr(s_in_fd, &termios) == 0, perror("tcgetattr"); error = kThreadError_Error);
106 
107  // Set up the termios settings for raw mode. This turns
108  // off input/output processing, line processing, and character processing.
109  cfmakeraw(&termios);
110 
111  // Set up our cflags for local use. Turn on hangup-on-close.
112  termios.c_cflag |= HUPCL | CREAD | CLOCAL;
113 
114  // "Minimum number of characters for noncanonical read"
115  termios.c_cc[VMIN] = 1;
116 
117  // "Timeout in deciseconds for noncanonical read"
118  termios.c_cc[VTIME] = 0;
119 
120  // configure baud rate
121  VerifyOrExit(cfsetispeed(&termios, B115200) == 0, perror("cfsetispeed"); error = kThreadError_Error);
122 
123  // set configuration
124  VerifyOrExit(tcsetattr(s_in_fd, TCSANOW, &termios) == 0, perror("tcsetattr"); error = kThreadError_Error);
125  }
126 
127  if (isatty(s_out_fd))
128  {
129  // get current configuration
130  VerifyOrExit(tcgetattr(s_out_fd, &termios) == 0, perror("tcgetattr"); error = kThreadError_Error);
131 
132  // Set up the termios settings for raw mode. This turns
133  // off input/output processing, line processing, and character processing.
134  cfmakeraw(&termios);
135 
136  // Absolutely obliterate all output processing.
137  termios.c_oflag = 0;
138 
139  // Set up our cflags for local use. Turn on hangup-on-close.
140  termios.c_cflag |= HUPCL | CREAD | CLOCAL;
141 
142  // configure baud rate
143  VerifyOrExit(cfsetospeed(&termios, B115200) == 0, perror("cfsetospeed"); error = kThreadError_Error);
144 
145  // set configuration
146  VerifyOrExit(tcsetattr(s_out_fd, TCSANOW, &termios) == 0, perror("tcsetattr"); error = kThreadError_Error);
147  }
148 
149  return error;
150 
151 exit:
152  close(s_in_fd);
153  close(s_out_fd);
154  return error;
155 }

213 void platformUartProcess(void)
214 {
215  ssize_t rval;
216  const int error_flags = POLLERR | POLLNVAL | POLLHUP;
217  struct pollfd pollfd[] =
218  {
219  { s_in_fd, POLLIN | error_flags, 0 },
220  { s_out_fd, POLLOUT | error_flags, 0 },
221  };
222 
223  errno = 0;
224 
225  rval = poll(pollfd, sizeof(pollfd) / sizeof(*pollfd), 0);
226 
227  if (rval < 0)
228  {
229  perror("poll");
230  exit(EXIT_FAILURE);
231  }
232 
233  if (rval > 0)
234  {
235  if ((pollfd[0].revents & error_flags) != 0)
236  {
237  perror("s_in_fd");
238  exit(EXIT_FAILURE);
239  }
240 
241  if ((pollfd[1].revents & error_flags) != 0)
242  {
243  perror("s_out_fd");
244  exit(EXIT_FAILURE);
245  }
246 
247  if (pollfd[0].revents & POLLIN)
248  {
249  rval = read(s_in_fd, s_receive_buffer, sizeof(s_receive_buffer));
250 
251  if (rval <= 0)
252  {
253  perror("read");
254  exit(EXIT_FAILURE);
255  }
256 
257  otPlatUartReceived(s_receive_buffer, (uint16_t)rval);
258  }
259 
260  if ((s_write_length > 0) && (pollfd[1].revents & POLLOUT))
261  {
262  rval = write(s_out_fd, s_write_buffer, s_write_length);
263 
264  if (rval <= 0)
265  {
266  perror("write");
267  exit(EXIT_FAILURE);
268  }
269 
270  s_write_buffer += (uint16_t)rval;
271  s_write_length -= (uint16_t)rval;
272 
273  if (s_write_length == 0)
274  {
275  otPlatUartSendDone();
276  }
277  }
278  }
