136 static void processTransmit(otInstance *aInstance, int argc, char *argv[], char *aOutput,
137  size_t aOutputMaxLen)
138 {
139  ThreadError error = kThreadError_None;
140 
141  VerifyOrExit(otPlatDiagModeGet(), error = kThreadError_InvalidState);
142 
143  if (argc == 0)
144  {
145  snprintf(aOutput, aOutputMaxLen, "transmit will send %" PRId32 " diagnostic messages with %" PRIu32
146  " ms interval\r\nstatus 0x%02x\r\n",
147  sTxRequestedCount, sTxPeriod, error);
148  }
149  else if (strcmp(argv[0], "stop") == 0)
150  {
151  otPlatAlarmStop(aInstance);
152  snprintf(aOutput, aOutputMaxLen, "diagnostic message transmission is stopped\r\nstatus 0x%02x\r\n", error);
153  sTransmitActive = false;
154  }
155  else if (strcmp(argv[0], "start") == 0)
156  {
157  otPlatAlarmStop(aInstance);
158  sTransmitActive = true;
159  sTxCount = sTxRequestedCount;
160  uint32_t now = otPlatAlarmGetNow();
161  otPlatAlarmStartAt(aInstance, now, sTxPeriod);
162  snprintf(aOutput, aOutputMaxLen, "sending %" PRId32 " diagnostic messages with %" PRIu32
163  " ms interval\r\nstatus 0x%02x\r\n",
164  sTxRequestedCount, sTxPeriod, error);
165  }
166  else if (strcmp(argv[0], "interval") == 0)
167  {
168  long value;
169 
170  VerifyOrExit(argc == 2, error = kThreadError_InvalidArgs);
171 
172  SuccessOrExit(error = parseLong(argv[1], &value));
173  VerifyOrExit(value > 0, error = kThreadError_InvalidArgs);
174  sTxPeriod = (uint32_t)(value);
175  snprintf(aOutput, aOutputMaxLen, "set diagnostic messages interval to %" PRIu32 " ms\r\nstatus 0x%02x\r\n", sTxPeriod,
176  error);
177  }
178  else if (strcmp(argv[0], "count") == 0)
179  {
180  long value;
181 
182  VerifyOrExit(argc == 2, error = kThreadError_InvalidArgs);
183 
184  SuccessOrExit(error = parseLong(argv[1], &value));
185  VerifyOrExit((value > 0) || (value == -1), error = kThreadError_InvalidArgs);
186  sTxRequestedCount = (uint32_t)(value);
187  snprintf(aOutput, aOutputMaxLen, "set diagnostic messages count to %" PRId32 "\r\nstatus 0x%02x\r\n", sTxRequestedCount,
188  error);
189  }
190 
191 exit:
192  appendErrorResult(error, aOutput, aOutputMaxLen);
193 }

203 {

205 
206  for (i = 0; i < sizeof(sCommands) / sizeof(sCommands[0]); i++)
207  {
208  if (strcmp(argv[0], sCommands[i].mName) == 0)
209  {
210  sCommands[i].mCommand(aInstance, argc - 1, argc > 1 ? &argv[1] : NULL, aOutput, aOutputMaxLen);

212  }
213  }
214

216  {

218  }

274 void otPlatDiagAlarmCallback(otInstance *aInstance)
275 {
276  if (sTransmitActive)
277  {
278  if ((sTxCount > 0) || (sTxCount == -1))
279  {
280  RadioPacket *sTxPacket = otPlatRadioGetTransmitBuffer(aInstance);
281 
282  sTxPacket->mLength = sizeof(struct PlatformDiagMessage);
283  sTxPacket->mChannel = sChannel;
284  sTxPacket->mPower = sTxPower;
285 
286  sDiagMessage.mChannel = sTxPacket->mChannel;
287  sDiagMessage.mID = sID;
288 
289  memcpy(sTxPacket->mPsdu, &sDiagMessage, sizeof(struct PlatformDiagMessage));
290  otPlatRadioTransmit(aInstance, sTxPacket);
291 
292  sDiagMessage.mCnt++;
293 
294  if (sTxCount != -1)
295  {
296  sTxCount--;
297  }
298 
299  uint32_t now = otPlatAlarmGetNow();
300  otPlatAlarmStartAt(aInstance, now, sTxPeriod);
301  }
302  else
303  {
304  sTransmitActive = false;
305  otPlatAlarmStop(aInstance);
306  otPlatLog(kLogLevelDebg, kLogRegionPlatform, "Transmit done");
307  }
308  }
