134 static inline bool isDataRequest(const uint8_t *frame)
135 {
136  const uint8_t *cur = frame;
137  uint8_t securityControl;
138  bool rval;
139 
140  // FCF + DSN
141  cur += 2 + 1;
142 
143  VerifyOrExit(isFrameTypeMacCmd(frame), rval = false);
144 
145  // Destination PAN + Address
146  switch (frame[1] & IEEE802154_DST_ADDR_MASK)
147  {
148  case IEEE802154_DST_ADDR_SHORT:
149  cur += sizeof(otPanId) + sizeof(otShortAddress);
150  break;
151 
152  case IEEE802154_DST_ADDR_EXT:
153  cur += sizeof(otPanId) + sizeof(otExtAddress);
154  break;
155 
156  default:
157  ExitNow(rval = false);
158  }
159 
160  // Source PAN + Address
161  switch (frame[1] & IEEE802154_SRC_ADDR_MASK)
162  {
163  case IEEE802154_SRC_ADDR_SHORT:
164  if (!isPanIdCompressed(frame))
165  {
166  cur += sizeof(otPanId);
167  }
168 
169  cur += sizeof(otShortAddress);
170  break;
171 
172  case IEEE802154_SRC_ADDR_EXT:
173  if (!isPanIdCompressed(frame))
174  {
175  cur += sizeof(otPanId);
176  }
177 
178  cur += sizeof(otExtAddress);
179  break;
180 
181  default:
182  ExitNow(rval = false);
183  }
184 
185  // Security Control + Frame Counter + Key Identifier
186  if (isSecurityEnabled(frame))
187  {
188  securityControl = *cur;
189 
190  if (securityControl & IEEE802154_SEC_LEVEL_MASK)
191  {
192  cur += 1 + 4;
193  }
194 
195  switch (securityControl & IEEE802154_KEY_ID_MODE_MASK)
196  {
197  case IEEE802154_KEY_ID_MODE_0:
198  cur += 0;
199  break;
200 
201  case IEEE802154_KEY_ID_MODE_1:
202  cur += 1;
203  break;
204 
205  case IEEE802154_KEY_ID_MODE_2:
206  cur += 5;
207  break;
208 
209  case IEEE802154_KEY_ID_MODE_3:
210  cur += 9;
211  break;
212  }
213  }
214 
215  // Command ID
216  rval = cur[0] == IEEE802154_MACCMD_DATA_REQ;
217 
218 exit:
219  return rval;
220 }

331 void platformRadioInit(void)
332 {
333  struct sockaddr_in sockaddr;
334  char *offset;
335  memset(&sockaddr, 0, sizeof(sockaddr));
336  sockaddr.sin_family = AF_INET;
337 
338  offset = getenv("PORT_OFFSET");
339 
340  if (offset)
341  {
342  char *endptr;
343 
344  sPortOffset = (uint16_t)strtol(offset, &endptr, 0);
345 
346  if (*endptr != '\0')
347  {
348  fprintf(stderr, "Invalid PORT_OFFSET: %s\n", offset);
349  exit(1);
350  }
351 
352  sPortOffset *= WELLKNOWN_NODE_ID;
353  }
354 
355  if (sPromiscuous)
356  {
357  sockaddr.sin_port = htons(9000 + sPortOffset + WELLKNOWN_NODE_ID);
358  }
359  else
360  {
361  sockaddr.sin_port = htons(9000 + sPortOffset + NODE_ID);
362  }
363 
364  sockaddr.sin_addr.s_addr = INADDR_ANY;
365 
366  sSockFd = (int)socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
367  bind(sSockFd, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
368 
369  sReceiveFrame.mPsdu = sReceiveMessage.mPsdu;
370  sTransmitFrame.mPsdu = sTransmitMessage.mPsdu;
371  sAckFrame.mPsdu = sAckMessage.mPsdu;
372 }

469 void radioReceive(otInstance *aInstance)
470 {
471  ssize_t rval = recvfrom(sSockFd, (char *)&sReceiveMessage, sizeof(sReceiveMessage), 0, NULL, NULL);
472 
473  if (rval < 0)
474  {
475  perror("recvfrom");
476  exit(EXIT_FAILURE);
477  }
478 
479  sReceiveFrame.mLength = (uint8_t)(rval - 1);
480 
481  if (sAckWait &&
482  sTransmitFrame.mChannel == sReceiveMessage.mChannel &&
483  isFrameTypeAck(sReceiveFrame.mPsdu) &&
484  getDsn(sReceiveFrame.mPsdu) == getDsn(sTransmitFrame.mPsdu))
485  {
486  sState = kStateReceive;
487  sAckWait = false;
488 
489 #if OPENTHREAD_ENABLE_DIAG
490 
491  if (otPlatDiagModeGet())
492  {
493  otPlatDiagRadioTransmitDone(aInstance, &sTransmitFrame, isFramePending(sReceiveFrame.mPsdu), kThreadError_None);
494  }
495  else
496 #endif
497  {
498  otPlatRadioTransmitDone(aInstance, &sTransmitFrame, isFramePending(sReceiveFrame.mPsdu), kThreadError_None);
499  }
500  }
501  else if ((sState == kStateReceive || sState == kStateTransmit) &&
502  (sReceiveFrame.mChannel == sReceiveMessage.mChannel))
503  {
504  radioProcessFrame(aInstance);
505  }
506 }

508 void radioSendMessage(otInstance *aInstance)
509 {
510  sTransmitMessage.mChannel = sTransmitFrame.mChannel;
511 
512  radioTransmit(&sTransmitMessage, &sTransmitFrame);
513 
514  sAckWait = isAckRequested(sTransmitFrame.mPsdu);
515 
516  if (!sAckWait)
517  {
518  sState = kStateReceive;
519 
520 #if OPENTHREAD_ENABLE_DIAG
521 
522  if (otPlatDiagModeGet())
523  {
524  otPlatDiagRadioTransmitDone(aInstance, &sTransmitFrame, false, kThreadError_None);
525  }
526  else
527 #endif
528  {
529  otPlatRadioTransmitDone(aInstance, &sTransmitFrame, false, kThreadError_None);
530  }
531  }

557 void platformRadioProcess(otInstance *aInstance)
558 {
559  const int flags = POLLIN | POLLRDNORM | POLLERR | POLLNVAL | POLLHUP;
560  struct pollfd pollfd = { sSockFd, flags, 0 };

573 void radioTransmit(struct RadioMessage *msg, const struct RadioPacket *pkt)
574 {
575  uint32_t i;
576  struct sockaddr_in sockaddr;
577 
578  uint16_t crc = 0;
579  uint16_t crc_offset = pkt->mLength - sizeof(uint16_t);
580 
581  for (i = 0; i < crc_offset; i++)
582  {
583  crc = crc16_citt(crc, msg->mPsdu[i]);
584  }
585 
586  msg->mPsdu[crc_offset] = crc & 0xff;
587  msg->mPsdu[crc_offset + 1] = crc >> 8;
588 
589  memset(&sockaddr, 0, sizeof(sockaddr));
590  sockaddr.sin_family = AF_INET;
591  inet_pton(AF_INET, "127.0.0.1", &sockaddr.sin_addr);
592 
593  for (i = 1; i <= WELLKNOWN_NODE_ID; i++)
594  {
595  ssize_t rval;
596 
597  if (NODE_ID == i)
598  {
599  continue;
600  }
601 
602  sockaddr.sin_port = htons(9000 + sPortOffset + i);
603  rval = sendto(sSockFd, (const char *)msg, 1 + pkt->mLength,
604  0, (struct sockaddr *)&sockaddr, sizeof(sockaddr));
605 
606  if (rval < 0)
607  {
608  perror("recvfrom");
609  exit(EXIT_FAILURE);
610  }
611  }
612 }

614 void radioSendAck(void)
615 {
616  sAckFrame.mLength = IEEE802154_ACK_LENGTH;
617  sAckMessage.mPsdu[0] = IEEE802154_FRAME_TYPE_ACK;
618 
619  if (isDataRequest(sReceiveFrame.mPsdu))
620  {
621  sAckMessage.mPsdu[0] |= IEEE802154_FRAME_PENDING;

632 void radioProcessFrame(otInstance *aInstance)
633 {
634  ThreadError error = kThreadError_None;
635  otPanId dstpan;
636  otShortAddress short_address;
637  otExtAddress ext_address;
638 
639  VerifyOrExit(sPromiscuous == false, error = kThreadError_None);
640 
641  switch (sReceiveFrame.mPsdu[1] & IEEE802154_DST_ADDR_MASK)
642  {
643  case IEEE802154_DST_ADDR_NONE:
644  break;
645 
646  case IEEE802154_DST_ADDR_SHORT:
647  dstpan = getDstPan(sReceiveFrame.mPsdu);
648  short_address = getShortAddress(sReceiveFrame.mPsdu);
649  VerifyOrExit((dstpan == IEEE802154_BROADCAST || dstpan == sPanid) &&
650  (short_address == IEEE802154_BROADCAST || short_address == sShortAddress),
651  error = kThreadError_Abort);
652  break;
653 
654  case IEEE802154_DST_ADDR_EXT:
655  dstpan = getDstPan(sReceiveFrame.mPsdu);
656  getExtAddress(sReceiveFrame.mPsdu, &ext_address);
657  VerifyOrExit((dstpan == IEEE802154_BROADCAST || dstpan == sPanid) &&
658  memcmp(&ext_address, sExtendedAddress, sizeof(ext_address)) == 0,
659  error = kThreadError_Abort);
660  break;
661 
662  default:
663  ExitNow(error = kThreadError_Abort);
664  }
665 
666  sReceiveFrame.mPower = -20;
667  sReceiveFrame.mLqi = kPhyNoLqi;
668 
669  // generate acknowledgment
670  if (isAckRequested(sReceiveFrame.mPsdu))
671  {
672  radioSendAck();
673  }
674 
675 exit:
676 
677 #if OPENTHREAD_ENABLE_DIAG
678 
679  if (otPlatDiagModeGet())
680  {
681  otPlatDiagRadioReceiveDone(aInstance, error == kThreadError_None ? &sReceiveFrame : NULL, error);
682  }
683  else
684 #endif
