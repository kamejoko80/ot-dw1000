152 void JoinerRouter::HandleUdpReceive(Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
153 {
154  ThreadError error;
155  Message *message = NULL;
156  Coap::Header header;
157  Ip6::MessageInfo messageInfo;
158  JoinerUdpPortTlv udpPort;
159  JoinerIidTlv iid;
160  JoinerRouterLocatorTlv rloc;
161  ExtendedTlv tlv;
162  uint16_t borderAgentRloc;
163 
164  otLogFuncEntryMsg("from peer: %llX",
165  HostSwap64(*reinterpret_cast<const uint64_t *>(aMessageInfo.GetPeerAddr().mFields.m8 + 8)));
166  otLogInfoMeshCoP("JoinerRouter::HandleUdpReceive");
167 
168  SuccessOrExit(error = GetBorderAgentRloc(borderAgentRloc));
169 
170  header.Init(kCoapTypeNonConfirmable, kCoapRequestPost);
171  header.SetToken(Coap::Header::kDefaultTokenLength);
172  header.AppendUriPathOptions(OPENTHREAD_URI_RELAY_RX);
173  header.SetPayloadMarker();
174 
175  VerifyOrExit((message = mNetif.GetCoapClient().NewMeshCoPMessage(header)) != NULL, error = kThreadError_NoBufs);
176 
177  udpPort.Init();
178  udpPort.SetUdpPort(aMessageInfo.GetPeerPort());
179  SuccessOrExit(error = message->Append(&udpPort, sizeof(udpPort)));
180 
181  iid.Init();
182  iid.SetIid(aMessageInfo.GetPeerAddr().mFields.m8 + 8);
183  SuccessOrExit(error = message->Append(&iid, sizeof(iid)));
184 
185  rloc.Init();
186  rloc.SetJoinerRouterLocator(mNetif.GetMle().GetRloc16());
187  SuccessOrExit(error = message->Append(&rloc, sizeof(rloc)));
188 
189  tlv.SetType(Tlv::kJoinerDtlsEncapsulation);
190  tlv.SetLength(aMessage.GetLength() - aMessage.GetOffset());
191  SuccessOrExit(error = message->Append(&tlv, sizeof(tlv)));
192 
193  while (aMessage.GetOffset() < aMessage.GetLength())
194  {
195  uint16_t length = aMessage.GetLength() - aMessage.GetOffset();
196  uint8_t tmp[16];
197 
198  if (length >= sizeof(tmp))
199  {
200  length = sizeof(tmp);
201  }
202 
203  aMessage.Read(aMessage.GetOffset(), length, tmp);
204  aMessage.MoveOffset(length);
205 
206  SuccessOrExit(error = message->Append(tmp, length));
207  }
208 
209  messageInfo.SetPeerAddr(mNetif.GetMle().GetMeshLocal16());
210  messageInfo.GetPeerAddr().mFields.m16[7] = HostSwap16(borderAgentRloc);
211  messageInfo.SetPeerPort(kCoapUdpPort);
212 
213  SuccessOrExit(error = mNetif.GetCoapClient().SendMessage(*message, messageInfo));
214 
215  otLogInfoMeshCoP("Sent relay rx");
216 
217 exit:
218 
219  if (error != kThreadError_None && message != NULL)
220  {
221  message->Free();
222  }
223 
224  otLogFuncExitErr(error);
225 }

235 void JoinerRouter::HandleRelayTransmit(Coap::Header &aHeader, Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
236 {
237  ThreadError error;
238  JoinerUdpPortTlv joinerPort;
239  JoinerIidTlv joinerIid;
240  JoinerRouterKekTlv kek;
241  uint16_t offset;
242  uint16_t length;
243  Message *message = NULL;
244  Ip6::MessageInfo messageInfo;
245 
246  otLogFuncEntry();
247  VerifyOrExit(aHeader.GetType() == kCoapTypeNonConfirmable &&
248  aHeader.GetCode() == kCoapRequestPost, error = kThreadError_Drop);
249 
250  otLogInfoMeshCoP("Received relay transmit");
251 
252  SuccessOrExit(error = Tlv::GetTlv(aMessage, Tlv::kJoinerUdpPort, sizeof(joinerPort), joinerPort));
253  VerifyOrExit(joinerPort.IsValid(), error = kThreadError_Parse);
254 
255  SuccessOrExit(error = Tlv::GetTlv(aMessage, Tlv::kJoinerIid, sizeof(joinerIid), joinerIid));
256  VerifyOrExit(joinerIid.IsValid(), error = kThreadError_Parse);
257 
258  SuccessOrExit(error = Tlv::GetValueOffset(aMessage, Tlv::kJoinerDtlsEncapsulation, offset, length));
259 
260  VerifyOrExit((message = mSocket.NewMessage(0)) != NULL, error = kThreadError_NoBufs);
261  message->SetPriority(kMeshCoPMessagePriority);
262  message->SetLinkSecurityEnabled(false);
263 
264  while (length)
265  {
266  uint16_t copyLength = length;
267  uint8_t tmp[16];
268 
269  if (copyLength >= sizeof(tmp))
270  {
271  copyLength = sizeof(tmp);
272  }
273 
274  aMessage.Read(offset, copyLength, tmp);
275  SuccessOrExit(error = message->Append(tmp, copyLength));
276 
277  offset += copyLength;
278  length -= copyLength;
279  }
280 
281  aMessage.CopyTo(offset, 0, length, *message);
282 
283  messageInfo.mPeerAddr.mFields.m16[0] = HostSwap16(0xfe80);
284  memcpy(messageInfo.mPeerAddr.mFields.m8 + 8, joinerIid.GetIid(), 8);
285  messageInfo.SetPeerPort(joinerPort.GetUdpPort());
286  messageInfo.SetInterfaceId(mNetif.GetInterfaceId());
287 
288  SuccessOrExit(error = mSocket.SendTo(*message, messageInfo));
289 
290  if (Tlv::GetTlv(aMessage, Tlv::kJoinerRouterKek, sizeof(kek), kek) == kThreadError_None)
291  {
292  otLogInfoMeshCoP("Received kek");
293  mNetif.GetKeyManager().SetKek(kek.GetKek());
294  SendJoinerEntrust(messageInfo);
295  }
296 
297 exit:
298  (void)aMessageInfo;
299 
300  if (error != kThreadError_None && message != NULL)
301  {
302  message->Free();
303  }
304 
305  otLogFuncExitErr(error);
306 }
