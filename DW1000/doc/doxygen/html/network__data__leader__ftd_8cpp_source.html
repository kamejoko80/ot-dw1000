185 void Leader::HandleCommissioningSet(Coap::Header &aHeader, Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
186 {
187  uint16_t offset = aMessage.GetOffset();
188  uint8_t length = static_cast<uint8_t>(aMessage.GetLength() - aMessage.GetOffset());
189  uint8_t tlvs[NetworkData::kMaxSize];
190  MeshCoP::StateTlv::State state = MeshCoP::StateTlv::kAccept;
191  bool hasSessionId = false;
192  bool hasValidTlv = false;
193  uint16_t sessionId = 0;
194 
195  VerifyOrExit(mNetif.GetMle().GetDeviceState() == Mle::kDeviceStateLeader, state = MeshCoP::StateTlv::kReject);
196 
197  aMessage.Read(offset, length, tlvs);
198 
199  // Session Id and Border Router Locator MUST NOT be set, but accept including unexpected or
200  // unknown TLV as long as there is at least one valid TLV.
201  for (MeshCoP::Tlv *cur = reinterpret_cast<MeshCoP::Tlv *>(tlvs);
202  cur < reinterpret_cast<MeshCoP::Tlv *>(tlvs + length);
203  cur = cur->GetNext())
204  {
205  MeshCoP::Tlv::Type type = cur->GetType();
206 
207  if (type == MeshCoP::Tlv::kJoinerUdpPort || type == MeshCoP::Tlv::kSteeringData)
208  {
209  hasValidTlv = true;
210  }
211  else if (type == MeshCoP::Tlv::kBorderAgentLocator)
212  {
213  ExitNow(state = MeshCoP::StateTlv::kReject);
214  }
215  else if (type == MeshCoP::Tlv::kCommissionerSessionId)
216  {
217  hasSessionId = true;
218  sessionId = static_cast<MeshCoP::CommissionerSessionIdTlv *>(cur)->GetCommissionerSessionId();
219  }
220  else
221  {
222  // do nothing for unexpected or unknown TLV
223  }
224  }
225 
226  // verify whether or not commissioner session id TLV is included
227  VerifyOrExit(hasSessionId, state = MeshCoP::StateTlv::kReject);
228 
229  // verify whether or not MGMT_COMM_SET.req includes at least one valid TLV
230  VerifyOrExit(hasValidTlv, state = MeshCoP::StateTlv::kReject);
231 
232  for (MeshCoP::Tlv *cur = reinterpret_cast<MeshCoP::Tlv *>(mTlvs + sizeof(CommissioningDataTlv));
233  cur < reinterpret_cast<MeshCoP::Tlv *>(mTlvs + mLength);
234  cur = cur->GetNext())
235  {
236  if (cur->GetType() == MeshCoP::Tlv::kCommissionerSessionId)
237  {
238  VerifyOrExit(sessionId ==
239  static_cast<MeshCoP::CommissionerSessionIdTlv *>(cur)->GetCommissionerSessionId(),
240  state = MeshCoP::StateTlv::kReject);
241  }
242  else if (cur->GetType() == MeshCoP::Tlv::kBorderAgentLocator)
243  {
244  memcpy(tlvs + length, reinterpret_cast<uint8_t *>(cur), cur->GetLength() + sizeof(MeshCoP::Tlv));
245  length += (cur->GetLength() + sizeof(MeshCoP::Tlv));
246  }
247  }
248 
249  SetCommissioningData(tlvs, length);
250 
251 exit:
252 
253  if (mNetif.GetMle().GetDeviceState() == Mle::kDeviceStateLeader)
254  {
255  SendCommissioningSetResponse(aHeader, aMessageInfo, state);
256  }
257 
258  return;
259 }

293 void Leader::SendCommissioningGetResponse(const Coap::Header &aRequestHeader, const Ip6::MessageInfo &aMessageInfo,
294  uint8_t *aTlvs, uint8_t aLength)
295 {
296  ThreadError error = kThreadError_None;
297  Coap::Header responseHeader;
298  Message *message;
299  uint8_t index;
300  uint8_t *data = NULL;
301  uint8_t length = 0;
302 
303  VerifyOrExit((message = mNetif.GetCoapServer().NewMeshCoPMessage(0)) != NULL, error = kThreadError_NoBufs);
304 
305  responseHeader.SetDefaultResponseHeader(aRequestHeader);
306  responseHeader.SetPayloadMarker();
307 
308  SuccessOrExit(error = message->Append(responseHeader.GetBytes(), responseHeader.GetLength()));
309 
310  for (NetworkDataTlv *cur = reinterpret_cast<NetworkDataTlv *>(mTlvs);
311  cur < reinterpret_cast<NetworkDataTlv *>(mTlvs + mLength);
312  cur = cur->GetNext())
313  {
314  if (cur->GetType() == NetworkDataTlv::kTypeCommissioningData)
315  {
316  data = cur->GetValue();
317  length = cur->GetLength();
318  break;
319  }
320  }
321 
322  VerifyOrExit(data && length, error = kThreadError_Drop);
323 
324  if (aLength == 0)
325  {
326  SuccessOrExit(error = message->Append(data, length));
327  }
328  else
329  {
330  for (index = 0; index < aLength; index++)
331  {
332  for (MeshCoP::Tlv *cur = reinterpret_cast<MeshCoP::Tlv *>(data);
333  cur < reinterpret_cast<MeshCoP::Tlv *>(data + length);
334  cur = cur->GetNext())
335  {
336  if (cur->GetType() == aTlvs[index])
337  {
338  SuccessOrExit(error = message->Append(cur, sizeof(NetworkDataTlv) + cur->GetLength()));
339  break;
340  }
341  }
342  }
343  }
344 
345  if (message->GetLength() == responseHeader.GetLength())
346  {
347  // no payload, remove coap payload marker
348  message->SetLength(message->GetLength() - 1);
349  }
350 
351  SuccessOrExit(error = mNetif.GetCoapServer().SendMessage(*message, aMessageInfo));
352 
353  otLogInfoMeshCoP("sent commissioning dataset get response");
354 
355 exit:
356 
357  if (error != kThreadError_None && message != NULL)
358  {
359  message->Free();
360  }
361 }

394 void Leader::RlocLookup(uint16_t aRloc16, bool &aIn, bool &aStable, uint8_t *aTlvs, uint8_t aTlvsLength)
395 {
396  NetworkDataTlv *cur = reinterpret_cast<NetworkDataTlv *>(aTlvs);
397  NetworkDataTlv *end = reinterpret_cast<NetworkDataTlv *>(aTlvs + aTlvsLength);
398  NetworkDataTlv *subCur;
399  NetworkDataTlv *subEnd;
400  PrefixTlv *prefix;
401  BorderRouterTlv *borderRouter;
402  HasRouteTlv *hasRoute;
403  BorderRouterEntry *borderRouterEntry;
404  HasRouteEntry *hasRouteEntry;
405 
406  while (cur < end)
407  {
408  if (cur->GetType() == NetworkDataTlv::kTypePrefix)
409  {
410  prefix = static_cast<PrefixTlv *>(cur);
411  subCur = prefix->GetSubTlvs();
412  subEnd = prefix->GetNext();
413 
414  while (subCur < subEnd)
415  {
416  switch (subCur->GetType())
417  {
418  case NetworkDataTlv::kTypeBorderRouter:
419  borderRouter = FindBorderRouter(*prefix);
420 
421  for (uint8_t i = 0; i < borderRouter->GetNumEntries(); i++)
422  {
423  borderRouterEntry = borderRouter->GetEntry(i);
424 
425  if (borderRouterEntry->GetRloc() == aRloc16)
426  {
427  aIn = true;
428 
429  if (borderRouter->IsStable())
430  {
431  aStable = true;
432  }
433  }
434  }
435 
436  break;
437 
438  case NetworkDataTlv::kTypeHasRoute:
439  hasRoute = FindHasRoute(*prefix);
440 
441  for (uint8_t i = 0; i < hasRoute->GetNumEntries(); i++)
442  {
443  hasRouteEntry = hasRoute->GetEntry(i);
444 
445  if (hasRouteEntry->GetRloc() == aRloc16)
446  {
447  aIn = true;
448 
449  if (hasRoute->IsStable())
450  {
451  aStable = true;
452  }
453  }
454  }
455 
456  break;
457 
458  default:
459  break;
460  }
461 
462  if (aIn && aStable)
463  {
464  ExitNow();
465  }
466 
467  subCur = subCur->GetNext();
468  }
469  }
470 
471  cur = cur->GetNext();
472  }
473 
474 exit:
475  return;
476 }

478 bool Leader::IsStableUpdated(uint16_t aRloc16, uint8_t *aTlvs, uint8_t aTlvsLength, uint8_t *aTlvsBase,
479  uint8_t aTlvsBaseLength)
480 {
481  bool rval = false;
482  NetworkDataTlv *cur = reinterpret_cast<NetworkDataTlv *>(aTlvs);
483  NetworkDataTlv *end = reinterpret_cast<NetworkDataTlv *>(aTlvs + aTlvsLength);
484  PrefixTlv *prefix;
485  PrefixTlv *prefixBase;
486  BorderRouterTlv *borderRouter;
487  HasRouteTlv *hasRoute;
488  ContextTlv *context;
489 
490  while (cur < end)
491  {
492  if (cur->GetType() == NetworkDataTlv::kTypePrefix)
493  {
494  prefix = static_cast<PrefixTlv *>(cur);
495  context = FindContext(*prefix);
496  borderRouter = FindBorderRouter(*prefix);
497  hasRoute = FindHasRoute(*prefix);
498 
499  if (cur->IsStable() && (!context || borderRouter))
500  {
501  prefixBase = FindPrefix(prefix->GetPrefix(), prefix->GetPrefixLength(), aTlvsBase, aTlvsBaseLength);
502 
503  if (!prefixBase)
504  {
505  ExitNow(rval = true);
506  }
507 
508  if (borderRouter && memcmp(borderRouter, FindBorderRouter(*prefixBase), borderRouter->GetLength()) != 0)
509  {
510  ExitNow(rval = true);
511  }
512 
513  if (hasRoute && (memcmp(hasRoute, FindHasRoute(*prefixBase), hasRoute->GetLength()) != 0))
514  {
515  ExitNow(rval = true);
516  }
517  }
518  }
519 
520  cur = cur->GetNext();
521  }
522 
523 exit:
524  (void)aRloc16;
525  return rval;
526 }
