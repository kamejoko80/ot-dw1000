213 ThreadError NetworkDiagnostic::AppendChildTable(Message &aMessage)
214 {
215  ThreadError error = kThreadError_None;
216  uint8_t count = 0;
217  uint8_t timeout = 0;
218  uint8_t numChildren;
219  const Child *children = mNetif.GetMle().GetChildren(&numChildren);
220  ChildTableTlv tlv;
221  ChildTableEntry entry;
222 
223  tlv.Init();
224 
225  for (int i = 0; i < numChildren; i++)
226  {
227  if (children[i].mState == Neighbor::kStateValid)
228  {
229  count++;
230  }
231  }
232 
233  tlv.SetLength(count * sizeof(ChildTableEntry));
234 
235  SuccessOrExit(error = aMessage.Append(&tlv, sizeof(ChildTableTlv)));
236 
237  for (int i = 0; i < numChildren; i++)
238  {
239  if (children[i].mState == Neighbor::kStateValid)
240  {
241  timeout = 0;
242 
243  while (static_cast<uint32_t>(1 << timeout) < children[i].mTimeout) { timeout++; }
244 
245  entry.SetReserved(0);
246  entry.SetTimeout(timeout + 4);
247  entry.SetChildId(mNetif.GetMle().GetChildId(children[i].mValid.mRloc16));
248  entry.SetMode(children[i].mMode);
249 
250  SuccessOrExit(error = aMessage.Append(&entry, sizeof(ChildTableEntry)));
251  }
252  }
253 
254 exit:
255 
256  return error;

271

273

275  {

277  {
278  ExtMacAddressTlv tlv;

283  }
284 
285  case NetworkDiagnosticTlv::kAddress16:
286  {
287  Address16Tlv tlv;
288  tlv.Init();
289  tlv.SetRloc16(mNetif.GetMle().GetRloc16());
290  SuccessOrExit(error = aResponse.Append(&tlv, sizeof(tlv)));
291  break;
292  }
293 
294  case NetworkDiagnosticTlv::kMode:
295  {
296  ModeTlv tlv;
297  tlv.Init();
298  tlv.SetMode(mNetif.GetMle().GetDeviceMode());
299  SuccessOrExit(error = aResponse.Append(&tlv, sizeof(tlv)));
300  break;
301  }
302 
303  case NetworkDiagnosticTlv::kTimeout:
304  {
305  if ((mNetif.GetMle().GetDeviceMode() & ModeTlv::kModeRxOnWhenIdle) == 0)
306  {
307  TimeoutTlv tlv;
308  tlv.Init();
309  tlv.SetTimeout(mNetif.GetMle().GetTimeout());
310  SuccessOrExit(error = aResponse.Append(&tlv, sizeof(tlv)));
311  }
312 
313  break;
314  }
315 
316  case NetworkDiagnosticTlv::kConnectivity:
317  {
318  ConnectivityTlv tlv;
319  tlv.Init();
320  mNetif.GetMle().FillConnectivityTlv(*reinterpret_cast<Mle::ConnectivityTlv *>(&tlv));
321  SuccessOrExit(error = aResponse.Append(&tlv, sizeof(tlv)));
322  break;
323  }

327  RouteTlv tlv;
328  tlv.Init();
329  mNetif.GetMle().FillRouteTlv(*reinterpret_cast<Mle::RouteTlv *>(&tlv));

332  }

351 
352  case NetworkDiagnosticTlv::kIPv6AddressList:

354  SuccessOrExit(error = AppendIPv6AddressList(aResponse));

356  }
357 
358  case NetworkDiagnosticTlv::kMacCounters:
359  {
360  MacCountersTlv tlv;
361  memset(&tlv, 0, sizeof(tlv));
362  tlv.Init();
363  mNetif.GetMac().FillMacCountersTlv(tlv);
364  SuccessOrExit(error = aResponse.Append(&tlv, tlv.GetSize()));
365  break;
366  }
367 
368  case NetworkDiagnosticTlv::kBatteryLevel:
369  {
370  // TODO Need more api from driver
371  BatteryLevelTlv tlv;
372  tlv.Init();
373  tlv.SetBatteryLevel(100);
374  SuccessOrExit(error = aResponse.Append(&tlv, tlv.GetSize()));
375  break;
376  }

379  {
380  // TODO Need more api from driver
381  SupplyVoltageTlv tlv;
382  tlv.Init();
383  tlv.SetSupplyVoltage(0);
384  SuccessOrExit(error = aResponse.Append(&tlv, tlv.GetSize()));
385  break;
386  }
387 
388  case NetworkDiagnosticTlv::kChildTable:
389  {
390  SuccessOrExit(error = AppendChildTable(aResponse));

392  }
393 
394  case NetworkDiagnosticTlv::kChannelPages:
395  {
396  ChannelPagesTlv tlv;
397  tlv.Init();

464  messageInfo.SetSockAddr(mNetif.GetMle().GetMeshLocal16());
465  messageInfo.SetPeerPort(kCoapUdpPort);

467 
468  SuccessOrExit(error = FillRequestedTlvs(aMessage, *message, networkDiagnosticTlv));
469 
470  SuccessOrExit(error = mNetif.GetCoapClient().SendMessage(*message, messageInfo, NULL, this));
471 
472  otLogInfoNetDiag("Sent diagnostic get answer");
473 
474 exit:
