453 __STATIC_INLINE NRF_GPIO_Type * nrf_gpio_pin_port_decode(uint32_t * p_pin)
454 {
455  ASSERT(*p_pin < NUMBER_OF_PINS);
456 #if (GPIO_COUNT == 1)
457  // The oldest definition case
458  return NRF_GPIO;
459 #else
460  if (*p_pin < P0_PIN_NUM)
461  {
462  return NRF_P0;
463  }
464  else
465  {
466  *p_pin = *p_pin & (P0_PIN_NUM - 1);
467  return NRF_P1;
468  }
469 #endif
470 }

495 __STATIC_INLINE void nrf_gpio_cfg(
496  uint32_t pin_number,
497  nrf_gpio_pin_dir_t dir,
498  nrf_gpio_pin_input_t input,
499  nrf_gpio_pin_pull_t pull,
500  nrf_gpio_pin_drive_t drive,
501  nrf_gpio_pin_sense_t sense)
502 {
503  NRF_GPIO_Type * reg = nrf_gpio_pin_port_decode(&pin_number);
504 
505  reg->PIN_CNF[pin_number] = ((uint32_t)dir << GPIO_PIN_CNF_DIR_Pos)
506  | ((uint32_t)input << GPIO_PIN_CNF_INPUT_Pos)
507  | ((uint32_t)pull << GPIO_PIN_CNF_PULL_Pos)
508  | ((uint32_t)drive << GPIO_PIN_CNF_DRIVE_Pos)
509  | ((uint32_t)sense << GPIO_PIN_CNF_SENSE_Pos);
510 }
