357 static int do_spi_xfer(int len)
358  {
359  int ret;
360 
361  struct spi_ioc_transfer xfer[2] =
362  {
363  { // This part is the delay between C̅S̅ being
364  // asserted and the SPI clock starting. This
365  // is not supported by all Linux SPI drivers.
366  .tx_buf = 0,
367  .rx_buf = 0,
368  .len = 0,
369  .delay_usecs = (uint16_t)sSpiCsDelay,
370  .speed_hz = (uint32_t)sSpiSpeed,
371  .bits_per_word = 8,
372  .cs_change = false,
373  },
374  { // This part is the actual SPI transfer.
375  .tx_buf = (unsigned long)sSpiTxFrameBuffer,
376  .rx_buf = (unsigned long)sSpiRxFrameBuffer,
377  .len = (uint32_t)(len + HEADER_LEN + sSpiRxAlignAllowance),
378  .delay_usecs = 0,
379  .speed_hz = (uint32_t)sSpiSpeed,
380  .bits_per_word = 8,
381  .cs_change = false,
382  }
383  };
384 
385  if (sSpiCsDelay > 0)
386  {
387  // A C̅S̅ delay has been specified. Start transactions
388  // with both parts.
389  ret = ioctl(sSpiDevFd, SPI_IOC_MESSAGE(2), &xfer[0]);
390  }
391  else
392  {
393  // No C̅S̅ delay has been specified, so we skip the first
394  // part because it causes some SPI drivers to croak.
395  ret = ioctl(sSpiDevFd, SPI_IOC_MESSAGE(1), &xfer[1]);
396  }
397 
398  if (ret != -1)
399  {
400  log_debug_buffer("SPI-TX", sSpiTxFrameBuffer, (int)xfer[1].len);
401  log_debug_buffer("SPI-RX", sSpiRxFrameBuffer, (int)xfer[1].len);
402 
403  if (spi_header_get_flag_byte(sSpiRxFrameBuffer) != 0xFF)
404  {
405  if (spi_header_get_flag_byte(sSpiRxFrameBuffer) & SPI_HEADER_RESET_FLAG)
406  {
407  sSlaveDidReset = true;
408  }
409  }
410 
411  sSpiFrameCount++;
412  }
413 
414  return ret;
415 }

439 static int push_pull_spi(void)
440 {
441  int ret;
442  uint8_t slave_header;
443  uint16_t slave_max_rx;
444  uint16_t slave_data_len;
445  uint16_t spi_xfer_bytes = 5;
446  const uint8_t* spiRxFrameBuffer = NULL;
447 
448  sSpiTxFlowControl = false;
449 
451 
452  // The purpose of the first transaction is to attempt to
453  // send any transactions we have queued and fetch the slave's
454  // buffer sizes.
455 
456  if (sSpiValidFrameCount == 0)
457  {
458  spi_header_set_flag_byte(sSpiTxFrameBuffer, SPI_HEADER_RESET_FLAG|SPI_HEADER_PATTERN_VALUE);
459  }
460  else
461  {
462  spi_header_set_flag_byte(sSpiTxFrameBuffer, SPI_HEADER_PATTERN_VALUE);
463  }
464 
465  // Zero out our max rx and data len
466  // so that the slave doesn't think
467  // we are actually trying to transfer
468  // data.
469  spi_header_set_accept_len(sSpiTxFrameBuffer, 0);
470  spi_header_set_data_len(sSpiTxFrameBuffer, 0);
471 
472  if (sSpiTxIsReady)
473  {
474  // Go ahead and try to immediately send a frame if we have it queued up.
475  spi_header_set_data_len(sSpiTxFrameBuffer, sSpiTxPayloadSize);
476 
477  if (sSpiTxPayloadSize > spi_xfer_bytes)
478  {
479  spi_xfer_bytes = sSpiTxPayloadSize;
480  }
481  }
482 
483  // If we aren't already processing a received frame, we
484  // can also handle receiving the next frame if its length
485  // is equal to or less than the size of what we are
486  // trying to transmit above.
487  if (sSpiRxPayloadSize == 0) {
488  spi_header_set_accept_len(sSpiTxFrameBuffer, spi_xfer_bytes);
489  }
490 
491  // Perform the first SPI transaction.
492  ret = do_spi_xfer(spi_xfer_bytes);
493  if (ret < 0)
494  {
495  perror("do_spi_xfer");
496 
497  // Print out a helpful error message for
498  // a common error.
499  if ( (sSpiCsDelay != 0)
500  && (errno == EINVAL)
501  ) {
502  syslog(LOG_ERR, "SPI ioctl failed with EINVAL. Try adding `--spi-cs-delay=0` to command line arguments.");
503  }
504  goto bail;
505  }
506 
507  // Account for misalignment (0xFF bytes at the start)
508  spiRxFrameBuffer = get_real_rx_frame_start();
509 
510  debug_spi_header("push_pull_1");
511 
512  slave_header = spi_header_get_flag_byte(spiRxFrameBuffer);
513 
514  if ((slave_header == 0xFF) || (slave_header == 0x00))
515  {
516  // Device is off or in a bad state.
517  sSpiTxFlowControl = true;
518 
519  syslog(LOG_DEBUG, "Discarded frame. (1)");
520  goto bail;
521  }
522 
523  slave_max_rx = spi_header_get_accept_len(spiRxFrameBuffer);
524  slave_data_len = spi_header_get_data_len(spiRxFrameBuffer);
525 
526  if ( ((slave_header & SPI_HEADER_PATTERN_MASK) != SPI_HEADER_PATTERN_VALUE)
527  || (slave_max_rx > MAX_FRAME_SIZE)
528  || (slave_data_len > MAX_FRAME_SIZE)
529  )
530  {
531  sSpiTxFlowControl = true;
532  syslog(
533  LOG_INFO,
534  "Gibberish in header (h:0x%02X, max_rx:0x%04X, data_len:0x%04X)",
535  slave_header,
536  slave_max_rx,
537  slave_data_len
538  );
539  goto bail;
540  }
541 
542  sSpiValidFrameCount++;
543 
544  // Handle received packet, if any.
545  if ( (sSpiRxPayloadSize == 0)
546  && (slave_data_len != 0)
547  && (slave_data_len <= spi_header_get_accept_len(sSpiTxFrameBuffer))
548  ) {
549  // We have received a packet. Set sSpiRxPayloadSize so that
550  // the packet will eventually get queued up by push_hdlc().
551  sSpiRxPayloadSize = slave_data_len;
552 
553  slave_data_len = 0;
554  }
555 
556  if (sSpiTxIsReady)
557  {
558  // Handle transmitted packet.
559  if (spi_header_get_data_len(sSpiTxFrameBuffer) <= slave_max_rx)
560  {
561  // Outbound packet has been successfully transmitted. Clear
562  // sSpiTxPayloadSize and sSpiTxIsReady so that pull_hdlc() can
563  // pull another packet for us to send.
564  sSpiTxIsReady = false;
565  sSpiTxPayloadSize = 0;
566  spi_header_set_data_len(sSpiTxFrameBuffer, 0);
567  } else {
568  // The slave Wasn't ready for what we had to
569  // send them. Turn on rate limiting so that we
570  // don't waste a ton of CPU bombarding them
571  // with useless SPI transfers.
572  sSpiTxFlowControl = true;
573  }
574  }
575 
576  if (slave_data_len == 0)
577  {
578  // Nothing else to do.
579  goto bail;
580  }
581 
583 
584  // The purpose of the second transaction is to attempt to
585  // fetch any packets that the slave has for us that didn't
586  // fit in the first transaction.
587 
588  spi_header_set_flag_byte(sSpiTxFrameBuffer, SPI_HEADER_PATTERN_VALUE);
589  spi_header_set_accept_len(sSpiTxFrameBuffer, 0);
590 
591  if (sSpiTxIsReady)
592  {
593  spi_xfer_bytes = sSpiTxPayloadSize;
594  spi_header_set_data_len(sSpiTxFrameBuffer, sSpiTxPayloadSize);
595 
596  } else {
597  spi_xfer_bytes = 0;
598  }
599 
600  if ( (slave_data_len != 0)
601  && (sSpiRxPayloadSize == 0)
602  )
603  {
604  spi_header_set_accept_len(sSpiTxFrameBuffer, slave_data_len);
605  if (slave_data_len > spi_xfer_bytes)
606  {
607  spi_xfer_bytes = slave_data_len;
608  }
609  }
610 
611  // Optionally delay a short period to give
612  // the slave time to get its affairs in order.
613  usleep((unsigned int)sSpiTransactionDelay);
614 
615  // Perform the second SPI transaction.
616  ret = do_spi_xfer(spi_xfer_bytes);
617  if (ret < 0)
618  {
619  perror("do_spi_xfer");
620 
621  // Print out a helpful error message for
622  // a common error.
623  if ( (sSpiCsDelay != 0)
624  && (errno == EINVAL)
625  ) {
626  syslog(LOG_ERR, "SPI ioctl failed with EINVAL. Try adding `--spi-cs-delay=0` to command line arguments.");
627  }
628  goto bail;
629  }
630 
631  // Account for misalignment (0xFF bytes at the start)
632  spiRxFrameBuffer = get_real_rx_frame_start();
633 
634  debug_spi_header("push_pull_2");
635 
636  slave_header = spi_header_get_flag_byte(spiRxFrameBuffer);
637 
638  if ((slave_header == 0xFF) || (slave_header == 0x00))
639  {
640  // Device is off or in a bad state.
641  sSpiTxFlowControl = true;
642 
643  syslog(LOG_DEBUG, "Discarded frame. (2)");
644  goto bail;
645  }
646 
647  slave_max_rx = spi_header_get_accept_len(spiRxFrameBuffer);
648  slave_data_len = spi_header_get_data_len(spiRxFrameBuffer);
649 
650  if ( (slave_header != SPI_HEADER_PATTERN_VALUE)
651  || (slave_max_rx > MAX_FRAME_SIZE)
652  || (slave_data_len > MAX_FRAME_SIZE)
653  )
654  {
655  sSpiTxFlowControl = true;
656  syslog(
657  LOG_INFO,
658  "Gibberish in header (h:0x%02X, max_rx:0x%04X, data_len:0x%04X) (2)",
659  slave_header,
660  slave_max_rx,
661  slave_data_len
662  );
663  goto bail;
664  }
665 
666  sSpiValidFrameCount++;
667 
668  if ( (sSpiRxPayloadSize == 0)
669  && (slave_data_len <= spi_header_get_accept_len(sSpiTxFrameBuffer))
670  ) {
671  // We have received a packet. Set sSpiRxPayloadSize so that
672  // the packet will eventually get queued up by push_hdlc().
673  sSpiRxPayloadSize = slave_data_len;
674  }
675 
676  if ( (sSpiTxPayloadSize == spi_header_get_data_len(sSpiTxFrameBuffer))
677  && (spi_header_get_data_len(sSpiTxFrameBuffer) <= slave_max_rx)
678  ) {

791 static int push_hdlc(void)
792 {
793  int ret = 0;
794  const uint8_t* spiRxFrameBuffer = get_real_rx_frame_start();
795  static uint8_t escaped_frame_buffer[MAX_FRAME_SIZE*2];
796  static uint16_t escaped_frame_len;
797  static uint16_t escaped_frame_sent;
798 
799  if (escaped_frame_len == 0)
800  {
801  if (sSlaveDidReset)
802  {
803  // Indicate an MCU reset.
804  memcpy(escaped_frame_buffer, kHdlcResetSignal, sizeof(kHdlcResetSignal));
805  escaped_frame_len = sizeof(kHdlcResetSignal);
806  sSlaveDidReset = false;
807  }
808  else if (sSpiRxPayloadSize != 0)
809  {
810  // Escape the frame.
811  uint8_t c;
812  uint16_t fcs = kHdlcCrcResetValue;
813  uint16_t i;
814 
815  for (i = 0; i < sSpiRxPayloadSize; i++)
816  {
817  c = spiRxFrameBuffer[i + HEADER_LEN];
818  fcs = hdlc_crc16(fcs, c);
819  if (hdlc_byte_needs_escape(c))
820  {
821  escaped_frame_buffer[escaped_frame_len++] = HDLC_BYTE_ESC;
822  escaped_frame_buffer[escaped_frame_len++] = c ^ HDLC_ESCAPE_XFORM;
823  }
824  else
825  {
826  escaped_frame_buffer[escaped_frame_len++] = c;
827  }
828  }
829 
830  fcs ^= 0xFFFF;
831 
832  c = fcs & 0xFF;
833  if (hdlc_byte_needs_escape(c))
834  {
835  escaped_frame_buffer[escaped_frame_len++] = HDLC_BYTE_ESC;
836  escaped_frame_buffer[escaped_frame_len++] = c ^ HDLC_ESCAPE_XFORM;
837  }
838  else
839  {
840  escaped_frame_buffer[escaped_frame_len++] = c;
841  }
842 
843  c = (fcs >> 8) & 0xFF;
844  if (hdlc_byte_needs_escape(c))
845  {
846  escaped_frame_buffer[escaped_frame_len++] = HDLC_BYTE_ESC;
847  escaped_frame_buffer[escaped_frame_len++] = c ^ HDLC_ESCAPE_XFORM;
848  }
849  else
850  {
851  escaped_frame_buffer[escaped_frame_len++] = c;
852  }
853 
854  escaped_frame_buffer[escaped_frame_len++] = HDLC_BYTE_FLAG;
855  escaped_frame_sent = 0;
856  sSpiRxPayloadSize = 0;
857 
858  }
859  else
860  {
861  // Nothing to do.
862  goto bail;
863  }
864  }
865 
866  ret = (int)write(
867  sHdlcOutputFd,
868  escaped_frame_buffer + escaped_frame_sent,
869  escaped_frame_len - escaped_frame_sent
870  );
871 
872  if (ret < 0)
873  {
874  if (errno == EAGAIN)
875  {
876  ret = 0;
877  }
878  else
879  {
880  perror("push_hdlc:write");
881  syslog(LOG_ERR, "push_hdlc:write: errno=%d (%s)", errno, strerror(errno));
882  }
883  goto bail;
884  }
885 
886  escaped_frame_sent += ret;
887 
888  // Reset state once we have sent the entire frame.
889  if (escaped_frame_len == escaped_frame_sent)
890  {
891  escaped_frame_len = escaped_frame_sent = 0;
892  }
893 
894  ret = 0;
895 
896 bail:
897  return ret;
898 }

900 static int pull_hdlc(void)
901 {
902  int ret = 0;
903  static uint16_t fcs;
904  static bool unescape_next_byte = false;
905 
906  if (!sSpiTxIsReady)
907  {
908  uint8_t byte;
909  while ((ret = (int)read(sHdlcInputFd, &byte, 1)) == 1)
910  {
911  if (sSpiTxPayloadSize >= (MAX_FRAME_SIZE - HEADER_LEN))
912  {
913  syslog(LOG_WARNING, "HDLC frame was too big");
914  unescape_next_byte = false;
915  sSpiTxPayloadSize = 0;
916  fcs = kHdlcCrcResetValue;
917 
918  }
919  else if (byte == HDLC_BYTE_FLAG)
920  {
921  if (sSpiTxPayloadSize <= 2)
922  {
923  unescape_next_byte = false;
924  sSpiTxPayloadSize = 0;
925  fcs = kHdlcCrcResetValue;
926  continue;
927 
928  }
929  else if (fcs != kHdlcCrcCheckValue)
930  {
931  syslog(LOG_WARNING, "HDLC frame with bad CRC (LEN:%d, FCS:0x%04X)", sSpiTxPayloadSize, fcs);
932  unescape_next_byte = false;
933  sSpiTxPayloadSize = 0;
934  fcs = kHdlcCrcResetValue;
935  continue;
936  }
937 
938  // Clip off the CRC
939  sSpiTxPayloadSize -= 2;
940 
941  // Indicate that a frame is ready to go out
942  sSpiTxIsReady = true;
943 
944  // Clean up for the next frame
945  unescape_next_byte = false;
946  fcs = kHdlcCrcResetValue;
947  break;
948 
949  }
950  else if (byte == HDLC_BYTE_ESC)
951  {
952  unescape_next_byte = true;
953  continue;
954 
955  }
956  else if (hdlc_byte_needs_escape(byte))
957  {
958  // Skip all other control codes.
959  continue;
960 
961  }
962  else if (unescape_next_byte)
963  {
964  byte = byte ^ HDLC_ESCAPE_XFORM;
965  unescape_next_byte = false;
966  }
967 
968  fcs = hdlc_crc16(fcs, byte);
969  sSpiTxFrameBuffer[HEADER_LEN + sSpiTxPayloadSize++] = byte;
970  }
971  }
972 
973  if (ret < 0)
974  {
975  if (errno == EAGAIN)
976  {
977  ret = 0;
978  }
979  else
980  {
981  perror("pull_hdlc:read");
982  syslog(LOG_ERR, "pull_hdlc:read: errno=%d (%s)", errno, strerror(errno));
983  }
984  }
985 
986  return ret < 0
987  ? ret
988  : 0;
989 }
