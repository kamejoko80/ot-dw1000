64 ThreadError Dhcp6Server::UpdateService(void)
65 {
66  ThreadError error = kThreadError_None;
67  bool found;
68  uint8_t i;
69  uint16_t rloc16 = mNetif.GetMle().GetRloc16();
70  Ip6::Address *address = NULL;
71  otNetworkDataIterator iterator;
72  otBorderRouterConfig config;
73  Lowpan::Context lowpanContext;
74 
75  // remove dhcp agent aloc and prefix delegation
76  for (i = 0; i < OPENTHREAD_CONFIG_NUM_DHCP_PREFIXES; i++)
77  {
78  found = false;
79 
80  if (!mAgentsAloc[i].mValid)
81  {
82  continue;
83  }
84 
85  address = &(mAgentsAloc[i].GetAddress());
86  iterator = OT_NETWORK_DATA_ITERATOR_INIT;
87 
88  while (mNetif.GetNetworkDataLeader().GetNextOnMeshPrefix(&iterator, rloc16, &config) == kThreadError_None)
89  {
90  if (!config.mDhcp)
91  {
92  continue;
93  }
94 
95  mNetif.GetNetworkDataLeader().GetContext(*static_cast<const Ip6::Address *>(&(config.mPrefix.mPrefix)),
96  lowpanContext);
97 
98  if (address->mFields.m8[15] == lowpanContext.mContextId)
99  {
100  // still in network data
101  found = true;
102  break;
103  }
104  }
105 
106  if (!found)
107  {
108  mNetif.GetNetworkDataLeader().GetContext(address->mFields.m8[15], lowpanContext);
109  mNetif.RemoveUnicastAddress(mAgentsAloc[i]);
110  mAgentsAloc[i].mValid = false;
111  RemovePrefixAgent(lowpanContext.mPrefix);
112  }
113  }
114 
115  // add dhcp agent aloc and prefix delegation
116  iterator = OT_NETWORK_DATA_ITERATOR_INIT;
117 
118  while (mNetif.GetNetworkDataLeader().GetNextOnMeshPrefix(&iterator, rloc16, &config) == kThreadError_None)
119  {
120  found = false;
121 
122  if (!config.mDhcp)
123  {
124  continue;
125  }
126 
127  mNetif.GetNetworkDataLeader().GetContext(*static_cast<const Ip6::Address *>(&config.mPrefix.mPrefix),
128  lowpanContext);
129 
130  for (i = 0; i < OPENTHREAD_CONFIG_NUM_DHCP_PREFIXES; i++)
131  {
132  address = &(mAgentsAloc[i].GetAddress());
133 
134  if ((mAgentsAloc[i].mValid) && (address->mFields.m8[15] == lowpanContext.mContextId))
135  {
136  found = true;
137  break;
138  }
139  }
140 
141  // alreay added
142  if (found)
143  {
144  continue;
145  }
146 
147  for (i = 0; i < OPENTHREAD_CONFIG_NUM_DHCP_PREFIXES; i++)
148  {
149  if (!mAgentsAloc[i].mValid)
150  {
151  address = &(mAgentsAloc[i].GetAddress());
152  memcpy(address, mNetif.GetMle().GetMeshLocalPrefix(), 8);
153  address->mFields.m16[4] = HostSwap16(0x0000);
154  address->mFields.m16[5] = HostSwap16(0x00ff);
155  address->mFields.m16[6] = HostSwap16(0xfe00);
156  address->mFields.m8[14] = Mle::kAloc16Mask;
157  address->mFields.m8[15] = lowpanContext.mContextId;
158  mAgentsAloc[i].mPrefixLength = 128;
159  mAgentsAloc[i].mPreferred = true;
160  mAgentsAloc[i].mValid = true;
161  mNetif.AddUnicastAddress(mAgentsAloc[i]);
162  AddPrefixAgent(config.mPrefix);
163  break;
164  }
165  }
166 
167  // if no available Dhcp Agent Aloc resources
168  if (i == OPENTHREAD_CONFIG_NUM_DHCP_PREFIXES)
169  {
170  ExitNow(error = kThreadError_NoBufs);
171  }
172  }
173 
174  if (mPrefixAgentsCount > 0)
175  {
176  Start();
177  }
178  else
179  {
180  Stop();
181  }
182 
183 exit:
184  return error;
185 }

272 void Dhcp6Server::ProcessSolicit(Message &aMessage, otIp6Address &aDst, uint8_t *aTransactionId)
273 {
274  IaNa iana;
275  ClientIdentifier clientIdentifier;
276  uint16_t optionOffset;
277  uint16_t offset = aMessage.GetOffset();
278  uint16_t length = aMessage.GetLength() - aMessage.GetOffset();
279 
280  // Client Identifier (discard if not present)
281  VerifyOrExit((optionOffset = FindOption(aMessage, offset, length, kOptionClientIdentifier)) > 0, ;);
282  SuccessOrExit(ProcessClientIdentifier(aMessage, optionOffset, clientIdentifier));
283 
284  // Server Identifier (assuming Rapid Commit, discard if present)
285  VerifyOrExit(FindOption(aMessage, offset, length, kOptionServerIdentifier) == 0, ;);
286 
287  // Rapid Commit (assuming Rapid Commit, discard if not present)
288  VerifyOrExit(FindOption(aMessage, offset, length, kOptionRapidCommit) > 0, ;);
289 
290  // Elapsed Time if present
291  if ((optionOffset = FindOption(aMessage, offset, length, kOptionElapsedTime)) > 0)
292  {
293  SuccessOrExit(ProcessElapsedTime(aMessage, optionOffset));
294  }
295 
296  // IA_NA (discard if not present)
297  VerifyOrExit((optionOffset = FindOption(aMessage, offset, length, kOptionIaNa)) > 0, ;);
298  SuccessOrExit(ProcessIaNa(aMessage, optionOffset, iana));
299 
300  SuccessOrExit(SendReply(aDst, aTransactionId, clientIdentifier, iana));
301 
302 exit:
303  {}
304 }

354  uint16_t optionOffset;
355  uint16_t length;

379 ThreadError Dhcp6Server::ProcessIaAddress(Message &aMessage, uint16_t aOffset)
380 {
381  ThreadError error = kThreadError_None;
382  otIp6Prefix *prefix = NULL;
383  IaAddress option;
384 
385  VerifyOrExit(((aMessage.Read(aOffset, sizeof(option), &option) == sizeof(option)) &&
386  option.GetLength() == (sizeof(option) - sizeof(Dhcp6Option))),
387  error = kThreadError_Parse);
388 
389  // mask matching prefix
390  for (uint8_t i = 0; i < OPENTHREAD_CONFIG_NUM_DHCP_PREFIXES; i++)

410 ThreadError Dhcp6Server::SendReply(otIp6Address &aDst, uint8_t *aTransactionId, ClientIdentifier &aClient, IaNa &aIaNa)
411 {
412  ThreadError error = kThreadError_None;
413  Ip6::MessageInfo messageInfo;
414  Message *message;
415 
416  VerifyOrExit((message = mSocket.NewMessage(0)) != NULL, error = kThreadError_NoBufs);
417  SuccessOrExit(error = AppendHeader(*message, aTransactionId));
418  SuccessOrExit(error = AppendServerIdentifier(*message));
419  SuccessOrExit(error = AppendClientIdentifier(*message, aClient));
420  SuccessOrExit(error = AppendIaNa(*message, aIaNa));
421  SuccessOrExit(error = AppendStatusCode(*message, kStatusSuccess));
422  SuccessOrExit(error = AppendIaAddress(*message, aClient));
423  SuccessOrExit(error = AppendRapidCommit(*message));
424 
425  memset(&messageInfo, 0, sizeof(messageInfo));
426  memcpy(&messageInfo.GetPeerAddr().mFields.m8, &aDst, sizeof(otIp6Address));
427  messageInfo.mPeerPort = kDhcpClientPort;
428  SuccessOrExit(error = mSocket.SendTo(*message, messageInfo));
429 
430 exit:
431 
432  if (message != NULL && error != kThreadError_None)
433  {
434  message->Free();
435  }
436 
437  return error;
438 }
