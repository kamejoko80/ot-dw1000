101 NDIS_STATUS RequestQuery( _In_ PNDIS_OID_REQUEST OidRequest, const _Datum & value )
102 {
103  PAGED_CODE();
104  NT_ASSERT( (OidRequest->RequestType == NdisRequestQueryInformation) ||
105  (OidRequest->RequestType == NdisRequestQueryStatistics) );
106 
107  if (OidRequest->DATA.QUERY_INFORMATION.InformationBufferLength < sizeof( _Datum ))
108  {
109  OidRequest->DATA.QUERY_INFORMATION.BytesNeeded = sizeof( _Datum );
110  OidRequest->DATA.QUERY_INFORMATION.BytesWritten = 0;
111  return NDIS_STATUS_INVALID_LENGTH;
112  }
113  else
114  {
115  OidRequest->DATA.QUERY_INFORMATION.BytesNeeded = sizeof( _Datum );
116  OidRequest->DATA.QUERY_INFORMATION.BytesWritten = sizeof( _Datum );
117  _Datum * pData = reinterpret_cast<_Datum *>(OidRequest->DATA.QUERY_INFORMATION.InformationBuffer);
118  *pData = value;
119  return NDIS_STATUS_SUCCESS;
120  }
121 }

179 NDIS_STATUS RequestQueryGenStatistics(_In_ PNDIS_OID_REQUEST OidRequest, _In_ POTTMP_ADAPTER_CONTEXT AdapterContext)
180 {
181  PAGED_CODE();
182 
183  NDIS_STATISTICS_INFO statistics = { 0 };
184  ASSIGN_NDIS_OBJECT_HEADER(statistics.Header, NDIS_OBJECT_TYPE_DEFAULT, NDIS_SIZEOF_STATISTICS_INFO_REVISION_1, NDIS_STATISTICS_INFO_REVISION_1);
185 
186  statistics.SupportedStatistics = NIC_SUPPORTED_STATISTICS;
187 
188  /* Bytes in */
189  statistics.ifHCInOctets = AdapterContext->BytesRxDirected +
190  AdapterContext->BytesRxMulticast +
191  AdapterContext->BytesRxBroadcast;
192 
193  statistics.ifHCInUcastOctets = AdapterContext->BytesRxDirected;
194  statistics.ifHCInMulticastOctets = AdapterContext->BytesRxMulticast;
195  statistics.ifHCInBroadcastOctets = AdapterContext->BytesRxBroadcast;
196 
197  /* Packets in */
198  statistics.ifHCInUcastPkts = AdapterContext->FramesRxDirected;
199  statistics.ifHCInMulticastPkts = AdapterContext->FramesRxMulticast;
200  statistics.ifHCInBroadcastPkts = AdapterContext->FramesRxBroadcast;
201 
202  /* Errors in */
203  statistics.ifInErrors = AdapterContext->RxCrcErrors +
204  AdapterContext->RxAlignmentErrors +
205  AdapterContext->RxDmaOverrunErrors +
206  AdapterContext->RxRuntErrors;
207 
208  statistics.ifInDiscards = AdapterContext->RxResourceErrors;
209 
210  /* Bytes out */
211  statistics.ifHCOutOctets = AdapterContext->BytesTxDirected +
212  AdapterContext->BytesTxMulticast +
213  AdapterContext->BytesTxBroadcast;
214 
215  statistics.ifHCOutUcastOctets = AdapterContext->BytesTxDirected;
216  statistics.ifHCOutMulticastOctets = AdapterContext->BytesTxMulticast;
217  statistics.ifHCOutBroadcastOctets = AdapterContext->BytesTxBroadcast;
218 
219  /* Packets out */
220  statistics.ifHCOutUcastPkts = AdapterContext->FramesTxDirected;
221  statistics.ifHCOutMulticastPkts = AdapterContext->FramesTxMulticast;
222  statistics.ifHCOutBroadcastPkts = AdapterContext->FramesTxBroadcast;
223 
224  /* Errors out */
225  statistics.ifOutErrors = AdapterContext->TxAbortExcessCollisions +
226  AdapterContext->TxDmaUnderrun +
227  AdapterContext->TxLostCRS +
228  AdapterContext->TxLateCollisions+
229  AdapterContext->TransmitFailuresOther;
230 
231  statistics.ifOutDiscards = 0ULL;
232 
233  return RequestQuery(OidRequest, statistics);
234 }

238 NDIS_STATUS
239 MPOidRequest(
240  _In_ NDIS_HANDLE MiniportAdapterContext,
241  _In_ PNDIS_OID_REQUEST OidRequest
242  )
243 {
244  POTTMP_ADAPTER_CONTEXT AdapterContext = (POTTMP_ADAPTER_CONTEXT)MiniportAdapterContext;
245  NDIS_STATUS status = NDIS_STATUS_NOT_SUPPORTED;
246  bool fFailExpected = false;
247 
248  LogFuncEntry(DRIVER_DEFAULT);
249 
250  PAGED_CODE();
251 
252  switch (OidRequest->RequestType)
253  {
254  case NdisRequestSetInformation:
255  switch (OidRequest->DATA.QUERY_INFORMATION.Oid)
256  {
257  case OID_802_3_MULTICAST_LIST:
258  // TODO: Check with Jeffrey how to better handle this
259  status = NDIS_STATUS_MULTICAST_FULL;
260  fFailExpected = true;
261  break;
262 
263  //
264  // Fake it until we make it :)
265  // We can't bind unless we report success for these OIDs
266  //
267  case OID_GEN_CURRENT_PACKET_FILTER:
268  case OID_PM_ADD_WOL_PATTERN:
269  case OID_PM_REMOVE_WOL_PATTERN:
270  case OID_GEN_CURRENT_LOOKAHEAD:
271  // TODO: Implement this
272  status = NDIS_STATUS_SUCCESS;
273  break;
274 
275  // Explicitly not supported
276  case OID_GEN_INTERRUPT_MODERATION:
277  status = NDIS_STATUS_NOT_SUPPORTED;
278  fFailExpected = true;
279  break;
280 
281  // Unknown
282  default:
283  status = NDIS_STATUS_NOT_SUPPORTED;
284  break;
285  }
286  break;
287  case NdisRequestQueryInformation:
288  case NdisRequestQueryStatistics:
289  switch (OidRequest->DATA.QUERY_INFORMATION.Oid)
290  {
291  case OID_GEN_INTERRUPT_MODERATION:
292  {
293  static const NDIS_INTERRUPT_MODERATION_PARAMETERS nimp = {
294  { NDIS_OBJECT_TYPE_DEFAULT, NDIS_INTERRUPT_MODERATION_PARAMETERS_REVISION_1, NDIS_SIZEOF_INTERRUPT_MODERATION_PARAMETERS_REVISION_1 },
295  0,
296  NdisInterruptModerationNotSupported };
297  status = RequestQuery( OidRequest, nimp );
298  break;
299  }
300 
301  case OID_GEN_RCV_OK:
302  status = RequestQuery32or64( OidRequest, AdapterContext->FramesRxBroadcast
303  + AdapterContext->FramesRxMulticast
304  + AdapterContext->FramesRxDirected );
305  break;
306 
307  case OID_GEN_MAXIMUM_TOTAL_SIZE:
308  case OID_GEN_TRANSMIT_BLOCK_SIZE:
309  case OID_GEN_RECEIVE_BLOCK_SIZE:
310  status = RequestQuery<ULONG>( OidRequest, HW_MAX_FRAME_SIZE );
311  break;
312 
313  case OID_GEN_RECEIVE_BUFFER_SPACE:
314  status = RequestQuery<ULONG>( OidRequest, HW_MAX_FRAME_SIZE * AdapterContext->ulMaxBusyRecvs );
315  break;
316 
317  case OID_GEN_STATISTICS:
318  status = RequestQueryGenStatistics(OidRequest, AdapterContext);
319  break;
320 
321  case OID_GEN_TRANSMIT_BUFFER_SPACE:
322  status = RequestQuery<ULONG>( OidRequest, HW_MAX_FRAME_SIZE * AdapterContext->ulMaxBusySends );
323  break;
324 
325  case OID_GEN_VENDOR_DESCRIPTION:
326  case OID_GEN_VENDOR_DRIVER_VERSION:
327  case OID_GEN_VENDOR_ID:
328  if (OidRequest->DATA.QUERY_INFORMATION.InformationBufferLength < sizeof(NIC_VENDOR_DESC))
329  {
330  OidRequest->DATA.QUERY_INFORMATION.BytesNeeded = sizeof( NIC_VENDOR_DESC );
331  OidRequest->DATA.QUERY_INFORMATION.BytesWritten = 0;
332  status = NDIS_STATUS_INVALID_LENGTH;
333  }
334  else
335  {
336  OidRequest->DATA.QUERY_INFORMATION.BytesNeeded = sizeof( NIC_VENDOR_DESC );
337  OidRequest->DATA.QUERY_INFORMATION.BytesWritten = sizeof( NIC_VENDOR_DESC );
338  memcpy(OidRequest->DATA.QUERY_INFORMATION.InformationBuffer, NIC_VENDOR_DESC, sizeof(NIC_VENDOR_DESC));
339  status = NDIS_STATUS_SUCCESS;
340  }
341  break;
342 
343  case OID_GEN_XMIT_OK:
344  status = RequestQuery32or64( OidRequest, AdapterContext->FramesTxBroadcast
345  + AdapterContext->FramesTxMulticast
346  + AdapterContext->FramesTxDirected );
347  break;
348 
349  case OID_802_3_CURRENT_ADDRESS:
350  status = RequestQuery( OidRequest, AdapterContext->ExtendedAddress );
351  break;
352 
353  case OID_802_3_PERMANENT_ADDRESS:
354  status = RequestQuery( OidRequest, AdapterContext->ExtendedAddress );
355  break;
356 
357  case OID_PNP_CAPABILITIES:
358  // We do not support low power
359  status = NDIS_STATUS_NOT_SUPPORTED;
360  fFailExpected = true;
361  break;
362 
363  case OID_PNP_QUERY_POWER:
364  status = NDIS_STATUS_NOT_ACCEPTED;
365  fFailExpected = true;
366  break;
367 
368  case OID_OT_CAPABILITIES:
369  status = RequestQueryThreadCapabilities( OidRequest );
370  break;
371 
372  default:
373  status = NDIS_STATUS_NOT_SUPPORTED;
374  break;
375  }
376  break;
377  case NdisRequestMethod:
378  break;
379  default:
380  status = NDIS_STATUS_INVALID_OID;
381  break;
382 
383  }
384  UNREFERENCED_PARAMETER( OidRequest );
385 
386  if (!fFailExpected && (status != NDIS_STATUS_SUCCESS))
387  {
388  LogFuncExitMsg(DRIVER_DEFAULT, " Type:%u Oid:%u Status:%!NDIS_STATUS!", OidRequest->RequestType, OidRequest->DATA.QUERY_INFORMATION.Oid, status);
389  }
390  else
391  {
392  // By not using LogFuncExitNDIS, this won't get auto-promoted to a warning on non-0 statuses
393  LogFuncExitMsg(DRIVER_DEFAULT, " Type:%u Oid:%u Status:%!NDIS_STATUS!", OidRequest->RequestType, OidRequest->DATA.QUERY_INFORMATION.Oid, status);
394  }
395  return status;
396 }
