434 #if QUAD_MODE

436  if (flash_config->quad_page_program_address == false)
437  {
438  qspi_set_bus_mode(HW_QSPI_BUS_MODE_QUAD);
439  }

1091

1102

1104

1106  {

1113  }
1114  }

1117  flash_config = &flash_autodetect_config;
1118 #else

1120  device_density = dg_configFLASH_DENSITY;
1121 #endif
1122 
1123  /*
1124  * Copy the selected flash struct from flash into retram
1125  */

1127

1129

1133  qspi_cfg.address_size = HW_QSPI_ADDR_SIZE_32; // It will be set when hw_qspi_init() is called
1134  }
1135  else
1136  {
1137  read_opcode = CMD_FAST_READ_QUAD;
1138  }
1139 
1140  /*
1141  * Setup erase instruction that will be sent by QSPI controller to erase sector in automode.
1142  */

1161  HW_QSPI_BUS_MODE_SINGLE, flash_config->erase_in_progress_bit, flash_config->erase_in_progress_bit_high_level ? 1 : 0,

1181  }
1182 
1183 #ifdef ENABLE_AUTOMODE_CONFIGURATION_FOR_READ
1184 
1185  flash_activate_command_entry_mode();
1186 
1187  hw_qspi_set_read_instruction(read_opcode, flash_config->send_once, flash_config->get_dummy_bytes(),
1188  HW_QSPI_BUS_MODE_SINGLE,
1189  HW_QSPI_BUS_MODE_QUAD, HW_QSPI_BUS_MODE_QUAD,
1190  HW_QSPI_BUS_MODE_QUAD);
1191  hw_qspi_set_extra_byte(flash_config->extra_byte, HW_QSPI_BUS_MODE_QUAD, 0);
1192  hw_qspi_set_address_size(flash_config->address_size);
1193 
1194  flash_deactivate_command_entry_mode();
1195 
1196  HW_QSPIC_REG_SETF(BURSTCMDB, CS_HIGH_MIN, 0);
1197 #else

1199 #endif
