61  uint32_t ui32StopCondition)
62 {
63  //
64  // Check the arguments.
65  //
66  ASSERT(AUXTDCBaseValid(ui32Base));

68  //
69  // Make sure the AUX TDC is in the idle state before changing the
70  // configuration.
71  //
72  while(!((HWREG(ui32Base + AUX_TDC_O_STAT) & AUX_TDC_STAT_STATE_M) ==
73  AUX_TDC_STAT_STATE_IDLE))
74  {
75  }
76 
77  //
78  // Clear previous results.
79  //
80  HWREG(ui32Base + AUX_TDC_O_CTL) = 0x0;
81 
82  //
83  // Change the configuration.
84  //
85  HWREG(ui32Base + AUX_TDC_O_TRIGSRC) = ui32StartCondition | ui32StopCondition;
86 }
