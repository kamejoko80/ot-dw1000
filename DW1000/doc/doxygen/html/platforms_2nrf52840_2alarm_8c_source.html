73 static inline uint32_t TimeToTicks(AlarmTime aTime)
74 {
75  uint64_t microseconds = US_PER_MS * (uint64_t)aTime.ms + (uint64_t)aTime.us;
76  return (uint32_t)((microseconds * RTC_FREQUENCY) / US_PER_S) & RTC_CC_COMPARE_Msk;
77 }

79 static inline AlarmTime TicksToTime(uint32_t aTicks)
80 {
81  uint64_t microseconds = (US_PER_S * (uint64_t)aTicks) / RTC_FREQUENCY;
82  return (AlarmTime) {microseconds / US_PER_MS, microseconds % US_PER_MS};
83 }

85 static inline AlarmTime TimeAdd(AlarmTime aTime1, AlarmTime aTime2)
86 {
87  AlarmTime result = { aTime1.ms + aTime2.ms, aTime1.us + aTime2.us };
88 
89  assert(result.us < 2 * US_PER_MS);
90 
91  if (result.us >= US_PER_MS)
92  {
93  result.ms++;
94  result.us -= US_PER_MS;
95  }
96 
97  return result;
98 }

100 static inline AlarmTime AlarmGetCurrentTime(void)
101 {
102  uint32_t rtcValue1;
103  uint32_t rtcValue2;
104  AlarmTime offset;
105 
106  rtcValue1 = nrf_rtc_counter_get(RTC_INSTANCE);
107 
108  __DMB();
109 
110  offset = sTimeOffset;
111 
112  __DMB();
113 
114  rtcValue2 = nrf_rtc_counter_get(RTC_INSTANCE);
115 
116  if ((rtcValue2 < rtcValue1) || (rtcValue1 == 0))
117  {
118  // Overflow detected. Additional condition (rtcValue1 == 0) covers situation when overflow occurred in
119  // interrupt state, before this function was entered. But in general, this function shall not be called
120  // from interrupt other than alarm interrupt.
121 
122  // Wait at least 20 cycles, to ensure that if interrupt is going to be called, it will be called now.
123  for (uint32_t i = 0; i < 4; i++)
124  {
125  __NOP();
126  __NOP();
127  __NOP();
128  }
129 
130  // If the event flag is still on, it means that the interrupt was not called, as we are in interrupt state.
131  if (nrf_rtc_event_pending(RTC_INSTANCE, NRF_RTC_EVENT_OVERFLOW))
132  {
133  HandleOverflow();
134  }
135 
136  offset = sTimeOffset;
137  }
138 
139  return TimeAdd(offset, TicksToTime(rtcValue2));
140 }

153 static inline bool AlarmShallStrike(AlarmTime aNow)
154 {
155  if (TimeIsLower(sTargetTime, sT0Time))
156  {
157  // Handle situation when timespan included timer overflow.
158  if (TimeIsLower(aNow, sT0Time) && !TimeIsLower(aNow, sTargetTime))
159  {
160  return true;
161  }
162  }
163  else
164  {
165  if (TimeIsLower(aNow, sT0Time) || !TimeIsLower(aNow, sTargetTime))
166  {
167  return true;
168  }
169  }
170 
171  return false;
172 }

174 static void HandleCompare0Match(void)
175 {
176  nrf_rtc_event_clear(RTC_INSTANCE, NRF_RTC_EVENT_COMPARE_0);
177 
178  AlarmTime now = AlarmGetCurrentTimeRtcProtected();
179 
180  // In case the target time was larger than single overflow,
181  // we should only strike the timer on final compare event.
182  if (AlarmShallStrike(now))
183  {
184  nrf_rtc_event_disable(RTC_INSTANCE, RTC_EVTEN_COMPARE0_Msk);
185  nrf_rtc_int_disable(RTC_INSTANCE, NRF_RTC_INT_COMPARE0_MASK);
186 
187  sFireAlarm = true;
188  }
189 }

192 {
193  nrf_rtc_event_clear(RTC_INSTANCE, NRF_RTC_EVENT_OVERFLOW);

195  // Increment counter on overflow.

256 void otPlatAlarmStartAt(otInstance *aInstance, uint32_t aT0, uint32_t aDt)
257 {
258  (void)aInstance;
259  uint32_t targetCounter;
260  AlarmTime now;
261 
262  nrf_rtc_int_disable(RTC_INSTANCE, NRF_RTC_INT_COMPARE0_MASK);
263  nrf_rtc_event_enable(RTC_INSTANCE, RTC_EVTEN_COMPARE0_Msk);
264 
265  sT0Time = (AlarmTime) {aT0, 0};
266  sTargetTime = (AlarmTime) {aT0 + aDt, 0};
267 
268  targetCounter = TimeToTicks(sTargetTime);
269 
270  nrf_rtc_cc_set(RTC_INSTANCE, 0, targetCounter);
271 
272  now = AlarmGetCurrentTimeRtcProtected();
273 
274  if (AlarmShallStrike(now))
275  {
276  HandleCompare0Match();
277  }
278  else
279  {
280  nrf_rtc_int_enable(RTC_INSTANCE, NRF_RTC_INT_COMPARE0_MASK);
