103 static void setPendingEvent(RadioPendingEvents aEvent)
104 {
105  volatile uint32_t pendingEvents;
106  uint32_t bitToSet = 1UL << aEvent;
107 
108  do
109  {
110  pendingEvents = __LDREXW(&sPendingEvents);
111  pendingEvents |= bitToSet;
112  }
113  while (__STREXW(pendingEvents, &sPendingEvents));
114 }

116 static void resetPendingEvent(RadioPendingEvents aEvent)
117 {

129 static inline void clearPendingEvents(void)
130 {
131  // Clear pending events that could cause race in the MAC layer.
132  volatile uint32_t pendingEvents;
133  uint32_t bitsToRemain = ~(0UL);
134 
135  bitsToRemain &= ~(1UL << kPendingEventFrameTransmitted);
136  bitsToRemain &= ~(1UL << kPendingEventChannelAccessFailure);
137 
138  bitsToRemain &= ~(1UL << kPendingEventSleep);
139 
140  do
141  {
142  pendingEvents = __LDREXW(&sPendingEvents);
143  pendingEvents &= bitsToRemain;
144  }
145  while (__STREXW(pendingEvents, &sPendingEvents));
146 }

437 void nrf5RadioProcess(otInstance *aInstance)
438 {
439  for (uint32_t i = 0; i < RADIO_RX_BUFFERS; i++)
440  {
441  if (sReceivedFrames[i].mPsdu != NULL)
442  {
443 #if OPENTHREAD_ENABLE_DIAG
444 
445  if (otPlatDiagModeGet())
446  {
447  otPlatDiagRadioReceiveDone(aInstance, &sReceivedFrames[i], kThreadError_None);
448  }
449  else
450 #endif
451  {
452  otPlatRadioReceiveDone(aInstance, &sReceivedFrames[i], kThreadError_None);
453  }
454 
455  uint8_t *bufferAddress = &sReceivedFrames[i].mPsdu[-1];
456  sReceivedFrames[i].mPsdu = NULL;
457  nrf_drv_radio802154_buffer_free(bufferAddress);
458  }
459  }
460 
461  if (isPendingEventSet(kPendingEventFrameTransmitted))
462  {
463 #if OPENTHREAD_ENABLE_DIAG
464 
465  if (otPlatDiagModeGet())
466  {
467  otPlatDiagRadioTransmitDone(aInstance, &sTransmitFrame, sTransmitPendingBit, kThreadError_None);
468  }
469  else
470 #endif
471  {
472  otPlatRadioTransmitDone(aInstance, &sTransmitFrame, sTransmitPendingBit, kThreadError_None);
473  }
474 
475  resetPendingEvent(kPendingEventFrameTransmitted);
476  }
477 
478  if (isPendingEventSet(kPendingEventChannelAccessFailure))
479  {
480 #if OPENTHREAD_ENABLE_DIAG
481 
482  if (otPlatDiagModeGet())
483  {
484  otPlatDiagRadioTransmitDone(aInstance, &sTransmitFrame, false, kThreadError_ChannelAccessFailure);
485  }
486  else
487 #endif
488  {
489  otPlatRadioTransmitDone(aInstance, &sTransmitFrame, false, kThreadError_ChannelAccessFailure);
490  }
491 
492  resetPendingEvent(kPendingEventChannelAccessFailure);
493  }
494 
495  if (isPendingEventSet(kPendingEventEnergyDetected))
496  {
497  otPlatRadioEnergyScanDone(aInstance, sEnergyDetected);
498 
499  resetPendingEvent(kPendingEventEnergyDetected);
500  }
501 
502  if (isPendingEventSet(kPendingEventSleep))
503  {
504  if (nrf_drv_radio802154_sleep())
505  {
506  resetPendingEvent(kPendingEventSleep);
507  }
508  }
509 
510  if (isPendingEventSet(kPendingEventEnergyDetectionStart))
511  {
512  if (nrf_drv_radio802154_energy_detection(sEnergyDetectionChannel, sEnergyDetectionTime))
513  {
514  resetPendingEvent(kPendingEventEnergyDetectionStart);
515  }
516  }
517 }

538  receivedFrame->mPower = power;
