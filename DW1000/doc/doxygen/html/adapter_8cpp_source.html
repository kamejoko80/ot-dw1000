253 
254 Routine Description:
255 
256  When a miniport receives a restart request, it enters into a Restarting
257  state. The miniport may begin indicating received data (e.g., using
258  NdisMIndicateReceiveNetBufferLists), handling status indications, and
259  processing OID requests in the Restarting state. However, no sends will be
260  requested while the miniport is in the Restarting state.
261 
262  Once the miniport is ready to send data, it has entered the Running state.
263  The miniport informs NDIS that it is in the Running state by returning
264  NDIS_STATUS_SUCCESS from this MiniportRestart function; or if this function
265  has already returned NDIS_STATUS_PENDING, by calling NdisMRestartComplete.
266 
267 
268  MiniportRestart runs at IRQL = PASSIVE_LEVEL.
269 
270 Arguments:
271 
272  MiniportAdapterContext Pointer to the Adapter
273  RestartParameters Additional information about the restart operation
274 
275 Return Value:
276 
277  If the miniport is able to immediately enter the Running state, it should
278  return NDIS_STATUS_SUCCESS.
279 
280  If the miniport is still in the Restarting state, it should return
281  NDIS_STATUS_PENDING now, and call NdisMRestartComplete when the miniport
282  has entered the Running state.
283 
284  Other NDIS_STATUS codes indicate errors. If an error is encountered, the
285  miniport must return to the Paused state (i.e., stop indicating receives).
286 
287 --*/

295  PAGED_CODE();
296 
297  // Set the running flag
298  AdapterContext->IsRunning = true;
299 
300  LogFuncExitNDIS(DRIVER_DEFAULT, status);
301 
302  return status;

313 /*++
314 
315 Routine Description:
316 
317  When a miniport receives a pause request, it enters into a Pausing state.
318  The miniport should not indicate up any more network data. Any pending
319  send requests must be completed, and new requests must be rejected with
320  NDIS_STATUS_PAUSED.
321 
322  Once all sends have been completed and all recieve NBLs have returned to
323  the miniport, the miniport enters the Paused state.
324 
325  While paused, the miniport can still service interrupts from the hardware
326  (to, for example, continue to indicate NDIS_STATUS_MEDIA_CONNECT
327  notifications).
328 
329  The miniport must continue to be able to handle status indications and OID
330  requests. MiniportPause is different from MiniportHalt because, in
331  general, the MiniportPause operation won't release any resources.
332  MiniportPause must not attempt to acquire any resources where allocation
333  can fail, since MiniportPause itself must not fail.
334 
335 
336  MiniportPause runs at IRQL = PASSIVE_LEVEL.
337 
338 Arguments:
339 
340  MiniportAdapterContext Pointer to the Adapter
341  MiniportPauseParameters Additional information about the pause operation
342 
343 Return Value:
344 
345  If the miniport is able to immediately enter the Paused state, it should
346  return NDIS_STATUS_SUCCESS.
347 
348  If the miniport must wait for send completions or pending receive NBLs, it
349  should return NDIS_STATUS_PENDING now, and call NDISMPauseComplete when the
350  miniport has entered the Paused state.
351 
352  No other return value is permitted. The pause operation must not fail.
353 
354 --*/
355 {
356  NDIS_STATUS status = NDIS_STATUS_SUCCESS;
357  POTTMP_ADAPTER_CONTEXT AdapterContext = (POTTMP_ADAPTER_CONTEXT)MiniportAdapterContext;
358 
359  LogFuncEntry(DRIVER_DEFAULT);
360 
361  PAGED_CODE();
362 
363  // Clear the flag to indicate we are no longer running
364  AdapterContext->IsRunning = false;
365 
366  LogFuncExitNDIS(DRIVER_DEFAULT, status);
367 
368  return status;
369 }

372 MPSendNetBufferLists(
373  _In_ NDIS_HANDLE MiniportAdapterContext,
374  _In_ PNET_BUFFER_LIST NetBufferLists,
375  _In_ NDIS_PORT_NUMBER /* PortNumber */,
376  _In_ ULONG SendFlags
377  )
378 /*++
379 
380 Routine Description:
381 
382  Send Packet Array handler. Called by NDIS whenever a protocol
383  bound to our miniport sends one or more packets.
384 
385  The input packet descriptor pointers have been ordered according
386  to the order in which the packets should be sent over the network
387  by the protocol driver that set up the packet array. The NDIS
388  library preserves the protocol-determined ordering when it submits
389  each packet array to MiniportSendPackets
390 
391  As a deserialized driver, we are responsible for holding incoming send
392  packets in our internal queue until they can be transmitted over the
393  network and for preserving the protocol-determined ordering of packet
394  descriptors incoming to its MiniportSendPackets function.
395  A deserialized miniport driver must complete each incoming send packet
396  with NdisMSendComplete, and it cannot call NdisMSendResourcesAvailable.
397 
398  Runs at IRQL <= DISPATCH_LEVEL
399 
400 Arguments:
401 
402  MiniportAdapterContext Pointer to our adapter
403  NetBufferLists Head of a list of NBLs to send
404  PortNumber A miniport adapter port. Default is 0.
405  SendFlags Additional flags for the send operation
406 
407 Return Value:
408 
409  None. Write status directly into each NBL with the NET_BUFFER_LIST_STATUS
410  macro.
411 
412 --*/
413 {
414  POTTMP_ADAPTER_CONTEXT AdapterContext = (POTTMP_ADAPTER_CONTEXT)MiniportAdapterContext;
415  PNET_BUFFER_LIST FailedNbls = NULL;
416 
417  LogFuncEntryMsg(DRIVER_DEFAULT, "NetBufferList: %p", NetBufferLists);
418 
419  PNET_BUFFER_LIST CurrNbl = NetBufferLists;
420  while (CurrNbl)
421  {
422  PNET_BUFFER_LIST NextNbl = CurrNbl->Next;
423  CurrNbl->Next = NULL;
424 
425  // Only allow one NB per NBL
426  if (CurrNbl->FirstNetBuffer == NULL ||
427  CurrNbl->FirstNetBuffer->Next != NULL)
428  {
429  CurrNbl->Status = STATUS_INVALID_PARAMETER;
430  }
431  else
432  {
433  // Try to queue up for send
434  NTSTATUS status = SerialSendData(AdapterContext, CurrNbl);
435 
436  if (!NT_SUCCESS(status)) {
437  CurrNbl->Status = status;
438  }
439  else {
440  CurrNbl = NULL;
441  }
442  }
443 
444  // If we still have the CurrNbl, it failed, so move it to the failure list
445  if (CurrNbl) {
446  CurrNbl->Next = FailedNbls;
447  FailedNbls = CurrNbl;
448  }
449 
450  CurrNbl = NextNbl;
451  }
452 
453  // Complete any failures
454  if (FailedNbls) {
455  NdisMSendNetBufferListsComplete(AdapterContext->Adapter, FailedNbls, (SendFlags & NDIS_SEND_COMPLETE_FLAGS_DISPATCH_LEVEL));
456  }
457 
458  LogFuncExit(DRIVER_DEFAULT);
459 }
