223 __STATIC_INLINE void
224 I2CMasterControl(uint32_t ui32Base, uint32_t ui32Cmd)
225 {
226  //
227  // Check the arguments.
228  //
229  ASSERT(I2CBaseValid(ui32Base));
230  (void)ui32Base;
231  ASSERT((ui32Cmd == I2C_MASTER_CMD_SINGLE_SEND) ||
232  // (ui32Cmd == I2C_MASTER_CMD_SINGLE_RECEIVE) || -> Equal to SINGLE_SEND
233  (ui32Cmd == I2C_MASTER_CMD_BURST_SEND_START) ||
234  (ui32Cmd == I2C_MASTER_CMD_BURST_SEND_CONT) ||
235  (ui32Cmd == I2C_MASTER_CMD_BURST_SEND_FINISH) ||
236  (ui32Cmd == I2C_MASTER_CMD_BURST_SEND_ERROR_STOP) ||
237  (ui32Cmd == I2C_MASTER_CMD_BURST_RECEIVE_START) ||
238  (ui32Cmd == I2C_MASTER_CMD_BURST_RECEIVE_CONT) ||
239  (ui32Cmd == I2C_MASTER_CMD_BURST_RECEIVE_FINISH) ||
240  (ui32Cmd == I2C_MASTER_CMD_BURST_RECEIVE_ERROR_STOP));
241 
242  //
243  // Send the command.
244  //
245  HWREG(I2C0_BASE + I2C_O_MCTRL) = ui32Cmd;
246 
247  //
248  // Delay minimum four cycles in order to ensure that the I2C_O_MSTAT
249  // register has been correctly updated before function exit
250  //
251  CPUdelay(2);
252 }
