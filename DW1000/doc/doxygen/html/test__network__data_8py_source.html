58  data = struct.pack(">HBB", border_router.border_router_16,
59  (border_router.o & 0x01) |
60  ((border_router.r & 0x01) << 1) |
61  ((border_router.c & 0x01) << 2) |
62  ((border_router.d & 0x01) << 3) |
63  ((border_router.s & 0x01) << 4) |
64  ((border_router.p & 0x01) << 5) |
65  ((border_router.prf & 0x03) << 6),
66  ((border_router.n & 0x01) << 7))
67 
68  return data
