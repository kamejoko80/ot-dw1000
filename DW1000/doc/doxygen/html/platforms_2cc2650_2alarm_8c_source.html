61 uint32_t otPlatAlarmGetNow(void)
62 {
63  /*
64  * This is current value of RTC as it appears in the register.
65  * With seconds as the upper 32 bytes and fractions of a second as the
66  * lower 32 bytes <32.32>.
67  */
68  uint64_t rtcVal = AONRTCCurrent64BitValueGet();
69  return ((rtcVal * 1000) >> 32);
70 }

95 void cc2650AlarmProcess(otInstance *aInstance)
96 {
97  uint32_t offsetTime;
98 
99  if (sIsRunning)
100  {
101  /* unsinged subtraction will result in the absolute offset */
102  offsetTime = otPlatAlarmGetNow() - sTime0;
103 
104  if (sAlarmTime <= offsetTime)
105  {
106  sIsRunning = false;
107 #if OPENTHREAD_ENABLE_DIAG
108 
109  if (otPlatDiagModeGet())
110  {
111  otPlatDiagAlarmFired(aInstance);
112  }
113  else
114 #endif /* OPENTHREAD_ENABLE_DIAG */
115  {
116  otPlatAlarmFired(aInstance);
117  }
118  }
119  }
