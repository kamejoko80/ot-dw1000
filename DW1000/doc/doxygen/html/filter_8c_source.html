54 _Use_decl_annotations_
55 NDIS_STATUS
56 FilterAttach(
57  NDIS_HANDLE NdisFilterHandle,
58  NDIS_HANDLE FilterDriverContext,
59  PNDIS_FILTER_ATTACH_PARAMETERS AttachParameters
60  )
61 /*++
62 
63 Routine Description:
64 
65  Filter attach routine.
66  Create filter's context, allocate NetBufferLists and NetBuffer pools and any
67  other resources, and read configuration if needed.
68 
69 Arguments:
70 
71  NdisFilterHandle - Specify a handle identifying this instance of the filter. FilterAttach
72  should save this handle. It is a required parameter in subsequent calls
73  to NdisFxxx functions.
74  FilterDriverContext - Filter driver context passed to NdisFRegisterFilterDriver.
75 
76  AttachParameters - attach parameters
77 
78 Return Value:
79 
80  NDIS_STATUS_SUCCESS: FilterAttach successfully allocated and initialize data structures
81  for this filter instance.
82  NDIS_STATUS_RESOURCES: FilterAttach failed due to insufficient resources.
83  NDIS_STATUS_FAILURE: FilterAttach could not set up this instance of this filter and it has called
84  NdisWriteErrorLogEntry with parameters specifying the reason for failure.
85 
86 N.B.: FILTER can use NdisRegisterDeviceEx to create a device, so the upper
87  layer can send Irps to the filter.
88 
89 --*/
90 {
91  PMS_FILTER pFilter = NULL;
92  NDIS_STATUS Status = NDIS_STATUS_SUCCESS;
93  NTSTATUS NtStatus;
94  NDIS_FILTER_ATTRIBUTES FilterAttributes;
95  ULONG Size;
96  COMPARTMENT_ID OriginalCompartmentID;
97  OBJECT_ATTRIBUTES ObjectAttributes = {0};
98 
99  const ULONG RegKeyOffset = ARRAYSIZE(L"\\REGISTRY\\MACHINE\\SYSTEM\\CurrentControlSet\\Services\\otlwf\\Parameters\\NdisAdapters\\") - 1;
100  DECLARE_CONST_UNICODE_STRING(RegKeyPath, L"\\REGISTRY\\MACHINE\\SYSTEM\\CurrentControlSet\\Services\\otlwf\\Parameters\\NdisAdapters\\{00000000-0000-0000-0000-000000000000}");
101  RtlCopyMemory(RegKeyPath.Buffer + RegKeyOffset, AttachParameters->BaseMiniportName->Buffer + 8, sizeof(L"{00000000-0000-0000-0000-000000000000}"));
102 
103  LogFuncEntry(DRIVER_DEFAULT);
104 
105  do
106  {
107  ASSERT(FilterDriverContext == (NDIS_HANDLE)FilterDriverObject);
108  if (FilterDriverContext != (NDIS_HANDLE)FilterDriverObject)
109  {
110  Status = NDIS_STATUS_INVALID_PARAMETER;
111  break;
112  }
113 
114  // Verify the media type is supported. This is a last resort; the
115  // the filter should never have been bound to an unsupported miniport
116  // to begin with.
117  if (AttachParameters->MiniportMediaType != NdisMediumIP)
118  {
119  LogError(DRIVER_DEFAULT, "Unsupported media type, 0x%x.", (ULONG)AttachParameters->MiniportMediaType);
120  Status = NDIS_STATUS_INVALID_PARAMETER;
121  break;
122  }
123 
124  Size = sizeof(MS_FILTER) + AttachParameters->BaseMiniportInstanceName->Length;
125 
126  pFilter = (PMS_FILTER)FILTER_ALLOC_MEM(NdisFilterHandle, Size);
127  if (pFilter == NULL)
128  {
129  LogWarning(DRIVER_DEFAULT, "Failed to allocate context structure, 0x%x bytes", Size);
130  Status = NDIS_STATUS_RESOURCES;
131  break;
132  }
133 
134  NdisZeroMemory(pFilter, sizeof(MS_FILTER));
135 
136  LogVerbose(DRIVER_DEFAULT, "Opening interface registry key %S", RegKeyPath.Buffer);
137 
138  InitializeObjectAttributes(
139  &ObjectAttributes,
140  (PUNICODE_STRING)&RegKeyPath,
141  OBJ_CASE_INSENSITIVE | OBJ_KERNEL_HANDLE,
142  NULL,
143  NULL);
144 
145  // Open the registry key
146  NtStatus = ZwOpenKey(&pFilter->InterfaceRegKey, KEY_ALL_ACCESS, &ObjectAttributes);
147  if (!NT_SUCCESS(NtStatus))
148  {
149  LogError(DRIVER_DEFAULT, "ZwOpenKey failed to open %S, %!STATUS!", RegKeyPath.Buffer, NtStatus);
150  Status = NDIS_STATUS_FAILURE;
151  break;
152  }
153 
154  // Format of "\DEVICE\{5BA90C49-0D7E-455B-8D3B-614F6714A212}"
155  AttachParameters->BaseMiniportName->Buffer += 8;
156  AttachParameters->BaseMiniportName->Length -= 8 * sizeof(WCHAR);
157  NtStatus = RtlGUIDFromString(AttachParameters->BaseMiniportName, &pFilter->InterfaceGuid);
158  AttachParameters->BaseMiniportName->Buffer -= 8;
159  AttachParameters->BaseMiniportName->Length += 8 * sizeof(WCHAR);
160  if (!NT_SUCCESS(NtStatus))
161  {
162  LogError(DRIVER_DEFAULT, "Failed to convert FilterModuleGuidName to a GUID, %!STATUS!", NtStatus);
163  Status = NDIS_STATUS_FAILURE;
164  break;
165  }
166 
167  pFilter->InterfaceFriendlyName.Length = pFilter->InterfaceFriendlyName.MaximumLength = AttachParameters->BaseMiniportInstanceName->Length;
168  pFilter->InterfaceFriendlyName.Buffer = (PWSTR)((PUCHAR)pFilter + sizeof(MS_FILTER));
169  NdisMoveMemory(pFilter->InterfaceFriendlyName.Buffer,
170  AttachParameters->BaseMiniportInstanceName->Buffer,
171  pFilter->InterfaceFriendlyName.Length);
172 
173  pFilter->InterfaceIndex = AttachParameters->BaseMiniportIfIndex;
174  pFilter->InterfaceLuid = AttachParameters->BaseMiniportNetLuid;
175  pFilter->InterfaceCompartmentID = UNSPECIFIED_COMPARTMENT_ID;
176  pFilter->FilterHandle = NdisFilterHandle;
177 
178  NdisZeroMemory(&FilterAttributes, sizeof(NDIS_FILTER_ATTRIBUTES));
179  FilterAttributes.Header.Revision = NDIS_FILTER_ATTRIBUTES_REVISION_1;
180  FilterAttributes.Header.Size = sizeof(NDIS_FILTER_ATTRIBUTES);
181  FilterAttributes.Header.Type = NDIS_OBJECT_TYPE_FILTER_ATTRIBUTES;
182  FilterAttributes.Flags = 0;
183 
184  NDIS_DECLARE_FILTER_MODULE_CONTEXT(MS_FILTER);
185  Status = NdisFSetAttributes(NdisFilterHandle, pFilter, &FilterAttributes);
186  if (Status != NDIS_STATUS_SUCCESS)
187  {
188  LogError(DRIVER_DEFAULT, "Failed to set attributes, %!NDIS_STATUS!", Status);
189  break;
190  }
191 
192  // Filter initially in Paused state
193  pFilter->State = FilterPaused;
194 
195  // Initialize rundowns to disabled with no active references
196  pFilter->ExternalRefs.Count = EX_RUNDOWN_ACTIVE;
197  pFilter->cmdRundown.Count = EX_RUNDOWN_ACTIVE;
198 
199  // Query the compartment ID for this interface to use for the IP stack
200  pFilter->InterfaceCompartmentID = GetInterfaceCompartmentID(&pFilter->InterfaceLuid);
201  LogVerbose(DRIVER_DEFAULT, "Interface %!GUID! is in Compartment %u", &pFilter->InterfaceGuid, (ULONG)pFilter->InterfaceCompartmentID);
202 
203  // Make sure we are in the right compartment
204  (VOID)otLwfSetCompartment(pFilter, &OriginalCompartmentID);
205 
206  // Register for address changed notifications
207  NtStatus =
208  NotifyUnicastIpAddressChange(
209  AF_INET6,
210  otLwfAddressChangeCallback,
211  pFilter,
212  FALSE,
213  &pFilter->AddressChangeHandle
214  );
215 
216  // Revert the compartment, now that we have the table
217  otLwfRevertCompartment(OriginalCompartmentID);
218 
219  if (!NT_SUCCESS(NtStatus))
220  {
221  LogError(DRIVER_DEFAULT, "NotifyUnicastIpAddressChange failed, %!STATUS!", NtStatus);
222  Status = NDIS_STATUS_FAILURE;
223  break;
224  }
225 
226  // Add Filter to global list of Thread Filters
227  NdisAcquireSpinLock(&FilterListLock);
228  InsertTailList(&FilterModuleList, &pFilter->FilterModuleLink);
229  NdisReleaseSpinLock(&FilterListLock);
230 
231  LogVerbose(DRIVER_DEFAULT, "Created Filter: %p", pFilter);
232 
233  } while (FALSE);
234 
235  // Clean up on failure
236  if (Status != NDIS_STATUS_SUCCESS)
237  {
238  if (pFilter != NULL)
239  {
240  if (pFilter->AddressChangeHandle != NULL)
241  {
242  CancelMibChangeNotify2(pFilter->AddressChangeHandle);
243  pFilter->AddressChangeHandle = NULL;
244  }
245 
246  NdisFreeMemory(pFilter, 0, 0);
247  }
248  }
249 
250  LogFuncExitNDIS(DRIVER_DEFAULT, Status);
251 
252  return Status;
253 }

255 _Use_decl_annotations_
256 VOID
257 FilterDetach(
258  NDIS_HANDLE FilterModuleContext
259  )
260 /*++
261 
262 Routine Description:
263 
264  Filter detach routine.
265  This is a required function that will deallocate all the resources allocated during
266  FilterAttach. NDIS calls FilterAttach to remove a filter instance from a filter stack.
267 
268 Arguments:
269 
270  FilterModuleContext - pointer to the filter context area.
271 
272 Return Value:
273  None.
274 
275 NOTE: Called at PASSIVE_LEVEL and the filter is in paused state
276 
277 --*/
278 {
279  PMS_FILTER pFilter = (PMS_FILTER)FilterModuleContext;
280 
281  LogFuncEntryMsg(DRIVER_DEFAULT, "Filter: %p", FilterModuleContext);
282 
283  // Filter must be in paused state and pretty much inactive
284  NT_ASSERT(pFilter->State == FilterPaused);
285  NT_ASSERT(pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_UNINTIALIZED);
286 
287  //
288  // Detach must not fail, so do not put any code here that can possibly fail.
289  //
290 
291  // Remove this Filter from the global list
292  NdisAcquireSpinLock(&FilterListLock);
293  RemoveEntryList(&pFilter->FilterModuleLink);
294  NdisReleaseSpinLock(&FilterListLock);
295 
296  // Unregister from address change notifications
297  CancelMibChangeNotify2(pFilter->AddressChangeHandle);
298  pFilter->AddressChangeHandle = NULL;
299 
300  // Close the registry key
301  if (pFilter->InterfaceRegKey)
302  {
303  ZwClose(pFilter->InterfaceRegKey);
304  pFilter->InterfaceRegKey = NULL;
305  }
306 
307  // Free the memory allocated
308  NdisFreeMemory(pFilter, 0, 0);
309 
310  LogFuncExit(DRIVER_DEFAULT);
311 }

334 NTSTATUS
335 GetRegDWORDValue(
336  _In_ PMS_FILTER pFilter,
337  _In_ PCWSTR ValueName,
338  _Out_ PULONG ValueData
339 )
340 {
341  NTSTATUS status;
342  ULONG resultLength;
343  UCHAR keybuf[128] = {0};
344  UNICODE_STRING UValueName;
345 
346  PAGED_CODE();
347 
348  RtlInitUnicodeString(&UValueName, ValueName);
349 
350  status = ZwQueryValueKey(
351  pFilter->InterfaceRegKey,
352  &UValueName,
353  KeyValueFullInformation,
354  keybuf,
355  sizeof(keybuf),
356  &resultLength);
357 
358  if (NT_SUCCESS(status))
359  {
360  PKEY_VALUE_FULL_INFORMATION keyInfo = (PKEY_VALUE_FULL_INFORMATION)keybuf;
361 
362  if (keyInfo->Type != REG_DWORD)
363  {
364  status = STATUS_INVALID_PARAMETER_MIX;
365  }
366  else
367  {
368  *ValueData = *((ULONG UNALIGNED *)(keybuf + keyInfo->DataOffset));
369  }
370  }
371 
372  return status;
373 }

401 _Use_decl_annotations_
402 NDIS_STATUS
403 FilterRestart(
404  NDIS_HANDLE FilterModuleContext,
405  PNDIS_FILTER_RESTART_PARAMETERS RestartParameters
406  )
407 /*++
408 
409 Routine Description:
410 
411  Filter restart routine.
412  Start the datapath - begin sending and receiving NBLs.
413 
414 Arguments:
415 
416  FilterModuleContext - pointer to the filter context stucture.
417  RestartParameters - additional information about the restart operation.
418 
419 Return Value:
420 
421  NDIS_STATUS_SUCCESS: if filter restarts successfully
422  NDIS_STATUS_XXX: Otherwise.
423 
424 --*/
425 {
426  NTSTATUS NtStatus = STATUS_SUCCESS;
427  NDIS_STATUS NdisStatus = NDIS_STATUS_SUCCESS;
428  PMS_FILTER pFilter = (PMS_FILTER)FilterModuleContext;
429  PVOID SpinelCapsDataBuffer = NULL;
430  const uint8_t* SpinelCapsPtr = NULL;
431  spinel_size_t SpinelCapsLen = 0;
432  NL_INTERFACE_KEY key = {0};
433  NL_INTERFACE_RW interfaceRw;
434  ULONG ThreadOnHost = TRUE;
435 
436  PNDIS_RESTART_GENERAL_ATTRIBUTES NdisGeneralAttributes;
437  PNDIS_RESTART_ATTRIBUTES NdisRestartAttributes;
438 
439  LogFuncEntryMsg(DRIVER_DEFAULT, "Filter: %p", FilterModuleContext);
440 
441  NT_ASSERT(pFilter->State == FilterPaused);
442 
443  NdisRestartAttributes = RestartParameters->RestartAttributes;
444 
445  //
446  // If NdisRestartAttributes is not NULL, then the filter can modify generic
447  // attributes and add new media specific info attributes at the end.
448  // Otherwise, if NdisRestartAttributes is NULL, the filter should not try to
449  // modify/add attributes.
450  //
451  if (NdisRestartAttributes != NULL)
452  {
453  ASSERT(NdisRestartAttributes->Oid == OID_GEN_MINIPORT_RESTART_ATTRIBUTES);
454 
455  NdisGeneralAttributes = (PNDIS_RESTART_GENERAL_ATTRIBUTES)NdisRestartAttributes->Data;
456 
457  //
458  // Check to see if we need to change any attributes. For example, the
459  // driver can change the current MAC address here. Or the driver can add
460  // media specific info attributes.
461  //
462  NdisGeneralAttributes->LookaheadSize = 128;
463  }
464 
465  // Initialize the Spinel command processing
466  NdisStatus = otLwfCmdInitialize(pFilter);
467  if (NdisStatus != NDIS_STATUS_SUCCESS)
468  {
469  LogError(DRIVER_DEFAULT, "otLwfCmdInitialize failed, %!NDIS_STATUS!", NdisStatus);
470  goto exit;
471  }
472 
473  // Query the device capabilities
474  NtStatus = otLwfCmdGetProp(pFilter, SpinelCapsDataBuffer, SPINEL_PROP_CAPS, SPINEL_DATATYPE_DATA_S, &SpinelCapsPtr, &SpinelCapsLen);
475  if (!NT_SUCCESS(NtStatus))
476  {
477  NdisStatus = NDIS_STATUS_NOT_SUPPORTED;
478  LogError(DRIVER_DEFAULT, "Failed to query SPINEL_PROP_CAPS, %!STATUS!", NtStatus);
479  goto exit;
480  }
481 
482  // Iterate and process returned capabilities
483  while (SpinelCapsLen > 0)
484  {
485  ULONG SpinelCap = 0;
486  spinel_ssize_t len = spinel_datatype_unpack(SpinelCapsPtr, SpinelCapsLen, SPINEL_DATATYPE_UINT_PACKED_S, &SpinelCap);
487  if (len < 1) break;
488  SpinelCapsLen -= (spinel_size_t)len;
489 
490  switch (SpinelCap)
491  {
492  case SPINEL_CAP_MAC_RAW:
493  pFilter->DeviceCapabilities |= OTLWF_DEVICE_CAP_RADIO;
494  pFilter->DeviceCapabilities |= OTLWF_DEVICE_CAP_RADIO_ACK_TIMEOUT;
495  pFilter->DeviceCapabilities |= OTLWF_DEVICE_CAP_RADIO_ENERGY_SCAN;
496  break;
497  case SPINEL_CAP_NET_THREAD_1_0:
498  pFilter->DeviceCapabilities |= OTLWF_DEVICE_CAP_THREAD_1_0;
499  break;
500  default:
501  break;
502  }
503  }
504 
505  // Set the state indicating where we should be running the Thread logic (Host or Device).
506  if (!NT_SUCCESS(GetRegDWORDValue(pFilter, L"RunOnHost", &ThreadOnHost)))
507  {
508  // Default to running on the host if the key isn't present
509  ThreadOnHost = TRUE;
510  SetRegDWORDValue(pFilter, L"RunOnHost", ThreadOnHost);
511  }
512 
513  LogInfo(DRIVER_DEFAULT, "Filter: %p initializing ThreadOnHost=%d", FilterModuleContext, ThreadOnHost);
514 
515  // Initialize the processing logic
516  if (ThreadOnHost)
517  {
518  // Ensure the device has the capabilities to support raw radio commands
519  if ((pFilter->DeviceCapabilities & OTLWF_DEVICE_CAP_RADIO) == 0)
520  {
521  LogError(DRIVER_DEFAULT, "Failed to start because device doesn't support raw radio commands");
522  NdisStatus = NDIS_STATUS_NOT_SUPPORTED;
523  goto exit;
524  }
525 
526  pFilter->DeviceStatus = OTLWF_DEVICE_STATUS_RADIO_MODE;
527  NtStatus = otLwfInitializeThreadMode(pFilter);
528  if (!NT_SUCCESS(NtStatus))
529  {
530  LogError(DRIVER_DEFAULT, "otLwfInitializeThreadMode failed, %!STATUS!", NtStatus);
531  NdisStatus = NDIS_STATUS_FAILURE;
532  pFilter->DeviceStatus = OTLWF_DEVICE_STATUS_UNINTIALIZED;
533  goto exit;
534  }
535  }
536  else
537  {
538  // Ensure the device has the capabilities to support Thread commands
539  if ((pFilter->DeviceCapabilities & OTLWF_DEVICE_CAP_THREAD_1_0) == 0)
540  {
541  LogError(DRIVER_DEFAULT, "Failed to start because device doesn't support thread commands");
542  NdisStatus = NDIS_STATUS_NOT_SUPPORTED;
543  goto exit;
544  }
545 
546  pFilter->DeviceStatus = OTLWF_DEVICE_STATUS_THREAD_MODE;
547  NtStatus = otLwfTunInitialize(pFilter);
548  if (!NT_SUCCESS(NtStatus))
549  {
550  LogError(DRIVER_DEFAULT, "otLwfInitializeTunnelMode failed, %!STATUS!", NtStatus);
551  NdisStatus = NDIS_STATUS_FAILURE;
552  pFilter->DeviceStatus = OTLWF_DEVICE_STATUS_UNINTIALIZED;
553  goto exit;
554  }
555  }
556 
557  //
558  // Disable DAD and Neighbor advertisements
559  //
560  key.Luid = pFilter->InterfaceLuid;
561  NlInitializeInterfaceRw(&interfaceRw);
562  interfaceRw.DadTransmits = 0;
563  interfaceRw.SendUnsolicitedNeighborAdvertisementOnDad = FALSE;
564 
565  NtStatus =
566  NsiSetAllParameters(
567  NsiActive,
568  NsiSetDefault,
569  &NPI_MS_IPV6_MODULEID,
570  NlInterfaceObject,
571  &key,
572  sizeof(key),
573  &interfaceRw,
574  sizeof(interfaceRw));
575  if (!NT_SUCCESS(NtStatus))
576  {
577  LogError(DRIVER_DEFAULT, "NsiSetAllParameters (NlInterfaceObject) failed, %!STATUS!", NtStatus);
578  NdisStatus = NDIS_STATUS_FAILURE;
579  goto exit;
580  }
581 
582  //
583  // Enable the external references to the filter
584  //
585  ExReInitializeRundownProtection(&pFilter->ExternalRefs);
586 
587  //
588  // If everything is OK, set the filter in running state.
589  //
590  pFilter->State = FilterRunning; // when successful
591  otLwfNotifyDeviceAvailabilityChange(pFilter, TRUE);
592  LogInfo(DRIVER_DEFAULT, "Interface %!GUID! arrival, Filter=%p", &pFilter->InterfaceGuid, pFilter);
593 
594 exit:
595 
596  //
597  // Ensure the state is Paused if restart failed.
598  //
599  if (NdisStatus != NDIS_STATUS_SUCCESS)
600  {
601  pFilter->State = FilterPaused;
602 
603  if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_RADIO_MODE)
604  {
605  otLwfUninitializeThreadMode(pFilter);
606  }
607  else if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_THREAD_MODE)
608  {
609  otLwfTunUninitialize(pFilter);
610  }
611 
612  pFilter->DeviceStatus = OTLWF_DEVICE_STATUS_UNINTIALIZED;
613 
614  // Clean up Spinel command processing
615  otLwfCmdUninitialize(pFilter);
616  }
617 
618  // Free the buffer for the capabilities we queried
619  if (SpinelCapsDataBuffer != NULL)
620  {
621  FILTER_FREE_MEM(SpinelCapsDataBuffer);
622  }
623 
624  LogFuncExitNDIS(DRIVER_DEFAULT, NdisStatus);
625  return NdisStatus;
626 }

628 _Use_decl_annotations_
629 NDIS_STATUS
630 FilterPause(
631  NDIS_HANDLE FilterModuleContext,
632  PNDIS_FILTER_PAUSE_PARAMETERS PauseParameters
633  )
634 /*++
635 
636 Routine Description:
637 
638  Filter pause routine.
639  Complete all the outstanding sends and queued sends,
640  wait for all the outstanding recvs to be returned
641  and return all the queued receives.
642 
643 Arguments:
644 
645  FilterModuleContext - pointer to the filter context stucture
646  PauseParameters - additional information about the pause
647 
648 Return Value:
649 
650  NDIS_STATUS_SUCCESS if filter pauses successfully, NDIS_STATUS_PENDING
651  if not. No other return value is allowed (pause must succeed, eventually).
652 
653 N.B.: When the filter is in Pausing state, it can still process OID requests,
654  complete sending, and returning packets to NDIS, and also indicate status.
655  After this function completes, the filter must not attempt to send or
656  receive packets, but it may still process OID requests and status
657  indications.
658 
659 --*/
660 {
661  PMS_FILTER pFilter = (PMS_FILTER)(FilterModuleContext);
662  NDIS_STATUS Status = STATUS_SUCCESS;
663 
664  UNREFERENCED_PARAMETER(PauseParameters);
665 
666  LogFuncEntryMsg(DRIVER_DEFAULT, "Filter: %p", FilterModuleContext);
667 
668  //
669  // Set the flag that the filter is going to pause
670  //
671  NT_ASSERT(pFilter->State == FilterRunning);
672  NdisAcquireSpinLock(&FilterListLock);
673  pFilter->State = FilterPausing;
674  NdisReleaseSpinLock(&FilterListLock);
675 
676  //
677  // Send final notification of interface removal
678  //
679  otLwfNotifyDeviceAvailabilityChange(pFilter, FALSE);
680  LogInfo(DRIVER_DEFAULT, "Interface %!GUID! removal.", &pFilter->InterfaceGuid);
681 
682  //
683  // Disable external references and wait for existing calls to complete
684  //
685  LogInfo(DRIVER_DEFAULT, "Disabling and waiting for external references to release");
