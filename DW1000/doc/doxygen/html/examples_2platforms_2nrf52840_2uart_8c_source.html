87 static void processReceive(void)
88 {
89  VerifyOrExit(isRxBufferEmpty() == false, ;);
90 
91  // Set head position to not be changed during read procedure.
92  uint16_t head = sReceiveHead;
93 
94  // In case head roll back to the beginning of the buffer, notify about left
95  // bytes from the end of the buffer.
96  if (head < sReceiveTail)
97  {
98  otPlatUartReceived(&sReceiveBuffer[sReceiveTail],
99  (UART_RX_BUFFER_SIZE - sReceiveTail));
100  sReceiveTail = 0;
101  }
102 
103  // Notify about received bytes.
104  if (head > sReceiveTail)
105  {
106  otPlatUartReceived(&sReceiveBuffer[sReceiveTail],
107  (head - sReceiveTail));
108  sReceiveTail = head;
109  }
110 
111 exit:
112  return;
113 }

118 static void processTransmit(void)
119 {
120  VerifyOrExit(sTransmitBuffer != NULL, ;);
121 
122  if (sTransmitDone)
123  {
124  // Clear Transmition transaction and notify application.
125  sTransmitBuffer = NULL;
126  sTransmitLength = 0;
127  sTransmitDone = false;
128  otPlatUartSendDone();
129  }
130 
131 exit:
132  return;
133 }

214 ThreadError otPlatUartSend(const uint8_t *aBuf, uint16_t aBufLength)
215 {
216  ThreadError error = kThreadError_None;
217 
218  VerifyOrExit(sTransmitBuffer == NULL, error = kThreadError_Busy);
219 
220  // Set up transmit buffer and its size without counting first triggered byte.
221  sTransmitBuffer = aBuf;
222  sTransmitLength = aBufLength - 1;
223 
224  // Initiate Transmission process.
225  nrf_uart_event_clear(UART_INSTANCE, NRF_UART_EVENT_TXDRDY);
226  nrf_uart_txd_set(UART_INSTANCE, *sTransmitBuffer++);
227  nrf_uart_task_trigger(UART_INSTANCE, NRF_UART_TASK_STARTTX);
228 
229 exit:
230  return error;
231 }

236 void UARTE0_UART0_IRQHandler(void)
237 {
238  // Check if any error has been detected.
239  if (nrf_uart_event_check(UART_INSTANCE, NRF_UART_EVENT_ERROR))
240  {
241  // Clear error event and ignore erronous byte in RXD register.
242  nrf_uart_event_clear(UART_INSTANCE, NRF_UART_EVENT_ERROR);
243  nrf_uart_event_clear(UART_INSTANCE, NRF_UART_EVENT_RXDRDY);
244  }
245  else if (nrf_uart_event_check(UART_INSTANCE, NRF_UART_EVENT_RXDRDY))
246  {
247  // Clear RXDRDY event.
248  nrf_uart_event_clear(UART_INSTANCE, NRF_UART_EVENT_RXDRDY);
249 
250  // Read byte from the UART buffer.
251  uint8_t byte = nrf_uart_rxd_get(UART_INSTANCE);
252 
253  if (!isRxBufferFull())
254  {
255  sReceiveBuffer[sReceiveHead] = byte;
256  sReceiveHead = (sReceiveHead + 1) % UART_RX_BUFFER_SIZE;
257  }
258  }
259 
260  if (nrf_uart_event_check(UART_INSTANCE, NRF_UART_EVENT_TXDRDY))
261  {
262  // Clear TXDRDY event.
263  nrf_uart_event_clear(UART_INSTANCE, NRF_UART_EVENT_TXDRDY);
264 
265  // Send any more bytes if available or call application about TX done.
266  if (sTransmitLength)
267  {
268  nrf_uart_txd_set(UART_INSTANCE, *sTransmitBuffer++);
269  sTransmitLength--;
270  }
271  else
272  {
273  sTransmitDone = true;
274  nrf_uart_task_trigger(UART_INSTANCE, NRF_UART_TASK_STOPTX);
275  }
276  }
