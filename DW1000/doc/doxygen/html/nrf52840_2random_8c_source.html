118 void nrf5RandomInit(void)
119 {
120  memset(sBuffer, 0, sizeof(sBuffer));
121  sReadPosition = 0;
122  sWritePosition = 0;
123 
124  NVIC_SetPriority(RNG_IRQn, RNG_IRQ_PRIORITY);
125  NVIC_ClearPendingIRQ(RNG_IRQn);
126  NVIC_EnableIRQ(RNG_IRQn);
127 
128  nrf_rng_error_correction_enable();
129  nrf_rng_shorts_disable(NRF_RNG_SHORT_VALRDY_STOP_MASK);
130  generatorStart();
131 
132  // Wait for the first randomized 4 bytes, to randomize software generator seed.
133  while (!bufferIsUint32Ready()) ;
134 
135  uint32_t seed = bufferGetUint32();
136  srand(seed);
137 }

144 ThreadError otPlatRandomSecureGet(uint16_t aInputLength, uint8_t *aOutput, uint16_t *aOutputLength)
145 {
146  ThreadError error = kThreadError_None;
147 
148  VerifyOrExit(aOutput && aOutputLength, error = kThreadError_InvalidArgs);
149  VerifyOrExit(!bufferIsEmpty(), error = kThreadError_Failed);
150 
151  uint16_t copyLength = (uint16_t)bufferCount();
152 
153  if (copyLength > aInputLength)
154  {
155  copyLength = aInputLength;
156  }
157 
158  for (uint32_t i = 0; i < copyLength; i++)
159  {
160  aOutput[i] = bufferGet();
161  }
162 
163  *aOutputLength = copyLength;
164 
165  generatorStart();
166 
167 exit:
168  return error;
169 }
