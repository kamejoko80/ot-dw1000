57 void cstartup_rwdata()
58 {
59 
60  // Get the addresses for the .data section (initialized data section)
61  uint32_t *data_rom = (uint32_t *) &__rwdata_start__;
62  uint32_t *data_ram = (uint32_t *) &_sdata;
63  uint32_t *data_ram_end = (uint32_t *) &_edata;
64 
65  // Copy initialized data from ROM to RAM
66  while (data_ram < data_ram_end)
67  {
68  *data_ram++ = *data_rom++;
69  }
70 }
