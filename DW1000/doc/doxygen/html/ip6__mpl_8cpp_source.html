85 ThreadError Mpl::UpdateSeedSet(uint16_t aSeedId, uint8_t aSequence)
86 {
87  ThreadError error = kThreadError_None;
88  MplSeedEntry *entry = NULL;
89  int8_t diff;
90 
91  for (uint32_t i = 0; i < kNumSeedEntries; i++)
92  {
93  if (mSeedSet[i].GetLifetime() == 0)
94  {
95  // Start allocating from the first possible entry to speed up process of searching.
96  if (entry == NULL)
97  {
98  entry = &mSeedSet[i];
99  }
100  }
101  else if (mSeedSet[i].GetSeedId() == aSeedId)
102  {
103  entry = &mSeedSet[i];
104  diff = static_cast<int8_t>(aSequence - entry->GetSequence());
105 
106  VerifyOrExit(diff > 0, error = kThreadError_Drop);
107 
108  break;
109  }
110  }
111 
112  VerifyOrExit(entry != NULL, error = kThreadError_Drop);
113 
114  entry->SetSeedId(aSeedId);
115  entry->SetSequence(aSequence);
116  entry->SetLifetime(kSeedEntryLifetime);
117  mSeedSetTimer.Start(kSeedEntryLifetimeDt);
118 
119 exit:

123 void Mpl::UpdateBufferedSet(uint16_t aSeedId, uint8_t aSequence)
124 {
125  int8_t diff;
126  MplBufferedMessageMetadata messageMetadata;
127 
128  Message *message = mBufferedMessageSet.GetHead();
129  Message *nextMessage = NULL;
130 
131  // Check if multicast forwarding is enabled.
132  VerifyOrExit(GetTimerExpirations() > 0, ;);
133 
134  while (message != NULL)
135  {
136  nextMessage = message->GetNext();
137  messageMetadata.ReadFrom(*message);
138 
139  if (messageMetadata.GetSeedId() == aSeedId)
140  {
141  diff = static_cast<int8_t>(aSequence - messageMetadata.GetSequence());
142 
143  if (diff > 0)
144  {
145  // Stop retransmitting MPL Data Message that is consider to be old.
146  mBufferedMessageSet.Dequeue(*message);
147  message->Free();
148  }
149 
150  break;
151  }
152 
153  message = nextMessage;
154  }
155 
156 exit:
157  return;

160 void Mpl::AddBufferedMessage(Message &aMessage, uint16_t aSeedId, uint8_t aSequence, bool aIsOutbound)
161 {
162  uint32_t now = Timer::GetNow();
163  ThreadError error = kThreadError_None;
164  Message *messageCopy = NULL;
165  MplBufferedMessageMetadata messageMetadata;
166  uint32_t nextTransmissionTime;
167  uint8_t hopLimit = 0;
168 
169  VerifyOrExit(GetTimerExpirations() > 0,);
170  VerifyOrExit((messageCopy = aMessage.Clone()) != NULL, error = kThreadError_NoBufs);
171 
172  if (!aIsOutbound)
173  {
174  aMessage.Read(Header::GetHopLimitOffset(), Header::GetHopLimitSize(), &hopLimit);
175  VerifyOrExit(hopLimit-- > 1, error = kThreadError_Drop);
176  messageCopy->Write(Header::GetHopLimitOffset(), Header::GetHopLimitSize(), &hopLimit);
177  }
178 
179  messageMetadata.SetSeedId(aSeedId);
180  messageMetadata.SetSequence(aSequence);
181  messageMetadata.SetTransmissionCount(aIsOutbound ? 1 : 0);
182  messageMetadata.GenerateNextTransmissionTime(now, kDataMessageInterval);
183 
184  // Append the message with MplBufferedMessageMetadata and add it to the queue.
185  SuccessOrExit(error = messageMetadata.AppendTo(*messageCopy));
186  mBufferedMessageSet.Enqueue(*messageCopy);
187 
188  if (mRetransmissionTimer.IsRunning())
189  {
190  // If timer is already running, check if it should be restarted with earlier fire time.
191  nextTransmissionTime = mRetransmissionTimer.Gett0() + mRetransmissionTimer.Getdt();
192 
193  if (messageMetadata.IsEarlier(nextTransmissionTime))
194  {
195  mRetransmissionTimer.Start(messageMetadata.GetTransmissionTime() - now);
196  }
197  }
198  else
199  {
200  // Otherwise just set the timer.
201  mRetransmissionTimer.Start(messageMetadata.GetTransmissionTime() - now);
202  }
203 
204 exit:
205 
206  if (error != kThreadError_None && messageCopy != NULL)
207  {
208  messageCopy->Free();
209  }
210 
211  return;
212 }

214 ThreadError Mpl::ProcessOption(Message &aMessage, const Address &aAddress, bool aIsOutbound)
215 {
216  ThreadError error;
217  OptionMpl option;
218 
219  VerifyOrExit(aMessage.Read(aMessage.GetOffset(), sizeof(option), &option) >= OptionMpl::kMinLength &&
220  (option.GetSeedIdLength() == OptionMpl::kSeedIdLength0 ||
221  option.GetSeedIdLength() == OptionMpl::kSeedIdLength2),
222  error = kThreadError_Drop);
223 
224  if (option.GetSeedIdLength() == OptionMpl::kSeedIdLength0)
225  {
226  // Retrieve MPL Seed Id from the IPv6 Source Address.
227  option.SetSeedId(HostSwap16(aAddress.mFields.m16[7]));
228  }
229 
230  // Check MPL Data Messages in the MPL Buffered Set against sequence number.
231  UpdateBufferedSet(option.GetSeedId(), option.GetSequence());
232 
233  // Check if the MPL Data Message is new.
234  error = UpdateSeedSet(option.GetSeedId(), option.GetSequence());
235 
236  if (error == kThreadError_None)
237  {
238  AddBufferedMessage(aMessage, option.GetSeedId(), option.GetSequence(), aIsOutbound);
239  }
240  else if (aIsOutbound)
241  {
242  // In case MPL Data Message is generated locally, ignore potential error of the MPL Seed Set
243  // to allow subsequent retransmissions with the same sequence number.
244  ExitNow(error = kThreadError_None);
245  }
246 
247 exit:
248  return error;
249 }

256 void Mpl::HandleRetransmissionTimer()
257 {
258  uint32_t now = Timer::GetNow();
259  uint32_t nextDelta = 0xffffffff;
260  MplBufferedMessageMetadata messageMetadata;
261 
262  Message *message = mBufferedMessageSet.GetHead();
263  Message *nextMessage = NULL;
264 
265  while (message != NULL)
266  {
267  nextMessage = message->GetNext();
268  messageMetadata.ReadFrom(*message);
269 
270  if (messageMetadata.IsLater(now))
271  {
272  // Calculate the next retransmission time and choose the lowest.
273  if (messageMetadata.GetTransmissionTime() - now < nextDelta)
274  {
275  nextDelta = messageMetadata.GetTransmissionTime() - now;
276  }
277  }
278  else
279  {
280  // Update the number of transmission timer expirations.
281  messageMetadata.SetTransmissionCount(messageMetadata.GetTransmissionCount() + 1);
282 
283  if (messageMetadata.GetTransmissionCount() < GetTimerExpirations())
284  {
285  Message *messageCopy = message->Clone(message->GetLength() - sizeof(MplBufferedMessageMetadata));
286 
287  if (messageCopy != NULL)
288  {
289  if (messageMetadata.GetTransmissionCount() > 1)
290  {
291  messageCopy->SetSubType(Message::kSubTypeMplRetransmission);
292  }
293 
294  mIp6.EnqueueDatagram(*messageCopy);
295  }
296 
297  messageMetadata.GenerateNextTransmissionTime(now, kDataMessageInterval);
298  messageMetadata.UpdateIn(*message);
299 
300  // Check if retransmission time is lower than the current lowest one.
301  if (messageMetadata.GetTransmissionTime() - now < nextDelta)
302  {
303  nextDelta = messageMetadata.GetTransmissionTime() - now;
304  }
305  }
306  else
307  {
308  mBufferedMessageSet.Dequeue(*message);
309 
310  if (messageMetadata.GetTransmissionCount() == GetTimerExpirations())
311  {
312  if (messageMetadata.GetTransmissionCount() > 1)
313  {
314  message->SetSubType(Message::kSubTypeMplRetransmission);
315  }
316 
317  // Remove the extra metadata from the MPL Data Message.
318  messageMetadata.RemoveFrom(*message);
319  mIp6.EnqueueDatagram(*message);
320  }
321  else
322  {
323  // Stop retransmitting if the number of timer expirations is already exceeded.
324  message->Free();
325  }
326  }
327  }
328 
329  message = nextMessage;
330  }
331 
332  if (nextDelta != 0xffffffff)
333  {
334  mRetransmissionTimer.Start(nextDelta);
335  }
336 }
