88 int8_t Routes::Lookup(const Address &aSource, const Address &aDestination)
89 {
90  int8_t maxPrefixMatch = -1;
91  uint8_t prefixMatch;
92  int8_t rval = -1;
93 
94  for (Route *cur = mRoutes; cur; cur = cur->mNext)
95  {
96  prefixMatch = cur->mPrefix.PrefixMatch(aDestination);
97 
98  if (prefixMatch < cur->mPrefixLength)
99  {
100  continue;
101  }
102 
103  if (prefixMatch > cur->mPrefixLength)
104  {
105  prefixMatch = cur->mPrefixLength;
106  }
107 
108  if (maxPrefixMatch > static_cast<int8_t>(prefixMatch))
109  {
110  continue;
111  }
112 
113  maxPrefixMatch = static_cast<int8_t>(prefixMatch);
114  rval = cur->mInterfaceId;
115  }
116 
117  for (Netif *netif = mIp6.GetNetifList(); netif; netif = netif->GetNext())
118  {
119  if (netif->RouteLookup(aSource, aDestination, &prefixMatch) == kThreadError_None &&
120  static_cast<int8_t>(prefixMatch) > maxPrefixMatch)
121  {
122  maxPrefixMatch = static_cast<int8_t>(prefixMatch);
123  rval = netif->GetInterfaceId();
124  }
125  }
126
