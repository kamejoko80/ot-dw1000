161 ret_code_t nrf_drv_spi_init(nrf_drv_spi_t const * const p_instance,
162  nrf_drv_spi_config_t const * p_config,
163  nrf_drv_spi_evt_handler_t handler,
164  void * p_context)
165 {
166  ASSERT(p_config);
167  spi_control_block_t * p_cb = &m_cb[p_instance->drv_inst_idx];
168  ret_code_t err_code;
169 
170  if (p_cb->state != NRF_DRV_STATE_UNINITIALIZED)
171  {
172  err_code = NRF_ERROR_INVALID_STATE;
173  return err_code;
174  }
175 
176 #if NRF_MODULE_ENABLED(PERIPHERAL_RESOURCE_SHARING)
177  if (nrf_drv_common_per_res_acquire(p_instance->p_registers,
178  m_irq_handlers[p_instance->drv_inst_idx]) != NRF_SUCCESS)
179  {
180  err_code = NRF_ERROR_BUSY;
181  return err_code;
182  }
183 #endif
184 
185  p_cb->handler = handler;
186  p_cb->p_context = p_context;
187 
188  uint32_t mosi_pin;
189  uint32_t miso_pin;
190  // Configure pins used by the peripheral:
191  // - SCK - output with initial value corresponding with the SPI mode used:
192  // 0 - for modes 0 and 1 (CPOL = 0), 1 - for modes 2 and 3 (CPOL = 1);
193  // according to the reference manual guidelines this pin and its input
194  // buffer must always be connected for the SPI to work.
195  if (p_config->mode <= NRF_DRV_SPI_MODE_1)
196  {
197  nrf_gpio_pin_clear(p_config->sck_pin);
198  }
199  else
200  {
201  nrf_gpio_pin_set(p_config->sck_pin);
202  }
203  nrf_gpio_cfg(p_config->sck_pin,
204  NRF_GPIO_PIN_DIR_OUTPUT,
205  NRF_GPIO_PIN_INPUT_CONNECT,
206  NRF_GPIO_PIN_NOPULL,
207  NRF_GPIO_PIN_S0S1,
208  NRF_GPIO_PIN_NOSENSE);
209  // - MOSI (optional) - output with initial value 0,
210  if (p_config->mosi_pin != NRF_DRV_SPI_PIN_NOT_USED)
211  {
212  mosi_pin = p_config->mosi_pin;
213  nrf_gpio_pin_clear(mosi_pin);
214  nrf_gpio_cfg_output(mosi_pin);
215  }
216  else
217  {
218  mosi_pin = NRF_SPI_PIN_NOT_CONNECTED;
219  }
220  // - MISO (optional) - input,
221  if (p_config->miso_pin != NRF_DRV_SPI_PIN_NOT_USED)
222  {
223  miso_pin = p_config->miso_pin;
224  nrf_gpio_cfg_input(miso_pin, NRF_GPIO_PIN_NOPULL);
225  }
226  else
227  {
228  miso_pin = NRF_SPI_PIN_NOT_CONNECTED;
229  }
230  // - Slave Select (optional) - output with initial value 1 (inactive).
231  if (p_config->ss_pin != NRF_DRV_SPI_PIN_NOT_USED)
232  {
233  nrf_gpio_pin_set(p_config->ss_pin);
234  nrf_gpio_cfg_output(p_config->ss_pin);
235  }
236  m_cb[p_instance->drv_inst_idx].ss_pin = p_config->ss_pin;
237 
238  CODE_FOR_SPIM
239  (
240  NRF_SPIM_Type * p_spim = (NRF_SPIM_Type *)p_instance->p_registers;
241  nrf_spim_pins_set(p_spim, p_config->sck_pin, mosi_pin, miso_pin);
242  nrf_spim_frequency_set(p_spim,
243  (nrf_spim_frequency_t)p_config->frequency);
244 
245  nrf_spim_configure(p_spim,
246  (nrf_spim_mode_t)p_config->mode,
247  (nrf_spim_bit_order_t)p_config->bit_order);
248 
249  nrf_spim_orc_set(p_spim, p_config->orc);
250 
251  if (p_cb->handler)
252  {
253  nrf_spim_int_enable(p_spim, END_INT_MASK);
254  }
255 
256  nrf_spim_enable(p_spim);
257  )
258  CODE_FOR_SPI
259  (
260  NRF_SPI_Type * p_spi = p_instance->p_registers;
261  nrf_spi_pins_set(p_spi, p_config->sck_pin, mosi_pin, miso_pin);
262  nrf_spi_frequency_set(p_spi,
263  (nrf_spi_frequency_t)p_config->frequency);
264  nrf_spi_configure(p_spi,
265  (nrf_spi_mode_t)p_config->mode,
266  (nrf_spi_bit_order_t)p_config->bit_order);
267 
268  m_cb[p_instance->drv_inst_idx].orc = p_config->orc;
269 
270  if (p_cb->handler)
271  {
272  nrf_spi_int_enable(p_spi, NRF_SPI_INT_READY_MASK);
273  }
274 
275  nrf_spi_enable(p_spi);
276  )
277 
278  if (p_cb->handler)
279  {
280  }
281 
282  p_cb->transfer_in_progress = false;
283  p_cb->state = NRF_DRV_STATE_INITIALIZED;
284 
285  err_code = NRF_SUCCESS;
286  return err_code;
287 }

290 {
291  spi_control_block_t * p_cb = &m_cb[p_instance->drv_inst_idx];
292  ASSERT(p_cb->state != NRF_DRV_STATE_UNINITIALIZED);
293 
294  if (p_cb->handler)
295  {
296  }
297 
298  #define DISABLE_ALL 0xFFFFFFFF
299 
300  CODE_FOR_SPIM
301  (
302  NRF_SPIM_Type * p_spim = (NRF_SPIM_Type *)p_instance->p_registers;
303  if (p_cb->handler)
304  {
305  nrf_spim_int_disable(p_spim, DISABLE_ALL);
306  if (p_cb->transfer_in_progress)
307  {
308  // Ensure that SPI is not performing any transfer.
309  nrf_spim_task_trigger(p_spim, NRF_SPIM_TASK_STOP);
310  while (!nrf_spim_event_check(p_spim, NRF_SPIM_EVENT_STOPPED)) {}
311  p_cb->transfer_in_progress = false;
312  }
313  }
314  nrf_spim_disable(p_spim);
315  )
316  CODE_FOR_SPI
317  (
318  NRF_SPI_Type * p_spi = p_instance->p_registers;
319  if (p_cb->handler)
320  {
321  nrf_spi_int_disable(p_spi, DISABLE_ALL);
322  }
323  nrf_spi_disable(p_spi);
324  )
325  #undef DISABLE_ALL
326 
327 #if NRF_MODULE_ENABLED(PERIPHERAL_RESOURCE_SHARING)
328  nrf_drv_common_per_res_release(p_instance->p_registers);
329 #endif
330 
331  p_cb->state = NRF_DRV_STATE_UNINITIALIZED;
332 }

349 static void finish_transfer(spi_control_block_t * p_cb)
350 {
351  // If Slave Select signal is used, this is the time to deactivate it.
352  if (p_cb->ss_pin != NRF_DRV_SPI_PIN_NOT_USED)
353  {
354  nrf_gpio_pin_set(p_cb->ss_pin);
355  }
356 
357  // By clearing this flag before calling the handler we allow subsequent
358  // transfers to be started directly from the handler function.
359  p_cb->transfer_in_progress = false;
360  p_cb->evt.type = NRF_DRV_SPI_EVENT_DONE;
361  p_cb->handler(&p_cb->evt, p_cb->p_context);
362 }

364 #ifdef SPI_IN_USE
365 // This function is called from IRQ handler or, in blocking mode, directly
366 // from the 'nrf_drv_spi_transfer' function.
367 // It returns true as long as the transfer should be continued, otherwise (when
368 // there is nothing more to send/receive) it returns false.
369 static bool transfer_byte(NRF_SPI_Type * p_spi, spi_control_block_t * p_cb)
370 {
371  // Read the data byte received in this transfer and store it in RX buffer,
372  // if needed.
373  volatile uint8_t rx_data = nrf_spi_rxd_get(p_spi);
374  if (p_cb->bytes_transferred < p_cb->evt.data.done.rx_length)
375  {
376  p_cb->evt.data.done.p_rx_buffer[p_cb->bytes_transferred] = rx_data;
377  }
378 
379  ++p_cb->bytes_transferred;
380 
381  // Check if there are more bytes to send or receive and write proper data
382  // byte (next one from TX buffer or over-run character) to the TXD register
383  // when needed.
384  // NOTE - we've already used 'p_cb->bytes_transferred + 1' bytes from our
385  // buffers, because we take advantage of double buffering of TXD
386  // register (so in effect one byte is still being transmitted now);
387  // see how the transfer is started in the 'nrf_drv_spi_transfer'
388  // function.
389  uint16_t bytes_used = p_cb->bytes_transferred + 1;
390 
391  if (p_cb->abort)
392  {
393  if (bytes_used < p_cb->evt.data.done.tx_length)
394  {
395  p_cb->evt.data.done.tx_length = bytes_used;
396  }
397  if (bytes_used < p_cb->evt.data.done.rx_length)
398  {
399  p_cb->evt.data.done.rx_length = bytes_used;
400  }
401  }
402 
403  if (bytes_used < p_cb->evt.data.done.tx_length)
404  {
405  nrf_spi_txd_set(p_spi, p_cb->evt.data.done.p_tx_buffer[bytes_used]);
406  return true;
407  }
408  else if (bytes_used < p_cb->evt.data.done.rx_length)
409  {
410  nrf_spi_txd_set(p_spi, p_cb->orc);
411  return true;
412  }
413 
414  return (p_cb->bytes_transferred < p_cb->evt.data.done.tx_length ||
415  p_cb->bytes_transferred < p_cb->evt.data.done.rx_length);
416 }

418 static void spi_xfer(NRF_SPI_Type * p_spi,
419  spi_control_block_t * p_cb,
420  nrf_drv_spi_xfer_desc_t const * p_xfer_desc)
421 {
422  p_cb->bytes_transferred = 0;
423  nrf_spi_int_disable(p_spi, NRF_SPI_INT_READY_MASK);
424 
425  nrf_spi_event_clear(p_spi, NRF_SPI_EVENT_READY);
426 
427  // Start the transfer by writing some byte to the TXD register;
428  // if TX buffer is not empty, take the first byte from this buffer,
429  // otherwise - use over-run character.
430  nrf_spi_txd_set(p_spi,
431  (p_xfer_desc->tx_length > 0 ? p_xfer_desc->p_tx_buffer[0] : p_cb->orc));
432 
433  // TXD register is double buffered, so next byte to be transmitted can
434  // be written immediately, if needed, i.e. if TX or RX transfer is to
435  // be more that 1 byte long. Again - if there is something more in TX
436  // buffer send it, otherwise use over-run character.
437  if (p_xfer_desc->tx_length > 1)
438  {
439  nrf_spi_txd_set(p_spi, p_xfer_desc->p_tx_buffer[1]);
440  }
441  else if (p_xfer_desc->rx_length > 1)
442  {
443  nrf_spi_txd_set(p_spi, p_cb->orc);
444  }
445 
446  // For blocking mode (user handler not provided) wait here for READY
447  // events (indicating that the byte from TXD register was transmitted
448  // and a new incoming byte was moved to the RXD register) and continue
449  // transaction until all requested bytes are transferred.
450  // In non-blocking mode - IRQ service routine will do this stuff.
451  if (p_cb->handler)
452  {
453  nrf_spi_int_enable(p_spi, NRF_SPI_INT_READY_MASK);
454  }
455  else
456  {
457  do {
458  while (!nrf_spi_event_check(p_spi, NRF_SPI_EVENT_READY)) {}
459  nrf_spi_event_clear(p_spi, NRF_SPI_EVENT_READY);
460  //NRF_LOG_DEBUG("SPI: Event: NRF_SPI_EVENT_READY.\r\n");
461  } while (transfer_byte(p_spi, p_cb));
462  if (p_cb->ss_pin != NRF_DRV_SPI_PIN_NOT_USED)
463  {
464  nrf_gpio_pin_set(p_cb->ss_pin);
465  }
466  }
467 }
468 #endif // SPI_IN_USE

504 static ret_code_t spim_xfer(NRF_SPIM_Type * p_spim,
505  spi_control_block_t * p_cb,
506  nrf_drv_spi_xfer_desc_t const * p_xfer_desc,
507  uint32_t flags)
508 {
509  ret_code_t err_code;
510  // EasyDMA requires that transfer buffers are placed in Data RAM region;
511  // signal error if they are not.
512  if ((p_xfer_desc->p_tx_buffer != NULL && !nrf_drv_is_in_RAM(p_xfer_desc->p_tx_buffer)) ||
513  (p_xfer_desc->p_rx_buffer != NULL && !nrf_drv_is_in_RAM(p_xfer_desc->p_rx_buffer)))
514  {
515  p_cb->transfer_in_progress = false;
516  err_code = NRF_ERROR_INVALID_ADDR;
517  return err_code;
518  }
519 
520 #if NRF_MODULE_ENABLED(SPIM_NRF52_ANOMALY_109_WORKAROUND)
521  p_cb->tx_length = 0;
522  p_cb->rx_length = 0;
523 #endif
524 
525  nrf_spim_tx_buffer_set(p_spim, p_xfer_desc->p_tx_buffer, p_xfer_desc->tx_length);
526  nrf_spim_rx_buffer_set(p_spim, p_xfer_desc->p_rx_buffer, p_xfer_desc->rx_length);
527 
528  nrf_spim_event_clear(p_spim, NRF_SPIM_EVENT_END);
529 
530  spim_list_enable_handle(p_spim, flags);
531 
532  if (!(flags & NRF_DRV_SPI_FLAG_HOLD_XFER))
533  {
534  nrf_spim_task_trigger(p_spim, NRF_SPIM_TASK_START);
535  }
536 #if NRF_MODULE_ENABLED(SPIM_NRF52_ANOMALY_109_WORKAROUND)
537  if (flags & NRF_DRV_SPI_FLAG_HOLD_XFER)
538  {
539  nrf_spim_event_clear(p_spim, NRF_SPIM_EVENT_STARTED);
540  p_cb->tx_length = p_xfer_desc->tx_length;
541  p_cb->rx_length = p_xfer_desc->rx_length;
542  nrf_spim_tx_buffer_set(p_spim, p_xfer_desc->p_tx_buffer, 0);
543  nrf_spim_rx_buffer_set(p_spim, p_xfer_desc->p_rx_buffer, 0);
544  nrf_spim_int_enable(p_spim, NRF_SPIM_INT_STARTED_MASK);
545  }
546 #endif
547 
548  if (!p_cb->handler)
549  {
550  while (!nrf_spim_event_check(p_spim, NRF_SPIM_EVENT_END)){}
551  if (p_cb->ss_pin != NRF_DRV_SPI_PIN_NOT_USED)
552  {
553  nrf_gpio_pin_set(p_cb->ss_pin);
554  }
555  }
556  else
557  {
558  spim_int_enable(p_spim, !(flags & NRF_DRV_SPI_FLAG_NO_XFER_EVT_HANDLER));
559  }
560  err_code = NRF_SUCCESS;
561  return err_code;
562 }
563 #endif

565 ret_code_t nrf_drv_spi_xfer(nrf_drv_spi_t const * const p_instance,
566  nrf_drv_spi_xfer_desc_t const * p_xfer_desc,
567  uint32_t flags)
568 {
569  spi_control_block_t * p_cb = &m_cb[p_instance->drv_inst_idx];
570  ASSERT(p_cb->state != NRF_DRV_STATE_UNINITIALIZED);
571  ASSERT(p_xfer_desc->p_tx_buffer != NULL || p_xfer_desc->tx_length == 0);
572  ASSERT(p_xfer_desc->p_rx_buffer != NULL || p_xfer_desc->rx_length == 0);
573 
574  ret_code_t err_code = NRF_SUCCESS;
575 
576  if (p_cb->transfer_in_progress)
577  {
578  err_code = NRF_ERROR_BUSY;
579  return err_code;
580  }
581  else
582  {
583  if (p_cb->handler && !(flags & (NRF_DRV_SPI_FLAG_REPEATED_XFER |
584  NRF_DRV_SPI_FLAG_NO_XFER_EVT_HANDLER)))
585  {
586  p_cb->transfer_in_progress = true;
587  }
588  }
589 
590  p_cb->evt.data.done = *p_xfer_desc;
591  p_cb->tx_done = false;
592  p_cb->rx_done = false;
593  p_cb->abort = false;
594 
595  if (p_cb->ss_pin != NRF_DRV_SPI_PIN_NOT_USED)
596  {
597  nrf_gpio_pin_clear(p_cb->ss_pin);
598  }
599  CODE_FOR_SPIM
600  (
601  return spim_xfer(p_instance->p_registers, p_cb, p_xfer_desc, flags);
602  )
603  CODE_FOR_SPI
604  (
605  if (flags)
606  {
607  p_cb->transfer_in_progress = false;
608  err_code = NRF_ERROR_NOT_SUPPORTED;
609  }
610  else
611  {
612  spi_xfer(p_instance->p_registers, p_cb, p_xfer_desc);
613  }
614  return err_code;
615  )
616 }

641 #if NRF_MODULE_ENABLED(SPIM_NRF52_ANOMALY_109_WORKAROUND)

643  (nrf_spim_event_check(p_spim, NRF_SPIM_EVENT_STARTED)) )
644  {
645  /* Handle first, zero-length, auxiliary transmission. */
646  nrf_spim_event_clear(p_spim, NRF_SPIM_EVENT_STARTED);

657

659  nrf_spim_task_trigger(p_spim, NRF_SPIM_TASK_START);
660  return;
661  }

666  nrf_spim_event_clear(p_spim, NRF_SPIM_EVENT_END);
667  ASSERT(p_cb->handler);
668  finish_transfer(p_cb);
669  }
