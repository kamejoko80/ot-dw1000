85  * Example function. Blink LED according to node state
86  * Leader - 5Hz
87  * Router - 2Hz
88  * Child - 0.5Hz
89  */

91 void ExampleProcess(otInstance *aInstance)
92 {
93  otDeviceRole devRole;
94  static int thrValue;
95 
96  devRole = otGetDeviceRole(aInstance);
97 
98  if (sBlink == false && otPlatAlarmGetNow() != 0)
99  {
100  sMsCounterInit = otPlatAlarmGetNow();
101  sBlink = true;
102  }
103 
104  sMsCounter = otPlatAlarmGetNow() - sMsCounterInit;
105 
106  switch (devRole)
107  {
108  case kDeviceRoleLeader:
109  thrValue = LEADER_BLINK_TIME;
110  break;
111 
112  case kDeviceRoleRouter:
113  thrValue = ROUTER_BLINK_TIME;
114  break;
115 
116  case kDeviceRoleChild:
117  thrValue = CHILD_BLINK_TIME;
118  break;
119 
120  default:
121  thrValue = 0x00;
122  }
123 
124  if ((thrValue != 0x00) && (sMsCounter >= thrValue))
125  {
126  hw_gpio_toggle(HW_GPIO_PORT_1, HW_GPIO_PIN_5);
127  sMsCounterInit = otPlatAlarmGetNow();
128  }
129 
130  if (thrValue == 0)
131  {
132  hw_gpio_set_inactive(HW_GPIO_PORT_1, HW_GPIO_PIN_5);
133  }
134 }
