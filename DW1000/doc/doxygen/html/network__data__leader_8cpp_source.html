106

116  continue;

284 ThreadError LeaderBase::ExternalRouteLookup(uint8_t aDomainId, const Ip6::Address &aDestination,
285  uint8_t *aPrefixMatch, uint16_t *aRloc16)
286 {
287  ThreadError error = kThreadError_NoRoute;
288  PrefixTlv *prefix;
289  HasRouteTlv *hasRoute;
290  HasRouteEntry *entry;
291  HasRouteEntry *rvalRoute = NULL;
292  uint8_t rval_plen = 0;
293  int8_t plen;
294  NetworkDataTlv *cur;
295  NetworkDataTlv *subCur;
296 
297  for (cur = reinterpret_cast<NetworkDataTlv *>(mTlvs);
298  cur < reinterpret_cast<NetworkDataTlv *>(mTlvs + mLength);
299  cur = cur->GetNext())
300  {
301  if (cur->GetType() != NetworkDataTlv::kTypePrefix)
302  {
303  continue;
304  }
305 
306  prefix = static_cast<PrefixTlv *>(cur);
307 
308  if (prefix->GetDomainId() != aDomainId)
309  {
310  continue;
311  }
312 
313  plen = PrefixMatch(prefix->GetPrefix(), aDestination.mFields.m8, prefix->GetPrefixLength());
314 
315  if (plen > rval_plen)
316  {
317  // select border router
318  for (subCur = prefix->GetSubTlvs(); subCur < prefix->GetNext(); subCur = subCur->GetNext())
319  {
320  if (subCur->GetType() != NetworkDataTlv::kTypeHasRoute)
321  {
322  continue;
323  }
324 
325  hasRoute = static_cast<HasRouteTlv *>(subCur);
326 
327  for (uint8_t i = 0; i < hasRoute->GetNumEntries(); i++)
328  {
329  entry = hasRoute->GetEntry(i);
330 
331  if (rvalRoute == NULL ||
332  entry->GetPreference() > rvalRoute->GetPreference() ||
333  (entry->GetPreference() == rvalRoute->GetPreference() &&
334  mNetif.GetMle().GetRouteCost(entry->GetRloc()) <
335  mNetif.GetMle().GetRouteCost(rvalRoute->GetRloc())))
336  {
337  rvalRoute = entry;
338  rval_plen = static_cast<uint8_t>(plen);
339  }
340  }
341 
342  }
343  }
344  }
345 
346  if (rvalRoute != NULL)
347  {
348  if (aRloc16 != NULL)
349  {
350  *aRloc16 = rvalRoute->GetRloc();
351  }
352 
353  if (aPrefixMatch != NULL)
354  {
355  *aPrefixMatch = rval_plen;
356  }
357 
358  error = kThreadError_None;
359  }
360 
361  return error;
362 }

364 ThreadError LeaderBase::DefaultRouteLookup(PrefixTlv &aPrefix, uint16_t *aRloc16)
365 {
366  ThreadError error = kThreadError_NoRoute;
367  BorderRouterTlv *borderRouter;
368  BorderRouterEntry *entry;
369  BorderRouterEntry *route = NULL;
370 
371  for (NetworkDataTlv *cur = aPrefix.GetSubTlvs(); cur < aPrefix.GetNext(); cur = cur->GetNext())
372  {
373  if (cur->GetType() != NetworkDataTlv::kTypeBorderRouter)
374  {
375  continue;
376  }
377 
378  borderRouter = static_cast<BorderRouterTlv *>(cur);
379 
380  for (uint8_t i = 0; i < borderRouter->GetNumEntries(); i++)
381  {
382  entry = borderRouter->GetEntry(i);
383 
384  if (entry->IsDefaultRoute() == false)
385  {
386  continue;
387  }
388 
389  if (route == NULL ||
390  entry->GetPreference() > route->GetPreference() ||
391  (entry->GetPreference() == route->GetPreference() &&
392  mNetif.GetMle().GetRouteCost(entry->GetRloc()) < mNetif.GetMle().GetRouteCost(route->GetRloc())))
393  {
394  route = entry;
395  }
396  }
397  }
398 
399  if (route != NULL)
400  {
401  if (aRloc16 != NULL)
402  {
403  *aRloc16 = route->GetRloc();
404  }
