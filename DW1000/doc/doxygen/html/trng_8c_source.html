59 void
60 TRNGConfigure(uint32_t ui32MinSamplesPerCycle,
61  uint32_t ui32MaxSamplesPerCycle,
62  uint32_t ui32ClocksPerSample)
63 {
64  uint32_t ui32Val;
65 
66  //
67  // Make sure the TRNG is disabled.
68  //
69  ui32Val = HWREG(TRNG_BASE + TRNG_O_CTL) & ~TRNG_CTL_TRNG_EN;
70  HWREG(TRNG_BASE + TRNG_O_CTL) = ui32Val;
71 
72  //
73  // Configure the startup number of samples.
74  //
75  ui32Val &= ~TRNG_CTL_STARTUP_CYCLES_M;
76  ui32Val |= ((( ui32MaxSamplesPerCycle >> 8 ) << TRNG_CTL_STARTUP_CYCLES_S ) & TRNG_CTL_STARTUP_CYCLES_M );
77  HWREG(TRNG_BASE + TRNG_O_CTL) = ui32Val;
78 
79  //
80  // Configure the minimum and maximum number of samples pr generated number
81  // and the number of clocks per sample.
82  //
83  HWREG(TRNG_BASE + TRNG_O_CFG0) = (
84  ((( ui32MaxSamplesPerCycle >> 8 ) << TRNG_CFG0_MAX_REFILL_CYCLES_S ) & TRNG_CFG0_MAX_REFILL_CYCLES_M ) |
85  ((( ui32ClocksPerSample ) << TRNG_CFG0_SMPL_DIV_S ) & TRNG_CFG0_SMPL_DIV_M ) |
86  ((( ui32MinSamplesPerCycle >> 6 ) << TRNG_CFG0_MIN_REFILL_CYCLES_S ) & TRNG_CFG0_MIN_REFILL_CYCLES_M ) );
87 }
