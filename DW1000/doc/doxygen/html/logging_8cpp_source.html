63 static void DumpLine(otLogLevel aLogLevel, otLogRegion aLogRegion, const void *aBuf, const size_t aLength)
64 {
65  char buf[80];
66  char *cur = buf;
67 
68  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), "|");
69  cur += strlen(cur);
70 
71  for (size_t i = 0; i < 16; i++)
72  {
73  if (i < aLength)
74  {
75  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), " %02X", ((uint8_t *)(aBuf))[i]);
76  cur += strlen(cur);
77  }
78  else
79  {
80  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), " ..");
81  cur += strlen(cur);
82  }
83 
84  if (!((i + 1) % 8))
85  {
86  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), " |");
87  cur += strlen(cur);
88  }
89  }
90 
91  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), " ");
92  cur += strlen(cur);
93 
94  for (size_t i = 0; i < 16; i++)
95  {
96  char c = 0x7f & ((char *)(aBuf))[i];
97 
98  if (i < aLength && isprint(c))
99  {
100  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), "%c", c);
101  cur += strlen(cur);
102  }
103  else
104  {
105  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), ".");
106  cur += strlen(cur);
107  }
108  }
109 
110  otLogDump("%s", buf);
111 }

114 {

121  {

123  cur += strlen(cur);
124  }
125 
126  snprintf(cur, sizeof(buf) - static_cast<size_t>(cur - buf), "[%s len=%03u]", aId, static_cast<uint16_t>(aLength));
127  cur += strlen(cur);
128 
129  for (size_t i = 0; i < (width - idlen) / 2 - 4; i++)
130  {

137  for (size_t i = 0; i < aLength; i += 16)
