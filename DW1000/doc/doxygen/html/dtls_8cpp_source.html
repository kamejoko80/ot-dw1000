369 void Dtls::Process(void)
370 {
371  uint8_t buf[MBEDTLS_SSL_MAX_CONTENT_LEN];
372  bool shouldClose = false;
373  int rval;
374 
375  while (mStarted)
376  {
377  if (mSsl.state != MBEDTLS_SSL_HANDSHAKE_OVER)
378  {
379  rval = mbedtls_ssl_handshake(&mSsl);
380 
381  if ((mSsl.state == MBEDTLS_SSL_HANDSHAKE_OVER) && (mConnectedHandler != NULL))
382  {
383  mConnectedHandler(mContext, true);
384  }
385  }
386  else
387  {
388  rval = mbedtls_ssl_read(&mSsl, buf, sizeof(buf));
389  }
390 
391  if (rval > 0)
392  {
393  mReceiveHandler(mContext, buf, static_cast<uint16_t>(rval));
394  }
395  else if (rval == 0 || rval == MBEDTLS_ERR_SSL_WANT_READ || rval == MBEDTLS_ERR_SSL_WANT_WRITE)
396  {
397  break;
398  }
399  else
400  {
401  switch (rval)
402  {
403  case MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY:
404  mbedtls_ssl_close_notify(&mSsl);
405  ExitNow(shouldClose = true);
406  break;
407 
408  case MBEDTLS_ERR_SSL_HELLO_VERIFY_REQUIRED:
409  break;
410 
411  case MBEDTLS_ERR_SSL_FATAL_ALERT_MESSAGE:
412  mbedtls_ssl_close_notify(&mSsl);
413  ExitNow(shouldClose = true);
414  break;
415 
416  case MBEDTLS_ERR_SSL_INVALID_MAC:
417  if (mSsl.state != MBEDTLS_SSL_HANDSHAKE_OVER)
418  {
419  mbedtls_ssl_send_alert_message(&mSsl, MBEDTLS_SSL_ALERT_LEVEL_FATAL,
420  MBEDTLS_SSL_ALERT_MSG_BAD_RECORD_MAC);
421  ExitNow(shouldClose = true);
422  }
423 
424  break;
425 
426  default:
427  if (mSsl.state != MBEDTLS_SSL_HANDSHAKE_OVER)
428  {
429  mbedtls_ssl_send_alert_message(&mSsl, MBEDTLS_SSL_ALERT_LEVEL_FATAL,
430  MBEDTLS_SSL_ALERT_MSG_HANDSHAKE_FAILURE);
431  ExitNow(shouldClose = true);
432  }
433 
434  break;
435  }
436 
437  mbedtls_ssl_session_reset(&mSsl);
438  mbedtls_ssl_set_hs_ecjpake_password(&mSsl, mPsk, mPskLength);
439  break;
440  }
441  }
442 
443 exit:
444 
445  if (shouldClose)
446  {
447  Close();
448  }
449 }
