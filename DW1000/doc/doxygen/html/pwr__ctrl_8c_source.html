59 PowerCtrlSourceSet(uint32_t ui32PowerConfig)
60 {
61  //
62  // Check the arguments.
63  //
64  ASSERT((ui32PowerConfig == PWRCTRL_PWRSRC_DCDC) ||
65  (ui32PowerConfig == PWRCTRL_PWRSRC_GLDO) ||
66  (ui32PowerConfig == PWRCTRL_PWRSRC_ULDO));
67 
68  //
69  // Configure the power.
70  //
71  if(ui32PowerConfig == PWRCTRL_PWRSRC_DCDC) {
72  HWREG(AON_SYSCTL_BASE + AON_SYSCTL_O_PWRCTL) |=
73  (AON_SYSCTL_PWRCTL_DCDC_EN | AON_SYSCTL_PWRCTL_DCDC_ACTIVE);
74  }
75  else if (ui32PowerConfig == PWRCTRL_PWRSRC_GLDO)
76  {
77  HWREG(AON_SYSCTL_BASE + AON_SYSCTL_O_PWRCTL) &=
78  ~(AON_SYSCTL_PWRCTL_DCDC_EN | AON_SYSCTL_PWRCTL_DCDC_ACTIVE);
79  }
80  else
81  {
82  PRCMMcuUldoConfigure(true);
83  }
84 }
