36 /* This file is given to prevent your SW from not compiling with the name changes between nRF51 or nRF52832 and nRF52840 devices.
37  * It redefines the old nRF51 or nRF52832 names into the new ones as long as the functionality is still supported. If the
38  * functionality is gone, there old names are not defined, so compilation will fail. Note that also includes macros
39  * from the nrf52_namechange.h file. */

41 /* Differences between latest nRF52 headers and nRF52840 headers. */
42 
43 /* UART */
44 /* The registers PSELRTS, PSELTXD, PSELCTS, PSELRXD were restructured into a struct. */
45 #define PSELRTS PSEL.RTS
46 #define PSELTXD PSEL.TXD
47 #define PSELCTS PSEL.CTS
48 #define PSELRXD PSEL.RXD
49 
50 /* TWI */
51 /* The registers PSELSCL, PSELSDA were restructured into a struct. */
52 #define PSELSCL PSEL.SCL
53 #define PSELSDA PSEL.SDA
