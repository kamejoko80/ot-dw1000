143 // This method encodes a frame from the tx frame buffer (mTxFrameBuffer) into the uart buffer and sends it over uart.
144 // If the uart buffer gets full, it sends the current encoded portion. This method remembers current state, so on
145 // sub-sequent calls, it restarts encoding the bytes from where it left of in the frame .
146 void NcpUart::EncodeAndSendToUart(void)
147 {
148  uint16_t len;
149 
150  while (!mTxFrameBuffer.IsEmpty())
151  {
152  switch (mState)
153  {
154  case kStartingFrame:
155 
156  SuccessOrExit(mFrameEncoder.Init(mUartBuffer));
157 
158  mTxFrameBuffer.OutFrameBegin();
159 
160  mState = kEncodingFrame;
161 
162  while (!mTxFrameBuffer.OutFrameHasEnded())
163  {
164  mByte = mTxFrameBuffer.OutFrameReadByte();
165 
166  case kEncodingFrame:
167 
168  SuccessOrExit(mFrameEncoder.Encode(mByte, mUartBuffer));
169  }
170 
171  mTxFrameBuffer.OutFrameRemove();
172 
173  // Notify the super/base class that there is space available in tx frame buffer for a new frame.
174  super_t::HandleSpaceAvailableInTxBuffer();
175 
176  mState = kFinalizingFrame;
177 
178  // fall through
179 
180  case kFinalizingFrame:
181 
182  SuccessOrExit(mFrameEncoder.Finalize(mUartBuffer));
183 
184  mState = kStartingFrame;
185  }
186  }
187 
188 exit:
189  len = mUartBuffer.GetLength();
190 
191  if (len > 0)
192  {
193  otPlatUartSend(mUartBuffer.GetBuffer(), len);
194  }
195 }

236 void NcpUart::HandleError(ThreadError aError, uint8_t *aBuf, uint16_t aBufLength)
237 {
238  char hexbuf[128];
239  uint16_t i = 0;
240 
241  super_t::IncrementFrameErrorCounter();
242 
243  // We can get away with sprintf because we know
244  // `hexbuf` is large enough.
245  sprintf(hexbuf, "Framing error %d: [", aError);
246 
247  // Write out the first part of our log message.
248  otNcpStreamWrite(0, reinterpret_cast<uint8_t*>(hexbuf), static_cast<int>(strlen(hexbuf)));
249 
250  // The first '3' comes from the trailing "]\n\000" at the end o the string.
251  // The second '3' comes from the length of two hex digits and a space.
252  for (i = 0; (i < aBufLength) && (i < (sizeof(hexbuf) - 3) / 3); i++)
253  {
254  // We can get away with sprintf because we know
255  // `hexbuf` is large enough, based on our calculations
256  // above.
257  sprintf(&hexbuf[i*3], " %02X", static_cast<uint8_t>(aBuf[i]));
258  }
259 
260  // Append a final closing bracket and newline character
261  // so our log line looks nice.
262  sprintf(&hexbuf[i*3], "]\n");
263 
264  // Write out the second part of our log message.
265  // We skip the first byte since it has a space in it.
266  otNcpStreamWrite(0, reinterpret_cast<uint8_t*>(hexbuf + 1), static_cast<int>(strlen(hexbuf) - 1));
267 }
