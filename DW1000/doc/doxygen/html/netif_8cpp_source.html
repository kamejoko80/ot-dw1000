218 ThreadError Netif::SubscribeExternalMulticast(const Address &aAddress)
219 {
220  ThreadError error = kThreadError_None;
221  NetifMulticastAddress *entry;
222  size_t num = sizeof(mExtMulticastAddresses) / sizeof(mExtMulticastAddresses[0]);
223 
224  for (entry = mMulticastAddresses; entry; entry = entry->GetNext())
225  {
226  if (memcmp(&entry->mAddress, &aAddress, sizeof(otIp6Address)) == 0)
227  {
228  VerifyOrExit((entry >= &mExtMulticastAddresses[0]) && (entry < &mExtMulticastAddresses[num]),
229  error = kThreadError_InvalidArgs);
230 
231  ExitNow(error = kThreadError_Already);
232  }
233  }
234 
235  // Find an available entry in the `mExtMulticastAddresses` array.
236  for (entry = &mExtMulticastAddresses[0]; num > 0; num--, entry++)
237  {
238  // In an unused/available entry, `mNext` points back to the entry itself.
239  if (entry->mNext == entry)
240  {
241  break;
242  }
243  }
244 
245  VerifyOrExit(num > 0, error = kThreadError_NoBufs);
246 
247  // Copy the address into the available entry and add it to linked-list.
248  entry->mAddress = aAddress;
249  entry->mNext = mMulticastAddresses;
250  mMulticastAddresses = entry;
251 
252 exit:
253  return error;
254 }
