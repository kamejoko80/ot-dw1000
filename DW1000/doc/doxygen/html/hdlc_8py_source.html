56  @classmethod
57  def mkfcstab(cls):
58  """ Make a static lookup table for byte value to FCS16 result. """
59  polynomial = HDLC_FCS_POLY
60 
61  def valiter():
62  """ Helper to yield FCS16 table entries for each byte value. """
63  for byte in range(256):
64  fcs = byte
65  i = 8
66  while i:
67  fcs = (fcs >> 1) ^ polynomial if fcs & 1 else fcs >> 1
68  i -= 1
69 
70  yield fcs & 0xFFFF
71 
72  return tuple(valiter())

74  def fcs16(self, byte, fcs):
75  """
76  Return the next iteration of an fcs16 calculation
77  given the next data byte and current fcs accumulator.
78  """
79  fcs = (fcs >> 8) ^ self.fcstab[(fcs ^ byte) & 0xff]
80  return fcs

82  def collect(self):
83  """ Return the next valid packet to pass HDLC decoding on the stream. """
84  fcs = HDLC_FCS_INIT
85  packet = []
86  raw = []
87 
88  # Synchronize
89  while 1:
90  byte = self.stream.read()
91  if CONFIG.DEBUG_HDLC:
92  raw.append(byte)
93  if byte == HDLC_FLAG:
94  break
95 
96  # Read packet, updating fcs, and escaping bytes as needed
97  while 1:
98  byte = self.stream.read()
99  if CONFIG.DEBUG_HDLC:
100  raw.append(byte)
101  if byte == HDLC_FLAG:
102  if len(packet) != 0:
103  break
104  else:
105  # If multiple FLAG bytes in a row, keep looking for data.
106  continue
107  if byte == HDLC_ESCAPE:
108  byte = self.stream.read()
109  if CONFIG.DEBUG_HDLC:
110  raw.append(byte)
111  byte ^= 0x20
112  packet.append(byte)
113  fcs = self.fcs16(byte, fcs)
114 
115  if CONFIG.DEBUG_HDLC:
116  logging.debug("RX Hdlc: " + str(map(hexify_int, raw)))
117 
118  if fcs != HDLC_FCS_GOOD:
119  packet = None
120  else:
121  packet = packet[:-2] # remove FCS16 from end
122 
123  return packet

135  def encode(self, payload=""):
136  """ Return the HDLC encoding of the given packet. """
137  fcs = HDLC_FCS_INIT
138  packet = []
139  packet.append(HDLC_FLAG)
140  for byte in payload:
141  byte = ord(byte)
142  fcs = self.fcs16(byte, fcs)
143  packet = self.encode_byte(byte, packet)
144 
145  fcs ^= 0xffff
146  byte = fcs & 0xFF
147  packet = self.encode_byte(byte, packet)
148  byte = fcs >> 8
149  packet = self.encode_byte(byte, packet)
150  packet.append(HDLC_FLAG)
151  packet = pack("%dB" % len(packet), *packet)
152 
153  if CONFIG.DEBUG_HDLC:
