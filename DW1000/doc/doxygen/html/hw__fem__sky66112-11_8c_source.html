504  /* CSD GPIO Config */

508  hw_gpio_configure_pin(dg_configFEM_SKY66112_11_CSD_PORT, dg_configFEM_SKY66112_11_CSD_PIN,

533  */
534

537  hw_gpio_set_pin_function(dg_configFEM_SKY66112_11_ANT_TRIM_0_PORT, dg_configFEM_SKY66112_11_ANT_TRIM_0_PIN,

539 #endif

551 #endif
552 
553  /******************************************************
554  * Setup DCFs
555  */
556 
557  /* assign values to the timer registers for CTX/CRX (in usec) */
558  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_27_REG, SET_OFFSET, dg_configFEM_SKY66112_11_TXSET_DCF);
559  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_27_REG, RESET_OFFSET, dg_configFEM_SKY66112_11_TXRESET_DCF);
560  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_28_REG, SET_OFFSET, dg_configFEM_SKY66112_11_RXSET_DCF);
561  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_28_REG, RESET_OFFSET, dg_configFEM_SKY66112_11_RXRESET_DCF);
562 
563  rf_port_en = 0x6; /* Start with Port 0: TX, Port 1: RX */
564 
565  /* Compute set/reset delays to use for CSD, CPS, CHL DCFs: For setting delay,
566  * smaller of TXSET, RXSET, and for resetting delay, larger of TXSET, RXSET
567  */
568 #if dg_configFEM_SKY66112_11_RXSET_DCF > dg_configFEM_SKY66112_11_TXSET_DCF
569  set_delay = dg_configFEM_SKY66112_11_TXSET_DCF;
570 #else
571  set_delay = dg_configFEM_SKY66112_11_RXSET_DCF;
572 #endif
573 
574 #if dg_configFEM_SKY66112_11_RXRESET_DCF > dg_configFEM_SKY66112_11_TXRESET_DCF
575  reset_delay = dg_configFEM_SKY66112_11_RXRESET_DCF;
576 #else
577  reset_delay = dg_configFEM_SKY66112_11_TXRESET_DCF;
578 #endif
579 
580  /* CSD DCF (if enabled) configuration */
581 #if defined(dg_configFEM_SKY66112_11_CSD_PORT) && defined(dg_configFEM_SKY66112_11_CSD_PIN)
582 # if dg_configFEM_SKY66112_11_CSD_USE_DCF != 0
583  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_29_REG, SET_OFFSET, set_delay);
584  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_29_REG, RESET_OFFSET, reset_delay);
585

587  rf_port_en |= 0x30;
588

590 #endif
591

593  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_30_REG, SET_OFFSET, set_delay);
594  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_30_REG, RESET_OFFSET, reset_delay);
595 
596  /* Set TX Power (CHL) DCF timers (but don't enable yet) */
597  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_31_REG, SET_OFFSET, dg_configFEM_SKY66112_11_TXSET_DCF);
598  REG_SETF(RFCU_POWER, RF_CNTRL_TIMER_31_REG, RESET_OFFSET, dg_configFEM_SKY66112_11_TXRESET_DCF);
599 
600  /* Enable DCFs */
601 #if dg_configBLACK_ORCA_IC_REV == BLACK_ORCA_IC_REV_A
602  RFCU_POWER->RF_PORT_EN_REG = rf_port_en;
603  hw_fem_set_txpower(fem_config.tx_power);
