82 {

84  {
85  // If the radio supports the retry and back off logic, immediately schedule the send,

101 
102  mBackoffTimer.Start(backoff);
103  }

280 void Mac::EnergyScanDone(int8_t aEnergyScanMaxRssi)
281 {
282  // Trigger a energy scan handler callback if necessary
283  if (aEnergyScanMaxRssi != kInvalidRssiValue)
284  {
285  otEnergyScanResult result;
286 
287  result.mChannel = mScanChannel;
288  result.mMaxRssi = aEnergyScanMaxRssi;
289  mEnergyScanHandler(mScanContext, &result);
290  }
291 
292  // Update to the next scan channel
293  do
294  {
295  mScanChannels >>= 1;
296  mScanChannel++;
297 
298  // If we have scanned all the channels, then fire the final callback
299  // and start the next transmission task
300  if (mScanChannels == 0 || mScanChannel > kPhyMaxChannel)
301  {
302  otPlatRadioReceive(mNetif.GetInstance(), mChannel);
303  mEnergyScanHandler(mScanContext, NULL);
304  ScheduleNextTransmission();
305  ExitNow();

307  }

311  StartEnergyScan();

661 void Mac::ProcessTransmitSecurity(Frame &aFrame)
662 {
663  uint32_t frameCounter = 0;
664  uint8_t securityLevel;
665  uint8_t keyIdMode;
666  uint8_t nonce[kNonceSize];
667  uint8_t tagLength;
668  Crypto::AesCcm aesCcm;
669  const uint8_t *key = NULL;
670  const ExtAddress *extAddress = NULL;
671 
672  if (aFrame.GetSecurityEnabled() == false)
673  {
674  ExitNow();
675  }
676 
677  aFrame.GetKeyIdMode(keyIdMode);
678 
679  switch (keyIdMode)
680  {
681  case Frame::kKeyIdMode0:
682  key = mNetif.GetKeyManager().GetKek();
683  frameCounter = mNetif.GetKeyManager().GetKekFrameCounter();
684  mNetif.GetKeyManager().IncrementKekFrameCounter();
685  extAddress = &mExtAddress;
686  break;
687 
688  case Frame::kKeyIdMode1:
689  key = mNetif.GetKeyManager().GetCurrentMacKey();
690  frameCounter = mNetif.GetKeyManager().GetMacFrameCounter();
691  mNetif.GetKeyManager().IncrementMacFrameCounter();
692  aFrame.SetKeyId((mNetif.GetKeyManager().GetCurrentKeySequence() & 0x7f) + 1);
693  extAddress = &mExtAddress;
694  break;
695 
696  case Frame::kKeyIdMode2:
697  {
698  const uint8_t keySource[] = {0xff, 0xff, 0xff, 0xff};
699  key = sMode2Key;
700  frameCounter = mKeyIdMode2FrameCounter++;
701  aFrame.SetKeySource(keySource);
702  aFrame.SetKeyId(0xff);
703  extAddress = static_cast<const ExtAddress *>(&sMode2ExtAddress);
704  break;
705  }
706 
707  default:
708  assert(false);
709  break;
710  }
711 
712  aFrame.GetSecurityLevel(securityLevel);
713  aFrame.SetFrameCounter(frameCounter);
714 
715  GenerateNonce(*extAddress, frameCounter, securityLevel, nonce);
716 
717  aesCcm.SetKey(key, 16);
718  tagLength = aFrame.GetFooterLength() - Frame::kFcsSize;
719 
720  aesCcm.Init(aFrame.GetHeaderLength(), aFrame.GetPayloadLength(), tagLength, nonce, sizeof(nonce));
721 
722  aesCcm.Header(aFrame.GetHeader(), aFrame.GetHeaderLength());
723  aesCcm.Payload(aFrame.GetPayload(), aFrame.GetPayload(), aFrame.GetPayloadLength(), true);
724  aesCcm.Finalize(aFrame.GetFooter(), &tagLength);
725 
726 exit:
727  {}
728 }

1061 ThreadError Mac::ProcessReceiveSecurity(Frame &aFrame, const Address &aSrcAddr, Neighbor *aNeighbor)
1062 {
1063  ThreadError error = kThreadError_None;
1064  uint8_t securityLevel;
1065  uint8_t keyIdMode;
1066  uint32_t frameCounter;
1067  uint8_t nonce[kNonceSize];
1068  uint8_t tag[Frame::kMaxMicSize];
1069  uint8_t tagLength;
1070  uint8_t keyid;
1071  uint32_t keySequence = 0;
1072  const uint8_t *macKey;
1073  const ExtAddress *extAddress;
1074  Crypto::AesCcm aesCcm;
1075 
1076  aFrame.SetSecurityValid(false);
1077 
1078  if (aFrame.GetSecurityEnabled() == false)
1079  {
1080  ExitNow();
1081  }
1082 
1083  aFrame.GetSecurityLevel(securityLevel);
1084  aFrame.GetFrameCounter(frameCounter);
1085  otLogDebgMac("Frame counter %u", frameCounter);
1086 
1087  aFrame.GetKeyIdMode(keyIdMode);
1088 
1089  switch (keyIdMode)
1090  {
1091  case Frame::kKeyIdMode0:
1092  VerifyOrExit((macKey = mNetif.GetKeyManager().GetKek()) != NULL, error = kThreadError_Security);
1093  extAddress = &aSrcAddr.mExtAddress;
1094  break;
1095 
1096  case Frame::kKeyIdMode1:
1097  VerifyOrExit(aNeighbor != NULL, error = kThreadError_Security);
1098 
1099  aFrame.GetKeyId(keyid);
1100  keyid--;
1101 
1102  if (keyid == (mNetif.GetKeyManager().GetCurrentKeySequence() & 0x7f))
1103  {
1104  // same key index
1105  keySequence = mNetif.GetKeyManager().GetCurrentKeySequence();
1106  macKey = mNetif.GetKeyManager().GetCurrentMacKey();
1107  }
1108  else if (keyid == ((mNetif.GetKeyManager().GetCurrentKeySequence() - 1) & 0x7f))
1109  {
1110  // previous key index
1111  keySequence = mNetif.GetKeyManager().GetCurrentKeySequence() - 1;
1112  macKey = mNetif.GetKeyManager().GetTemporaryMacKey(keySequence);
1113  }
1114  else if (keyid == ((mNetif.GetKeyManager().GetCurrentKeySequence() + 1) & 0x7f))
1115  {
1116  // next key index
1117  keySequence = mNetif.GetKeyManager().GetCurrentKeySequence() + 1;
1118  macKey = mNetif.GetKeyManager().GetTemporaryMacKey(keySequence);
1119  }
1120  else
1121  {
1122  ExitNow(error = kThreadError_Security);
1123  }
1124 
1125  // If the frame is from a neighbor not in valid state (e.g., it is from a child being
1126  // restored), skip the key sequence and frame counter checks but continue to verify
1127  // the tag/MIC. Such a frame is later filtered in `RxDoneTask` which only allows MAC
1128  // Data Request frames from a child being restored.
1129 
1130  if (aNeighbor->mState == Neighbor::kStateValid)
1131  {
1132  if (keySequence < aNeighbor->mKeySequence)
1133  {
1134  ExitNow(error = kThreadError_Security);
1135  }
1136  else if (keySequence == aNeighbor->mKeySequence)
1137  {
1138  if ((frameCounter + 1) < aNeighbor->mValid.mLinkFrameCounter)
1139  {
1140  ExitNow(error = kThreadError_Security);
1141  }
1142  else if ((frameCounter + 1) == aNeighbor->mValid.mLinkFrameCounter)
1143  {
1144  // drop duplicated packets
1145  ExitNow(error = kThreadError_Duplicated);
1146  }
1147  }
1148  }
1149 
1150  extAddress = &aSrcAddr.mExtAddress;
1151 
1152  break;
1153 
1154  case Frame::kKeyIdMode2:
1155  macKey = sMode2Key;
1156  extAddress = static_cast<const ExtAddress *>(&sMode2ExtAddress);
1157  break;
1158 
1159  default:
1160  ExitNow(error = kThreadError_Security);
1161  break;
1162  }
1163 
1164  GenerateNonce(*extAddress, frameCounter, securityLevel, nonce);
1165  tagLength = aFrame.GetFooterLength() - Frame::kFcsSize;
1166 
1167  aesCcm.SetKey(macKey, 16);
1168  aesCcm.Init(aFrame.GetHeaderLength(), aFrame.GetPayloadLength(), tagLength, nonce, sizeof(nonce));
1169  aesCcm.Header(aFrame.GetHeader(), aFrame.GetHeaderLength());
1170  aesCcm.Payload(aFrame.GetPayload(), aFrame.GetPayload(), aFrame.GetPayloadLength(), false);
1171  aesCcm.Finalize(tag, &tagLength);
1172 
1173  VerifyOrExit(memcmp(tag, aFrame.GetFooter(), tagLength) == 0, error = kThreadError_Security);
1174 
1175  if ((keyIdMode == Frame::kKeyIdMode1) && (aNeighbor->mState == Neighbor::kStateValid))
1176  {
1177  if (aNeighbor->mKeySequence != keySequence)
1178  {
1179  aNeighbor->mKeySequence = keySequence;
1180  aNeighbor->mValid.mMleFrameCounter = 0;
1181  }
1182 
1183  aNeighbor->mValid.mLinkFrameCounter = frameCounter + 1;
1184 
1185  if (keySequence > mNetif.GetKeyManager().GetCurrentKeySequence())
1186  {
1187  mNetif.GetKeyManager().SetCurrentKeySequence(keySequence);
1188  }
1189  }
1190 
1191  aFrame.SetSecurityValid(true);
1192 
1193 exit:
1194  return error;
1195 }
