199 static inline void LPTMR_EnableInterrupts(LPTMR_Type *base, uint32_t mask)
200 {
201  uint32_t reg = base->CSR;
202 
203  /* Clear the TCF bit so that we don't clear this w1c bit when writing back */
204  reg &= ~(LPTMR_CSR_TCF_MASK);
205  reg |= mask;
206  base->CSR = reg;

309 static inline uint32_t LPTMR_GetCurrentTimerCount(LPTMR_Type *base)
310 {
311  /* Must first write any value to the CNR. This synchronizes and registers the current value
312  * of the CNR into a temporary register which can then be read
313  */
314  base->CNR = 0U;
