84 char *Diag::ProcessCmd(int argc, char *argv[])
85 {
86  unsigned int i;
87 
88  if (argc == 0)
89  {
90  snprintf(sDiagOutput, sizeof(sDiagOutput), "diagnostics mode is %s\r\n", otPlatDiagModeGet() ? "enabled" : "disabled");
91  }
92  else
93  {
94  for (i = 0; i < sizeof(sCommands) / sizeof(sCommands[0]); i++)
95  {
96  if (strcmp(argv[0], sCommands[i].mName) == 0)
97  {
98  sCommands[i].mCommand(argc - 1, argc > 1 ? &argv[1] : NULL, sDiagOutput, sizeof(sDiagOutput));
99  break;
100  }
101  }
102 
103  // more platform specific features will be processed under platform layer
104  if (i == sizeof(sCommands) / sizeof(sCommands[0]))
105  {
106  otPlatDiagProcess(sContext, argc, argv, sDiagOutput, sizeof(sDiagOutput));
107  }
108  }
109 
110  return sDiagOutput;
111 }

201 {
202  ThreadError error = kThreadError_None;
203 
204  VerifyOrExit(otPlatDiagModeGet(), error = kThreadError_InvalidState);
205 
206  if (argc == 0)
207  {
208  snprintf(aOutput, aOutputMaxLen, "channel: %d\r\n", sChannel);
209  }
210  else
211  {
212  long value;
213 
214  SuccessOrExit(error = ParseLong(argv[0], value));

274 void Diag::ProcessRepeat(int argc, char *argv[], char *aOutput, size_t aOutputMaxLen)
275 {
276  ThreadError error = kThreadError_None;
277 
278  VerifyOrExit(otPlatDiagModeGet(), error = kThreadError_InvalidState);
279  VerifyOrExit(argc > 0, error = kThreadError_InvalidArgs);
280 
281  if (strcmp(argv[0], "stop") == 0)
282  {
283  otPlatAlarmStop(sContext);
284  sRepeatActive = false;
285  snprintf(aOutput, aOutputMaxLen, "repeated packet transmission is stopped\r\nstatus 0x%02x\r\n", error);
286  }
287  else
288  {
289  long value;
290 
291  VerifyOrExit(argc == 2, error = kThreadError_InvalidArgs);
292 
293  SuccessOrExit(error = ParseLong(argv[0], value));
294  sTxPeriod = static_cast<uint32_t>(value);
295 
296  SuccessOrExit(error = ParseLong(argv[1], value));
297  VerifyOrExit(value <= kMaxPHYPacketSize, error = kThreadError_InvalidArgs);
298  sTxLen = static_cast<uint8_t>(value);
299 
300  sRepeatActive = true;
301  uint32_t now = otPlatAlarmGetNow();
302  otPlatAlarmStartAt(sContext, now, sTxPeriod);
303  snprintf(aOutput, aOutputMaxLen, "sending packets of length %#x at the delay of %#x ms\r\nstatus 0x%02x\r\n", static_cast<int>(sTxLen), static_cast<int>(sTxPeriod), error);
304  }
305 
306 exit:
307  AppendErrorResult(error, aOutput, aOutputMaxLen);
