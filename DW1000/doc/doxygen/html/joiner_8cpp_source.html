80 ThreadError Joiner::Start(const char *aPSKd, const char *aProvisioningUrl,
81  const char *aVendorName, const char *aVendorModel, const char *aVendorSwVersion,
82  const char *aVendorData, otJoinerCallback aCallback, void *aContext)
83 {

85  Mac::ExtAddress extAddress;
86 
87  otLogFuncEntry();
88 
89  VerifyOrExit(mState == kStateIdle, error = kThreadError_Busy);
90 
91  // use extended address based on factory-assigned IEEE EUI-64
92  mNetif.GetMac().GetHashMacAddress(&extAddress);
93  mNetif.GetMac().SetExtAddress(extAddress);
94  mNetif.GetMle().UpdateLinkLocalAddress();
95 
96  error = mNetif.GetSecureCoapClient().GetDtls().SetPsk(reinterpret_cast<const uint8_t *>(aPSKd),
97  static_cast<uint8_t>(strlen(aPSKd)));

99  error = mNetif.GetSecureCoapClient().GetDtls().mProvisioningUrl.SetProvisioningUrl(aProvisioningUrl);
100  SuccessOrExit(error);
101 
102  mJoinerRouterPanId = Mac::kPanIdBroadcast;
103  SuccessOrExit(error = mNetif.GetMle().Discover(0, 0, mNetif.GetMac().GetPanId(), true, HandleDiscoverResult, this));
104 
105  mVendorName = aVendorName;
106  mVendorModel = aVendorModel;
107  mVendorSwVersion = aVendorSwVersion;
108  mVendorData = aVendorData;
109  mCallback = aCallback;
110  mContext = aContext;
111  mState = kStateDiscover;
112 
113 exit:

155 void Joiner::HandleDiscoverResult(otActiveScanResult *aResult)
156 {
157  Ip6::MessageInfo messageInfo;
158 
159  otLogFuncEntry();
160 
161  if (aResult != NULL)
162  {
163  otLogFuncEntryMsg("aResult = %llX", HostSwap64(*reinterpret_cast<uint64_t *>(&aResult->mExtAddress)));
164 
165  // Joining is disabled if the Steering Data is not included
166  if (aResult->mSteeringData.mLength == 0)
167  {
168  otLogDebgMeshCoP("No steering data, joining disabled");
169  ExitNow();
170  }
171 
172  SteeringDataTlv steeringData;
173  Mac::ExtAddress extAddress;
174  Crc16 ccitt(Crc16::kCcitt);
175  Crc16 ansi(Crc16::kAnsi);
176 
177  mNetif.GetMac().GetHashMacAddress(&extAddress);
178 
179  for (size_t i = 0; i < sizeof(extAddress); i++)
180  {
181  ccitt.Update(extAddress.m8[i]);
182  ansi.Update(extAddress.m8[i]);
183  }
184 
185  steeringData.SetLength(aResult->mSteeringData.mLength);
186  memcpy(steeringData.GetValue(), aResult->mSteeringData.m8, steeringData.GetLength());
187 
188  if (steeringData.DoesAllowAny() ||
189  (steeringData.GetBit(ccitt.Get() % steeringData.GetNumBits()) &&
190  steeringData.GetBit(ansi.Get() % steeringData.GetNumBits())))
191  {
192  mJoinerUdpPort = aResult->mJoinerUdpPort;
193  mJoinerRouterPanId = aResult->mPanId;
194  mJoinerRouterChannel = aResult->mChannel;
195  memcpy(&mJoinerRouter, &aResult->mExtAddress, sizeof(mJoinerRouter));
196  }
197  else
198  {
199  otLogDebgMeshCoP("Steering data does not include this device");
200  }
201  }
202  else if (mJoinerRouterPanId != Mac::kPanIdBroadcast)
203  {
204  otLogFuncEntryMsg("aResult = NULL");
205 
206  mNetif.GetMac().SetPanId(mJoinerRouterPanId);
207  mNetif.GetMac().SetChannel(mJoinerRouterChannel);
208  mNetif.GetIp6Filter().AddUnsecurePort(mNetif.GetSecureCoapClient().GetPort());
209 
210  messageInfo.GetPeerAddr().mFields.m16[0] = HostSwap16(0xfe80);
211  messageInfo.GetPeerAddr().SetIid(mJoinerRouter);
212  messageInfo.mPeerPort = mJoinerUdpPort;
213  messageInfo.mInterfaceId = OT_NETIF_INTERFACE_ID_THREAD;
214 
215  mNetif.GetSecureCoapClient().Connect(messageInfo, Joiner::HandleSecureCoapClientConnect, this);
216  mState = kStateConnect;
217  }
218  else
219  {
220  otLogDebgMeshCoP("No joinable network found");
221  Complete(kThreadError_NotFound);
222  }
223 
224 exit:
225  otLogFuncExit();
226 }

256 void Joiner::SendJoinerFinalize(void)
257 {
258  Coap::Header header;
259  ThreadError error = kThreadError_None;
260  Message *message = NULL;
261  StateTlv stateTlv;
262  VendorNameTlv vendorNameTlv;
263  VendorModelTlv vendorModelTlv;
264  VendorSwVersionTlv vendorSwVersionTlv;
265  VendorStackVersionTlv vendorStackVersionTlv;
266 
267  otLogFuncEntry();
268 
269  header.Init(kCoapTypeConfirmable, kCoapRequestPost);
270  header.AppendUriPathOptions(OPENTHREAD_URI_JOINER_FINALIZE);
271  header.SetPayloadMarker();
272 
273  VerifyOrExit((message = mNetif.GetSecureCoapClient().NewMeshCoPMessage(header)) != NULL,
274  error = kThreadError_NoBufs);
275 
276  stateTlv.Init();
277  stateTlv.SetState(MeshCoP::StateTlv::kAccept);
278  SuccessOrExit(error = message->Append(&stateTlv, sizeof(stateTlv)));
279 
280  vendorNameTlv.Init();
281  vendorNameTlv.SetVendorName(mVendorName);
282  SuccessOrExit(error = message->Append(&vendorNameTlv, vendorNameTlv.GetSize()));
283 
284  vendorModelTlv.Init();
285  vendorModelTlv.SetVendorModel(mVendorModel);
286  SuccessOrExit(error = message->Append(&vendorModelTlv, vendorModelTlv.GetSize()));
287 
288  vendorSwVersionTlv.Init();
289  vendorSwVersionTlv.SetVendorSwVersion(mVendorSwVersion);
290  SuccessOrExit(error = message->Append(&vendorSwVersionTlv, vendorSwVersionTlv.GetSize()));
291 
292  vendorStackVersionTlv.Init();
293  vendorStackVersionTlv.SetOui(OPENTHREAD_CONFIG_STACK_VENDOR_OUI);
294  vendorStackVersionTlv.SetMajor(OPENTHREAD_CONFIG_STACK_VERSION_MAJOR);
295  vendorStackVersionTlv.SetMinor(OPENTHREAD_CONFIG_STACK_VERSION_MINOR);
296  vendorStackVersionTlv.SetRevision(OPENTHREAD_CONFIG_STACK_VERSION_REV);
297  SuccessOrExit(error = message->Append(&vendorStackVersionTlv, vendorStackVersionTlv.GetSize()));
298 
299  if (mVendorData != NULL)
300  {
301  VendorDataTlv vendorDataTlv;
302  vendorDataTlv.Init();
303  vendorDataTlv.SetVendorData(mVendorData);
304  SuccessOrExit(error = message->Append(&vendorDataTlv, vendorDataTlv.GetSize()));
305  }
306 
307  if (mNetif.GetSecureCoapClient().GetDtls().mProvisioningUrl.GetLength() > 0)
308  {
309  SuccessOrExit(error = message->Append(&mNetif.GetSecureCoapClient().GetDtls().mProvisioningUrl,
310  mNetif.GetSecureCoapClient().GetDtls().mProvisioningUrl.GetSize()));
311  }
312 
313 #if OPENTHREAD_ENABLE_CERT_LOG
314  uint8_t buf[OPENTHREAD_CONFIG_MESSAGE_BUFFER_SIZE];
315  VerifyOrExit(message->GetLength() <= sizeof(buf), ;);
316  message->Read(header.GetLength(), message->GetLength() - header.GetLength(), buf);
317  otDumpCertMeshCoP("[THCI] direction=send | type=JOIN_FIN.req |", buf, message->GetLength() - header.GetLength());
318 #endif
319 
320  mNetif.GetSecureCoapClient().SendMessage(*message, Joiner::HandleJoinerFinalizeResponse, this);
321 
322  otLogInfoMeshCoP("Sent joiner finalize");
323 
324 exit:
325 
326  if (error != kThreadError_None && message != NULL)
327  {
328  message->Free();
329  }
330 
331  otLogFuncExit();
332 }

377 void Joiner::HandleJoinerEntrust(Coap::Header &aHeader, Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
378 {
379  ThreadError error;
380 
381  NetworkMasterKeyTlv masterKey;
382  MeshLocalPrefixTlv meshLocalPrefix;
383  ExtendedPanIdTlv extendedPanId;

385  ActiveTimestampTlv activeTimestamp;
386  NetworkKeySequenceTlv networkKeySeq;
387 
388  otLogFuncEntry();
389 
390  VerifyOrExit(mState == kStateEntrust &&
391  aHeader.GetType() == kCoapTypeConfirmable &&
392  aHeader.GetCode() == kCoapRequestPost, error = kThreadError_Drop);
393 
394  otLogInfoMeshCoP("Received joiner entrust");
395  otLogCertMeshCoP("[THCI] direction=recv | type=JOIN_ENT.ntf");

398  VerifyOrExit(masterKey.IsValid(), error = kThreadError_Parse);
399

401  VerifyOrExit(meshLocalPrefix.IsValid(), error = kThreadError_Parse);
402 
403  SuccessOrExit(error = Tlv::GetTlv(aMessage, Tlv::kExtendedPanId, sizeof(extendedPanId), extendedPanId));

405 
406  SuccessOrExit(error = Tlv::GetTlv(aMessage, Tlv::kNetworkName, sizeof(networkName), networkName));
407  VerifyOrExit(networkName.IsValid(), error = kThreadError_Parse);
408 
409  SuccessOrExit(error = Tlv::GetTlv(aMessage, Tlv::kActiveTimestamp, sizeof(activeTimestamp), activeTimestamp));
410  VerifyOrExit(activeTimestamp.IsValid(), error = kThreadError_Parse);
411 
412  SuccessOrExit(error = Tlv::GetTlv(aMessage, Tlv::kNetworkKeySequence, sizeof(networkKeySeq), networkKeySeq));
413  VerifyOrExit(networkKeySeq.IsValid(), error = kThreadError_Parse);
414 
415  mNetif.GetKeyManager().SetMasterKey(masterKey.GetNetworkMasterKey(), masterKey.GetLength());
416  mNetif.GetKeyManager().SetCurrentKeySequence(networkKeySeq.GetNetworkKeySequence());
417  mNetif.GetMle().SetMeshLocalPrefix(meshLocalPrefix.GetMeshLocalPrefix());
418  mNetif.GetMac().SetExtendedPanId(extendedPanId.GetExtendedPanId());
419  mNetif.GetMac().SetNetworkName(networkName.GetNetworkName());
420 
421  otLogInfoMeshCoP("join success!");
422 
423  // Send dummy response.
424  SendJoinerEntrustResponse(aHeader, aMessageInfo);
425 
426  // Delay extended address configuration to allow DTLS wrap up.
427  mTimer.Start(kConfigExtAddressDelay);
428 
429 exit:
430  otLogFuncExit();
431 }
