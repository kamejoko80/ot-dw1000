73 ThreadError ActiveDataset::GenerateLocal(void)
74 {
75  ThreadError error = kThreadError_None;
76  otOperationalDataset dataset;
77 
78  VerifyOrExit(mNetif.GetMle().IsAttached(), error = kThreadError_InvalidState);
79 
80  memset(&dataset, 0, sizeof(dataset));
81 
82  // Active Timestamp
83  if (!IsTlvInitialized(Tlv::kActiveTimestamp))
84  {
85  ActiveTimestampTlv activeTimestampTlv;
86  activeTimestampTlv.Init();
87  activeTimestampTlv.SetSeconds(0);
88  activeTimestampTlv.SetTicks(0);
89  mLocal.Set(activeTimestampTlv);
90  }
91 
92  // Channel
93  if (!IsTlvInitialized(Tlv::kChannel))
94  {
95  ChannelTlv tlv;
96  tlv.Init();
97  tlv.SetChannelPage(0);
98  tlv.SetChannel(mNetif.GetMac().GetChannel());
99  mLocal.Set(tlv);
100  }
101 
102  // channelMask
103  if (!IsTlvInitialized(Tlv::kChannelMask))
104  {
105  ChannelMask0Tlv tlv;
106  tlv.Init();
107  tlv.SetMask(kPhySupportedChannelMask);
108  mLocal.Set(tlv);
109  }
110 
111  // Extended PAN ID
112  if (!IsTlvInitialized(Tlv::kExtendedPanId))
113  {
114  ExtendedPanIdTlv tlv;
115  tlv.Init();
116  tlv.SetExtendedPanId(mNetif.GetMac().GetExtendedPanId());
117  mLocal.Set(tlv);
118  }
119 
120  // Mesh-Local Prefix
121  if (!IsTlvInitialized(Tlv::kMeshLocalPrefix))
122  {
123  MeshLocalPrefixTlv tlv;
124  tlv.Init();
125  tlv.SetMeshLocalPrefix(mNetif.GetMle().GetMeshLocalPrefix());
126  mLocal.Set(tlv);
127  }
128 
129  // Master Key
130  if (!IsTlvInitialized(Tlv::kNetworkMasterKey))
131  {
132  NetworkMasterKeyTlv tlv;
133  tlv.Init();
134  tlv.SetNetworkMasterKey(mNetif.GetKeyManager().GetMasterKey(NULL));
135  mLocal.Set(tlv);
136  }
137 
138  // Network Name
139  if (!IsTlvInitialized(Tlv::kNetworkName))
140  {
141  NetworkNameTlv tlv;
142  tlv.Init();
143  tlv.SetNetworkName(mNetif.GetMac().GetNetworkName());
144  mLocal.Set(tlv);
145  }
146 
147  // Pan ID
148  if (!IsTlvInitialized(Tlv::kPanId))
149  {
150  PanIdTlv tlv;
151  tlv.Init();
152  tlv.SetPanId(mNetif.GetMac().GetPanId());
153  mLocal.Set(tlv);
154  }
155 
156  // PSKc
157  if (!IsTlvInitialized(Tlv::kPSKc))
158  {
159  const uint8_t pskc[OT_PSKC_MAX_SIZE] = {0};
160  PSKcTlv tlv;
161  tlv.Init();
162  tlv.SetPSKc(pskc);
163  mLocal.Set(tlv);
164  }
165 
166  // Security Policy
167  if (!IsTlvInitialized(Tlv::kSecurityPolicy))
168  {
169  SecurityPolicyTlv tlv;
170  tlv.Init();
171  tlv.SetRotationTime(static_cast<uint16_t>(mNetif.GetKeyManager().GetKeyRotation()));
172  tlv.SetFlags(mNetif.GetKeyManager().GetSecurityPolicyFlags());
173  mLocal.Set(tlv);
174  }
175 
176 exit:
177  return error;
178 }
