60 class SubTlvsFactory(object):
61 
62  def __init__(self, sub_tlvs_factories):
63  self._sub_tlvs_factories = sub_tlvs_factories
64 
65  def _get_factory(self, _type):
66  try:
67  return self._sub_tlvs_factories[_type]
68  except KeyError:
69  raise RuntimeError("Could not find factory. Factory type = {}.".format(_type))
70 
71  def parse(self, data, message_info):
72  sub_tlvs = []
73 
74  while data.tell() < len(data.getvalue()):
75  data_byte = ord(data.read(1))
76 
77  stable = data_byte & 0x01
78  _type = (data_byte >> 1) & 0x7f
79 
80  length = ord(data.read(1))
81  value = data.read(length)
82 
83  factory = self._get_factory(_type)
84 
85  message_info.stable = stable
86  tlv = factory.parse(io.BytesIO(value), message_info)
87 
88  sub_tlvs.append(tlv)
89 
90  return sub_tlvs
