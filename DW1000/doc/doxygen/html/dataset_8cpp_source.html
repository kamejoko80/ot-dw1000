105 void Dataset::Get(otOperationalDataset &aDataset) const
106 {
107  const Tlv *cur = reinterpret_cast<const Tlv *>(mTlvs);
108  const Tlv *end = reinterpret_cast<const Tlv *>(mTlvs + mLength);
109 
110  memset(&aDataset, 0, sizeof(aDataset));
111 
112  while (cur < end)
113  {
114  switch (cur->GetType())
115  {
116 
117  case Tlv::kActiveTimestamp:
118  {
119  const ActiveTimestampTlv *tlv = static_cast<const ActiveTimestampTlv *>(cur);
120  aDataset.mActiveTimestamp = tlv->GetSeconds();
121  aDataset.mIsActiveTimestampSet = true;
122  break;
123  }
124 
125  case Tlv::kChannel:
126  {
127  const ChannelTlv *tlv = static_cast<const ChannelTlv *>(cur);
128  aDataset.mChannel = tlv->GetChannel();
129  aDataset.mIsChannelSet = true;
130  break;
131  }
132 
133  case Tlv::kChannelMask:
134  {
135  uint8_t length = cur->GetLength();
136  const uint8_t *entry = reinterpret_cast<const uint8_t *>(cur) + sizeof(Tlv);
137  const uint8_t *entryEnd = entry + length;
138 
139  while (entry < entryEnd)
140  {
141  if (reinterpret_cast<const ChannelMaskEntry *>(entry)->GetChannelPage() == 0)
142  {
143  uint8_t i = sizeof(ChannelMaskEntry);
144  aDataset.mChannelMaskPage0 = static_cast<uint32_t>(entry[i] | (entry[i + 1] << 8) |
145  (entry[i + 2] << 16) | (entry[i + 3] << 24));
146  aDataset.mIsChannelMaskPage0Set = true;
147  break;
148  }
149 
150  entry += (reinterpret_cast<const ChannelMaskEntry *>(entry)->GetMaskLength() +
151  sizeof(ChannelMaskEntry));
152  }
153 
154  break;
155  }
156 
157  case Tlv::kDelayTimer:
158  {
159  const DelayTimerTlv *tlv = static_cast<const DelayTimerTlv *>(cur);
160  aDataset.mDelay = tlv->GetDelayTimer();
161  aDataset.mIsDelaySet = true;
162  break;
163  }
164 
165  case Tlv::kExtendedPanId:
166  {
167  const ExtendedPanIdTlv *tlv = static_cast<const ExtendedPanIdTlv *>(cur);
168  memcpy(aDataset.mExtendedPanId.m8, tlv->GetExtendedPanId(), sizeof(aDataset.mExtendedPanId));
169  aDataset.mIsExtendedPanIdSet = true;
170  break;
171  }
172 
173  case Tlv::kMeshLocalPrefix:
174  {
175  const MeshLocalPrefixTlv *tlv = static_cast<const MeshLocalPrefixTlv *>(cur);
176  memcpy(aDataset.mMeshLocalPrefix.m8, tlv->GetMeshLocalPrefix(), sizeof(aDataset.mMeshLocalPrefix));
177  aDataset.mIsMeshLocalPrefixSet = true;
178  break;
179  }
180 
181  case Tlv::kNetworkMasterKey:
182  {
183  const NetworkMasterKeyTlv *tlv = static_cast<const NetworkMasterKeyTlv *>(cur);
184  memcpy(aDataset.mMasterKey.m8, tlv->GetNetworkMasterKey(), sizeof(aDataset.mMasterKey));
185  aDataset.mIsMasterKeySet = true;
186  break;
187  }
188 
189  case Tlv::kNetworkName:
190  {
191  const NetworkNameTlv *tlv = static_cast<const NetworkNameTlv *>(cur);
192  memcpy(aDataset.mNetworkName.m8, tlv->GetNetworkName(), tlv->GetLength());
193  aDataset.mIsNetworkNameSet = true;
194  break;
195  }
196 
197  case Tlv::kPanId:
198  {
199  const PanIdTlv *panid = static_cast<const PanIdTlv *>(cur);
200  aDataset.mPanId = panid->GetPanId();
201  aDataset.mIsPanIdSet = true;
202  break;
203  }
204 
205  case Tlv::kPendingTimestamp:
206  {
207  const PendingTimestampTlv *tlv = static_cast<const PendingTimestampTlv *>(cur);
208  aDataset.mPendingTimestamp = tlv->GetSeconds();
209  aDataset.mIsPendingTimestampSet = true;
210  break;
211  }
212 
213  case Tlv::kPSKc:
214  {
215  const PSKcTlv *tlv = static_cast<const PSKcTlv *>(cur);
216  memcpy(aDataset.mPSKc.m8, tlv->GetPSKc(), tlv->GetLength());
217  aDataset.mIsPSKcSet = true;
218  break;
219  }
220 
221  case Tlv::kSecurityPolicy:
222  {
223  const SecurityPolicyTlv *tlv = static_cast<const SecurityPolicyTlv *>(cur);
224  aDataset.mSecurityPolicy.mRotationTime = tlv->GetRotationTime();
225  aDataset.mSecurityPolicy.mFlags = tlv->GetFlags();
226  aDataset.mIsSecurityPolicySet = true;
227  break;
228  }
229 
230  default:
231  {
232  break;
233  }
234  }
235 
236  cur = cur->GetNext();

404 {

406  const Timestamp *compareTimestamp = aCompare.GetTimestamp();
407  int rval;
408 
409  if (compareTimestamp == NULL && thisTimestamp == NULL)
410  {
411  rval = 0;
412  }
413  else if (compareTimestamp == NULL && thisTimestamp != NULL)
414  {
415  rval = -1;
416  }
417  else if (compareTimestamp != NULL && thisTimestamp == NULL)

458 ThreadError Dataset::Set(const Tlv &aTlv)
459 {
460  ThreadError error = kThreadError_None;
461  uint16_t bytesAvailable = sizeof(mTlvs) - mLength;
462  Tlv *old = Get(aTlv.GetType());
463 
464  if (old != NULL)
465  {
466  bytesAvailable += sizeof(Tlv) + old->GetLength();
467  }
468 
469  VerifyOrExit(sizeof(Tlv) + aTlv.GetLength() <= bytesAvailable, error = kThreadError_NoBufs);
470 
471  // remove old TLV
472  if (old != NULL)
473  {
474  Remove(reinterpret_cast<uint8_t *>(old), sizeof(Tlv) + old->GetLength());
475  }
476 
477  // add new TLV
478  memcpy(mTlvs + mLength, &aTlv, sizeof(Tlv) + aTlv.GetLength());
479  mLength += sizeof(Tlv) + aTlv.GetLength();
480 
481 exit:
482  return error;
483 }
