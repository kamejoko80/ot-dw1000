44 def convert_aux_sec_hdr_to_bytearray(aux_sec_hdr):
45  data = bytearray([aux_sec_hdr.security_level | ((aux_sec_hdr.key_id_mode & 0x03) << 3)])
46  data += struct.pack("<L", aux_sec_hdr.frame_counter)
47  data += aux_sec_hdr.key_id
48  return data
