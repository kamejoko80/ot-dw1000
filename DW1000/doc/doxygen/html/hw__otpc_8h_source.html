194 static inline void hw_otpc_init(void)
195 {
196  GLOBAL_INT_DISABLE();
197  if ((dg_configBLACK_ORCA_IC_REV == BLACK_ORCA_IC_REV_A)
198  || ((dg_configUSE_AUTO_CHIP_DETECTION == 1) && (CHIP_IS_AE))) {
199  /*
200  * Reset OTP controller
201  */
202  OTPC->OTPC_MODE_REG = HW_OTPC_MODE_STBY; // Set OTPC to standby
203  REG_SET_BIT(CRG_TOP, SYS_CTRL_REG, OTPC_RESET_REQ); // Reset the OTPC
204  REG_CLR_BIT(CRG_TOP, SYS_CTRL_REG, OTPC_RESET_REQ); // Get the OTPC out of Reset
205 
206  /*
207  * Initialize the POR registers
208  */
209  OTPC->OTPC_NWORDS_REG = OTPC_NWORDS_REG_RESET;
210  OTPC->OTPC_TIM1_REG = OTPC_TIM1_REG_RESET;
211  OTPC->OTPC_TIM2_REG = OTPC_TIM2_REG_RESET;
212  }
213 
214  /*
215  * Enable OTPC clock
216  */
217  CRG_TOP->CLK_AMBA_REG |= 1 << REG_POS(CRG_TOP, CLK_AMBA_REG, OTP_ENABLE);
218  GLOBAL_INT_RESTORE();
219 }
