105  break;

137  break;
138  }
139 
140  default:
141  {
142  break;
143  }
144  }
145 
146  cur = cur->GetNext();
147  }

337 ThreadError DatasetManager::Set(Coap::Header &aHeader, Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
338 {
339  Tlv tlv;
340  Timestamp *timestamp;
341  uint16_t offset = aMessage.GetOffset();
342  Tlv::Type type;
343  bool isUpdateFromCommissioner = false;
344  bool doesAffectConnectivity = false;
345  bool doesAffectMasterKey = false;
346  StateTlv::State state = StateTlv::kAccept;
347 
348  ActiveTimestampTlv activeTimestamp;
349  PendingTimestampTlv pendingTimestamp;
350  ChannelTlv channel;
351  CommissionerSessionIdTlv sessionId;
352  MeshLocalPrefixTlv meshLocalPrefix;
353  NetworkMasterKeyTlv masterKey;
354  PanIdTlv panId;
355 
356  activeTimestamp.SetLength(0);
357  pendingTimestamp.SetLength(0);
358  channel.SetLength(0);
359  masterKey.SetLength(0);
360  meshLocalPrefix.SetLength(0);
361  panId.SetLength(0);
362  pendingTimestamp.SetLength(0);
363  sessionId.SetLength(0);
364 
365  VerifyOrExit(mNetif.GetMle().GetDeviceState() == Mle::kDeviceStateLeader, state = StateTlv::kReject);
366 
367  // verify that TLV data size is less than maximum TLV value size
368  while (offset < aMessage.GetLength())
369  {
370  aMessage.Read(offset, sizeof(tlv), &tlv);
371  VerifyOrExit(tlv.GetLength() <= Dataset::kMaxValueSize, state = StateTlv::kReject);
372  offset += sizeof(tlv) + tlv.GetLength();
373  }
374 
375  // verify that does not overflow dataset buffer
376  VerifyOrExit((offset - aMessage.GetOffset()) <= Dataset::kMaxSize, state = StateTlv::kReject);
377 
378  type = (strcmp(mUriSet, OPENTHREAD_URI_ACTIVE_SET) == 0 ? Tlv::kActiveTimestamp : Tlv::kPendingTimestamp);
379 
380  if (Tlv::GetTlv(aMessage, Tlv::kActiveTimestamp, sizeof(activeTimestamp), activeTimestamp) != kThreadError_None)
381  {
382  ExitNow(state = StateTlv::kReject);
383  }
384 
385  VerifyOrExit(activeTimestamp.IsValid(), state = StateTlv::kReject);
386 
387  if (Tlv::GetTlv(aMessage, Tlv::kPendingTimestamp, sizeof(pendingTimestamp), pendingTimestamp) == kThreadError_None)
388  {
389  VerifyOrExit(pendingTimestamp.IsValid(), state = StateTlv::kReject);
390  }
391 
392  // verify the request includes a timestamp that is ahead of the locally stored value
393  timestamp = (type == Tlv::kActiveTimestamp) ?
394  static_cast<Timestamp *>(&activeTimestamp) :
395  static_cast<Timestamp *>(&pendingTimestamp);
396 
397  VerifyOrExit(mLocal.GetTimestamp() == NULL || mLocal.GetTimestamp()->Compare(*timestamp) > 0,
398  state = StateTlv::kReject);
399 
400  // check channel
401  if (Tlv::GetTlv(aMessage, Tlv::kChannel, sizeof(channel), channel) == kThreadError_None)
402  {
403  VerifyOrExit(channel.IsValid() &&
404  channel.GetChannel() >= kPhyMinChannel &&
405  channel.GetChannel() <= kPhyMaxChannel,
406  state = StateTlv::kReject);
407 
408  if (channel.GetChannel() != mNetif.GetMac().GetChannel())
409  {
410  doesAffectConnectivity = true;
411  }
412  }
413 
414  // check PAN ID
415  if (Tlv::GetTlv(aMessage, Tlv::kPanId, sizeof(panId), panId) == kThreadError_None &&
416  panId.IsValid() &&
417  panId.GetPanId() != mNetif.GetMac().GetPanId())
418  {
419  doesAffectConnectivity = true;
420  }
421 
422  // check mesh local prefix
423  if (Tlv::GetTlv(aMessage, Tlv::kMeshLocalPrefix, sizeof(meshLocalPrefix), meshLocalPrefix) == kThreadError_None &&
424  memcmp(meshLocalPrefix.GetMeshLocalPrefix(), mNetif.GetMle().GetMeshLocalPrefix(),
425  meshLocalPrefix.GetLength()))
426  {
427  doesAffectConnectivity = true;
428  }
429 
430  // check network master key
431  if (Tlv::GetTlv(aMessage, Tlv::kNetworkMasterKey, sizeof(masterKey), masterKey) == kThreadError_None &&
432  memcmp(masterKey.GetNetworkMasterKey(), mNetif.GetKeyManager().GetMasterKey(NULL),
433  masterKey.GetLength()))
434  {
435  doesAffectConnectivity = true;
436  doesAffectMasterKey = true;
437  }
438 
439  // check active timestamp rollback
440  if (type == Tlv::kPendingTimestamp &&
441  (masterKey.GetLength() == 0 ||
442  memcmp(masterKey.GetNetworkMasterKey(), mNetif.GetKeyManager().GetMasterKey(NULL),
443  masterKey.GetLength()) == 0))
444  {
445  // no change to master key, active timestamp must be ahead
446  const Timestamp *localActiveTimestamp = mNetif.GetActiveDataset().GetNetwork().GetTimestamp();
447 
448  VerifyOrExit(localActiveTimestamp == NULL || localActiveTimestamp->Compare(activeTimestamp) > 0,
449  state = StateTlv::kReject);
450  }
451 
452  // check commissioner session id
453  if (Tlv::GetTlv(aMessage, Tlv::kCommissionerSessionId, sizeof(sessionId), sessionId) == kThreadError_None)
454  {
455  CommissionerSessionIdTlv *localId;
456 
457  isUpdateFromCommissioner = true;
458 
459  localId = static_cast<CommissionerSessionIdTlv *>(mNetif.GetNetworkDataLeader().GetCommissioningDataSubTlv(
460  Tlv::kCommissionerSessionId));
461 
462  VerifyOrExit(sessionId.IsValid() &&
463  localId != NULL &&
464  localId->GetCommissionerSessionId() == sessionId.GetCommissionerSessionId(),
465  state = StateTlv::kReject);
466  }
467 
468  // verify an MGMT_ACTIVE_SET.req from a Commissioner does not affect connectivity
469  VerifyOrExit(!isUpdateFromCommissioner || type == Tlv::kPendingTimestamp || !doesAffectConnectivity,
470  state = StateTlv::kReject);
471 
472  // update dataset
473  if (type == Tlv::kPendingTimestamp && isUpdateFromCommissioner)
474  {
475  mLocal.Clear(true);
476  mLocal.Set(mNetif.GetActiveDataset().GetNetwork());
477  }
478 
479  if (type == Tlv::kPendingTimestamp || !doesAffectConnectivity)
480  {
481  offset = aMessage.GetOffset();
482 
483  while (offset < aMessage.GetLength())
484  {
485  OT_TOOL_PACKED_BEGIN
486  struct
487  {
488  Tlv tlv;
489  uint8_t value[Dataset::kMaxValueSize];
490  } OT_TOOL_PACKED_END data;
491 
492  aMessage.Read(offset, sizeof(Tlv), &data.tlv);
493  aMessage.Read(offset + sizeof(Tlv), data.tlv.GetLength(), data.value);
494 
495  switch (data.tlv.GetType())
496  {
497  case Tlv::kCommissionerSessionId:
498  // do not store Commissioner Session ID TLV
499  break;
500 
501  case Tlv::kDelayTimer:
502  {
503  DelayTimerTlv *delayTimerTlv = reinterpret_cast<DelayTimerTlv *>(&tlv);
504 
505  if (doesAffectMasterKey && delayTimerTlv->GetDelayTimer() < DelayTimerTlv::kDelayTimerDefault)
506  {
507  delayTimerTlv->SetDelayTimer(DelayTimerTlv::kDelayTimerDefault);
508  }
509  else if (delayTimerTlv->GetDelayTimer() < mNetif.GetLeader().GetDelayTimerMinimal())
510  {
511  delayTimerTlv->SetDelayTimer(mNetif.GetLeader().GetDelayTimerMinimal());
512  }
513  }
514 
515  // fall through
516 
517  default:
518  mLocal.Set(data.tlv);
519  break;
520  }
521 
522  offset += sizeof(Tlv) + data.tlv.GetLength();
523  }
524 
525  mLocal.Store();
526  mNetwork = mLocal;
527  mNetif.GetNetworkDataLeader().IncrementVersion();
528  mNetif.GetNetworkDataLeader().IncrementStableVersion();
529  }
530  else
531  {
532  mNetif.GetPendingDataset().ApplyActiveDataset(activeTimestamp, aMessage);
533  }
534 
535  // notify commissioner if update is from thread device
536  if (!isUpdateFromCommissioner)
537  {
538  BorderAgentLocatorTlv *borderAgentLocator;
539  Ip6::Address destination;
540 
541  borderAgentLocator = static_cast<BorderAgentLocatorTlv *>(mNetif.GetNetworkDataLeader().GetCommissioningDataSubTlv(
542  Tlv::kBorderAgentLocator));
543  VerifyOrExit(borderAgentLocator != NULL,);
544 
545  memset(&destination, 0, sizeof(destination));
546  destination = mNetif.GetMle().GetMeshLocal16();
547  destination.mFields.m16[4] = HostSwap16(0x0000);
548  destination.mFields.m16[5] = HostSwap16(0x00ff);
549  destination.mFields.m16[6] = HostSwap16(0xfe00);
550  destination.mFields.m16[7] = HostSwap16(borderAgentLocator->GetBorderAgentLocator());
551 
552  mNetif.GetLeader().SendDatasetChanged(destination);
553  }
554 
555 exit:
556 
557  if (mNetif.GetMle().GetDeviceState() == Mle::kDeviceStateLeader)
558  {
559  SendSetResponse(aHeader, aMessageInfo, state);
560  }
561 
562  return state == StateTlv::kAccept ? kThreadError_None : kThreadError_Drop;
563 }
