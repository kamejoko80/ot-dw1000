128  """Wait until `what` return True
129 
130  Args:
131  what (Callable[bool]): Call `wait()` again and again until it returns True
132  times (int): Maximum times of trials before giving up
133 
134  Returns:
135  True if success, False if times threshold reached
136 
137  """

141  if what() is True:

144  logger.exception('Wait failed')
145  else:
146  logger.warning('Trial[%d] failed', times)

681  """Handle a dialog.
682 
683  Returns:
684  bool True if no more dialogs expected,
685  False if more dialogs needed, and
686  None if not handled
687  """
688  done = self.on_dialog(dialog, title)
689  if isinstance(done, bool):
690  return done
691 
692  if title.startswith('Start DUT'):
693  body = dialog.find_element_by_id('cnfrmMsg').text
694  if 'Sleepy End Device' in body:
695  self.dut.mode = 's'
696  self.dut.child_timeout = self.child_timeout
697  elif 'End Device' in body:
698  self.dut.mode = 'rsn'
699  self.dut.child_timeout = self.child_timeout
700  else:
701  self.dut.mode = 'rsdn'
702 
703  if 'at channel' in body:
704  self.channel = int(body.split(':')[1])
