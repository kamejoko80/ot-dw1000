108  _In_ PMS_FILTER pFilter
109  )
110 {
111  LogFuncEntry(DRIVER_DEFAULT);
112 
113  NT_ASSERT(pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_RADIO_MODE);
114 
115  // Initialize the OpenThread radio capability flags
116  pFilter->otRadioCapabilities = 0;
117  if ((pFilter->DeviceCapabilities & OTLWF_DEVICE_CAP_RADIO_ACK_TIMEOUT) != 0)
118  pFilter->otRadioCapabilities |= kRadioCapsAckTimeout;
119  if ((pFilter->DeviceCapabilities & OTLWF_DEVICE_CAP_RADIO_MAC_RETRY_AND_COLLISION_AVOIDANCE) != 0)
120  pFilter->otRadioCapabilities |= kRadioCapsTransmitRetries;
121  if ((pFilter->DeviceCapabilities & OTLWF_DEVICE_CAP_RADIO_ENERGY_SCAN) != 0)
122  pFilter->otRadioCapabilities |= kRadioCapsEnergyScan;
123 
124  pFilter->otPhyState = kStateDisabled;
125  pFilter->otCurrentListenChannel = 0xFF;
126  pFilter->otPromiscuous = false;
127 
128  pFilter->otReceiveFrame.mPsdu = pFilter->otReceiveMessage;
129  pFilter->otTransmitFrame.mPsdu = pFilter->otTransmitMessage;
130 
131  pFilter->otPendingMacOffloadEnabled = FALSE;
132  pFilter->otPendingShortAddressCount = 0;
133  pFilter->otPendingExtendedAddressCount = 0;
134 
135  // Cache the factory address
136  otLwfRadioGetFactoryAddress(pFilter);
137 
138  LogInfo(DRIVER_DEFAULT, "Filter %p PhyState = kStateDisabled.", pFilter);
139 
140  LogFuncExit(DRIVER_DEFAULT);

352 ThreadError otPlatRadioReceive(_In_ otInstance *otCtx, uint8_t aChannel)
353 {
354  NT_ASSERT(otCtx);
355  PMS_FILTER pFilter = otCtxToFilter(otCtx);
356 
357  NT_ASSERT(pFilter->otPhyState != kStateDisabled);
358  if (pFilter->otPhyState == kStateDisabled) return kThreadError_Busy;
359 
360  LogFuncEntryMsg(DRIVER_DATA_PATH, "Filter: %p", pFilter);
361 
362  // Update current channel if different
363  if (pFilter->otCurrentListenChannel != aChannel)
364  {
365  NTSTATUS status;
366 
367  NT_ASSERT(aChannel >= 11 && aChannel <= 26);
368 
369  LogInfo(DRIVER_DEFAULT, "Filter %p new Listen Channel = %u.", pFilter, aChannel);
370  pFilter->otCurrentListenChannel = aChannel;
371 
372  // Indicate to the miniport
373  status =
374  otLwfCmdSetProp(
375  pFilter,
376  SPINEL_PROP_PHY_CHAN,
377  SPINEL_DATATYPE_UINT8_S,
378  aChannel
379  );
380  if (!NT_SUCCESS(status))
381  {
382  LogError(DRIVER_DEFAULT, "Set SPINEL_PROP_PHY_CHAN failed, %!STATUS!", status);
383  }
384  }
385 
386  // Only transition to the receive state if we were sleeping; otherwise we
387  // are already in receive or transmit state.
388  if (pFilter->otPhyState == kStateSleep)
389  {
390  pFilter->otPhyState = kStateReceive;
391  LogInfo(DRIVER_DEFAULT, "Filter %p PhyState = kStateReceive.", pFilter);
392 
393  NTSTATUS status =
394  otLwfCmdSetProp(
395  pFilter,
396  SPINEL_PROP_MAC_RAW_STREAM_ENABLED,
397  SPINEL_DATATYPE_BOOL_S,
398  TRUE
399  );
400  if (!NT_SUCCESS(status))
401  {
402  LogError(DRIVER_DEFAULT, "Set SPINEL_PROP_MAC_RAW_STREAM_ENABLED (true) failed, %!STATUS!", status);
403  }
404 
405  // Set the event to indicate we can process NBLs
406  KeSetEvent(&pFilter->EventWorkerThreadProcessNBLs, 0, FALSE);
407  }
408 
409  LogFuncExit(DRIVER_DATA_PATH);
410 
411  return kThreadError_None;
412 }
