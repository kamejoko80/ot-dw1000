216 ThreadError Local::SendServerDataNotification(void)
217 {
218  ThreadError error = kThreadError_None;
219  uint16_t rloc = mNetif.GetMle().GetRloc16();
220 
221  if ((mNetif.GetMle().GetDeviceMode() & Mle::ModeTlv::kModeFFD) != 0 &&
222  (mNetif.GetMle().IsRouterRoleEnabled()) &&
223  (mNetif.GetMle().GetDeviceState() < Mle::kDeviceStateRouter) &&
224  (mNetif.GetMle().GetActiveRouterCount() < mNetif.GetMle().GetRouterUpgradeThreshold()))
225  {
226  ExitNow(error = kThreadError_InvalidState);
227  }
228 
229  UpdateRloc();
230 
231  VerifyOrExit(!IsOnMeshPrefixConsistent() || !IsExternalRouteConsistent(), ClearResubmitDelayTimer());
232 
233  if (mOldRloc == rloc)
234  {
235  mOldRloc = Mac::kShortAddrInvalid;
236  }
237 
238  SuccessOrExit(error = NetworkData::SendServerDataNotification(mOldRloc));
239  mOldRloc = rloc;
