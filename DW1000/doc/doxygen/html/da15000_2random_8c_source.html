50 int zhal_get_entropy(uint8_t *outEntropy, size_t inSize)
51 {
52  uint32_t randword;
53  unsigned char *pbuf = outEntropy;
54  const size_t req_words = inSize >> 2;
55  const unsigned char *pbuf_end = outEntropy + inSize;
56  const unsigned char *preq_words_end = (unsigned char *)(((uint32_t *)outEntropy) + req_words);
57  ptrdiff_t remainder_bytes = pbuf_end - preq_words_end;
58 
59  hw_trng_enable(NULL);
60 
61  for (pbuf = outEntropy; pbuf < preq_words_end; pbuf += 4)
62  {
63  /* Wait for a random word to become available in the TRNG FIFO. */
64  while ((TRNG->TRNG_FIFOLVL_REG & TRNG_TRNG_FIFOLVL_REG_TRNG_FIFOLVL_Msk) == 0);
65 
66  randword = *((volatile uint32_t *)HW_TRNG_RAM);
67  memcpy(pbuf, &randword, 4);
68  }
69 
70  if (remainder_bytes)
71  {
72  while ((TRNG->TRNG_FIFOLVL_REG & TRNG_TRNG_FIFOLVL_REG_TRNG_FIFOLVL_Msk) == 0);
73 
74  randword = *((volatile uint32_t *)HW_TRNG_RAM);
75  memcpy(pbuf, &randword, remainder_bytes);
76  }
77 
78  hw_trng_disable();
79 
80  return 0;
81 }
