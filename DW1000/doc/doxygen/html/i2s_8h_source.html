683 I2SIntStatus(uint32_t ui32Base, bool bMasked)
684 {
685  uint32_t ui32Mask;
686 
687  //
688  // Check the arguments.
689  //
690  ASSERT(I2SBaseValid(ui32Base));
691  (void)ui32Base;
692 
693  //
694  // Return either the interrupt status or the raw interrupt status as
695  // requested.
696  //
697  if(bMasked)
698  {
699  ui32Mask = HWREG(I2S0_BASE + I2S_O_IRQFLAGS);
700  return(ui32Mask & HWREG(I2S0_BASE + I2S_O_IRQMASK));
701  }
702  else
703  {
704  return(HWREG(I2S0_BASE + I2S_O_IRQFLAGS));
