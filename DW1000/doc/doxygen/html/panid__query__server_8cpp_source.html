66 void PanIdQueryServer::HandleQuery(Coap::Header &aHeader, Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
67 {
68  MeshCoP::PanIdTlv panId;
69  MeshCoP::ChannelMask0Tlv channelMask;
70  Ip6::MessageInfo responseInfo(aMessageInfo);
71 
72  VerifyOrExit(aHeader.GetCode() == kCoapRequestPost, ;);
73 
74  SuccessOrExit(MeshCoP::Tlv::GetTlv(aMessage, MeshCoP::Tlv::kChannelMask, sizeof(channelMask), channelMask));

118 ThreadError PanIdQueryServer::SendConflict(void)
119 {
120  ThreadError error = kThreadError_None;
121  Coap::Header header;
122  MeshCoP::ChannelMask0Tlv channelMask;
123  MeshCoP::PanIdTlv panId;
124  Ip6::MessageInfo messageInfo;
125  Message *message;
126 
127  header.Init(kCoapTypeConfirmable, kCoapRequestPost);
128  header.SetToken(Coap::Header::kDefaultTokenLength);
129  header.AppendUriPathOptions(OPENTHREAD_URI_PANID_CONFLICT);
130  header.SetPayloadMarker();
131 
132  VerifyOrExit((message = mNetif.GetCoapClient().NewMeshCoPMessage(header)) != NULL, error = kThreadError_NoBufs);
133 
134  channelMask.Init();
135  channelMask.SetMask(mChannelMask);
136  SuccessOrExit(error = message->Append(&channelMask, sizeof(channelMask)));
137 
138  panId.Init();
139  panId.SetPanId(mPanId);
140  SuccessOrExit(error = message->Append(&panId, sizeof(panId)));
141 
142  messageInfo.SetPeerAddr(mCommissioner);
143  messageInfo.SetPeerPort(kCoapUdpPort);
144  SuccessOrExit(error = mNetif.GetCoapClient().SendMessage(*message, messageInfo));
145 
146  otLogInfoMeshCoP("sent panid conflict");
147 
148 exit:
149 
150  if (error != kThreadError_None && message != NULL)
151  {
152  message->Free();
153  }
