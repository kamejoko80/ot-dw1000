88 static int TRNGPoll(unsigned char *aOutput, size_t aLen, size_t *oLen)
89 {
90  size_t length = 0;
91  union
92  {
93  uint32_t u32[2];
94  uint8_t u8[8];
95  } buffer;
96 
97  while (length < aLen)
98  {
99  if (length % 8 == 0)
100  {
101  /* we've run to the end of the buffer */
102  while (!(TRNGStatusGet() & TRNG_NUMBER_READY));
103 
104  /*
105  * don't use TRNGNumberGet here because it will tell the TRNG to
106  * refil the entropy pool, instad we do it ourself.
107  */
108  buffer.u32[0] = HWREG(TRNG_BASE + TRNG_O_OUT0);
109  buffer.u32[1] = HWREG(TRNG_BASE + TRNG_O_OUT1);
110  HWREG(TRNG_BASE + TRNG_O_IRQFLAGCLR) = 0x1;
111  }
112 
113  aOutput[length] = buffer.u8[length % 8];
114 
115  length++;
116  *oLen = length;
117  }
118 
119  return 0;
120 }

126 ThreadError otPlatRandomSecureGet(uint16_t aInputLength, uint8_t *aOutput, uint16_t *aOutputLength)
127 {
128  ThreadError error = kThreadError_None;
129  size_t temp_size;
130  size_t length = aInputLength;
131 
132  VerifyOrExit(aOutput && aOutputLength, error = kThreadError_InvalidArgs);
133 
134  VerifyOrExit(TRNGPoll((unsigned char *)aOutput, length, &temp_size) != 0, error = kThreadError_Failed);
135 
136 exit:
137  *aOutputLength = temp_size;
138  return error;
139 }
