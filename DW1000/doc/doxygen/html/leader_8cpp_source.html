161 void Leader::HandleKeepAlive(Coap::Header &aHeader, Message &aMessage, const Ip6::MessageInfo &aMessageInfo)
162 {
163  StateTlv state;
164  CommissionerSessionIdTlv sessionId;
165  StateTlv::State responseState;
166 
167  otLogInfoMeshCoP("received keep alive");
168 
169  SuccessOrExit(Tlv::GetTlv(aMessage, Tlv::kState, sizeof(state), state));
170  VerifyOrExit(state.IsValid(),);
171 
172  SuccessOrExit(Tlv::GetTlv(aMessage, Tlv::kCommissionerSessionId, sizeof(sessionId), sessionId));
173  VerifyOrExit(sessionId.IsValid(),);
174 
175  if (sessionId.GetCommissionerSessionId() != mSessionId)
176  {
177  responseState = StateTlv::kReject;
178  }
179  else if (state.GetState() != StateTlv::kAccept)
180  {
181  responseState = StateTlv::kReject;
182  mTimer.Stop();
183  ResignCommissioner();
184  }
185  else
186  {
187  responseState = StateTlv::kAccept;
188  mTimer.Start(Timer::SecToMsec(kTimeoutLeaderPetition));
189  }
190 
191  SendKeepAliveResponse(aHeader, aMessageInfo, responseState);
192 
193 exit:
194  return;
