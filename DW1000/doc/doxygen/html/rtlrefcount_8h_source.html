220 /*++
221 
222 Routine Description:
223 
224  This function increases the specified reference count by the specified bias,
225  unless the reference count was previously zero.
226 
227 Arguments:
228 
229  RefCount - Supplies a pointer to a reference count.
230 
231  Bias - Supplies a reference bias amount.
232 
233 Return Value:
234 
235  TRUE if the reference count was incremented, FALSE otherwise.
236 
237 --*/
238 
239 {
240 
241  RTL_REFERENCE_COUNT NewValue;
242  RTL_REFERENCE_COUNT OldValue;
243 
244  PrefetchForWrite(RefCount);
245  OldValue = ReadLongPtrNoFence(RefCount);
246  for (;;) {
247  NewValue = OldValue + Bias;
248  if ((ULONG_PTR)NewValue > Bias) {
249  NewValue = RtlCompareExchangeLongPtrNoFence(RefCount,
250  NewValue,
251  OldValue);
252 
253  if (NewValue == OldValue) {
254  return TRUE;
255  }
256 
257  OldValue = NewValue;
258 
259  } else if ((ULONG_PTR)NewValue == Bias) {
260  return FALSE;
261 
262  } else {
263  __fastfail(FAST_FAIL_INVALID_REFERENCE_COUNT);
264  }
265  }
266 }

276 Routine Description:
277 
278  This function reduces the specified reference count, potentially triggering

280 
281 Arguments:
282 
283  RefCount - Supplies a pointer to a reference count.
284 
285 Return Value:
286 
287  TRUE if the object should be destroyed, FALSE otherwise.
288 
289 --*/

291 {
292 
293  RTL_REFERENCE_COUNT NewValue;
294 
295  //
296  // A release fence is required to ensure all guarded memory accesses are
297  // complete before any thread can begin destroying the object.
298  //
299 
300  NewValue = RtlDecrementLongPtrRelease(RefCount);
301  if (NewValue > 0) {
302  return FALSE;
303 
304  } else if (NewValue == 0) {
305 
306  //
307  // An acquire fence is required before object destruction to ensure
308  // that the destructor cannot observe values changing on other threads.
309  //

312  return TRUE;
313  }
