66 //
67 // Description for a circular buffer (also called "ring buffer")
68 // which is used as up- (T->H) or down-buffer (H->T)
69 //
70 typedef struct
71 {
72  const char *sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
73  char *pBuffer; // Pointer to start of buffer
74  unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
75  volatile unsigned
76  WrOff; // Position of next item to be written by either host (down-buffer) or target (up-buffer). Must be volatile since it may be modified by host (down-buffer)
77  volatile unsigned
78  RdOff; // Position of next item to be read by target (down-buffer) or host (up-buffer). Must be volatile since it may be modified by host (up-buffer)
79  unsigned Flags; // Contains configuration flags
80 } SEGGER_RTT_RING_BUFFER;
81 
82 //
83 // RTT control block which describes the number of buffers available
84 // as well as the configuration for each buffer
85 //
86 //
87 typedef struct
88 {
89  char acID[16]; // Initialized to "SEGGER RTT"
90  int
91  MaxNumUpBuffers; // Initialized to SEGGER_RTT_MAX_NUM_UP_BUFFERS (type. 2)
92  int
93  MaxNumDownBuffers; // Initialized to SEGGER_RTT_MAX_NUM_DOWN_BUFFERS (type. 2)
94  SEGGER_RTT_RING_BUFFER
95  aUp[SEGGER_RTT_MAX_NUM_UP_BUFFERS]; // Up buffers, transferring information up from target via debug probe to host
96  SEGGER_RTT_RING_BUFFER
97  aDown[SEGGER_RTT_MAX_NUM_DOWN_BUFFERS]; // Down buffers, transferring information down from host via debug probe to target
98 } SEGGER_RTT_CB;
