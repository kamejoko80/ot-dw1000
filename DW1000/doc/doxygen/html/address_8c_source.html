33 BOOLEAN
34 otLwfOnAddressAdded(
35  _In_ PMS_FILTER pFilter,
36  _In_ const otNetifAddress* Addr,
37  _In_ BOOLEAN UpdateWindows
38  )
39 {
40  if (pFilter->otCachedAddrCount >= OT_MAX_ADDRESSES)
41  {
42  LogError(DRIVER_DEFAULT, "Failing to add new address as we have reached our max!");
43  return FALSE;
44  }
45 
46  LogInfo(DRIVER_DEFAULT, "Interface %!GUID! adding address: %!IPV6ADDR! (%u-bit prefix)",
47  &pFilter->InterfaceGuid,
48  (PIN6_ADDR)&Addr->mAddress,
49  Addr->mPrefixLength
50  );
51 
52  // Update local cache
53  memcpy(pFilter->otCachedAddr + pFilter->otCachedAddrCount, Addr, sizeof(IN6_ADDR));
54  pFilter->otCachedAddrCount++;
55 
56  // If this is link local, cache it as our link local address
57  if (IN6_IS_ADDR_LINKLOCAL((PIN6_ADDR)&Addr->mAddress))
58  {
59  memcpy(&pFilter->otLinkLocalAddr, Addr, sizeof(IN6_ADDR));
60  }
61 
62  // Update Windows if necessary
63  if (UpdateWindows)
64  {
65  NTSTATUS status;
66  MIB_UNICASTIPADDRESS_ROW newRow;
67  MIB_IPFORWARD_ROW2 newRouteRow;
68  COMPARTMENT_ID OriginalCompartmentID;
69  InitializeUnicastIpAddressEntry(&newRow);
70  InitializeIpForwardEntry(&newRouteRow);
71 
72  newRow.InterfaceIndex = pFilter->InterfaceIndex;
73  newRow.InterfaceLuid = pFilter->InterfaceLuid;
74  newRow.Address.si_family = AF_INET6;
75  newRow.Address.Ipv6.sin6_family = AF_INET6;
76 
77  static_assert(sizeof(IN6_ADDR) == sizeof(otIp6Address), "Windows and OpenThread IPv6 Addr Structs must be same size");
78 
79  memcpy(&newRow.Address.Ipv6.sin6_addr, &Addr->mAddress, sizeof(IN6_ADDR));
80  newRow.OnLinkPrefixLength = Addr->mPrefixLength;
81  newRow.PreferredLifetime = Addr->mPreferred ? 0xffffffff : 0;
82  newRow.ValidLifetime = Addr->mValid ? 0xffffffff : 0;
83  newRow.PrefixOrigin = IpPrefixOriginOther; // Derived from network XPANID
84  newRow.SkipAsSource = FALSE; // Allow automatic binding to this address (default)
85 
86  if (IN6_IS_ADDR_LINKLOCAL(&newRow.Address.Ipv6.sin6_addr))
87  {
88  newRow.SuffixOrigin = IpSuffixOriginLinkLayerAddress; // Derived from Extended MAC address
89  }
90  else
91  {
92  newRow.SuffixOrigin = IpSuffixOriginRandom; // Was created randomly
93  }
94 
95  // Make sure we are in the right compartment
96  (VOID)otLwfSetCompartment(pFilter, &OriginalCompartmentID);
97 
98  status = CreateUnicastIpAddressEntry(&newRow);
99  //NT_ASSERT(NT_SUCCESS(status));
100  if (!NT_SUCCESS(status))
101  {
102  LogError(DRIVER_DEFAULT, "CreateUnicastIpAddressEntry failed %!STATUS!", status);
103  }
104 
105  newRouteRow.InterfaceIndex = pFilter->InterfaceIndex;
106  newRouteRow.InterfaceLuid = pFilter->InterfaceLuid;
107  newRouteRow.DestinationPrefix.Prefix.si_family = AF_INET6;
108  newRouteRow.DestinationPrefix.PrefixLength = 0;
109 
110  status = CreateIpForwardEntry2(&newRouteRow);
111  if (!NT_SUCCESS(status) && status != STATUS_DUPLICATE_OBJECTID)
112  {
113  LogVerbose(DRIVER_DEFAULT, "CreateIpForwardEntry2 failed %!STATUS!", status);
114  }
115 
116  // Revert back to original compartment
117  otLwfRevertCompartment(OriginalCompartmentID);
118  }
119 
120  return TRUE;
121 }

125 otLwfOnAddressRemoved(
126  _In_ PMS_FILTER pFilter,
127  _In_ ULONG CachedIndex,
128  _In_ BOOLEAN UpdateWindows
129  )
130 {
131  // Cache address before we delete local cache
132  IN6_ADDR Addr = pFilter->otCachedAddr[CachedIndex];
133 
134  LogInfo(DRIVER_DEFAULT, "Interface %!GUID! removing address: %!IPV6ADDR!", &pFilter->InterfaceGuid, &Addr);
135 
136  NT_ASSERT(pFilter->otCachedAddrCount != 0);
137  NT_ASSERT(CachedIndex < pFilter->otCachedAddrCount);
138 
139  // Remove the cached entry
140  if (CachedIndex + 1 != pFilter->otCachedAddrCount)
141  memmove(pFilter->otCachedAddr + CachedIndex,
142  pFilter->otCachedAddr + CachedIndex + 1,
143  (pFilter->otCachedAddrCount - CachedIndex - 1) * sizeof(IN6_ADDR)
144  );
145  pFilter->otCachedAddrCount--;
146 
147  // Update Windows if necessary
148  if (UpdateWindows)
149  {
150  MIB_UNICASTIPADDRESS_ROW deleteRow;
151  COMPARTMENT_ID OriginalCompartmentID;
152  InitializeUnicastIpAddressEntry(&deleteRow);
153 
154  deleteRow.InterfaceIndex = pFilter->InterfaceIndex;
155  deleteRow.InterfaceLuid = pFilter->InterfaceLuid;
156  deleteRow.Address.si_family = AF_INET6;
157 
158  deleteRow.Address.Ipv6.sin6_addr = Addr;
159 
160  // Make sure we are in the right compartment
161  (VOID)otLwfSetCompartment(pFilter, &OriginalCompartmentID);
162 
163  // Best effort remove address from TCPIP
164  (VOID)DeleteUnicastIpAddressEntry(&deleteRow);
165 
166  // Revert back to original compartment
167  otLwfRevertCompartment(OriginalCompartmentID);
168  }

185 otLwfInitializeAddresses(
186  _In_ PMS_FILTER pFilter
187  )
188 {
189  NTSTATUS status = STATUS_SUCCESS;
190 
191  PMIB_UNICASTIPADDRESS_TABLE pMibUnicastAddressTable = NULL;
192  COMPARTMENT_ID OriginalCompartmentID;
193 
194  LogFuncEntry(DRIVER_DEFAULT);
195 
196  pFilter->otCachedAddrCount = 0;
197 
198  // Make sure we are in the right compartment
199  (VOID)otLwfSetCompartment(pFilter, &OriginalCompartmentID);
200 
201  // Query the table for the current compartment
202  status = GetUnicastIpAddressTable(AF_INET6, &pMibUnicastAddressTable);
203 
204  // Revert the compartment, now that we have the table
205  otLwfRevertCompartment(OriginalCompartmentID);
206 
207  if (!NT_SUCCESS(status))
208  {
209  LogError(DRIVER_DEFAULT, "GetUnicastIpAddressTable failed, %!STATUS!", status);
210  goto error;
211  }
212 
213  // Iterate through the addresses and delete (best effort) the ones for our interface
214  for (ULONG Index = 0; Index < pMibUnicastAddressTable->NumEntries; Index++)
215  {
216  MIB_UNICASTIPADDRESS_ROW* row = &pMibUnicastAddressTable->Table[Index];
217 
218  if ((0 == memcmp(&row->InterfaceLuid, &pFilter->InterfaceLuid, sizeof(NET_LUID))))
219  {
220  LogInfo(DRIVER_DEFAULT, "Caching initial address: %!IPV6ADDR!", &row->Address.Ipv6.sin6_addr);
221  memcpy(pFilter->otCachedAddr + pFilter->otCachedAddrCount, &row->Address.Ipv6.sin6_addr, sizeof(IN6_ADDR));
222  pFilter->otCachedAddrCount++;
223  }
224  }
225 
226 error:
227 
228  if (NULL != pMibUnicastAddressTable)
229  {
230  FreeMibTable(pMibUnicastAddressTable);
231  }
232 
233  LogFuncExitNT(DRIVER_DEFAULT, status);
234 
235  return status;
236 }

242 otLwfAddressChangeCallback(
243  _In_ PVOID CallerContext,
244  _In_opt_ PMIB_UNICASTIPADDRESS_ROW Row,
245  _In_ MIB_NOTIFICATION_TYPE NotificationType
246  )
247 {
248  PMS_FILTER pFilter = (PMS_FILTER)CallerContext;
249  if (Row == NULL || pFilter == NULL) return;
250 
251  // Ignore notifications that aren't for our interface
252  if (Row->InterfaceIndex != pFilter->InterfaceIndex) return;
253 
254  LogFuncEntryMsg(DRIVER_DEFAULT, "%p (%u) %!IPV6ADDR!", pFilter, NotificationType, &Row->Address.Ipv6.sin6_addr);
255 
256  // Since we don't pass in the initial flag, we shouldn't get this type
257  NT_ASSERT(NotificationType != MibInitialNotification);
258 
259  // Make sure we can reference the interface
260  if (ExAcquireRundownProtection(&pFilter->ExternalRefs))
261  {
262  if (pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_RADIO_MODE)
263  {
264  // Queue up the event for processing
265  otLwfEventProcessingIndicateAddressChange(
266  pFilter,
267  NotificationType,
268  &Row->Address.Ipv6.sin6_addr
269  );
270  }
271  else
272  {
273  NT_ASSERT(FALSE); // Need to add support for this in tunnel mode
274  }
275 
276  // Release reference on the interface
277  ExReleaseRundownProtection(&pFilter->ExternalRefs);
278  }
279 
280  LogFuncExit(DRIVER_DEFAULT);
281 }

283 // Callback on the OpenThread thread for processing an address change
284 _IRQL_requires_max_(PASSIVE_LEVEL)
285 VOID
286 otLwfEventProcessingAddressChanged(
287  _In_ PMS_FILTER pFilter,
288  _In_ MIB_NOTIFICATION_TYPE NotificationType,
289  _In_ PIN6_ADDR pAddr
290  )
291 {
292  LogFuncEntryMsg(DRIVER_DEFAULT, "%p (%u)", pFilter, NotificationType);
293 
294  if (NotificationType == MibAddInstance ||
295  NotificationType == MibParameterNotification)
296  {
297  MIB_UNICASTIPADDRESS_ROW Row;
298  InitializeUnicastIpAddressEntry(&Row);
299 
300  Row.Address.si_family = AF_INET6;
301  Row.Address.Ipv6.sin6_addr = *pAddr;
302  Row.InterfaceIndex = pFilter->InterfaceIndex;
303  Row.InterfaceLuid = pFilter->InterfaceLuid;
304 
305  NTSTATUS status = GetUnicastIpAddressEntry(&Row);
306  if (!NT_SUCCESS(status))
307  {
308  LogError(DRIVER_DEFAULT, "GetUnicastIpAddressEntry failed, %!STATUS!", status);
309  }
310  else
311  {
312  otNetifAddress otAddr = {0};
313  memcpy(&otAddr.mAddress, pAddr, sizeof(IN6_ADDR));
314  otAddr.mPreferred = Row.PreferredLifetime != 0;
315  otAddr.mPrefixLength = Row.OnLinkPrefixLength;
316  otAddr.mValid = Row.ValidLifetime != 0;
317 
318  BOOLEAN ShouldDelete = FALSE;
319  BOOLEAN AddedToCache = FALSE;
320  BOOLEAN IsCached = otLwfFindCachedAddrIndex(pFilter, pAddr) != -1;
321 
322  // Ignore link local addresses
323  if (IN6_IS_ADDR_LINKLOCAL(pAddr) && !IsCached)
324  {
325  ShouldDelete = TRUE;
326  goto add_complete;
327  }
328 
329  // Add to the cache if this is a new address
330  if (NotificationType == MibAddInstance && !IsCached)
331  {
332  AddedToCache = otLwfOnAddressAdded(pFilter, &otAddr, FALSE);
333  if (AddedToCache == FALSE)
334  {
335  ShouldDelete = TRUE;
336  goto add_complete;
337  }
338  }
339 
340  // Update OpenThread if we don't have this cached or it is being updated
341  if (!IsCached/* || NotificationType == MibParameterNotification*/)
342  {
343  LogInfo(DRIVER_DEFAULT, "Filter %p trying to add/update address: %!IPV6ADDR!", pFilter, pAddr);
344 
345  // Add (or update) the address to OpenThread
346  ThreadError otError = otAddUnicastAddress(pFilter->otCtx, &otAddr);
347  if (otError != kThreadError_None)
348  {
349  LogError(DRIVER_DEFAULT, "otAddUnicastAddress failed, %!otError!", otError);
350  ShouldDelete = otError == kThreadError_NoBufs ? TRUE : FALSE;
351  }
352  }
353 
354  add_complete:
355 
356  // Remove it from TCPIP if necessary
357  if (ShouldDelete)
358  {
359  LogInfo(DRIVER_DEFAULT, "Filter %p deleting recently added address: %!IPV6ADDR!", pFilter, pAddr);
360 
361  // Best effort remove address from TCPIP
362  (VOID)DeleteUnicastIpAddressEntry(&Row);
363  }
364  }
365  }
366  else if (NotificationType == MibDeleteInstance)
367  {
368  // Look for the address in our cache
369  int index = otLwfFindCachedAddrIndex(pFilter, pAddr);
370 
371  // If it's not already deleted from our cache, then Windows
372  // is deleting the adddress and we need to update OpenThread.
373  if (index != -1)
374  {
375  // Update our cache
376  otLwfOnAddressRemoved(pFilter, (ULONG)index, FALSE);
377 
378  LogInfo(DRIVER_DEFAULT, "Filter %p trying to remove address: %!IPV6ADDR!", pFilter, pAddr);
379 
380  // Find the correct address from OpenThread to remove (best effort)
381  (void)otRemoveUnicastAddress(pFilter->otCtx, (otIp6Address*)pAddr);
382  }
383  }
384 
385  LogFuncExit(DRIVER_DEFAULT);
386 }

389 VOID
390 otLwfRadioAddressesUpdated(
391  _In_ PMS_FILTER pFilter
392  )
393 {
394  LogFuncEntry(DRIVER_DEFAULT);
395 
396  ULONG FoundInOpenThread = 0; // Bit field
397  ULONG OriginalCacheLength = pFilter->otCachedAddrCount;
398 
399  NT_ASSERT(pFilter->DeviceStatus == OTLWF_DEVICE_STATUS_RADIO_MODE);
400 
401  const otNetifAddress* addr = otGetUnicastAddresses(pFilter->otCtx);
402 
403  // Process the addresses
404  while (addr)
405  {
406  int index = otLwfFindCachedAddrIndex(pFilter, (PIN6_ADDR)&addr->mAddress);
407  if (index == -1)
408  {
409  otLwfOnAddressAdded(pFilter, addr, TRUE);
410  }
411  else
412  {
413  NT_ASSERT(index < 8 * sizeof(FoundInOpenThread));
414  FoundInOpenThread |= 1 << index;
415  }
416  addr = addr->mNext;
417  }
418 
419  // Look for missing addresses and mark them as removed
420  for (int i = OriginalCacheLength - 1; i >= 0; i--)
421  {
422  if ((FoundInOpenThread & (1 << i)) == 0)
423  {
424  otLwfOnAddressRemoved(pFilter, (ULONG)i, TRUE);
425  }
426  }
427 
428  LogFuncExit(DRIVER_DEFAULT);
429 }
