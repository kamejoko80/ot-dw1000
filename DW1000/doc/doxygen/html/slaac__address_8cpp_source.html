54 void Slaac::UpdateAddresses(otInstance *aInstance, otNetifAddress *aAddresses, uint32_t aNumAddresses,
55  IidCreator aIidCreator, void *aContext)
56 {
57  otNetworkDataIterator iterator;
58  otBorderRouterConfig config;
59 
60  // remove addresses
61  for (size_t i = 0; i < aNumAddresses; i++)
62  {
63  otNetifAddress *address = &aAddresses[i];
64  bool found = false;
65 
66  if (!address->mValid)
67  {
68  continue;
69  }
70 
71  iterator = OT_NETWORK_DATA_ITERATOR_INIT;
72 
73  while (otGetNextOnMeshPrefix(aInstance, false, &iterator, &config) == kThreadError_None)
74  {
75  if (config.mSlaac == false)
76  {
77  continue;
78  }
79 
80  if (otIp6PrefixMatch(&config.mPrefix.mPrefix, &address->mAddress) >= config.mPrefix.mLength &&
81  config.mPrefix.mLength == address->mPrefixLength)
82  {
83  found = true;
84  break;
85  }
86  }
87 
88  if (!found)
89  {
90  otRemoveUnicastAddress(aInstance, &address->mAddress);
91  address->mValid = false;
92  }
93  }
94 
95  // add addresses
96  iterator = OT_NETWORK_DATA_ITERATOR_INIT;
97 
98  while (otGetNextOnMeshPrefix(aInstance, false, &iterator, &config) == kThreadError_None)
99  {
100  bool found = false;
101 
102  if (config.mSlaac == false)
103  {
104  continue;
105  }
106 
107  for (size_t i = 0; i < aNumAddresses; i++)
108  {
109  otNetifAddress *address = &aAddresses[i];
110 
111  if (!address->mValid)
112  {
113  continue;
114  }
115 
116  if (otIp6PrefixMatch(&config.mPrefix.mPrefix, &address->mAddress) >= config.mPrefix.mLength &&
117  config.mPrefix.mLength == address->mPrefixLength)
118  {
119  found = true;
120  break;
121  }
122  }
123 
124  if (!found)

126  for (size_t i = 0; i < aNumAddresses; i++)

128  otNetifAddress *address = &aAddresses[i];
129 
130  if (address->mValid)

132  continue;
133  }
134

137 
138  address->mPrefixLength = config.mPrefix.mLength;
139  address->mPreferred = config.mPreferred;
140  address->mValid = true;
141 
142  if (aIidCreator(aInstance, address, aContext) != kThreadError_None)
143  {
144  CreateRandomIid(aInstance, address, aContext);
145  }
146 
147  otAddUnicastAddress(aInstance, address);
148  break;
149  }
150  }
151  }

180 ThreadError SemanticallyOpaqueIidGenerator::CreateIidOnce(otInstance *aInstance, otNetifAddress *aAddress)
181 {
182  ThreadError error = kThreadError_None;
183  Crypto::Sha256 sha256;
184  uint8_t hash[Crypto::Sha256::kHashSize];
185  Ip6::Address *address = static_cast<Ip6::Address *>(&aAddress->mAddress);
186 
187  sha256.Start();
188 
189  sha256.Update(aAddress->mAddress.mFields.m8, aAddress->mPrefixLength / 8);
190 
191  VerifyOrExit(mInterfaceId != NULL, error = kThreadError_InvalidArgs);
192  sha256.Update(mInterfaceId, mInterfaceIdLength);
193 
194  if (mNetworkIdLength)
195  {
196  VerifyOrExit(mNetworkId != NULL, error = kThreadError_InvalidArgs);
197  sha256.Update(mNetworkId, mNetworkIdLength);
198  }
199 
200  sha256.Update(static_cast<uint8_t *>(&mDadCounter), sizeof(mDadCounter));
201 
202  VerifyOrExit(mSecretKey != NULL, error = kThreadError_InvalidArgs);
203  sha256.Update(mSecretKey, mSecretKeyLength);
204 
205  sha256.Finish(hash);
206 
207  memcpy(&aAddress->mAddress.mFields.m8[OT_IP6_ADDRESS_SIZE - OT_IP6_IID_SIZE],
208  &hash[sizeof(hash) - OT_IP6_IID_SIZE], OT_IP6_IID_SIZE);
209 
210  VerifyOrExit(!IsAddressRegistered(aInstance, aAddress), error = kThreadError_Ipv6AddressCreationFailure);
211  VerifyOrExit(!address->IsIidReserved(), error = kThreadError_Ipv6AddressCreationFailure);
212 
213 exit:
214  return error;
215 }

217 bool SemanticallyOpaqueIidGenerator::IsAddressRegistered(otInstance *aInstance, otNetifAddress *aCreatedAddress)
218 {
219  bool result = false;
220  const otNetifAddress *address = otGetUnicastAddresses(aInstance);
221 
222  while (address != NULL)
223  {
224  if (0 == memcmp(aCreatedAddress->mAddress.mFields.m8, address->mAddress.mFields.m8,
225  sizeof(address->mAddress.mFields)))
226  {
227  ExitNow(result = true);
228  }
229 
230  address = address->mNext;
231  }
232 
233 exit:
234  return result;
