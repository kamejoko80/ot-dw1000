72 int convert_pem_to_der( const unsigned char *input, size_t ilen,
73  unsigned char *output, size_t *olen )
74 {
75  int ret;
76  const unsigned char *s1, *s2, *end = input + ilen;
77  size_t len = 0;
78 
79  s1 = (unsigned char *) strstr( (const char *) input, "-----BEGIN" );
80  if( s1 == NULL )
81  return( -1 );
82 
83  s2 = (unsigned char *) strstr( (const char *) input, "-----END" );
84  if( s2 == NULL )
85  return( -1 );
86 
87  s1 += 10;
88  while( s1 < end && *s1 != '-' )
89  s1++;
90  while( s1 < end && *s1 == '-' )
91  s1++;
92  if( *s1 == '\r' ) s1++;
93  if( *s1 == '\n' ) s1++;
94 
95  if( s2 <= s1 || s2 > end )
96  return( -1 );
97 
98  ret = mbedtls_base64_decode( NULL, 0, &len, (const unsigned char *) s1, s2 - s1 );
99  if( ret == MBEDTLS_ERR_BASE64_INVALID_CHARACTER )
100  return( ret );
101 
102  if( len > *olen )
103  return( -1 );
104 
105  if( ( ret = mbedtls_base64_decode( output, len, &len, (const unsigned char *) s1,
106  s2 - s1 ) ) != 0 )
107  {
108  return( ret );
109  }
110 
111  *olen = len;
112 
113  return( 0 );
114 }

116 /*
117  * Load all data from a file into a given buffer.
118  */
119 static int load_file( const char *path, unsigned char **buf, size_t *n )
120 {
121  FILE *f;
122  long size;
123 
124  if( ( f = fopen( path, "rb" ) ) == NULL )
125  return( -1 );
126 
127  fseek( f, 0, SEEK_END );
128  if( ( size = ftell( f ) ) == -1 )
129  {
130  fclose( f );
131  return( -1 );
132  }
133  fseek( f, 0, SEEK_SET );
134 
135  *n = (size_t) size;
136 
137  if( *n + 1 == 0 ||
138  ( *buf = mbedtls_calloc( 1, *n + 1 ) ) == NULL )
139  {
140  fclose( f );
141  return( -1 );
142  }
143 
144  if( fread( *buf, 1, *n, f ) != *n )
145  {
146  fclose( f );
147  free( *buf );
148  *buf = NULL;
149  return( -1 );
150  }
151 
152  fclose( f );
153 
154  (*buf)[*n] = '\0';
155 
156  return( 0 );
157 }
