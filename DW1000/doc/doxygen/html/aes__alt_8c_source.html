52 void mbedtls_aes_init(mbedtls_aes_context *ctx)
53 {
54  if (sRefNum++ == 0)
55  {
56  /* enable the crypto core */
57  /* The TRNG should already be running before we ever ask the AES core
58  * to do anything, if there is any scenario that the TRNG powers off
59  * the peripheral power domain use this code to repower it
60 
61  PRCMPowerDomainOn(PRCM_DOMAIN_PERIPH);
62  while (PRCMPowerDomainStatus(PRCM_DOMAIN_PERIPH) != PRCM_DOMAIN_POWER_ON);
63  */
64  PRCMPeripheralRunEnable(PRCM_PERIPH_CRYPTO);
65  PRCMPeripheralSleepEnable(PRCM_PERIPH_CRYPTO);
66  PRCMPeripheralDeepSleepEnable(PRCM_PERIPH_CRYPTO);
67  PRCMLoadSet();
68 
69  while (!PRCMLoadGet());
70 
71  }
72 
73  ctx->magic = CC2650_AES_CTX_MAGIC;
74  ctx->key_idx = CC2650_AES_KEY_UNUSED;
75 }

77 void mbedtls_aes_free(mbedtls_aes_context *ctx)
78 {
79  VerifyOrExit(ctx->magic == CC2650_AES_CTX_MAGIC, ;);
80 
81  if (ctx->key_idx != CC2650_AES_KEY_UNUSED)
82  {
83  sUsedKeys &= ~(1 << ctx->key_idx);
84  }
85 
86  if (--sRefNum == 0)
87  {
88  /* disable the crypto core */
89  /* The TRNG core needs the peripheral power domain powered on to
90  * function. if there is a situation where the power domain must be
91  * powered off, use this code to do so.
92 
93  PRCMPowerDomainOff(PRCM_DOMAIN_PERIPH);
94  while (PRCMPowerDomainStatus(PRCM_DOMAIN_PERIPH) != PRCM_DOMAIN_POWER_OFF);
95  */
96  PRCMPeripheralRunDisable(PRCM_PERIPH_CRYPTO);
97  PRCMPeripheralSleepDisable(PRCM_PERIPH_CRYPTO);
98  PRCMPeripheralDeepSleepDisable(PRCM_PERIPH_CRYPTO);
99  PRCMLoadSet();
100 
101  while (!PRCMLoadGet());
102 
103  }
104 
105  memset((void *)ctx, 0x00, sizeof(ctx));
106 
107 exit:
108  return;
109 }

111 int mbedtls_aes_setkey_enc(mbedtls_aes_context *ctx, const unsigned char *key, unsigned int keybits)
112 {
113  unsigned char key_idx;
114  int retval = 0;
115 
116  VerifyOrExit(ctx->magic == CC2650_AES_CTX_MAGIC, retval = -1);
117 
118  if (ctx->key_idx != CC2650_AES_KEY_UNUSED)
119  {
120  sUsedKeys &= ~(1 << ctx->key_idx);
121  }
122 
123  /* our hardware only supports 128 bit keys */
124  VerifyOrExit(keybits == 128u, retval = MBEDTLS_ERR_AES_INVALID_KEY_LENGTH);
125 
126  for (key_idx = 0; ((sUsedKeys >> key_idx) & 0x01) != 0 && key_idx < 8; key_idx++);
127 
128  /* we have no more room for this key */
129  VerifyOrExit(key_idx < 8, retval = -2);
130 
131  VerifyOrExit(CRYPTOAesLoadKey((uint32_t *)key, key_idx) == AES_SUCCESS, retval = MBEDTLS_ERR_AES_INVALID_KEY_LENGTH);
132 
133  sUsedKeys |= (1 << key_idx);
134  ctx->key_idx = key_idx;
135 exit:
136  return retval;
137 }
