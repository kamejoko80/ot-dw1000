151  SetCallback(
152  vector<otCallback<CallbackT>*> &Callbacks,
153  const GUID& InterfaceGuid,
154  CallbackT Callback,
155  PVOID CallbackContext
156  )
157  {
158  bool alreadyExists = false;
159  otCallback<CallbackT>* CallbackToRelease = nullptr;
160 
161  EnterCriticalSection(&CallbackLock);
162 
163  if (Callback == nullptr)
164  {
165  for (size_t i = 0; i < Callbacks.size(); i++)
166  {
167  if (Callbacks[i]->InterfaceGuid == InterfaceGuid)
168  {
169  CallbackToRelease = Callbacks[i];
170  Callbacks.erase(Callbacks.begin() + i);
171  break;
172  }
173  }
174  }
175  else
176  {
177  for (size_t i = 0; i < Callbacks.size(); i++)
178  {
179  if (Callbacks[i]->InterfaceGuid == InterfaceGuid)
180  {
181  alreadyExists = true;
182  break;
183  }
184  }
185 
186  if (!alreadyExists)
187  {
188  Callbacks.push_back(new otCallback<CallbackT>(InterfaceGuid, Callback, CallbackContext));
189  }
190  }
191 
192  LeaveCriticalSection(&CallbackLock);
193 
194  if (CallbackToRelease)
195  {
196  CallbackToRelease->Release(true);
197  delete CallbackToRelease;
198  }
199 
200  return !alreadyExists;
201  }

236  if (aApitInstance == nullptr)
237  {
238  dwError = GetLastError();
239  otLogWarnApi("Failed to allocate otApiInstance");
240  goto error;

242 
243  // Open the pipe to the OpenThread driver
244  aApitInstance->DeviceHandle =
245  CreateFile(
246  OTLWF_IOCLT_PATH,
247  GENERIC_READ | GENERIC_WRITE,
248  0,
249  nullptr, // no SECURITY_ATTRIBUTES structure
250  OPEN_EXISTING, // No special create flags
251  FILE_FLAG_OVERLAPPED, // Allow asynchronous requests
252  nullptr
253  );
254  if (aApitInstance->DeviceHandle == INVALID_HANDLE_VALUE)
255  {
256  dwError = GetLastError();
257  otLogCritApi("CreateFile failed, %!WINERROR!", dwError);
258  goto error;
259  }
260 
261  // Create event for completion of async IO
262  aApitInstance->Overlapped.hEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
263  if (aApitInstance->Overlapped.hEvent == nullptr)
264  {
265  dwError = GetLastError();
266  otLogCritApi("CreateEvent (Overlapped.hEvent) failed, %!WINERROR!", dwError);
267  goto error;
268  }
269 
270  // Create the otpool wait
271  aApitInstance->ThreadpoolWait =
272  CreateThreadpoolWait(
273  otIoComplete,
274  aApitInstance,
275  nullptr
276  );
277  if (aApitInstance->ThreadpoolWait == nullptr)
278  {
279  dwError = GetLastError();
280  otLogCritApi("CreateThreadpoolWait failed, %!WINERROR!", dwError);
281  goto error;

285  SetThreadpoolWait(aApitInstance->ThreadpoolWait, aApitInstance->Overlapped.hEvent, nullptr);
286 
287 #ifdef DEBUG_ASYNC_IO

289 #endif

292  if (!DeviceIoControl(

301  if (dwError != ERROR_IO_PENDING)
302  {
303  otLogCritApi("DeviceIoControl for first notification failed, %!WINERROR!", dwError);
304  goto error;
305  }
306  dwError = ERROR_SUCCESS;
307  }
308 
309 error:
310 
311  if (dwError != ERROR_SUCCESS)
312  {

314  aApitInstance = nullptr;

441  {
442  CloseHandle(aApitInstance->Overlapped.hEvent);
443  }
444 
445  // Close the device handle
446  CloseHandle(aApitInstance->DeviceHandle);
447  }
448 
449  delete aApitInstance;
450 
451  otLogFuncExit();
452 }

708 otIoComplete(

710  _Inout_opt_ PVOID Context,
711  _Inout_ PTP_WAIT /* Wait */,
712  _In_ TP_WAIT_RESULT /* WaitResult */
713  )

716  otLogFuncEntry();
717 #endif
718

720  if (aApitInstance == nullptr) return;

723  DWORD dwBytesTransferred = 0;
724  if (!GetOverlappedResult(
725  aApitInstance->DeviceHandle,
726  &aApitInstance->Overlapped,
727  &dwBytesTransferred,
728  FALSE))
729  {

739  ProcessNotification(aApitInstance, &aApitInstance->NotificationBuffer);
740

742  EnterCriticalSection(&aApitInstance->CallbackLock);
743  PTP_WAIT tpWait = aApitInstance->ThreadpoolWait;
744  LeaveCriticalSection(&aApitInstance->CallbackLock);
745 
746  if (tpWait)
747  {

749  SetThreadpoolWait(tpWait, aApitInstance->Overlapped.hEvent, nullptr);
750

752  otLogDebgApi("Querying for next notification");
753 #endif
754 
755  // Request next notification
756  if (!DeviceIoControl(
757  aApitInstance->DeviceHandle,
758  IOCTL_OTLWF_QUERY_NOTIFICATION,
759  nullptr, 0,
