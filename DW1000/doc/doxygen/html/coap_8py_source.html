83  @classmethod
84  def _read_extended_value(cls, data, value):
85  if value == 13:
86  return ord(data.read(1)) + 13
87  elif value == 14:

94  def from_bytes(cls, data):
95  initial_byte = ord(data.read(1))
96 
97  delta = (initial_byte >> 4) & 0xf
98  length = initial_byte & 0xf
99

101  length = cls._read_extended_value(data, length)
102

126 class CoapOptionsFactory(object):
127 
128  """ Factory that produces CoAP options. """
129 
130  def parse(self, data, message_info):
131  options = []
132 
133  _type = 0
134  while data.tell() < len(data.getvalue()):
135  option_header = CoapOptionHeader.from_bytes(data)
136  if option_header.is_payload_marker:
137  break
138 
139  _type += option_header.delta
140  value = data.read(option_header.length)
141 
142  option = CoapOption(_type, value)
143  options.append(option)
144

342 class CoapMessageFactory(object):
343 
344  """ Factory that produces CoAP messages. """
345 
346  def __init__(self, options_factory, uri_path_based_payload_factories, message_id_to_uri_path_binder):
347  self._options_factory = options_factory
348  self._uri_path_based_payload_factories = uri_path_based_payload_factories
349  self._mid_to_uri_path_binder = message_id_to_uri_path_binder
350 
351  def _uri_path_from(self, options):
352  uri_path_options = []
353 
354  for option in options:
355  if option.type == CoapOptionsTypes.URI_PATH:
356  uri_path_options.append(option.value.decode("utf-8"))
357 
358  if not uri_path_options:
359  return None
360 
361  return "/" + "/".join(uri_path_options)
362 
363  def _parse_initial_byte(self, data, message_info):
364  initial_byte = ord(data.read(1))
365 
366  version = (initial_byte >> 6) & 0x3
367  _type = CoapMessageType((initial_byte >> 4) & 0x3)
368  token_length = initial_byte & 0xf
369 
370  return version, _type, token_length
371 
372  def parse(self, data, message_info):
373  version, _type, token_length = self._parse_initial_byte(data, message_info)
374 
375  code = CoapCode(ord(data.read(1)))
376  message_id = struct.unpack(">H", data.read(2))[0]
377  token = data.read(token_length)
378 
379  options = self._options_factory.parse(data, message_info)
380 
381  uri_path = self._uri_path_from(options)
382  if uri_path is not None:
383  self._mid_to_uri_path_binder.add_uri_path_for(message_id, token, uri_path)
384 
385  coap_message = CoapMessage(version, _type, code, message_id, token, options, data.read(), uri_path)
386 
387  return CoapMessageProxy(coap_message, message_info, self._mid_to_uri_path_binder, self._uri_path_based_payload_factories)
