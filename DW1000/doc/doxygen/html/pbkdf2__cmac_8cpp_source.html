45 void otPbkdf2Cmac(
46  const uint8_t *aPassword, uint16_t aPasswordLen,
47  const uint8_t *aSalt, uint16_t aSaltLen,
48  uint32_t aIterationCounter , uint16_t aKeyLen,
49  uint8_t *aKey)
50 {
51  uint32_t blockCounter = 0;
52  uint16_t useLen = 0;
53  uint16_t prfBlockLen = MBEDTLS_CIPHER_BLKSIZE_MAX;
54  uint8_t prfInput[OT_PBKDF2_SALT_MAX_LEN + 4]; // Salt || INT(), for U1 calculation
55  uint8_t prfOutput[MBEDTLS_CIPHER_BLKSIZE_MAX];
56  uint8_t keyBlock[MBEDTLS_CIPHER_BLKSIZE_MAX];
57  uint8_t *key = aKey;
58  uint16_t keyLen = aKeyLen;
59 
60  while (keyLen)
61  {
62  memcpy(prfInput, aSalt, aSaltLen);
63 
64  blockCounter++;
65  prfInput[aSaltLen + 0] = (uint8_t)(blockCounter >> 24);
66  prfInput[aSaltLen + 1] = (uint8_t)(blockCounter >> 16);
67  prfInput[aSaltLen + 2] = (uint8_t)(blockCounter >> 8);
68  prfInput[aSaltLen + 3] = (uint8_t)(blockCounter);
69 
70  // Calculate U_1
71  mbedtls_aes_cmac_prf_128(aPassword, aPasswordLen, prfInput, aSaltLen + 4, prfOutput);
72  memcpy(keyBlock, prfOutput, prfBlockLen);
73 
74  for (uint32_t i = 1; i < aIterationCounter; i++)
75  {
76  memcpy(prfInput, prfOutput, prfBlockLen);
77 
78  // Calculate U_i
79  mbedtls_aes_cmac_prf_128(aPassword, aPasswordLen, prfInput, prfBlockLen, prfOutput);
80 
81  // xor
82  for (uint32_t j = 0; j < prfBlockLen; j++)
83  {
84  keyBlock[j] ^= prfOutput[j];
85  }
86  }
87 
88  useLen = (keyLen < prfBlockLen) ? keyLen : prfBlockLen;
89  memcpy(key, keyBlock, useLen);
90  key += useLen;
91  keyLen -= useLen;
92  }
93 }
