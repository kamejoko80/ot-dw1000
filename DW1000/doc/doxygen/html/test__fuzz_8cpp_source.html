169  fuzzPacket.mLength = (uint8_t)(otPlatRandomGet() % 127);
170 
171  // Populate the length with random
172  for (uint8_t i = 0; i < fuzzPacket.mLength; i++)
173  {
174  fuzzRecvBuff[i] = (uint8_t)otPlatRandomGet();
175  }
176 
177  // Clear the global flag
178  g_RecvChannel = 0;
179 
180  // Indicate the receive complete
181  otPlatRadioReceiveDone(aInstance, &fuzzPacket, kThreadError_None);
182 
183  countRecv++;
184 #ifdef DBG_FUZZ
185  Log("<== receive (%llu, %u bytes)", countRecv, fuzzPacket.mLength);
186 #endif
187 
188  // Hack to get a receive poll immediately
189  otSetChannel(aInstance, 11);
190  }
191  }
192  }
193 
194  Log("%u packets received", countRecv);
195 
196  // Clean up the instance
197  otInstanceFinalize(aInstance);
198 
199 #ifdef OPENTHREAD_MULTIPLE_INSTANCE
200  free(otInstanceBuffer);
201 #endif
202 }
