65 def calculate_checksum(data):
66  """ Calculate checksum from data bytes.
67 
68  How to calculate checksum (RFC 2460):
69  https://tools.ietf.org/html/rfc2460#page-27
70 
71  Args:
72  data (bytes): input data from which checksum will be calculated
73 
74  Returns:
75  int: calculated checksum
76  """
77  # Create halfwords from data bytes. Example: data[0] = 0x01, data[1] = 0xb2 => 0x01b2
78  halfwords = [((byte0 << 8) | byte1) for byte0, byte1 in zip_longest(data[::2], data[1::2], fillvalue=0x00)]
79 
80  checksum = 0
81  for halfword in halfwords:
82  checksum += halfword
83  checksum = (checksum & 0xFFFF) + (checksum >> 16)
84 
85  checksum ^= 0xFFFF
86 
87  if checksum == 0:
88  return 0xFFFF
89  else:
90  return checksum

337 class IPv6Packet(ConvertibleToBytes):
338 
339  """ Class representing IPv6 packet.
340 
341  IPv6 packet consists of IPv6 header, optional extension header, and upper layer protocol.
342 
343  IPv6 packet
344 
345  +-------------+----------------------------------+----------------------------------------------+
346  | | | |
347  | IPv6 header | extension headers (zero or more) | upper layer protocol (e.g. UDP, TCP, ICMPv6) |
348  | | | |
349  +-------------+----------------------------------+----------------------------------------------+
350 
351  Extension headers:
352  - HopByHop
353  - Routing header (not implemented in this module)
354 
355  Upper layer protocols:
356  - ICMPv6
357  - UDP
358  - TCP (not implemented in this module)
359 
360  Example:
361  IPv6 packet construction without extension headers:
362 
363  ipv6_packet = IPv6Packet(IPv6Header("fd00:1234:4555::ff:fe00:1800", "ff03::1"),
364  ICMPv6(ICMPv6Header(128, 0),
365  ICMPv6EchoBody(0, 2, bytes([0x80, 0x00, 0xc7, 0xbf, 0x00, 0x00, 0x00, 0x01,
366  0x41, 0x41, 0x41, 0x41, 0x41, 0x41, 0x41, 0x41,
367  0x41, 0x41]))))
368 
369  IPv6 packet construction with extension headers:
370 
371  ipv6_packet = IPv6Packet(IPv6Header("fd00:1234:4555::ff:fe00:1800", "ff03::1"),
372  ICMPv6(ICMPv6Header(128, 0),
373  ICMPv6EchoBody(0, 2, bytes([0x80, 0x00, 0xc7, 0xbf, 0x00, 0x00, 0x00, 0x01,
374  0x41, 0x41, 0x41, 0x41, 0x41, 0x41, 0x41, 0x41,
375  0x41, 0x41])),
376  [HopByHop(options=[
377  HopByHopOption(HopByHopOptionHeader(_type=0x6d),
378  MPLOption(S=1, M=0, V=0, sequence=2, seed_id=bytes([0x00, 0x18])))
379  ])])
380

382

384  self.ipv6_header = ipv6_header
385 
386  self.upper_layer_protocol = upper_layer_protocol
387 
388  self.extension_headers = extension_headers if extension_headers is not None else []
389 
390  self._update_next_header_values_in_headers()
391 
392  if not upper_layer_protocol.is_valid_checksum():
393  self.upper_layer_protocol.checksum = self.calculate_checksum()
394 
395  def _validate_checksum(self):
396  checksum = self.calculate_checksum()
397 
398  if self.upper_layer_protocol.checksum != checksum:
399  raise RuntimeError("Could not create IPv6 packet. "
400  "Invalid checksum: {}!={}".format(self.upper_layer_protocol.checksum, checksum))
401 
402  self.upper_layer_protocol.checksum = checksum
403

405  self.ipv6_header.payload_length = len(self.upper_layer_protocol) + \
406  sum([len(extension_header) for extension_header in self.extension_headers])
407

409  last_header = self.ipv6_header
410

412  last_header.next_header = extension_header.type
413  last_header = extension_header
414

416

418  saved_checksum = self.upper_layer_protocol.checksum
419 
420  self.upper_layer_protocol.checksum = 0
421

423

427  self.ipv6_header.destination_address,
428  len(upper_layer_protocol_bytes),
429  self.upper_layer_protocol.type)
430

432

599 class HopByHop(ExtensionHeader):
600 
601  """ Class representing HopByHop extension header.
602 
603  HopByHop extension header consists of:
604  - next_header type
605  - extension header length which is multiple of 8
606  - options
607 
608  """
609  _one_byte_padding = 0x00
610  _many_bytes_padding = 0x01
611 
612  @property
613  def type(self):
614  return 0
615 
616  def __init__(self, next_header=None, options=None, hdr_ext_len=None):
617  super(HopByHop, self).__init__(next_header, hdr_ext_len)
618  self.options = options if options is not None else []
619 
620  if hdr_ext_len is not None:
621  self.hdr_ext_len = hdr_ext_len
622  else:
623  payload_length = self._calculate_payload_length()
624  self.hdr_ext_len = self._calculate_hdr_ext_len(payload_length)
625 
626  def _calculate_payload_length(self):
627  payload_length = 2
628 
629  for option in self.options:
630  payload_length += len(option)
631 
632  return payload_length
633 
634  def _calculate_hdr_ext_len(self, payload_length):
635  count = payload_length >> 3
636 
637  if (payload_length & 0x7) == 0 and count > 0:
638  return count - 1
639 
640  return count
641 
642  def to_bytes(self):
643  data = bytearray([self.next_header, self.hdr_ext_len])
644 
645  for option in self.options:
646  data += option.to_bytes()
647 
648  # Padding
649  #
650  # More details:
651  # https://tools.ietf.org/html/rfc2460#section-4.2
652  #
653  excess_bytes = len(data) & 0x7
654 
655  if excess_bytes > 0:
656  padding_length = 8 - excess_bytes
657 
658  if padding_length == 1:
659  data += bytearray([self._one_byte_padding])
660 
661  else:
662  padding_length -= 2
663  data += bytearray([self._many_bytes_padding, padding_length])
664  data += bytearray([0x00 for _ in range(padding_length)])
665 
666  return data
667 
668  def __len__(self):
669  """ HopByHop extension header length
670 
671  More details:
672  https://tools.ietf.org/html/rfc2460#section-4.3
673 
674  """
675  return (self.hdr_ext_len + 1) * 8
