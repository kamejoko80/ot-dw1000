147 ThreadError Icmp::HandleMessage(Message &aMessage, MessageInfo &aMessageInfo)
148 {
149  ThreadError error = kThreadError_None;
150  uint16_t payloadLength;
151  IcmpHeader icmp6Header;
152  uint16_t checksum;
153 
154  VerifyOrExit(aMessage.Read(aMessage.GetOffset(), sizeof(icmp6Header), &icmp6Header) == sizeof(icmp6Header),);
155  payloadLength = aMessage.GetLength() - aMessage.GetOffset();
156 
157  // verify checksum
158  checksum = Ip6::ComputePseudoheaderChecksum(aMessageInfo.GetPeerAddr(), aMessageInfo.GetSockAddr(),
159  payloadLength, kProtoIcmp6);
160  checksum = aMessage.UpdateChecksum(checksum, aMessage.GetOffset(), payloadLength);
161  VerifyOrExit(checksum == 0xffff, ;);
162 
163  switch (icmp6Header.GetType())
164  {
165  case IcmpHeader::kTypeEchoRequest:
166  return HandleEchoRequest(aMessage, aMessageInfo);
167 
168  case IcmpHeader::kTypeEchoReply:
169  return HandleEchoReply(aMessage, aMessageInfo);
170 
171  case IcmpHeader::kTypeDstUnreach:
172  return HandleDstUnreach(aMessage, aMessageInfo, icmp6Header);
173  }
174 
175 exit:
176  return error;
177 }

192 ThreadError Icmp::HandleEchoRequest(Message &aRequestMessage, const MessageInfo &aMessageInfo)
193 {
194  ThreadError error = kThreadError_None;
195  IcmpHeader icmp6Header;
196  Message *replyMessage = NULL;
197  MessageInfo replyMessageInfo;
198  uint16_t payloadLength;
199 
200  VerifyOrExit(mIsEchoEnabled, ;);
201 
202  otLogInfoIcmp("Received Echo Request");
203 
204  icmp6Header.Init();
205  icmp6Header.SetType(IcmpHeader::kTypeEchoReply);
206 
207  if ((replyMessage = mIp6.NewMessage(0)) == NULL)
208  {
209  otLogDebgIcmp("icmp fail");
210  ExitNow();
211  }
212 
213  payloadLength = aRequestMessage.GetLength() - aRequestMessage.GetOffset() - IcmpHeader::GetDataOffset();
214  SuccessOrExit(error = replyMessage->SetLength(IcmpHeader::GetDataOffset() + payloadLength));
215 
216  replyMessage->Write(0, IcmpHeader::GetDataOffset(), &icmp6Header);
217  aRequestMessage.CopyTo(aRequestMessage.GetOffset() + IcmpHeader::GetDataOffset(),
218  IcmpHeader::GetDataOffset(), payloadLength, *replyMessage);
219 
220  replyMessageInfo.SetPeerAddr(aMessageInfo.GetPeerAddr());
221 
222  if (!aMessageInfo.GetSockAddr().IsMulticast())
223  {
224  replyMessageInfo.SetSockAddr(aMessageInfo.GetSockAddr());
225  }
226 
227  replyMessageInfo.SetInterfaceId(aMessageInfo.mInterfaceId);
228 
229  SuccessOrExit(error = mIp6.SendDatagram(*replyMessage, replyMessageInfo, kProtoIcmp6));
230 
231  replyMessage->Read(replyMessage->GetOffset(), sizeof(icmp6Header), &icmp6Header);
232  otLogInfoIcmp("Sent Echo Reply (seq = %d)", icmp6Header.GetSequence());
233 
234 exit:
235 
236  if (error != kThreadError_None && replyMessage != NULL)
237  {
238  replyMessage->Free();
239  }
240 
241  return error;
242 }

254 ThreadError Icmp::UpdateChecksum(Message &aMessage, uint16_t aChecksum)
255 {
256  aChecksum = aMessage.UpdateChecksum(aChecksum, aMessage.GetOffset(),
257  aMessage.GetLength() - aMessage.GetOffset());
258 
259  if (aChecksum != 0xffff)
260  {
261  aChecksum = ~aChecksum;
262  }
263 
264  aChecksum = HostSwap16(aChecksum);
265  aMessage.Write(aMessage.GetOffset() + IcmpHeader::GetChecksumOffset(), sizeof(aChecksum), &aChecksum);
266  return kThreadError_None;
267 }
