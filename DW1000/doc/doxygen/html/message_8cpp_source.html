229 ThreadError Message::ResizeMessage(uint16_t aLength)
230 {
231  ThreadError error = kThreadError_None;
232 
233  // add buffers
234  Buffer *curBuffer = this;
235  Buffer *lastBuffer;
236  uint16_t curLength = kHeadBufferDataSize;
237 
238  while (curLength < aLength)
239  {
240  if (curBuffer->GetNextBuffer() == NULL)
241  {
242  curBuffer->SetNextBuffer(GetMessagePool()->NewBuffer());
243  VerifyOrExit(curBuffer->GetNextBuffer() != NULL, error = kThreadError_NoBufs);
244  }
245 
246  curBuffer = curBuffer->GetNextBuffer();
247  curLength += kBufferDataSize;
248  }
249 
250  // remove buffers
251  lastBuffer = curBuffer;
252  curBuffer = curBuffer->GetNextBuffer();
253  lastBuffer->SetNextBuffer(NULL);
254 
255  GetMessagePool()->FreeBuffers(curBuffer);
256 
257 exit:
258  return error;
259 }

295 ThreadError Message::SetLength(uint16_t aLength)
296 {
297  ThreadError error = kThreadError_None;
298  uint16_t totalLengthRequest = GetReserved() + aLength;
299  uint16_t totalLengthCurrent = GetReserved() + GetLength();
300  int bufs = 0;
301 
302  if (totalLengthRequest > kHeadBufferDataSize)
303  {
304  bufs = (((totalLengthRequest - kHeadBufferDataSize) - 1) / kBufferDataSize) + 1;
305  }
306 
307  if (totalLengthCurrent > kHeadBufferDataSize)
308  {
309  bufs -= (((totalLengthCurrent - kHeadBufferDataSize) - 1) / kBufferDataSize) + 1;
310  }
311 
312  SuccessOrExit(error = GetMessagePool()->ReclaimBuffers(bufs));
313 
314  SuccessOrExit(error = ResizeMessage(totalLengthRequest));
315  mInfo.mLength = aLength;
316 
317 exit:
318  return error;
319 }

457 {

510 uint16_t Message::Read(uint16_t aOffset, uint16_t aLength, void *aBuf) const
511 {
512  Buffer *curBuffer;
513  uint16_t bytesCopied = 0;
514  uint16_t bytesToCopy;
515 
516  if (aOffset >= GetLength())
517  {
518  ExitNow();
519  }
520 
521  if (aOffset + aLength >= GetLength())
522  {
523  aLength = GetLength() - aOffset;
524  }
525 
526  aOffset += GetReserved();
527 
528  // special case first buffer
529  if (aOffset < kHeadBufferDataSize)
530  {
531  bytesToCopy = kHeadBufferDataSize - aOffset;
532 
533  if (bytesToCopy > aLength)
534  {
535  bytesToCopy = aLength;
536  }
537 
538  memcpy(aBuf, GetFirstData() + aOffset, bytesToCopy);
539 
540  aLength -= bytesToCopy;
541  bytesCopied += bytesToCopy;
542  aBuf = static_cast<uint8_t *>(aBuf) + bytesToCopy;
543 
544  aOffset = 0;
545  }
546  else
547  {
548  aOffset -= kHeadBufferDataSize;
549  }
550 
551  // advance to offset
552  curBuffer = GetNextBuffer();
553 
554  while (aOffset >= kBufferDataSize)
555  {
556  assert(curBuffer != NULL);
557 
558  curBuffer = curBuffer->GetNextBuffer();
559  aOffset -= kBufferDataSize;
560  }
561 
562  // begin copy
563  while (aLength > 0)
564  {
565  assert(curBuffer != NULL);
566 
567  bytesToCopy = kBufferDataSize - aOffset;
568 
569  if (bytesToCopy > aLength)
570  {
571  bytesToCopy = aLength;
572  }
573 
574  memcpy(aBuf, curBuffer->GetData() + aOffset, bytesToCopy);
575 
576  aLength -= bytesToCopy;
577  bytesCopied += bytesToCopy;
578  aBuf = static_cast<uint8_t *>(aBuf) + bytesToCopy;
579 
580  curBuffer = curBuffer->GetNextBuffer();
581  aOffset = 0;
582  }
583 
584 exit:
585  return bytesCopied;
586 }
