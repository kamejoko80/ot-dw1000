100 ThreadError AddressResolver::Resolve(const Ip6::Address &aEid, uint16_t &aRloc16)
101 {
102  ThreadError error = kThreadError_None;
103  Cache *entry = NULL;
104 
105  for (int i = 0; i < kCacheEntries; i++)
106  {
107  if (mCache[i].mState != Cache::kStateInvalid)
108  {
109  if (memcmp(&mCache[i].mTarget, &aEid, sizeof(mCache[i].mTarget)) == 0)
110  {
111  entry = &mCache[i];
112  break;
113  }
114  }
115  else if (entry == NULL)
116  {
117  entry = &mCache[i];
118  }
119  }
120 
121  VerifyOrExit(entry != NULL, error = kThreadError_NoBufs);
122 
123  switch (entry->mState)
124  {
125  case Cache::kStateInvalid:
126  entry->mTarget = aEid;
127  entry->mRloc16 = Mac::kShortAddrInvalid;
128  entry->mTimeout = kAddressQueryTimeout;
129  entry->mFailures = 0;
130  entry->mRetryTimeout = kAddressQueryInitialRetryDelay;
131  entry->mState = Cache::kStateQuery;
132  SendAddressQuery(aEid);
133  error = kThreadError_AddressQuery;
134  break;
135 
136  case Cache::kStateQuery:
137  if (entry->mTimeout > 0)
138  {
139  error = kThreadError_AddressQuery;
140  }
141  else if (entry->mTimeout == 0 && entry->mRetryTimeout == 0)
142  {
143  entry->mTimeout = kAddressQueryTimeout;
144  SendAddressQuery(aEid);
145  error = kThreadError_AddressQuery;
146  }
147  else
148  {
149  error = kThreadError_Drop;
150  }
151 
152  break;
153 
154  case Cache::kStateCached:
155  aRloc16 = entry->mRloc16;
156  break;
157  }
158 
159 exit:
160  return error;
161 }

277  mCache[i].mTimeout = 0;
278  mCache[i].mFailures = 0;
279  mCache[i].mState = Cache::kStateCached;
280

282  {
283  otLogInfoArp("Sent address notification acknowledgment");
284  }
285

288  }
289  }
290 
291 exit:

349 void AddressResolver::HandleAddressError(Coap::Header &aHeader, Message &aMessage,
350  const Ip6::MessageInfo &aMessageInfo)
351 {
352  ThreadError error = kThreadError_None;
353  ThreadTargetTlv targetTlv;
354  ThreadMeshLocalEidTlv mlIidTlv;
355  Child *children;
356  uint8_t numChildren;
357  Mac::ExtAddress macAddr;
358  Ip6::Address destination;
359 
360  VerifyOrExit(aHeader.GetType() == kCoapTypeConfirmable &&
361  aHeader.GetCode() == kCoapRequestPost, error = kThreadError_Drop);
362 
363  otLogInfoArp("Received address error notification");
364 
365  if (!aMessageInfo.GetSockAddr().IsMulticast())
366  {
367  if (mNetif.GetCoapServer().SendEmptyAck(aHeader, aMessageInfo) == kThreadError_None)
368  {
369  otLogInfoArp("Sent address error notification acknowledgment");
370  }
371  }
372 
373  SuccessOrExit(error = ThreadTlv::GetTlv(aMessage, ThreadTlv::kTarget, sizeof(targetTlv), targetTlv));
374  VerifyOrExit(targetTlv.IsValid(), error = kThreadError_Parse);
375 
376  SuccessOrExit(error = ThreadTlv::GetTlv(aMessage, ThreadTlv::kMeshLocalEid, sizeof(mlIidTlv), mlIidTlv));
377  VerifyOrExit(mlIidTlv.IsValid(), error = kThreadError_Parse);
378 
379  for (const Ip6::NetifUnicastAddress *address = mNetif.GetUnicastAddresses(); address; address = address->GetNext())
380  {
381  if (memcmp(&address->mAddress, targetTlv.GetTarget(), sizeof(address->mAddress)) == 0 &&
382  memcmp(mNetif.GetMle().GetMeshLocal64().GetIid(), mlIidTlv.GetIid(), 8))
383  {
384  // Target EID matches address and Mesh Local EID differs
385  mNetif.RemoveUnicastAddress(*address);
386  ExitNow();
387  }
388  }
389 
390  children = mNetif.GetMle().GetChildren(&numChildren);
391 
392  memcpy(&macAddr, mlIidTlv.GetIid(), sizeof(macAddr));
393  macAddr.m8[0] ^= 0x2;
394 
395  for (int i = 0; i < numChildren; i++)
396  {
397  if (children[i].mState != Neighbor::kStateValid || (children[i].mMode & Mle::ModeTlv::kModeFFD) != 0)
398  {
399  continue;
400  }
401 
402  for (int j = 0; j < Child::kMaxIp6AddressPerChild; j++)
403  {
404  if (memcmp(&children[i].mIp6Address[j], targetTlv.GetTarget(), sizeof(children[i].mIp6Address[j])) == 0 &&
405  memcmp(&children[i].mMacAddr, &macAddr, sizeof(children[i].mMacAddr)))
406  {
407  // Target EID matches child address and Mesh Local EID differs on child
408  memset(&children[i].mIp6Address[j], 0, sizeof(children[i].mIp6Address[j]));
409 
410  memset(&destination, 0, sizeof(destination));
411  destination.mFields.m16[0] = HostSwap16(0xfe80);
412  destination.SetIid(children[i].mMacAddr);
413 
414  SendAddressError(targetTlv, mlIidTlv, &destination);
415  ExitNow();
416  }
417  }
418  }
419 
420 exit:
421  {}
422 }

541 void AddressResolver::HandleTimer()
542 {
543  bool continueTimer = false;
544 
545  for (int i = 0; i < kCacheEntries; i++)
546  {
547  if (mCache[i].mState != Cache::kStateQuery)
548  {
549  continue;
550  }
551 
552  continueTimer = true;
553 
554  if (mCache[i].mTimeout > 0)
555  {
556  mCache[i].mTimeout--;
557 
558  if (mCache[i].mTimeout == 0)
559  {
560  mCache[i].mRetryTimeout =
561  static_cast<uint16_t>(kAddressQueryInitialRetryDelay * (1 << mCache[i].mFailures));
562 
563  if (mCache[i].mRetryTimeout < kAddressQueryMaxRetryDelay)
564  {
565  mCache[i].mFailures++;
566  }
567  else
568  {
569  mCache[i].mRetryTimeout = kAddressQueryMaxRetryDelay;
570  }
571 
572  mNetif.GetMeshForwarder().HandleResolved(mCache[i].mTarget, kThreadError_Drop);
573  }
574  }
575  else if (mCache[i].mRetryTimeout > 0)
576  {
577  mCache[i].mRetryTimeout--;
578  }
579  }
580 
581  if (continueTimer)
582  {
583  mTimer.Start(kStateUpdatePeriod);
584  }
