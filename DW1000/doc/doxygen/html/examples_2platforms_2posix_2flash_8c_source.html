99 ThreadError utilsFlashErasePage(uint32_t aAddress)
100 {
101  ThreadError error = kThreadError_None;
102  uint8_t buf = 0xff;
103  uint32_t address;
104 
105  VerifyOrExit(sFlashFd >= 0, error = kThreadError_Failed);
106  VerifyOrExit(aAddress < FLASH_SIZE, error = kThreadError_InvalidArgs);
107 
108  // Get start address of the flash page that includes aAddress
109  address = aAddress & (~(uint32_t)(FLASH_PAGE_SIZE - 1));
110 
111  for (uint16_t offset = 0; offset < FLASH_PAGE_SIZE; offset++)
112  {
113  VerifyOrExit(pwrite(sFlashFd, &buf, 1, address + offset) == 1, error = kThreadError_Failed);
114  }
115 
116 exit:
117  return error;
118 }
