186 void MainPage::ShowInterfaceDetails(Platform::Guid InterfaceGuid)
187 {
188  if (ApiInstance == nullptr) return;
189 
190  GUID deviceGuid = InterfaceGuid;
191  auto device = otInstanceInit(ApiInstance, &deviceGuid);
192 
193  auto extendedAddress = otGetExtendedAddress(device);
194  if (extendedAddress)
195  {
196  WCHAR szMac[256] = { 0 };
197  swprintf_s(szMac, 256, MAC8_FORMAT, MAC8_ARG(extendedAddress));
198  InterfaceMacAddress->Text = ref new String(szMac);
199 
200  otFreeMemory(extendedAddress);
201  }
202  else
203  {
204  InterfaceMacAddress->Text = L"ERROR";
205  }
206 
207  auto ml_eid = otGetMeshLocalEid(device);
208  if (ml_eid)
209  {
210  WCHAR szAddress[46] = { 0 };
211  RtlIpv6AddressToStringW((const PIN6_ADDR)ml_eid, szAddress);
212  InterfaceML_EID->Text = ref new String(szAddress);
213 
214  otFreeMemory(ml_eid);
215  }
216  else
217  {
218  InterfaceML_EID->Text = L"ERROR";
219  }
220 
221  auto rloc16 = otGetRloc16(device);
222  WCHAR szRloc[16] = { 0 };
223  swprintf_s(szRloc, 16, L"%4x", rloc16);
224  InterfaceRLOC->Text = ref new String(szRloc);
225 
226  if (otGetDeviceRole(device) > kDeviceRoleChild)
227  {
228  uint8_t index = 0;
229  otChildInfo childInfo;
230  while (kThreadError_None == otGetChildInfoByIndex(device, index, &childInfo))
231  {
232  index++;
233  }
234 
235  WCHAR szText[64] = { 0 };
236  swprintf_s(szText, 64, L"%d", index);
237  InterfaceChildren->Text = ref new String(szText);
238 
239  InterfaceNeighbors->Text = L"unknown";
240 
241  InterfaceNeighbors->Visibility = Windows::UI::Xaml::Visibility::Visible;
242  InterfaceNeighborsText->Visibility = Windows::UI::Xaml::Visibility::Visible;
243  InterfaceChildren->Visibility = Windows::UI::Xaml::Visibility::Visible;
244  InterfaceChildrenText->Visibility = Windows::UI::Xaml::Visibility::Visible;
245  }
246 
247  // Show the details
248  InterfaceDetails->Visibility = Windows::UI::Xaml::Visibility::Visible;
249 
250  otFreeMemory(device);
251 }

321 void MainPage::BuildInterfaceList()
322 {
323  if (ApiInstance == nullptr) return;
324 
325  // Clear all existing children
326  InterfaceList->Items->Clear();
327 
328  // Clean up devices
329  for each (auto device in _devices)
330  {
331  // Unregister for callbacks for the device
332  otSetStateChangedCallback((otInstance*)device, nullptr, nullptr);
333 
334  // Free the device
335  otFreeMemory(device);
336  }
337  _devices.clear();
338 
339  // Enumerate the new device list
340  auto deviceList = otEnumerateDevices(ApiInstance);
341  if (deviceList)
342  {
343  // Dump the results to the console
344  for (DWORD dwIndex = 0; dwIndex < deviceList->aDevicesLength; dwIndex++)
345  {
346  InterfaceList->Items->Append(
347  CreateNewInterface(deviceList->aDevices[dwIndex]));
348  }
349 
350  otFreeMemory(deviceList);
351  }
352 }

354 void MainPage::ConnectNetwork(Platform::Guid InterfaceGuid)
355 {
356  if (ApiInstance == nullptr) return;
357 
358  GUID deviceGuid = InterfaceGuid;
359  auto device = otInstanceInit(ApiInstance, &deviceGuid);
360 
361  //
362  // Configure
363  //
364 
365  otNetworkName networkName = {};
366  wcstombs(networkName.m8, InterfaceConfigName->Text->Data(), sizeof(networkName.m8));
367  otSetNetworkName(device, networkName.m8);
368 
369  otMasterKey masterKey = {};
370  wcstombs((char*)masterKey.m8, InterfaceConfigKey->Text->Data(), sizeof(masterKey.m8));
371  otSetMasterKey(device, masterKey.m8, sizeof(masterKey.m8));
372 
373  otSetChannel(device, (uint8_t)InterfaceConfigChannel->Value);
374  otSetMaxAllowedChildren(device, (uint8_t)InterfaceConfigMaxChildren->Value);
375 
376  otSetPanId(device, 0x4567);
377 
378  //
379  // Bring up the interface and start the Thread logic
380  //
381 
382  otInterfaceUp(device);
383 
384  otThreadStart(device);
385 
386  // Cleanup
387  otFreeMemory(device);
