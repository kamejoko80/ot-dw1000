187  // Callback from OpenThread indicating arrival or removal of interfaces
188  static void OTCALL
189  ThreadDeviceAvailabilityCallback(
190  bool aAdded,
191  const GUID* aDeviceGuid,
192  _In_ void* aContext
193  )
194  {
195  IInspectable* pInspectable = (IInspectable*)aContext;
196  otApi^ pThis = reinterpret_cast<otApi^>(pInspectable);
197 
198  if (aAdded)
199  {
200  otAdapter^ adapter = nullptr;
201 
202  // Add the device to the list, if it isn't already there
203  EnterCriticalSection(&pThis->_cs);
204 
205  if (pThis->GetAdapter(*aDeviceGuid) == nullptr)
206  {
207  auto deviceInstance = otInstanceInit((otApiInstance*)pThis->_apiInstance, aDeviceGuid);
208  if (deviceInstance)
209  {
210  pThis->_adapters->Append(adapter = ref new otAdapter(deviceInstance));
211  }
212  }
213 
214  LeaveCriticalSection(&pThis->_cs);
215 
216  if (adapter)
217  {
218  // Send a notification of arrival
219  pThis->AdapterArrival(adapter);
220  }
221  }
222  else
223  {
224  otAdapter^ ret = nullptr;
225  Guid guid = *aDeviceGuid;
226 
227  EnterCriticalSection(&pThis->_cs);
228 
229  // Look up in the cached list of adapters to remove it
230  uint32_t i = 0;
231  for (auto&& adapter : pThis->_adapters)
232  {
233  if (adapter->InterfaceGuid == guid)
234  {
235  ret = adapter;
236  pThis->_adapters->RemoveAt(i);
237  break;
238  }
239  i++;
240  }
241 
242  LeaveCriticalSection(&pThis->_cs);
243 
244  if (ret)
245  {
246  ret->InvokeAdapterRemoval();
247  }
248  }
249  }
