61 ThreadError Header::FromMessage(const Message &aMessage, bool aCopiedMessage)
62 {
63  ThreadError error = kThreadError_Parse;
64  uint16_t offset = aMessage.GetOffset();
65  uint16_t length = aMessage.GetLength() - aMessage.GetOffset();
66  uint8_t tokenLength;
67  bool firstOption = true;
68  uint16_t optionDelta;
69  uint16_t optionLength;
70 
71  if (aCopiedMessage)
72  {
73  length -= sizeof(RequestMetadata);
74  }
75 
76  VerifyOrExit(length >= kTokenOffset, error = kThreadError_Parse);
77  aMessage.Read(offset, kTokenOffset, mHeader.mBytes);
78  mHeaderLength = kTokenOffset;
79  offset += kTokenOffset;
80  length -= kTokenOffset;
81 
82  VerifyOrExit(GetVersion() == 1, error = kThreadError_Parse);
83 
84  tokenLength = GetTokenLength();
85  VerifyOrExit(tokenLength <= kMaxTokenLength && tokenLength <= length, error = kThreadError_Parse);
86  aMessage.Read(offset, tokenLength, mHeader.mBytes + mHeaderLength);
87  mHeaderLength += tokenLength;
88  offset += tokenLength;
89  length -= tokenLength;
90 
91  while (length > 0)
92  {
93  aMessage.Read(offset, kMaxOptionHeaderSize, mHeader.mBytes + mHeaderLength);
94 
95  if (mHeader.mBytes[mHeaderLength] == 0xff)
96  {
97  mHeaderLength += sizeof(uint8_t);
98  ExitNow(error = kThreadError_None);
99  }
100 
101  optionDelta = mHeader.mBytes[mHeaderLength] >> 4;
102  optionLength = mHeader.mBytes[mHeaderLength] & 0xf;
103  mHeaderLength += sizeof(uint8_t);
104  offset += sizeof(uint8_t);
105  length -= sizeof(uint8_t);
106 
107  if (optionDelta < kOption1ByteExtension)
108  {
109  // do nothing
110  }
111  else if (optionDelta == kOption1ByteExtension)
112  {
113  optionDelta = kOption1ByteExtensionOffset + mHeader.mBytes[mHeaderLength];
114  mHeaderLength += sizeof(uint8_t);
115  offset += sizeof(uint8_t);
116  length -= sizeof(uint8_t);
117  }
118  else if (optionDelta == kOption2ByteExtension)
119  {
120  optionDelta = kOption2ByteExtensionOffset +
121  static_cast<uint16_t>((mHeader.mBytes[mHeaderLength] << 8) | mHeader.mBytes[mHeaderLength + 1]);
122  mHeaderLength += sizeof(uint16_t);
123  offset += sizeof(uint16_t);
124  length -= sizeof(uint16_t);
125  }
126  else
127  {
128  ExitNow(error = kThreadError_Parse);
129  }
130 
131  if (optionLength < kOption1ByteExtension)
132  {
133  // do nothing
134  }
135  else if (optionLength == kOption1ByteExtension)
136  {
137  optionLength = kOption1ByteExtensionOffset + mHeader.mBytes[mHeaderLength];
138  mHeaderLength += sizeof(uint8_t);
139  offset += sizeof(uint8_t);

146  mHeaderLength += sizeof(uint16_t);

165  aMessage.Read(offset, optionLength, mHeader.mBytes + mHeaderLength);
166  mHeaderLength += static_cast<uint8_t>(optionLength);
167  offset += optionLength;
168  length -= optionLength;
169  }
170

173  // No payload present - return success.
174  error = kThreadError_None;
175  }
176

181 ThreadError Header::AppendOption(const Option &aOption)
182 {
183  ThreadError error = kThreadError_None;
184  uint8_t *buf = mHeader.mBytes + mHeaderLength;
185  uint8_t *cur = buf + 1;
186  uint16_t optionDelta = aOption.mNumber - mOptionLast;
187  uint16_t optionLength;
188 
189  // Assure that no option is inserted out of order.
190  VerifyOrExit(aOption.mNumber >= mOptionLast, error = kThreadError_InvalidArgs);
191 
192  // Calculate the total option size and check the buffers.
193  optionLength = 1 + aOption.mLength;
194  optionLength += optionDelta < kOption1ByteExtensionOffset ? 0 :
195  (optionDelta < kOption2ByteExtensionOffset ? 1 : 2);
196  optionLength += aOption.mLength < kOption1ByteExtensionOffset ? 0 :
197  (aOption.mLength < kOption2ByteExtensionOffset ? 1 : 2);
198  VerifyOrExit(mHeaderLength + optionLength < kMaxHeaderLength, error = kThreadError_NoBufs);
199 
200  // Insert option delta.
201  if (optionDelta < kOption1ByteExtensionOffset)
202  {
203  *buf = (optionDelta << Option::kOptionDeltaOffset) & Option::kOptionDeltaMask;
204  }
205  else if (optionDelta < kOption2ByteExtensionOffset)
206  {
207  *buf |= kOption1ByteExtension << Option::kOptionDeltaOffset;
208  *cur++ = (optionDelta - kOption1ByteExtensionOffset) & 0xff;
209  }
210  else
211  {
212  *buf |= kOption2ByteExtension << Option::kOptionDeltaOffset;
213  optionDelta -= kOption2ByteExtensionOffset;
214  *cur++ = optionDelta >> 8;
215  *cur++ = optionDelta & 0xff;
216  }
217 
218  // Insert option length.
219  if (aOption.mLength < kOption1ByteExtensionOffset)
220  {
221  *buf |= aOption.mLength;
222  }
223  else if (aOption.mLength < kOption2ByteExtensionOffset)
224  {
225  *buf |= kOption1ByteExtension;
226  *cur++ = (aOption.mLength - kOption1ByteExtensionOffset) & 0xff;
227  }
228  else
229  {
230  *buf |= kOption2ByteExtension;
231  optionLength = aOption.mLength - kOption2ByteExtensionOffset;
232  *cur++ = optionLength >> 8;
233  *cur++ = optionLength & 0xff;
234  }
235 
236  // Insert option value.
237  memcpy(cur, aOption.mValue, aOption.mLength);
238  cur += aOption.mLength;
239 
240  mHeaderLength += static_cast<uint8_t>(cur - buf);
241  mOptionLast = aOption.mNumber;
242 
243 exit:
244  return error;
245 }

247 ThreadError Header::AppendUriPathOptions(const char *aUriPath)
248 {
249  ThreadError error = kThreadError_None;
250  const char *cur = aUriPath;
251  const char *end;
252  Header::Option coapOption;
253 
254  coapOption.mNumber = kCoapOptionUriPath;
255 
256  while ((end = strchr(cur, '/')) != NULL)
257  {
258  coapOption.mLength = static_cast<uint16_t>(end - cur);
