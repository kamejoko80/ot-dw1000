91 void mbedtls_sha256_finish(mbedtls_sha256_context *ctx, unsigned char output[32])
92 {
93  /* workaround for error in copy subroutine of SHA256 ROM implementation.
94  * Allocate an extra 64 bytes on the stack to make sure we have buffer
95  * room. This could be optomized out if you never call this function with
96  * a call stack shorter than 16 words, approx. 8 stack frames.
97  */
98  __asm("sub sp, #0x40 ");
99  SHA256_output(ctx, (uint8_t *)output);
100  __asm("add sp, #0x40 ");
101  return;
102 }
