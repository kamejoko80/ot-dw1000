142 spinel_ssize_t
143 spinel_packed_uint_decode(const uint8_t *bytes, spinel_size_t len, unsigned int *value_ptr)
144 {
145  spinel_ssize_t ret = 0;
146  unsigned int value = 0;
147 
148  int i = 0;
149 
150  do
151  {
152  if (len < sizeof(uint8_t))
153  {
154  ret = -1;
155  break;
156  }
157 
158  value |= (unsigned int)((bytes[0] & 0x7F) << i);
159  i += 7;
160  ret += sizeof(uint8_t);
161  bytes += sizeof(uint8_t);
162  len -= sizeof(uint8_t);
163  }
164  while ((bytes[-1] & 0x80) == 0x80);
165 
166  if ((ret > 0) && (value_ptr != NULL))
167  {
168  *value_ptr = value;
169  }
170 
171  return ret;
172 }

174 spinel_ssize_t
175 spinel_packed_uint_size(unsigned int value)
176 {
177  spinel_ssize_t ret;
178 
179  if (value < (1 << 7))
180  {
181  ret = 1;
182  }
183  else if (value < (1 << 14))
184  {
185  ret = 2;
186  }
187  else if (value < (1 << 21))
188  {
189  ret = 3;
190  }
191  else if (value < (1 << 28))
192  {
193  ret = 4;
194  }
195  else
196  {
197  ret = 5;
198  }
199 
200  return ret;
201 }

203 spinel_ssize_t
204 spinel_packed_uint_encode(uint8_t *bytes, spinel_size_t len, unsigned int value)
205 {
206  const spinel_ssize_t encoded_size = spinel_packed_uint_size(value);
207 
208  if ((spinel_ssize_t)len >= encoded_size)
209  {
210  spinel_ssize_t i;
211 
212  for (i = 0; i != encoded_size - 1; ++i)
213  {
214  *bytes++ = (value & 0x7F) | 0x80;
215  value = (value >> 7);
216  }
217 
218  *bytes++ = (value & 0x7F);
219  }
220 
221  return encoded_size;
222 }

224 const char *
225 spinel_next_packed_datatype(const char *pack_format)
226 {
227  int depth = 0;
228 
229  do
230  {
231  switch (*++pack_format)
232  {
233  case '(':
234  depth++;
235  break;
236 
237  case ')':
238  depth--;
239 
240  if (depth == 0)
241  {
242  pack_format++;
243  }
244 
245  break;
246  }
247  }
248  while ((depth > 0) && *pack_format != 0);
249 
250  return pack_format;
251 }

254 spinel_datatype_vunpack_(const uint8_t *data_ptr, spinel_size_t data_len, const char *pack_format, va_list_obj *args)
255 {

257 
258  for (; *pack_format != 0; pack_format = spinel_next_packed_datatype(pack_format))
259  {
260  if (*pack_format == ')')
261  {
262  // Don't go past the end of a struct.

285  case SPINEL_DATATYPE_UINT8_C:

297  data_len -= sizeof(uint8_t);
298  break;

302  case SPINEL_DATATYPE_UINT16_C:

402 
403  require_action((len <= data_len) || (data_ptr[data_len - 1] != 0), bail, (ret = -1, errno = EOVERFLOW));
404 
405  if (arg_ptr)
406  {
407  *arg_ptr = (const char *)data_ptr;
408  }
409 
410  ret += (spinel_size_t)len;
411  data_ptr += len;
412  data_len -= (spinel_size_t)len;
413  break;
414  }
415 
416  case SPINEL_DATATYPE_DATA_C:
417  {
418  spinel_ssize_t pui_len = 0;
419  uint16_t block_len = 0;
420  const uint8_t *block_ptr = data_ptr;
421  const uint8_t **block_ptr_ptr = va_arg(args->obj, const uint8_t **);
422  unsigned int *block_len_ptr = va_arg(args->obj, unsigned int *);
423  char nextformat = *spinel_next_packed_datatype(pack_format);
424 
425  if ((nextformat != 0) && (nextformat != ')'))
426  {
427  pui_len = spinel_datatype_unpack(data_ptr, data_len, SPINEL_DATATYPE_UINT16_S, &block_len);
428  //pui_len = spinel_packed_uint_decode(data_ptr, data_len, &block_len);
429  block_ptr += pui_len;
430 
431  require(pui_len > 0, bail);
432  require(block_len < SPINEL_FRAME_MAX_SIZE, bail);
433  }
434  else
435  {
436  block_len = (uint16_t)data_len;
437  pui_len = 0;
438  }
439 
440  require_action((spinel_ssize_t)data_len >= (block_len + pui_len), bail, (ret = -1, errno = EOVERFLOW));
441 
442  if (NULL != block_ptr_ptr)
443  {
444  *block_ptr_ptr = block_ptr;
445  }
446 
447  if (NULL != block_len_ptr)
448  {
449  *block_len_ptr = block_len;
450  }
451 
452  block_len += (uint16_t)pui_len;
453  ret += block_len;
454  data_ptr += block_len;
455  data_len -= block_len;
456  break;
457  }
458 
459  case SPINEL_DATATYPE_STRUCT_C:
460  {
461  spinel_ssize_t pui_len = 0;
462  uint16_t block_len = 0;
463  spinel_ssize_t actual_len = 0;
464  const uint8_t *block_ptr = data_ptr;
465  char nextformat = *spinel_next_packed_datatype(pack_format);
466 
467  if ((nextformat != 0) && (nextformat != ')'))
468  {
469  pui_len = spinel_datatype_unpack(data_ptr, data_len, SPINEL_DATATYPE_UINT16_S, &block_len);
470  block_ptr += pui_len;
471 
472  require(pui_len > 0, bail);
473  require(block_len < SPINEL_FRAME_MAX_SIZE, bail);
474  }
475  else
476  {
477  block_len = (uint16_t)data_len;
478  pui_len = 0;
479  }
480 
481  require_action((spinel_ssize_t)data_len >= (block_len + pui_len), bail, (ret = -1, errno = EOVERFLOW));
482 
483  actual_len = spinel_datatype_vunpack_(block_ptr, block_len, pack_format + 2, args);
484 
485  require_action(actual_len > -1, bail, (ret = -1, errno = EOVERFLOW));
486 
487  if (pui_len)
488  {
489  block_len += (uint16_t)pui_len;
490  }
491  else
492  {
493  block_len = (uint16_t)actual_len;
494  }
495 
496  ret += block_len;
497  data_ptr += block_len;
498  data_len -= block_len;
499  break;
500  }
501 
502  case '.':
503  // Skip.
504  break;
505 
506  case SPINEL_DATATYPE_ARRAY_C:
507  default:
508  // Unsupported Type!
509  ret = -1;
510  errno = EINVAL;
511  goto bail;
512  }
513  }
514 
515  return ret;
516 
517 bail:
518  return ret;
519 }

758  spinel_ssize_t size_len = 0;
759  char nextformat = *spinel_next_packed_datatype(pack_format);
760

762  {

806  require_action(size_len > 0, bail, {ret = -1; errno = EINVAL;});

832 
833  default:
834  // Unsupported Type!
835  ret = -1;
836  errno = EINVAL;
837  goto bail;
838 
839  }
840  }
841 
842 bail:
843  return ret;
844 }
