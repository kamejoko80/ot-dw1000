275 /*------------------------------------------------------------------------------
276 
277  clock_gate_t definition:
278 
279  31 16 0
280  -----------------------------------------------------------------
281  | SIM_SCGC register offset | control bit offset in SCGC |
282  -----------------------------------------------------------------
283 
284  For example, the SDHC clock gate is controlled by SIM_SCGC3[17], the
285  SIM_SCGC3 offset in SIM is 0x1030, then kClockGateSdhc0 is defined as
286 
287  kClockGateSdhc0 = (0x1030 << 16) | 17;
288 
289 ------------------------------------------------------------------------------*/
290 
291 #define CLK_GATE_REG_OFFSET_SHIFT 16U
292 #define CLK_GATE_REG_OFFSET_MASK 0xFFFF0000U
293 #define CLK_GATE_BIT_SHIFT_SHIFT 0U
294 #define CLK_GATE_BIT_SHIFT_MASK 0x0000FFFFU
295 
296 #define CLK_GATE_DEFINE(reg_offset, bit_shift) \
297  ((((reg_offset) << CLK_GATE_REG_OFFSET_SHIFT) & CLK_GATE_REG_OFFSET_MASK) | \
298  (((bit_shift) << CLK_GATE_BIT_SHIFT_SHIFT) & CLK_GATE_BIT_SHIFT_MASK))
299 
300 #define CLK_GATE_ABSTRACT_REG_OFFSET(x) (((x)&CLK_GATE_REG_OFFSET_MASK) >> CLK_GATE_REG_OFFSET_SHIFT)
301 #define CLK_GATE_ABSTRACT_BITS_SHIFT(x) (((x)&CLK_GATE_BIT_SHIFT_MASK) >> CLK_GATE_BIT_SHIFT_SHIFT)

554 static inline void CLOCK_EnableClock(clock_ip_name_t name)
555 {
556  uint32_t regAddr = SIM_BASE + CLK_GATE_ABSTRACT_REG_OFFSET((uint32_t)name);
557  (*(volatile uint32_t *)regAddr) |= (1U << CLK_GATE_ABSTRACT_BITS_SHIFT((uint32_t)name));
558 }

565 static inline void CLOCK_DisableClock(clock_ip_name_t name)
566 {
567  uint32_t regAddr = SIM_BASE + CLK_GATE_ABSTRACT_REG_OFFSET((uint32_t)name);
568  (*(volatile uint32_t *)regAddr) &= ~(1U << CLK_GATE_ABSTRACT_BITS_SHIFT((uint32_t)name));
569 }
