123 void TrickleTimer::StartNewInterval(void)
124 {
125  // Reset the counter and timer phase
126 
127 #ifdef ENABLE_TRICKLE_TIMER_SUPPRESSION_SUPPORT
128  c = 0;
129 #endif
130  mPhase = kPhaseTransmit;
131 
132  // Initialize t
133  if (I < 2)
134  {
135  // Immediate interval, just set t to 0
136  t = 0;
137  }
138  else if (mMode == kModeMPL)
139  {
140  // Initialize t to random value between (0, I]
141  t = otPlatRandomGet() % I;
142  }
143  else if (mMode == kModePlainTimer)
144  {
145  // Initialize t to I, which has already been randomized in Start
146  t = I;
147  }
148  else
149  {
150  // Initialize t to random value between (I/2, I]
151  t = (I / 2) + otPlatRandomGet() % (I / 2);
152  }
153 
154  // Start the timer for 't' milliseconds from now
155  mTimer.Start(t);
156 }

164 void TrickleTimer::HandleTimerFired(void)
165 {
166  Phase curPhase = mPhase;
167  bool shouldContinue = true;
168 
169  // Default the current state to Dormant
170  mPhase = kPhaseDormant;
171 
172  switch (curPhase)
173  {
174  // We have just reached time 't'
175  case kPhaseTransmit:
176  {
177  // Are we not using redundancy or is the counter still less than it?
178 #ifdef ENABLE_TRICKLE_TIMER_SUPPRESSION_SUPPORT
179  if (k == 0 || c < k)
180 #endif
181  {
182  // Invoke the transmission callback
183  shouldContinue = TransmitFired();
184  }
185 
186  // Wait for the rest of the interval to elapse
187  if (shouldContinue)
188  {
189  // If we are in plain timer mode, just randomize I and restart the interval
190  if (mMode == kModePlainTimer)
191  {
192  // Initialize I to [Imin, Imax]
193  I = Imin + otPlatRandomGet() % (Imax - Imin);
194 
195  // Start a new interval
196  StartNewInterval();
197  }
198  else
199  {
200  // Start next phase of the timer
201  mPhase = kPhaseInterval;
202 
203  // Start the time for 'I - t' milliseconds
204  mTimer.Start(I - t);
205  }
206  }
207 
208  break;
209  }
210 
211  // We have just reached time 'I'
212  case kPhaseInterval:
213  {
214  // Double 'I' to get the new interval length
215  uint32_t newI = I == 0 ? 1 : I << 1;
216 
217  if (newI > Imax) { newI = Imax; }
218 
219  I = newI;
220 
221  // Invoke the interval expiration callback
222  shouldContinue = IntervalExpiredFired();
223 
224  if (shouldContinue)
225  {
226  // Start a new interval
227  StartNewInterval();
228  }
229 
230  break;
231  }
232 
233  default:
234  assert(false);
235  }
236 }
