69 ThreadError otPlatUartEnable(void)
70 {
71  uint32_t div;
72 
73  sReceive.mHead = 0;
74  sReceive.mTail = 0;
75 
76  // clock
77  HWREG(SYS_CTRL_RCGCUART) = SYS_CTRL_RCGCUART_UART0;
78  HWREG(SYS_CTRL_SCGCUART) = SYS_CTRL_SCGCUART_UART0;
79  HWREG(SYS_CTRL_DCGCUART) = SYS_CTRL_DCGCUART_UART0;
80 
81  HWREG(UART0_BASE + UART_O_CC) = 0;
82 
83  // tx pin
84  HWREG(IOC_PA1_SEL) = IOC_MUX_OUT_SEL_UART0_TXD;
85  HWREG(IOC_PA1_OVER) = IOC_OVERRIDE_OE;
86  HWREG(GPIO_A_BASE + GPIO_O_AFSEL) |= GPIO_PIN_1;
87 
88  // rx pin
89  HWREG(IOC_PA0_SEL) = IOC_UARTRXD_UART0;
90  HWREG(IOC_PA0_OVER) = IOC_OVERRIDE_DIS;
91  HWREG(GPIO_A_BASE + GPIO_O_AFSEL) |= GPIO_PIN_0;
92 
93  HWREG(UART0_BASE + UART_O_CTL) = 0;
94 
95  // baud rate
96  div = (((kPlatformClock * 8) / kBaudRate) + 1) / 2;
97  HWREG(UART0_BASE + UART_O_IBRD) = div / 64;
98  HWREG(UART0_BASE + UART_O_FBRD) = div % 64;
99  HWREG(UART0_BASE + UART_O_LCRH) = UART_CONFIG_WLEN_8 | UART_CONFIG_STOP_ONE | UART_CONFIG_PAR_NONE;
100 
101  // configure interrupts
102  HWREG(UART0_BASE + UART_O_IM) |= UART_IM_RXIM | UART_IM_RTIM;
103 
104  // enable
105  HWREG(UART0_BASE + UART_O_CTL) = UART_CTL_UARTEN | UART_CTL_TXE | UART_CTL_RXE;
106 
107  // enable interrupts
108  HWREG(NVIC_EN0) = 1 << ((INT_UART0 - 16) & 31);
109 
110  return kThreadError_None;
111 }

131 void processReceive(void)
132 {
133  // Copy tail to prevent multiple reads
134  uint16_t tail = sReceive.mTail;
135 
136  // If the data wraps around, process the first part
137  if (sReceive.mHead > tail)
138  {
139  otPlatUartReceived(sReceive.mBuffer + sReceive.mHead, kReceiveBufferSize - sReceive.mHead);
140 
141  // Reset the buffer mHead back to zero.
142  sReceive.mHead = 0;
143  }
144 
145  // For any data remaining, process it
146  if (sReceive.mHead != tail)
147  {
148  otPlatUartReceived(sReceive.mBuffer + sReceive.mHead, tail - sReceive.mHead);
149 
150  // Set mHead to the local tail we have cached
151  sReceive.mHead = tail;
152  }
153 }

155 void processTransmit(void)
156 {
157  VerifyOrExit(sTransmitBuffer != NULL, ;);
158 
159  for (; sTransmitLength > 0; sTransmitLength--)
160  {
161  while (HWREG(UART0_BASE + UART_O_FR) & UART_FR_TXFF);
162 
163  HWREG(UART0_BASE + UART_O_DR) = *sTransmitBuffer++;
164  }
165 
166  sTransmitBuffer = NULL;
167  otPlatUartSendDone();
168 
169 exit:
170  return;
171 }

179 void UART0IntHandler(void)
180 {
181  uint32_t mis;
182  uint8_t byte;
183 
184  mis = HWREG(UART0_BASE + UART_O_MIS);
185  HWREG(UART0_BASE + UART_O_ICR) = mis;
186 
187  if (mis & (UART_IM_RXIM | UART_IM_RTIM))
188  {
189  while (!(HWREG(UART0_BASE + UART_O_FR) & UART_FR_RXFE))
190  {
191  byte = HWREG(UART0_BASE + UART_O_DR);
192 
193  // We can only write if incrementing mTail doesn't equal mHead
194  if (sReceive.mHead != (sReceive.mTail + 1) % kReceiveBufferSize)
195  {
196  sReceive.mBuffer[sReceive.mTail] = byte;
197  sReceive.mTail = (sReceive.mTail + 1) % kReceiveBufferSize;
198  }
199  }
200  }
