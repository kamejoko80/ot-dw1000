188 ThreadError SecureClient::HandleDtlsSend(const uint8_t *aBuf, uint16_t aLength)
189 {
190  ThreadError error = kThreadError_None;
191 
192  otLogFuncEntry();
193 
194  if (mTransmitMessage == NULL)
195  {
196  VerifyOrExit((mTransmitMessage = mSocket.NewMessage(0)) != NULL, error = kThreadError_NoBufs);
197  mTransmitMessage->SetLinkSecurityEnabled(false);
198  }
199 
200  VerifyOrExit(mTransmitMessage->Append(aBuf, aLength) == kThreadError_None, error = kThreadError_NoBufs);
201 
202  mTransmitTask.Post();
203 
204 exit:
205 
206  if (error != kThreadError_None && mTransmitMessage != NULL)
207  {
208  mTransmitMessage->Free();
209  }
210 
211  otLogFuncExitErr(error);
212 
213  return error;
214 }
