139 void
140 NcpSpi::SpiTransactionComplete(
141  uint8_t *aMISOBuf,
142  uint16_t aMISOBufLen,
143  uint8_t *aMOSIBuf,
144  uint16_t aMOSIBufLen,
145  uint16_t aTransactionLength
146 ) {
147  // This may be executed from an interrupt context.
148  // Must return as quickly as possible.
149 
150  uint16_t rx_data_len(0);
151  uint16_t rx_accept_len(0);
152  uint16_t tx_data_len(0);
153  uint16_t tx_accept_len(0);
154 
155  // TODO: Check `PATTERN` bits of `HDR` and ignore frame if not set.
156  // Holding off on implementing this so as to not cause immediate
157  // compatibility problems, even though it is required by the spec.
158 
159  if (aTransactionLength >= kSpiHeaderLength)
160  {
161  if (aMISOBufLen >= kSpiHeaderLength)
162  {
163  rx_accept_len = spi_header_get_accept_len(aMISOBuf);
164  tx_data_len = spi_header_get_data_len(aMISOBuf);
165  (void)spi_header_get_flag_byte(aMISOBuf);
166  }
167 
168  if (aMOSIBufLen >= kSpiHeaderLength)
169  {
170  rx_data_len = spi_header_get_data_len(aMOSIBuf);
171  tx_accept_len = spi_header_get_accept_len(aMOSIBuf);
172  }
173 
174  if ( !mHandlingRxFrame
175  && (rx_data_len > 0)
176  && (rx_data_len <= (aTransactionLength - kSpiHeaderLength))
177  && (rx_data_len <= rx_accept_len)
178  ) {
179  mHandlingRxFrame = true;
180  mHandleRxFrameTask.Post();
181  }
182 
183  if ( (mTxState == kTxStateSending)
184  && (tx_data_len > 0)
185  && (tx_data_len <= (aTransactionLength - kSpiHeaderLength))
186  && (tx_data_len <= tx_accept_len)
187  ) {
188  // Our transmission was successful.
189  mTxState = kTxStateHandlingSendDone;
190  mPrepareTxFrameTask.Post();
191  }
192  }
193 
194  if ( (aTransactionLength >= 1)
195  && (aMISOBufLen >= 1)
196  ) {
197  // Clear the reset flag.
198  spi_header_set_flag_byte(mSendFrame, SPI_PATTERN_VALUE);
199  spi_header_set_flag_byte(mEmptySendFrame, SPI_PATTERN_VALUE);
200  }
201 
202  if (mTxState == kTxStateSending)
203  {
204  aMISOBuf = mSendFrame;
205  aMISOBufLen = mSendFrameLen;
206  }
207  else
208  {
209  aMISOBuf = mEmptySendFrame;
210  aMISOBufLen = kSpiHeaderLength;
211  }
212 
213  if (mHandlingRxFrame)
214  {
215  aMOSIBuf = mEmptyReceiveFrame;
216  aMOSIBufLen = kSpiHeaderLength;
217  spi_header_set_accept_len(aMISOBuf, 0);
218  }
219  else
220  {
221  aMOSIBuf = mReceiveFrame;
222  aMOSIBufLen = sizeof(mReceiveFrame);
223  spi_header_set_accept_len(aMISOBuf, sizeof(mReceiveFrame) - kSpiHeaderLength);
224  }
225 
226  otPlatSpiSlavePrepareTransaction(
227  aMISOBuf,
228  aMISOBufLen,
229  aMOSIBuf,
230  aMOSIBufLen,
231  (mTxState == kTxStateSending)
232  );
233 }

268 ThreadError NcpSpi::PrepareNextSpiSendFrame(void)
269 {
270  ThreadError errorCode = kThreadError_None;
271  uint16_t frameLength;
272  uint16_t readLength;
273 
274  VerifyOrExit(!mTxFrameBuffer.IsEmpty(), ;);
275 
276  SuccessOrExit(errorCode = mTxFrameBuffer.OutFrameBegin());
277 
278  frameLength = mTxFrameBuffer.OutFrameGetLength();
279  VerifyOrExit(frameLength <= sizeof(mSendFrame) - kSpiHeaderLength, errorCode = kThreadError_NoBufs);
280 
281  spi_header_set_data_len(mSendFrame, frameLength);
282 
283  // Half-duplex to avoid race condition.
284  spi_header_set_accept_len(mSendFrame, 0);
285 
286  readLength = mTxFrameBuffer.OutFrameRead(frameLength, mSendFrame + kSpiHeaderLength);
287  VerifyOrExit(readLength == frameLength, errorCode = kThreadError_Failed);
288 
289  mSendFrameLen = frameLength + kSpiHeaderLength;
290 
291  mTxState = kTxStateSending;
292 
293  errorCode = otPlatSpiSlavePrepareTransaction(
294  mSendFrame,
295  mSendFrameLen,
296  mEmptyReceiveFrame,
297  sizeof(mEmptyReceiveFrame),
298  true
299  );
300 
301  if (errorCode == kThreadError_Busy)
302  {
303  // Being busy is OK. We will get the transaction
304  // set up properly when the current transaction
305  // is completed.
306  errorCode = kThreadError_None;
307  }
308 
309  if (errorCode != kThreadError_None)
310  {
311  mTxState = kTxStateIdle;
312  }
313 
314  // Remove the frame from tx buffer and inform the base
315  // class that space is now available for a new frame.
316  mTxFrameBuffer.OutFrameRemove();
317  super_t::HandleSpaceAvailableInTxBuffer();
318 
319 exit:
320  return errorCode;
321 }

328 void NcpSpi::PrepareTxFrame(void)
329 {
330  switch (mTxState)
331  {
332  case kTxStateHandlingSendDone:
333  mTxState = kTxStateIdle;
334 
335  // Fall-through to next case to prepare the next frame (if any).
336 
337  case kTxStateIdle:
338  PrepareNextSpiSendFrame();
339  break;
340 
341  case kTxStateSending:
342  // The next frame in queue (if any) will be prepared when the
343  // current frame is successfully sent and this task is posted
344  // again from the `SpiTransactionComplete()` callback.
345  break;
346  }
347 }
