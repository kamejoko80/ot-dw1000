195 /*++
196 
197 Routine Description:
198 
199  Halt handler is called when NDIS receives IRP_MN_STOP_DEVICE,
200  IRP_MN_SUPRISE_REMOVE or IRP_MN_REMOVE_DEVICE requests from the PNP
201  manager. Here, the driver should free all the resources acquired in
202  MiniportInitialize and stop access to the hardware. NDIS will not submit
203  any further request once this handler is invoked.
204 
205  1) Free and unmap all I/O resources.
206  2) Disable interrupt and deregister interrupt handler.
207  3) Deregister shutdown handler regsitered by
208  NdisMRegisterAdapterShutdownHandler .
209  4) Cancel all queued up timer callbacks.
210  5) Finally wait indefinitely for all the outstanding receive
211  packets indicated to the protocol to return.
212 
213  MiniportHalt runs at IRQL = PASSIVE_LEVEL.
214 
215 
216 Arguments:
217 
218  MiniportAdapterContext Pointer to the Adapter
219  HaltAction The reason for halting the adapter
220 
221 Return Value:
222 
223  None.
224 
225 --*/
226 {
227  POTTMP_ADAPTER_CONTEXT AdapterContext = (POTTMP_ADAPTER_CONTEXT)MiniportAdapterContext;
228 
229  LogFuncEntry(DRIVER_DEFAULT);
230 
231  PAGED_CODE();
232 
233  //
234  // Call Shutdown handler to disable interrupt and turn the hardware off
235  // by issuing a full reset
236  //
237  if (HaltAction != NdisHaltDeviceSurpriseRemoved)
238  {
239  MPShutdownEx(MiniportAdapterContext, NdisShutdownPowerOff);
240  }
241 
242  AdapterUninitialize(AdapterContext);
243 
244  LogFuncExit(DRIVER_DEFAULT);
245 }

309 
310  PAGED_CODE();
311 
312  switch (NetDevicePnPEvent->DevicePnPEvent)
313  {
314  case NdisDevicePnPEventQueryRemoved:
315  //
316  // Called when NDIS receives IRP_MN_QUERY_REMOVE_DEVICE.
317  //
318  LogInfo(DRIVER_DEFAULT, "MPPnPEventNotify: NdisDevicePnPEventQueryRemoved");
319  break;
320 
321  case NdisDevicePnPEventRemoved:
322  //
323  // Called when NDIS receives IRP_MN_REMOVE_DEVICE.
324  // NDIS calls MiniportHalt function after this call returns.
325  //
326  LogInfo(DRIVER_DEFAULT, "MPPnPEventNotify: NdisDevicePnPEventRemoved");
327  break;
328 
329  case NdisDevicePnPEventSurpriseRemoved:
330  //
331  // Called when NDIS receives IRP_MN_SUPRISE_REMOVAL.
332  // NDIS calls MiniportHalt function after this call returns.
333  //
334  LogInfo(DRIVER_DEFAULT, "MPDevicePnpEventNotify: NdisDevicePnPEventSurpriseRemoved");
335  break;
336 
337  case NdisDevicePnPEventQueryStopped:
338  //
339  // Called when NDIS receives IRP_MN_QUERY_STOP_DEVICE. ??
340  //
341  LogInfo(DRIVER_DEFAULT, "MPPnPEventNotify: NdisDevicePnPEventQueryStopped");
342  break;
343 
344  case NdisDevicePnPEventStopped:
345  //
346  // Called when NDIS receives IRP_MN_STOP_DEVICE.
347  // NDIS calls MiniportHalt function after this call returns.
348  //
349  //
350  LogInfo(DRIVER_DEFAULT, "MPPnPEventNotify: NdisDevicePnPEventStopped");
351  break;
352 
353  case NdisDevicePnPEventPowerProfileChanged:
354  //
355  // After initializing a miniport driver and after miniport driver
356  // receives an OID_PNP_SET_POWER notification that specifies
357  // a device power state of NdisDeviceStateD0 (the powered-on state),
358  // NDIS calls the miniport's MiniportPnPEventNotify function with
359  // PnPEvent set to NdisDevicePnPEventPowerProfileChanged.
360  //
361  LogInfo(DRIVER_DEFAULT, "MPDevicePnpEventNotify: NdisDevicePnPEventPowerProfileChanged");
362 
363  if (NetDevicePnPEvent->InformationBufferLength == sizeof( ULONG ))
364  {
365  const ULONG NdisPowerProfile = *((const ULONG *)NetDevicePnPEvent->InformationBuffer);
366 
367  if (NdisPowerProfile == NdisPowerProfileBattery)
368  {
369  LogInfo(DRIVER_DEFAULT, "The host system is running on battery power");
370  }
371  else if (NdisPowerProfile == NdisPowerProfileAcOnLine)
372  {
373  LogInfo(DRIVER_DEFAULT, "The host system is running on AC power");
374  }
375  }
376  break;
377 
378  default:
379  LogError(DRIVER_DEFAULT, "MPDevicePnpEventNotify: unknown PnP event 0x%x\n", NetDevicePnPEvent->DevicePnPEvent);
380  }
381 
382  LogFuncExit(DRIVER_DEFAULT);
383 }
