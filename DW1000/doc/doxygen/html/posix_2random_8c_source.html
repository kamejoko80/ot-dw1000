52 uint32_t otPlatRandomGet(void)
53 {
54  uint32_t mlcg, p, q;
55  uint64_t tmpstate;
56 
57  tmpstate = (uint64_t)33614 * (uint64_t)s_state;
58  q = tmpstate & 0xffffffff;
59  q = q >> 1;
60  p = tmpstate >> 32;
61  mlcg = p + q;
62 
63  if (mlcg & 0x80000000)
64  {
65  mlcg &= 0x7fffffff;
66  mlcg++;
67  }
68 
69  s_state = mlcg;
70 
71  return mlcg;
72 }
