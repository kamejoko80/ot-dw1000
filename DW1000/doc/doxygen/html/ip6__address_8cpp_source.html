182 uint8_t Address::PrefixMatch(const Address &aOther) const
183 {
184  uint8_t rval = 0;
185  uint8_t diff;
186 
187  for (uint8_t i = 0; i < sizeof(Address); i++)
188  {
189  diff = mFields.m8[i] ^ aOther.mFields.m8[i];
190 
191  if (diff == 0)
192  {
193  rval += 8;
194  }
195  else
196  {
197  while ((diff & 0x80) == 0)
198  {
199  rval++;
200  diff <<= 1;
201  }
202 
203  break;
204  }
205  }
206 
207  return rval;
208 }

220 ThreadError Address::FromString(const char *aBuf)
221 {
222  ThreadError error = kThreadError_None;
223  uint8_t *dst = reinterpret_cast<uint8_t *>(mFields.m8);
224  uint8_t *endp = reinterpret_cast<uint8_t *>(mFields.m8 + 15);
225  uint8_t *colonp = NULL;
226  uint16_t val = 0;
227  uint8_t count = 0;
228  bool first = true;
229  char ch;
230  uint8_t d;
231 
232  memset(mFields.m8, 0, 16);
233 
234  dst--;
235 
236  for (;;)
237  {
238  ch = *aBuf++;
239  d = ch & 0xf;
240 
241  if (('a' <= ch && ch <= 'f') || ('A' <= ch && ch <= 'F'))
242  {
243  d += 9;
244  }
245  else if (ch == ':' || ch == '\0' || ch == ' ')
246  {
247  if (count)
248  {
249  VerifyOrExit(dst + 2 <= endp, error = kThreadError_Parse);
250  *(dst + 1) = static_cast<uint8_t>(val >> 8);
251  *(dst + 2) = static_cast<uint8_t>(val);
252  dst += 2;
253  count = 0;
254  val = 0;
255  }
256  else if (ch == ':')
257  {
258  VerifyOrExit(colonp == NULL || first, error = kThreadError_Parse);
259  colonp = dst;
260  }
261 
262  if (ch == '\0' || ch == ' ')
263  {
264  break;
265  }
266 
267  continue;
268  }
269  else
270  {
271  VerifyOrExit('0' <= ch && ch <= '9', error = kThreadError_Parse);
272  }
273 
274  first = false;
275  val = static_cast<uint16_t>((val << 4) | d);
276  VerifyOrExit(++count <= 4, error = kThreadError_Parse);
277  }
278 
279  while (colonp && dst > colonp)
280  {
281  *endp-- = *dst--;
282  }
283 
284  while (endp > dst)
285  {
286  *endp-- = 0;
287  }
288 
289 exit:
290  return error;
291 }
