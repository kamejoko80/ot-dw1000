66 // Checks the encoded average RSS value to match the value from GetAverageRss().
67 void VerifyEncodedRssValue(LinkQualityInfo &aLinkInfo)
68 {
69  int8_t rss = aLinkInfo.GetAverageRss();
70  uint16_t encodedRss = aLinkInfo.GetAverageRssAsEncodedWord();
71 
72  if (rss != LinkQualityInfo::kUnknownRss)
73  {
74  VerifyOrQuit(rss == -static_cast<int16_t>((encodedRss + (kEncodedAverageMultiple / 2)) >> kEncodedAverageBitShift),
75  "TestLinkQualityInfo failed - Ecoded RSS does not match the value from GetAverageRss().");
76  }
77  else
78  {
79  VerifyOrQuit(encodedRss == 0,
80  "TestLinkQualityInfo failed - Ecoded RSS does not match the value from GetAverageRss().");
81  }
82 }

100 void TestLinkQualityData(RssTestData anRssData)
101 {
102  LinkQualityInfo linkInfo;
103  int8_t rss, ave, min, max;
104  size_t i;
105 
106  printf("- - - - - - - - - - - - - - - - - -\n");
107  min = kMinRssValue;
108  max = kMaxRssValue;
109 
110  for (i = 0; i < anRssData.mRssListSize; i++)
111  {
112  rss = anRssData.mRssList[i];
113  min = MIN_RSS(rss, min);
114  max = MAX_RSS(rss, max);
115  linkInfo.AddRss(sNoiseFloor, rss);
116  ave = linkInfo.GetAverageRss();
117  VerifyOrQuit(ave >= min,
118  "TestLinkQualityInfo failed - GetAverageRss() is smaller than min value.");
119  VerifyOrQuit(ave <= max,
120  "TestLinkQualityInfo failed - GetAverageRss() is larger than min value");
121  VerifyEncodedRssValue(linkInfo);
122  printf("%02u) AddRss(%4d): ", (unsigned int)i, rss);
123  PrintOutcome(linkInfo);
124  }
125 
126  VerifyOrQuit(linkInfo.GetLinkQuality(sNoiseFloor) == anRssData.mExpectedLinkQuality,
127  "TestLinkQualityInfo failed - GetLinkQuality() is incorrect");
128 }
