319

332  def _parse_security_control(self, security_control_byte):
333  security_level = (security_control_byte & 0x07)
334  key_id_mode = (security_control_byte >> 3) & 0x03
335 
336  return security_level, key_id_mode
337 
338  def _parse_frame_counter(self, frame_counter_bytes):
339  return struct.unpack("<I", frame_counter_bytes)[0]
340 
341  def _key_id_length(self, key_id_mode):
342  return self._key_id_lengths[key_id_mode]
343 
344  def parse(self, data, message_info):
345  security_control_bytes = bytearray(data.read(self._SECURITY_CONTROL_LENGTH))
346  frame_counter_bytes = bytearray(data.read(self._FRAME_COUNTER_LENGTH))
347 
348  security_level, key_id_mode = self._parse_security_control(security_control_bytes[0])
349  frame_counter = self._parse_frame_counter(frame_counter_bytes)
350 
351  key_id_length = self._key_id_length(key_id_mode)
352  key_id_bytes = bytearray(data.read(key_id_length))
353 
354  aux_sec_hdr = AuxiliarySecurityHeader(key_id_mode, security_level, frame_counter, key_id_bytes)
355
