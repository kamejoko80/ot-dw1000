338 void dw1000RadioProcess(otInstance *aInstance)
339 {
340  decaIrqStatus_t stat;
341 
342  stat = decamutexon();
343 
344  VerifyOrExit(sState == kStateReceive || sState == kStateTransmit, ;);
345 
346  if ((sState == kStateReceive && sReceiveFrame.mLength > 0) ||
347  (sState == kStateTransmit && sReceiveFrame.mLength > IEEE802154_ACK_LENGTH))
348  {
349 #if OPENTHREAD_ENABLE_DIAG
350 
351  if (otPlatDiagModeGet())
352  {
353  otPlatDiagRadioReceiveDone(aInstance, &sReceiveFrame, sReceiveError);
354  }
355  else
356 #endif
357  {
358  // signal MAC layer for each received frame if promiscous is enabled
359  // otherwise only signal MAC layer for non-ACK frame
360  if ((dw1000GetPromiscuous() ||
361  (sReceiveFrame.mLength > IEEE802154_ACK_LENGTH)) &&
362  (receiveDone==1))
363  {
364  otLogDebgPlat("Received %d bytes", sReceiveFrame.mLength);
365  otPlatRadioReceiveDone(aInstance,
366  &sReceiveFrame,
367  sReceiveError);
368  receiveDone=0;
369  }
370  }
371  }
372 
373  if (sState == kStateTransmit)
374  {
375  if (sTransmitError != kThreadError_None ||
376  (sTransmitFrame.mPsdu[0] & IEEE802154_ACK_REQUEST) == 0)
377  {
378  if (sTransmitError != kThreadError_None)
379  {
380  otLogDebgPlat("Transmit failed ErrorCode=%d", sTransmitError);
381  }
382 
383  sState = kStateReceive;
384 
385 #if OPENTHREAD_ENABLE_DIAG
386  if (otPlatDiagModeGet())
387  {
388  otPlatDiagRadioTransmitDone(aInstance,
389  &sTransmitFrame,
390  false,
391  sTransmitError);
392  }
393  else
394 #endif
395  {
396  otPlatRadioTransmitDone(aInstance,
397  &sTransmitFrame,
398  false,
399  sTransmitError);
400  }
401 
402  transmitDone=0;
403  }
404  else if (sReceiveFrame.mLength == IEEE802154_ACK_LENGTH &&
405  (sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_TYPE_MASK)
406  == IEEE802154_FRAME_TYPE_ACK &&
407  (sReceiveFrame.mPsdu[IEEE802154_DSN_OFFSET]
408  == sTransmitFrame.mPsdu[IEEE802154_DSN_OFFSET]))
409  {
410  sState = kStateReceive;
411 
412 #if OPENTHREAD_ENABLE_DIAG
413  if (otPlatDiagModeGet())
414  {
415  otPlatDiagRadioTransmitDone(
416  aInstance,
417  &sTransmitFrame,
418  (sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_PENDING) != 0,
419  sTransmitError);
420  }
421  else
422 #endif
423  {
424  otPlatRadioTransmitDone(
425  aInstance,
426  &sTransmitFrame,
427  (sReceiveFrame.mPsdu[0] & IEEE802154_FRAME_PENDING) != 0,
428  sTransmitError);
429  }
430 
431  receiveDone=0;
432  }
433  }
434 
435  sReceiveFrame.mLength = 0;
436 
437 exit:
438  decamutexoff(stat);
439 return;
440 }

554 /*
555  * GPIOTE_IRQ handler
556  *
557  * This handler is used to interrupt cpu on Receive Good Frame and Transmit
558  * Frame Sent
559  */
560 void GPIOTE_IRQHandler(void)
561 {
562  uint32_t sStatus = 0;
563 
564  sStatus = dwt_read32bitoffsetreg(SYS_STATUS_ID,0);
565 
566  /* Check Whether interrupt is to indicate Received Frame is Good */
567  if (sStatus & SYS_STATUS_RXFCG)
568  {
569  dwt_write32bitreg(
570  SYS_STATUS_ID,
571  SYS_STATUS_RXFCG | SYS_STATUS_RXPHD | SYS_STATUS_RXDFR);
572 
573  sReceiveFrame.mLength =
574  (uint8_t) dwt_read16bitoffsetreg(RX_FINFO_ID,0) & 0x7f;
575 
576  dwt_readrxdata(sReceiveFrame.mPsdu, sReceiveFrame.mLength, 0);
577 
578  if (sReceiveFrame.mPsdu[0] & IEEE802154_ACK_REQUEST)
579  {
580  if(sReceiveFrame.mPsdu[1] & SRC_ADDR_TYPE_EXTENDED)
581  ack_psdu[0]= 0x02;
582  else
583  ack_psdu[0]= 0x12;
584 
585  ack_psdu[3] = sReceiveFrame.mPsdu[3];
586  dwt_writetxdata(5, (uint8_t *)ack_psdu, 0);
587  dwt_writetxfctrl(5, 0);
588  dwt_starttx(DWT_START_TX_IMMEDIATE | DWT_RESPONSE_EXPECTED);
589  }
590  else
591  {
592  dw1000RxEnable();
593  }
594 
595  receiveDone=1;
596  }
597  /* Check whether interrupt is Transmit Frame Sent*/
598  if (sStatus & SYS_STATUS_TXFRS)
599  {
600  transmitDone=1;
601  dwt_write32bitreg(
602  SYS_STATUS_ID,
603  SYS_STATUS_TXFRS | SYS_STATUS_TXFRB | SYS_STATUS_TXPRS |
604  SYS_STATUS_TXPHS);
605  }
606 
607  /* Clear the Interrupt*/
608  NRF_GPIOTE->EVENTS_PORT = 0;
609 }
