78 void Uart::ReceiveTask(const uint8_t *aBuf, uint16_t aBufLength)
79 {
80  const uint8_t *end;
81 
82  end = aBuf + aBufLength;
83 
84  for (; aBuf < end; aBuf++)
85  {
86  switch (*aBuf)
87  {
88  case '\r':
89  case '\n':
90  Output(CRNL, sizeof(CRNL));
91 
92  if (mRxLength > 0)
93  {
94  mRxBuffer[mRxLength] = '\0';
95  ProcessCommand();
96  }
97 
98  Output(sCommandPrompt, sizeof(sCommandPrompt));
99 
100  break;
101 
102 #ifdef OPENTHREAD_EXAMPLES_POSIX
103 
104  case 0x03: // ASCII for Ctrl-C
105  exit(EXIT_SUCCESS);
106  break;
107 #endif
108 
109  case '\b':
110  case 127:
111  if (mRxLength > 0)
112  {
113  Output(sEraseString, sizeof(sEraseString));
114  mRxBuffer[--mRxLength] = '\0';
115  }
116 
117  break;
118 
119  default:
120  if (mRxLength < kRxBufferSize)
121  {
122  Output(reinterpret_cast<const char *>(aBuf), 1);
123  mRxBuffer[mRxLength++] = static_cast<char>(*aBuf);
124  }
125 
126  break;
127  }
128  }
129 }

152 int Uart::Output(const char *aBuf, uint16_t aBufLength)
153 {
154  uint16_t remaining = kTxBufferSize - mTxLength;
155  uint16_t tail;
156 
157  if (aBufLength > remaining)
158  {
159  aBufLength = remaining;
160  }
161 
162  for (int i = 0; i < aBufLength; i++)
163  {
164  tail = (mTxHead + mTxLength) % kTxBufferSize;
165  mTxBuffer[tail] = *aBuf++;
166  mTxLength++;
167  }
168 
169  Send();
170 
171  return aBufLength;
172 }
