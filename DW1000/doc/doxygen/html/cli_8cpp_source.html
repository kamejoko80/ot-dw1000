163 int Interpreter::Hex2Bin(const char *aHex, uint8_t *aBin, uint16_t aBinLength)
164 {
165  size_t hexLength = strlen(aHex);
166  const char *hexEnd = aHex + hexLength;
167  uint8_t *cur = aBin;
168  uint8_t numChars = hexLength & 1;
169  uint8_t byte = 0;
170 
171  if ((hexLength + 1) / 2 > aBinLength)
172  {
173  return -1;
174  }
175 
176  while (aHex < hexEnd)
177  {
178  if ('A' <= *aHex && *aHex <= 'F')
179  {
180  byte |= 10 + (*aHex - 'A');
181  }
182  else if ('a' <= *aHex && *aHex <= 'f')
183  {
184  byte |= 10 + (*aHex - 'a');
185  }
186  else if ('0' <= *aHex && *aHex <= '9')
187  {
188  byte |= *aHex - '0';
189  }
190  else
191  {
192  return -1;
193  }
194 
195  aHex++;
196  numChars++;
197 
198  if (numChars >= 2)
199  {
200  numChars = 0;
201  *cur++ = byte;
202  byte = 0;
203  }
204  else
205  {
206  byte <<= 4;
207  }
208  }
209 
210  return static_cast<int>(cur - aBin);
211 }

259 {
260  ThreadError error = kThreadError_None;
261  otMacBlacklistEntry entry;
262  int argcur = 0;
263  uint8_t extAddr[8];
264 
265  if (argcur >= argc)
266  {
267  if (otIsMacBlacklistEnabled(mInstance))
268  {
269  sServer->OutputFormat("Enabled\r\n");
270  }
271  else
272  {
273  sServer->OutputFormat("Disabled\r\n");
274  }
275 
276  for (uint8_t i = 0; ; i++)
277  {
278  if (otGetMacBlacklistEntry(mInstance, i, &entry) != kThreadError_None)
279  {
280  break;
281  }
282 
283  if (entry.mValid == false)
284  {
285  continue;
286  }
287 
288  OutputBytes(entry.mExtAddress.m8, OT_EXT_ADDRESS_SIZE);
289 
290  sServer->OutputFormat("\r\n");
291  }
292  }
293  else if (strcmp(argv[argcur], "add") == 0)
294  {
295  VerifyOrExit(++argcur < argc, error = kThreadError_Parse);
296  VerifyOrExit(Hex2Bin(argv[argcur], extAddr, sizeof(extAddr)) == sizeof(extAddr), error = kThreadError_Parse);
297 
298  otAddMacBlacklist(mInstance, extAddr);
299  VerifyOrExit(otAddMacBlacklist(mInstance, extAddr) == kThreadError_None, error = kThreadError_Parse);
300  }
301  else if (strcmp(argv[argcur], "clear") == 0)
302  {
303  otClearMacBlacklist(mInstance);
304  }
305  else if (strcmp(argv[argcur], "disable") == 0)
306  {
307  otDisableMacBlacklist(mInstance);
308  }
309  else if (strcmp(argv[argcur], "enable") == 0)
310  {
311  otEnableMacBlacklist(mInstance);
312  }
313  else if (strcmp(argv[argcur], "remove") == 0)
314  {
315  VerifyOrExit(++argcur < argc, error = kThreadError_Parse);
316  VerifyOrExit(Hex2Bin(argv[argcur], extAddr, sizeof(extAddr)) == sizeof(extAddr), error = kThreadError_Parse);
317  otRemoveMacBlacklist(mInstance, extAddr);
318  }
319 
320 exit:
321  AppendResult(error);
322 }
