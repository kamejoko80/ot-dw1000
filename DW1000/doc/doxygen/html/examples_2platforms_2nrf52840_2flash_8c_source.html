79 ThreadError utilsFlashStatusWait(uint32_t aTimeout)
80 {
81  ThreadError error = kThreadError_Busy;
82 
83  if (aTimeout == 0)
84  {
85  if (NRF_NVMC->READY == NVMC_READY_READY_Ready)
86  {
87  error = kThreadError_None;
88  }
89  }
90  else
91  {
92  uint32_t startTime = otPlatAlarmGetNow();
93 
94  do
95  {
96  if (NRF_NVMC->READY == NVMC_READY_READY_Ready)
97  {
98  error = kThreadError_None;
99  break;
100  }
101  }
102  while (otPlatAlarmGetNow() - startTime < aTimeout);
103  }
104 
105  return error;
106 }
