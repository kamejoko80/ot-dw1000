170 VOID
171 otLwfTunWorkerThread(
172  PVOID Context
173  )
174 {
175  PMS_FILTER pFilter = (PMS_FILTER)Context;
176  NT_ASSERT(pFilter);
177 
178  LogFuncEntry(DRIVER_DEFAULT);
179 
180  PKEVENT WaitEvents[] =
181  {
182  &pFilter->TunWorkerThreadStopEvent,
183  &pFilter->TunWorkerThreadAddressChangedEvent
184  };
185 
186  LogFuncExit(DRIVER_DEFAULT);
187 
188  while (true)
189  {
190  // Wait for event to stop or process event to fire
191  NTSTATUS status =
192  KeWaitForMultipleObjects(
193  ARRAYSIZE(WaitEvents),
194  (PVOID*)WaitEvents,
195  WaitAny,
196  Executive,
197  KernelMode,
198  FALSE,
199  NULL,
200  NULL);
201 
202  // If it is the first event, then we are shutting down. Exit loop and terminate thread
203  if (status == STATUS_WAIT_0)
204  {
205  LogInfo(DRIVER_DEFAULT, "Received tunnel worker thread shutdown event.");
206  break;
207  }
208  else if (status == STATUS_WAIT_0 + 1) // TunWorkerThreadAddressChangedEvent fired
209  {
210  PVOID DataBuffer = NULL;
211  const uint8_t* value_data_ptr = NULL;
212  spinel_size_t value_data_len = 0;
213 
214  // Query the current addresses
215  status =
216  otLwfCmdGetProp(
217  pFilter,
218  &DataBuffer,
219  SPINEL_PROP_IPV6_ADDRESS_TABLE,
220  SPINEL_DATATYPE_DATA_S,
221  &value_data_ptr,
222  &value_data_len);
223  if (NT_SUCCESS(status))
224  {
225  uint32_t aNotifFlags = 0;
226  otLwfTunAddressesUpdated(pFilter, value_data_ptr, value_data_len, &aNotifFlags);
227 
228  // Send notification
229  if (aNotifFlags != 0)
230  {
231  PFILTER_NOTIFICATION_ENTRY NotifEntry = FILTER_ALLOC_NOTIF(pFilter);
232  if (NotifEntry)
233  {
234  RtlZeroMemory(NotifEntry, sizeof(FILTER_NOTIFICATION_ENTRY));
235  NotifEntry->Notif.InterfaceGuid = pFilter->InterfaceGuid;
236  NotifEntry->Notif.NotifType = OTLWF_NOTIF_STATE_CHANGE;
237  NotifEntry->Notif.StateChangePayload.Flags = aNotifFlags;
238 
239  otLwfIndicateNotification(NotifEntry);
240  }
241  }
242  }
243  else
244  {
245  LogWarning(DRIVER_DEFAULT, "Failed to query addresses, %!STATUS!", status);
246  }
247 
248  if (DataBuffer) FILTER_FREE_MEM(DataBuffer);
249  }
250  else
251  {
252  LogWarning(DRIVER_DEFAULT, "Unexpected wait result, %!STATUS!", status);
253  }
254  }
255 
256  PsTerminateSystemThread(STATUS_SUCCESS);
257 }

260 VOID
261 otLwfIrpCommandHandler(
262  _In_ PMS_FILTER pFilter,
263  _In_ PVOID Context,
264  _In_ UINT Command,
265  _In_ spinel_prop_key_t Key,
266  _In_reads_bytes_(DataLength) const uint8_t* Data,
267  _In_ spinel_size_t DataLength
268  )
269 {
270  SPINEL_IRP_CMD_CONTEXT* CmdContext = (SPINEL_IRP_CMD_CONTEXT*)Context;
271  PIO_STACK_LOCATION IrpSp = IoGetCurrentIrpStackLocation(CmdContext->Irp);
272 
273  ULONG IoControlCode = IrpSp->Parameters.DeviceIoControl.IoControlCode;
274  PVOID OutBuffer = CmdContext->Irp->AssociatedIrp.SystemBuffer;
275  ULONG OutBufferLength = IrpSp->Parameters.DeviceIoControl.OutputBufferLength;
276  ULONG OrigOutBufferLength = OutBufferLength;
277 
278  NTSTATUS status;
279 
280  UNREFERENCED_PARAMETER(pFilter);
281 
282  // Clear the cancel routine
283  IoSetCancelRoutine(CmdContext->Irp, NULL);
284 
285  if (Data == NULL)
286  {
287  status = STATUS_CANCELLED;
288  OutBufferLength = 0;
289  }
290  else if (Command == SPINEL_CMD_PROP_VALUE_IS && Key == SPINEL_PROP_LAST_STATUS)
291  {
292  spinel_status_t spinel_status = SPINEL_STATUS_OK;
293  spinel_ssize_t packed_len = spinel_datatype_unpack(Data, DataLength, "i", &spinel_status);
294  if (packed_len < 0 || (ULONG)packed_len > DataLength)
295  {
296  status = STATUS_INSUFFICIENT_RESOURCES;
297  }
298  else
299  {
300  status = ThreadErrorToNtstatus(SpinelStatusToThreadError(spinel_status));
301  }
302  }
303  else if (CmdContext->Handler)
304  {
305  status = CmdContext->Handler(Key, Data, DataLength, OutBuffer, &OutBufferLength);
306  }
307  else // No handler, so no output
308  {
309  status = STATUS_SUCCESS;
310  OutBufferLength = 0;
311  }
312 
313  // Clear any leftover output buffer
314  if (OutBufferLength < OrigOutBufferLength)
315  {
316  RtlZeroMemory((PUCHAR)OutBuffer + OutBufferLength, OrigOutBufferLength - OutBufferLength);
317  }
318 
319  LogVerbose(DRIVER_IOCTL, "Completing Irp=%p, with %!STATUS! for %s (Out:%u)",
320  CmdContext->Irp, status, IoCtlString(IoControlCode), OutBufferLength);
321 
322  // Complete the IRP
323  CmdContext->Irp->IoStatus.Information = OutBufferLength;
324  CmdContext->Irp->IoStatus.Status = status;
325  IoCompleteRequest(CmdContext->Irp, IO_NO_INCREMENT);
326 
327  FILTER_FREE_MEM(Context);
328 }
