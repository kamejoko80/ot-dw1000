87 {
88  ThreadError error = kThreadError_None;
89  Router *routers;
90  Child *children;
91  uint8_t num;
92 
93  VerifyOrExit(aKeyLength <= sizeof(mMasterKey), error = kThreadError_InvalidArgs);
94  VerifyOrExit((mMasterKeyLength != aKeyLength) || (memcmp(mMasterKey, aKey, aKeyLength) != 0), ;);
95 
96  memcpy(mMasterKey, aKey, aKeyLength);

98  mKeySequence = 0;

100 
101  // reset parent frame counters
102  routers = mNetif.GetMle().GetParent();
103  routers->mKeySequence = 0;
104  routers->mValid.mLinkFrameCounter = 0;
105  routers->mValid.mMleFrameCounter = 0;

109

112  routers[i].mKeySequence = 0;

119 
120  for (uint8_t i = 0; i < num; i++)
121  {
122  children[i].mKeySequence = 0;
123  children[i].mValid.mLinkFrameCounter = 0;
124  children[i].mValid.mMleFrameCounter = 0;

133 ThreadError KeyManager::ComputeKey(uint32_t aKeySequence, uint8_t *aKey)
134 {
135  Crypto::HmacSha256 hmac;
136  uint8_t keySequenceBytes[4];
137 
138  hmac.Start(mMasterKey, mMasterKeyLength);
139 
140  keySequenceBytes[0] = (aKeySequence >> 24) & 0xff;
141  keySequenceBytes[1] = (aKeySequence >> 16) & 0xff;
142  keySequenceBytes[2] = (aKeySequence >> 8) & 0xff;
143  keySequenceBytes[3] = aKeySequence & 0xff;
144  hmac.Update(keySequenceBytes, sizeof(keySequenceBytes));
145  hmac.Update(kThreadString, sizeof(kThreadString));
146 
147  hmac.Finish(aKey);
148 
149  return kThreadError_None;
150 }

157 void KeyManager::SetCurrentKeySequence(uint32_t aKeySequence)
158 {
159  if (aKeySequence == mKeySequence)
160  {
161  ExitNow();
162  }
163 
164  // Check if the guard timer has expired if key rotation is requested.
165  if ((aKeySequence == (mKeySequence + 1)) &&
166  (mKeySwitchGuardTime != 0) &&
167  mKeyRotationTimer.IsRunning() &&
168  mKeySwitchGuardEnabled)
169  {
170  uint32_t now = Timer::GetNow();
171  uint32_t guardStartTimestamp = mKeyRotationTimer.Gett0();
172  uint32_t guardEndTimestamp = guardStartTimestamp + Timer::HoursToMsec(mKeySwitchGuardTime);
173 
174  // Check for timer overflow
175  if (guardEndTimestamp < mKeyRotationTimer.Gett0())
176  {
177  if ((now > guardStartTimestamp) || (now < guardEndTimestamp))
178  {
179  ExitNow();
180  }
181  }
182  else
183  {
184  if ((now > guardStartTimestamp) && (now < guardEndTimestamp))
185  {
186  ExitNow();
187  }
188  }
189  }
190 
191  mKeySequence = aKeySequence;
192  ComputeKey(mKeySequence, mKey);
193 
194  mMacFrameCounter = 0;
195  mMleFrameCounter = 0;
196 
197  if (mKeyRotationTimer.IsRunning())
198  {
199  mKeySwitchGuardEnabled = true;
200  mKeyRotationTimer.Start(Timer::HoursToMsec(mKeyRotationTime));
201  }
202 
203  mNetif.SetStateChangedFlags(OT_NET_KEY_SEQUENCE_COUNTER);
204 
205 exit:
206  return;
207 }
